CREATE OR REPLACE PACKAGE BODY JAVA_CREATECUSTOMERGROUP
AS
  -- 返回结果集为
    -- 卡号
  PROCEDURE CREATECUSTOMERGROUP
    (
      rc OUT RETCURSOR
    )
  AS
  BEGIN
    OPEN rc FOR
      select '111111','HYFZ','000000','' from dual
      union
      select '222222','HYFZ','000000','' from dual
      union
      select '333333','HYFZ','111111','' from dual;
  END CREATECUSTOMERGROUP;

END JAVA_CREATECUSTOMERGROUP;
