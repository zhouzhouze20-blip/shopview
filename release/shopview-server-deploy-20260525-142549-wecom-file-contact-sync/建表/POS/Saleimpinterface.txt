CREATE OR REPLACE Package Body Saleimpinterface Is

  b_Mkt Varchar2(20);

  b_Jygs Varchar2(20);

  --商品匹配模式：0-按导入商品编码，与系统中商品编码匹配
  --              1-自动寻找柜组下有效商品，优先取服务费编码，然后取租赁和联营
  b_Goods_Mode Varchar2(60);

  Function Fgetsyjid(v_Supid In Varchar2) Return Varchar2 As
  Begin
    Return '0000';
  End;

  Procedure Raiseerr(Msg In Varchar2) As
  Begin
    Rollback;
    Raise_Application_Error(Errcode, Msg);
  End;

  Function Fgetdjlb(v_Type In Varchar2) Return Varchar2 As
    l_Type Varchar2(1);
  Begin
    Select Decode(v_Type, '1', '1', '2', '2', '3', '4', '4', '5', '0')
    Into   l_Type
    From   Dual;
    Return l_Type;
  End;

  Function Fgetpaymodeconver(Vcode In Varchar2) Return Char As
    Vnewcode Paymodeconver.Npmcode%Type;
  Begin
    Select Npmcode Into Vnewcode From Paymodeconver Where Opmcode = Vcode;
    Return Vnewcode;
  Exception
    When Others Then
      Return Vnewcode;
  End;

  /*检查档案数据 商铺，商品，租户，商品是否匹配等*/
  Procedure Checkisdoc
  (
    v_Mfid    In Varchar2,
    v_Supid   In Varchar2,
    v_Mkt     In Varchar2,
    v_Goodsid In Out Varchar2
  ) As
    l_Mkt       Varchar2(20);
    l_Sup       Varchar2(30);
    l_Gbid      Varchar2(30);
    l_Exists    Char(1);
    l_Goodspara Char(1); -- 商品校验参数  0 -校验商品 1-自动查找商品,不做校验
  Begin
    l_Goodspara := Nvl(Fgetprivatepara('I8'), '0');
    Begin
      Select Mffcode Into l_Mkt From Manaframe Where Mfcode = v_Mfid;
    Exception
      When Others Then
        Raiseerr('C.1->铺位[' || v_Mfid || ']在系统中未定义,请检查!');
    End;
    If Nvl(v_Mkt, '#') <> '#' Then
      If l_Mkt <> v_Mkt Then
        Raiseerr('C.2->铺位[' || v_Mfid || ']对应的门店[' || l_Mkt || ']和导入的门店[' ||
                 v_Mkt || ']不一致,请检查!');
      End If;
    End If;
    If Nvl(v_Supid, '#') <> '#' Then
      Begin
        Select Sbid Into l_Sup From Supplierbase Where Sbid = v_Supid;
      Exception
        When Others Then
          Raiseerr('C.3->租户代码[' || v_Supid || ']已不存在,请检查!');
      End;
    End If;
    /*检查商品,需要和商铺，供应商，一起检查*/
    If Nvl(v_Goodsid, '#') <> '#' Then
      If l_Goodspara = '0' Then
        Begin
          Select Gbid, Gbsupid
          Into   l_Gbid, l_Sup
          From   Goodsbase
          Where  (Gbid = v_Goodsid Or Gbbarcode = v_Goodsid);
        Exception
          When Others Then
            Raiseerr('C.4->商品编码[' || v_Goodsid || ']不存在!');
        End;
        If l_Sup <> Nvl(v_Supid, '#') Then
          Raiseerr('C.5->商品编码[' || v_Goodsid || ']对应的租户[' || l_Sup ||
                   ']和当前导入的租户[' || v_Supid || ']不一致!');
        End If;
        Begin
          Select 'Y'
          Into   l_Exists
          From   Goodsmframe
          Where  Gmfgdid = l_Gbid And Gmfmfid = v_Mfid And Rownum = 1;
        Exception
          When Others Then
            Raiseerr('C.6->商品编码[' || v_Goodsid || ']没有铺位[' || v_Mfid ||
                     ']经营配置!');
        End;
      Else
        Begin
          Select Gbid
          Into   l_Gbid
          From   Goodsbase, Goodsmframe
          Where  Gbid = Gmfgdid And Gmfmfid = v_Mfid And Gmfsupid = v_Supid And
                 Rownum = 1;
        Exception
          When Others Then
            Raiseerr('C.7->没有找到商位[' || v_Mfid || ']租户[' || v_Supid ||
                     ']对应的商品编号!');
        End;
      End If;
      v_Goodsid := l_Gbid;
    End If;
  End;

  Procedure Checkpaycode(v_Code In Varchar2) As
    l_Paycode Varchar2(20);
  Begin
    Select Ppmcode
    Into   l_Paycode
    From   Pospaymode
    Where  Ppmcode = Fgetpaymodeconver(v_Code);
  Exception
    When Others Then
      Raiseerr('P.7->付款方式代码[||v_code||]未在系统中定义');
  End;

  /*检查导入的主表*/
  Procedure Checkimpmainrow_Ex(Vr In Ls_Sal_Main%Rowtype) As
    Lmsg        Varchar2(100);
    l_Payamount Number;
  Begin
    Lmsg := '序号[' || Vr.Impseq || ']店铺[' || Vr.Storeid || ']租户[' ||
            Vr.Posid || ']小票流水号[' || Vr.Billnum || ']';
    If Vr.Goodsnum < 0 Or Vr.Goodsprice < 0 Then
      Raiseerr(Lmsg || 'M.9数量/单价!');
    End If;
    If Abs(Vr.Goodsnum * Vr.Goodsprice - Vr.Amtbill) > 0.01 Then
      Raiseerr(Lmsg || 'M.10数量*单价与金额不符!');
    End If;
    Begin
      If Vr.Payamount01 <> 0 Then
        Checkpaycode('01');
      End If;
      If Vr.Payamount02 <> 0 Then
        Checkpaycode('02');
      End If;
      If Vr.Payamount03 <> 0 Then
        Checkpaycode('03');
      End If;
      If Vr.Payamount04 <> 0 Then
        Checkpaycode('04');
      End If;
      If Vr.Payamount05 <> 0 Then
        Checkpaycode('05');
      End If;
      If Vr.Payamount06 <> 0 Then
        Checkpaycode('06');
      End If;
    Exception
      When Others Then
        Raiseerr(Lmsg || Sqlerrm);
    End;
    l_Payamount := Vr.Payamount01 + Vr.Payamount02 + Vr.Payamount03 +
                   Vr.Payamount04 + Vr.Payamount05 + Vr.Payamount06;
    If l_Payamount <> Vr.Amtbill Then
      Raiseerr(Lmsg || 'M.11付款方式金额合计与总金额不符!');
    End If;
  End;

  Procedure Checkimpmainrow(Vr In Ls_Sal_Main%Rowtype) As
    l_Exists Ls_Sal_Main%Rowtype;
    Lmsg     Varchar2(200);
    l_Tmp    Varchar2(30);
  Begin
    /*检查重复*/
    Lmsg := '序号[' || Vr.Impseq || ']店铺[' || Vr.Storeid || ']租户[' ||
            Vr.Posid || ']小票流水号[' || Vr.Billnum || ']';
    Begin
      Select *
      Into   l_Exists
      From   Ls_Sal_Main
      Where  Impseq <> Vr.Impseq And Mktid = Vr.Mktid And
             Storeid = Vr.Storeid And Posid = Vr.Posid And
             Billnum = Vr.Billnum And Readflag In (0, 1);
    Exception
      When No_Data_Found Then
        Null;
      When Others Then
        Raiseerr(Lmsg || 'M.1->此行数据已经存在,不能再次导入!');
    End;
    If l_Exists.Impseq Is Not Null Then
      Raiseerr(Lmsg || 'M.2->此行数据已经存在,导入时间为[' || l_Exists.Impdate || ']!');
    End If;
    If Vr.Billtype Not In (1, 2, 3, 4) Then
      Raiseerr(Lmsg || 'M.5->小票交易类型[' || Vr.Billtype || ']错误,请检查!');
    End If;
    If Vr.Amtbill < 0 Or Vr.Amtdis < 0 Then
      Raiseerr(Lmsg || 'M.6->金额不能为负数,请检查!');
    End If;
    --如果明细行号为0，表示是单表导入,需要检查商品，付款方式等
    If Vr.Detailrow = -1 Then
      l_Tmp := Vr.Goodsid;
      Checkimpmainrow_Ex(Vr);
    Elsif Vr.Detailrow = 0 Or Vr.Payrow = 0 Then
      Raiseerr(Lmsg || 'M.8->必须填写明细行数和付款行数,请检查!');
      l_Tmp := '#';
    End If;
    Begin
      Checkisdoc(Vr.Storeid, Vr.Posid, Vr.Mktid, l_Tmp);
    Exception
      When Others Then
        Raiseerr(Lmsg || Sqlerrm);
    End;
    Update Ls_Sal_Main
    Set    Readflag = 0, Goodsid = Nvl(l_Tmp, '#')
    Where  Impseq = Vr.Impseq And Mktid = Vr.Mktid And Storeid = Vr.Storeid And
           Posid = Vr.Posid And Billnum = Vr.Billnum;
  End;

    /**/

  Procedure Checkimpdetrow(Vr In Ls_Sal_Detail%Rowtype) As
    Lmsg     Varchar2(200);
    l_Exists Ls_Sal_Detail%Rowtype;
    l_Gbid   Varchar2(30);
  Begin
    /*检查重复*/
    Lmsg := '序号[' || Vr.Impseq || ']店铺[' || Vr.Storeid || ']租户[' ||
            Vr.Posid || ']小票流水号[' || Vr.Billnum || ']明细标志[' ||
            Vr.Billdetailid || ' ]';
    Begin
      Select *
      Into   l_Exists
      From   Ls_Sal_Detail
      Where  Impseq <> Vr.Impseq And Storeid = Vr.Storeid And
             Posid = Vr.Posid And Billnum = Vr.Billnum And
             Billdetailid = Vr.Billdetailid And Readflag In (0, 1);
    Exception
      When No_Data_Found Then
        Null;
      When Others Then
        Raiseerr(Lmsg || 'D.1->此行数据已经存在,不能再次导入!');
    End;
    If l_Exists.Impseq Is Not Null Then
      Raiseerr(Lmsg || 'D.2->此行数据已经存在,导入时间为[' || l_Exists.Impdate || ']!');
    End If;
    l_Gbid := Vr.Goodscode;
    Begin
      Checkisdoc(Vr.Storeid, Vr.Posid, Vr.Mktid, l_Gbid);
    Exception
      When Others Then
        Raiseerr(Lmsg || Sqlerrm);
    End;
    If Vr.Goodsqty < 0 Or Vr.Goodsprice < 0 Or Vr.Amtdetail < 0 Then
      Raiseerr(Lmsg || 'D.4数量/单价/金额不能小于0!');
    End If;
    If Abs(Vr.Goodsprice * Vr.Goodsqty - Vr.Amtdetail) > 0.01 Then
      Raiseerr(Lmsg || 'D.3数量*单价与金额不符!');
    End If;
    Update Ls_Sal_Detail
    Set    Readflag = 0, Goodsid = l_Gbid
    Where  Impseq = Vr.Impseq And Storeid = Vr.Storeid And Posid = Vr.Posid And
           Billnum = Vr.Billnum And Billdetailid = Vr.Billdetailid;
  End;

  Procedure Checkimppayrow(Vr In Ls_Sal_Pay%Rowtype) As
    l_Exists    Char(1);
    Lmsg        Varchar2(200);
    l_Existsrow Ls_Sal_Pay%Rowtype;
  Begin
    /*检查重复*/
    Lmsg := '序号[' || Vr.Impseq || ']店铺[' || Vr.Storeid || ']租户[' ||
            Vr.Posid || ']小票流水号[' || Vr.Billnum || ']明细标志[' ||

      Where  Impseq <> Vr.Impseq And Storeid = Vr.Storeid And
             Posid = Vr.Posid And Billnum = Vr.Billnum And
             Paydetailid = Vr.Paydetailid And Readflag In (0, 1);
    Exception
      When No_Data_Found Then
        Null;
      When Others Then
        Raiseerr(Lmsg || 'P.1->此行数据已经存在,不能再次导入!');
    End;

      Raiseerr(Lmsg || 'P.2->此行数据已经存在,导入时间为[' || l_Existsrow.Impdate || ']!');

    Set    Readflag = 0
    Where  Impseq = Vr.Impseq And Storeid = Vr.Storeid And Posid = Vr.Posid And

  Procedure Sp_Checksalebill(Row_Main In Ls_Sal_Main%Rowtype) As
    Lmsg           Varchar2(200);
    Ln_Cnt         Number;
    Lsum_Amtdetail Number;
    Lsum_Detaildis Number;
    Lsum_Goodsqty  Number;
    Lsum_Payamt    Number;
  Begin
    Lmsg := '序号[' || Row_Main.Impseq || ']店铺[' || Row_Main.Storeid ||
            ']租户[' || Row_Main.Posid || ']小票流水号[' || Row_Main.Billnum || ']';
    Checkimpmainrow(Row_Main);
    /*检查明细*/
    If Row_Main.Detailrow = -1 Then
      Return;
    End If;
    Select Count(1), Sum(Nvl(Amtdetail, 0)), Sum(Nvl(Detaildis, 0)),
           Sum(Nvl(Goodsqty, 0))
    Into   Ln_Cnt, Lsum_Amtdetail, Lsum_Detaildis, Lsum_Goodsqty
    From   Ls_Sal_Detail
    Where  Storeid = Row_Main.Storeid And Posid = Row_Main.Posid And
           Billnum = Row_Main.Billnum And Readflag = 0;
    If Ln_Cnt = 0 Then
      Raiseerr(Lmsg || ':此小票明细行数为0!');
    End If;
    If Lsum_Amtdetail <> Row_Main.Amtbill Then
      Raiseerr(Lmsg || ':小票明细金额合计[' || Lsum_Amtdetail || ']与表头金额[' ||
               Row_Main.Amtbill || ']不相符');
    End If;
    If Lsum_Detaildis <> Row_Main.Amtdis Then
      Raiseerr(Lmsg || ':小票明细优惠金额合计[' || Lsum_Amtdetail || ']与表头优惠金额[' ||
               Row_Main.Amtdis || ']不相符');
    End If;
    /*检查明细商品*/
    For Vr In (Select Ls_Sal_Detail.*
               From   Ls_Sal_Detail
               Where  Storeid = Row_Main.Storeid And Posid = Row_Main.Posid And
                      Billnum = Row_Main.Billnum And Readflag = 0) Loop
      Checkimpdetrow(Vr);
    End Loop;
    /*检查付款方式*/
    Select Count(1), Sum(Nvl(Payamt, 0))
    Into   Ln_Cnt, Lsum_Payamt
    From   Ls_Sal_Pay
    Where  Storeid = Row_Main.Storeid And Posid = Row_Main.Posid And
           Billnum = Row_Main.Billnum And Readflag = 0;
    If Ln_Cnt = 0 Then
      Raiseerr(Lmsg || ':此小票付款行数为0!');
    End If;
    If Ln_Cnt <> Row_Main.Payrow Then
      Raiseerr(Lmsg || ':小票付款行数[' || Ln_Cnt || ']与表头付款行数[' ||
               Row_Main.Payrow || ']不相符');
    End If;
    If Lsum_Payamt <> Row_Main.Amtbill Then
      Raiseerr(Lmsg || ':小票付款金额合计[' || Lsum_Payamt || ']与表头金额[' ||
               Row_Main.Amtbill || ']不相符');
    End If;
    For Vr In (Select *
               From   Ls_Sal_Pay
               Where  Storeid = Row_Main.Storeid And Posid = Row_Main.Posid And
                      Billnum = Row_Main.Billnum And Readflag = 0) Loop
      Checkimppayrow(Vr);
    End Loop;
  End;

  Procedure Sp_Gensalebill
  (
    v_Seq     In Number,
    v_Mkt     In Varchar2,
    v_Storeid In Varchar2,
    v_Posid   In Varchar2,
    v_Billnum In Varchar2
  ) As
    Row_Main Ls_Sal_Main%Rowtype;
    Lmsg     Varchar2(200);
    l_Billno Number;
    l_Djlb   Varchar2(1);
    l_Row    Number;
  Begin
    Lmsg := '序号[' || v_Seq || ']店铺[' || v_Storeid || ']租户[' || v_Posid ||
            ']小票流水号[' || v_Billnum || ']';
    Begin
      Select *
      Into   Row_Main
      From   Ls_Sal_Main
      Where  Impseq = v_Seq And Mktid = v_Mkt And Storeid = v_Storeid And
             Posid = v_Posid And Billnum = v_Billnum
      For    Update;
    Exception
      When Others Then
        Raiseerr(Lmsg || '此小票不存在,请检查!');
    End;
    If Row_Main.Readflag <> 0 Then
      Raiseerr(Lmsg || '此小票已处理或未通过导入校验!');
    End If;
    Sp_Checksalebill(Row_Main);
    Begin
      l_Billno := Improtsaleinterface.Getsalebillno;
      l_Djlb   := Fgetdjlb(Row_Main.Billtype);
      Improtsaleinterface.Insertsalehead(l_Billno, v_Mkt, l_Djlb,
                                         Row_Main.Amtbill, Row_Main.Amtbill,
                                         '', 'import', Row_Main.Billdate,
                                         Fgetsyjid(Row_Main.Posid));
      If Row_Main.Detailrow = -1 Then
        /*插入明细*/
        Improtsaleinterface.Insertsalegoods(l_Billno, 1, v_Mkt,
                                            Row_Main.Storeid,
                                            Row_Main.Goodsid,
                                            Row_Main.Goodsnum,
                                            Row_Main.Goodsprice,
                                            Row_Main.Amtbill);
        l_Row := 0;
        If Row_Main.Payamount01 <> 0 Then
          l_Row := l_Row + 1;
          Improtsaleinterface.Insertsalepay(l_Billno, l_Row, v_Mkt,
                                            Fgetpaymodeconver('01'),
                                            Row_Main.Payamount01);
        End If;
        If Row_Main.Payamount02 <> 0 Then
          l_Row := l_Row + 1;
          Improtsaleinterface.Insertsalepay(l_Billno, l_Row, v_Mkt,
                                            Fgetpaymodeconver('02'),
                                            Row_Main.Payamount02);
        End If;
        If Row_Main.Payamount03 <> 0 Then
          l_Row := l_Row + 1;
          Improtsaleinterface.Insertsalepay(l_Billno, l_Row, v_Mkt,
                                            Fgetpaymodeconver('03'),
                                            Row_Main.Payamount03);
        End If;
        If Row_Main.Payamount04 <> 0 Then
          l_Row := l_Row + 1;
          Improtsaleinterface.Insertsalepay(l_Billno, l_Row, v_Mkt,
                                            Fgetpaymodeconver('04'),
                                            Row_Main.Payamount04);
        End If;
        If Row_Main.Payamount05 <> 0 Then
          l_Row := l_Row + 1;
          Improtsaleinterface.Insertsalepay(l_Billno, l_Row, v_Mkt,
                                            Fgetpaymodeconver('05'),
                                            Row_Main.Payamount05);
        End If;
        If Row_Main.Payamount06 <> 0 Then
          l_Row := l_Row + 1;
          Improtsaleinterface.Insertsalepay(l_Billno, l_Row, v_Mkt,
                                            Fgetpaymodeconver('06'),
                                            Row_Main.Payamount06);
        End If;
      Else
        /*插入明细*/
        l_Row := 0;
        For Vr In (Select *
                   From   Ls_Sal_Detail
                   Where  Storeid = Row_Main.Storeid And
                          Posid = Row_Main.Posid And
                          Billnum = Row_Main.Billnum and readflag = 0) Loop
          l_Row := l_Row + 1;
          Improtsaleinterface.Insertsalegoods(l_Billno, l_Row, v_Mkt,
                                              Vr.Storeid, Vr.Goodsid,
                                              Vr.Goodsqty, Vr.Goodsprice,
                                              Vr.Amtdetail);
        End Loop;
        /*插入付款方式*/
        l_Row := 0;
        For Vr In (Select *
                   From   Ls_Sal_Pay
                   Where  Storeid = Row_Main.Storeid And
                          Posid = Row_Main.Posid And
                          Billnum = Row_Main.Billnum and readflag = 0) Loop
          l_Row := l_Row + 1;
          Improtsaleinterface.Insertsalepay(l_Billno, l_Row, v_Mkt,
                                            Fgetpaymodeconver(Vr.Paytype),
                                            Vr.Payamt);
        End Loop;
      End If;
      Improtsaleinterface.Insertsaleok(l_Billno);
    Exception
      When Others Then
        Raiseerr(Lmsg || Sqlerrm);
    End;
    Update Ls_Sal_Main
    Set    Readflag = 1, Billno = l_Billno
    Where  Storeid = v_Storeid And Posid = v_Posid And Billnum = v_Billnum;
  End;

  Procedure Sp_Deal_Lssal As
    l_Err Varchar(2000);
  Begin
    For Vr In (Select *
               From   Ls_Sal_Main
               Where  Readflag = 0
               Order  By Storeid, Posid, Billnum) Loop
      Begin
        Sp_Gensalebill(Vr.Impseq, Vr.Mktid, Vr.Storeid, Vr.Posid,
                       Vr.Billnum);
        Insert Into Interfacelog
          (Ilseq, Ildate, Iltypeid, Ilmode, Ilkey, Ilmfid, Ilfilename)
          Select Seq_Interfacelog.Nextval, Sysdate, '1', '2',
                 '销售数据处理成功，序号[' || Vr.Impseq || ']商铺[' || Vr.Storeid ||
                  '],商户[' || Vr.Posid || '],流水号[' || Vr.Billnum || ']',
                 Vr.Storeid, Vr.Billmemo
          From   Dual;
        Commit;
      Exception
        When Others Then
          Rollback;
          l_Err := Substr(Sqlerrm, 1, 2000);
          Insert Into Interfacelog
            (Ilseq, Ildate, Iltypeid, Ilmode, Ilkey, Ilmfid, Ilfilename)
            Select Seq_Interfacelog.Nextval, Sysdate, '2', '2', l_Err,
                   Vr.Storeid, Vr.Billmemo
            From   Dual;
          Commit;
      End;
    End Loop;
  End;

Begin
  Select Fgetprivatepara('14'), Fgetprivatepara('1M'),

End Saleimpinterface;
