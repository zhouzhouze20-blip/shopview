CREATE OR REPLACE Package Body Ticketmana Is
  --审核单据  档期活动券分摊单,设置商品收券规则 C2605 C2103
  Procedure Bapprove_Ft_Sq(p_Billno In Char,
                           p_Person In Char,
                           p_Billid In Char,
                           p_Djlb   In Char) Is
    Vclass    Number(2);
    Row_h     Tktbgoodsfqhead%Rowtype;
    Row_h_Old Tktbgoodsfqhead%Rowtype;
    Row_d     Tktbgoodsfqdetail%Rowtype;
    v_Ppcode  Varchar2(20);
    v_Spz     Varchar2(20);
    i         Number;
    l_Ins     Number;
    Cursor Cur_d Is
      Select * From Tktbgoodsfqdetail Where Tbfdbillno = p_Billno;
  Begin
    --查找单据头
    Begin
      Select *
        Into Row_h
        From Tktbgoodsfqhead
       Where Tbfhbillno = p_Billno
         For Update;
    Exception
      When Others Then
        Raise_Application_Error(Errcode, p_Billno || '->1 未找到单据');
    End;
    --检查单据类型
    If p_Djlb Not In (Billtypeft, Billtypesq, BilltypeFKzkft, BilltypeFQzkft) Or p_Djlb <> Row_h.Tbfhdjlb Or
       p_Billid Not In (Billidft, Billidsq, BillidFKzkft, BillidFQzkft) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->2 单据类别错误,不能审核!');
    End If;
    --检查是否能审核
    If Row_h.Tbfhflag In (Flagcanceled, Flagapproved) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->3 单据已取消或已审核,不能修改!');
    End If;
    --检查经营公司
    Begin
      Select Muclass
        Into Vclass
        From Manaunit
       Where Mucode = Row_h.Tbfhmanaunit;
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->4 无效的经营公司[' ||
                                Row_h.Tbfhmanaunit || ']!');
    End;
    --通宵营业2010.01.15
    Row_h_Old := Row_h;
    If Row_h_Old.Tbfhendtime >= '24' Then
      l_Ins := 2;
    Else
      l_Ins := 1;
    End If;
    ----通宵营业2010.01.15
    Open Cur_d;
    Loop
      Fetch Cur_d
        Into Row_d;
      Exit When Cur_d%Notfound Or Cur_d%Notfound Is Null;
      --确定柜组的级次
      If Row_h.Tbfhmode = Modecomb Then
        If Row_d.Tbfdmfid <> 'ALL' Then
          Begin
            Select Mfclass
              Into Vclass
              From Manaframe
             Where Mfcode = Row_d.Tbfdmfid
               And Mfmanaunit = Row_h.Tbfhmanaunit;
          Exception
            When Others Then
              Raise_Application_Error(Errcode,
                                      p_Billno || '->4 无效的柜组或楼层[' ||
                                      Row_d.Tbfdmfid || ']!');
          End;
        End If;
        If Row_d.Tbfdcatid <> 'ALL' Then
          Begin
            Select Catclass
              Into Vclass
              From Goodscat
             Where Catcode = Row_d.Tbfdcatid
               And Catmanaunit = Row_h.Tbfhmanaunit;
          Exception
            When Others Then
              Raise_Application_Error(Errcode,
                                      p_Billno || '->5 无效的类别[' ||
                                      Row_d.Tbfdcatid || ']!');
          End;
        End If;
        If Row_d.Tbfdppcode <> 'ALL' Then
          Begin
            Select Ppcode
              Into v_Ppcode
              From Codepp
             Where Ppcode = Row_d.Tbfdppcode
               And Ppmanaunit = Row_h.Tbfhmanaunit;
          Exception
            When Others Then
              Raise_Application_Error(Errcode,
                                      p_Billno || '->5 无效的品牌[' ||
                                      Row_d.Tbfdppcode || ']!');
          End;
        End If;
        If Row_d.Tbfdrcode <> 'ALL' Then
          Begin
            Select Catcode
              Into v_Spz
              From Goodscat_Analyze
             Where Catcode = Row_d.Tbfdrcode
               And Jygs = Row_h.Tbfhmanaunit;
          Exception
            When Others Then
              Raise_Application_Error(Errcode,
                                      p_Billno || '->5 无效的商品组[' ||
                                      Row_d.Tbfdrcode || ']!');
          End;
        End If;
      End If;
      If p_Djlb In ( Billtypeft, BilltypeFKzkft, BilltypeFQzkft) Then
        --通宵营业2010.01.15
        For i In 1 .. l_Ins Loop
          Row_h := Row_h_Old;
          If l_Ins = 2 Then
            If i = 1 Then
              Row_h.Tbfhendtime := '2359';
            Else
              Row_h.Tbfhstartdate := Row_h.Tbfhstartdate + 1;
              Row_h.Tbfhenddate   := Row_h.Tbfhenddate + 1;
              Row_h.Tbfhstarttime := '0000';
              Row_h.Tbfhendtime   := Fgetrealtime(Row_h.Tbfhendtime);
            End If;
          End If;
          --通宵营业2010.01.15
          --审核券费用分摊定义单
          Begin
            Insert Into Tktgoodsfqft
              (Tgffseqno,
               Tgffpid,
               Tgffmode,
               Tgffbarcode,
               Tgffmfid,
               Tgffcatid,
               Tgffppcode,
               Tgfftype,
               Tgffrate,
               Tgffstartdate,
               Tgffenddate,
               Tgffstarttime,
               Tgffendtime,
               Tgffbillno,
               Tgffsupid,
               Tgffwmid,
               Tgffcalctype,
               Tgffspsx,
               Tgffstr1,
               Tgffstr2,
               Tgffmemo,
               TGFFNUM1)
            Values
              (Seq_Tktgoodsfqft.Nextval,
               --tgffseqno
               Row_h.Tbfhpid,
               --tgffpid
               Row_h.Tbfhmode,
               --tgffmode
               Decode(Row_h.Tbfhmode,
                      '1',
                      Row_d.Tbfdgdid,
                      Row_d.Tbfdbarcode),
               --tgffbarcode
               Decode(Row_h.Tbfhmode, '1', 'ALL', Row_d.Tbfdmfid),
               --tgffmfid
               Row_d.Tbfdcatid,
               --tgffcatid
               Row_d.Tbfdppcode,
               --tgffppcode
               --row_h.tbfhtype, --tgfftype 分摊方式(1-按销售额,2-按返券额,3-按收券额)
               Row_d.Tbfdtype,
               Row_d.Tbfdrate,
               --分摊比例(供应商承担比例)
               Row_h.Tbfhstartdate,
               --tgffstartdate
               Row_h.Tbfhenddate,
               --tgffenddate
               --'00:00', --tgffstarttime
               --'23:59', --tgffendtime
               Substr(Row_h.Tbfhstarttime, 1, 2) || ':' ||
               Substr(Row_h.Tbfhstarttime, 3, 4),
               --tgffstarttime
               Substr(Row_h.Tbfhendtime, 1, 2) || ':' ||
               Substr(Row_h.Tbfhendtime, 3, 4),
               --tgffendtime
               Row_h.Tbfhbillno,
               Row_d.Tbfdsupid,
               Row_d.Tbfdjyfs,
               Row_h.Tbfhcalctype,
               --计算方法
               Row_d.Tbfdspsx,
               --商品属性
               P_BillId,  --Billidft,
               decode(Row_d.Tbfdtype,
                      'A',
                      row_d.tbfdpaycode,
                      'B',
                      row_d.tbfdpaycode || ',' || row_d.tbfdqtype,
                      'C',
                      row_d.tbfdqtype,
                      ''),
               --单据类别
               Row_h.Tbfhmemo,
               Nvl(Row_h.Tbfhchar1, 1));
          End;
        End Loop;
      Elsif p_Djlb = Billtypesq Then
        For r_Tbyd In (Select *
                         From Tktbgoodsyqdetail
                        Where Tbfdbillno = p_Billno) Loop
          --审核商品收券规则
          If Row_h.Tbfhmode = Modesup Then
            --供应商
            For r_Sgs In (Select Sgsgdid
                            From Suppliergoods
                           Where Sgssupid = Row_d.Tbfdsupid
                             And Sgsmanaunit = Row_h.Tbfhmanaunit
                             And Sgsstatus In ('Y', 'C')) Loop
              Insert Into Tktgoodsyqrate
                (Tgyrseqno,
                 Tgyrpid,
                 --档期
                 Tgyrmode,
                 --设置模式 1商品 2组合 3供应商（转换为商品）
                 Tgyrbarcode,
                 --
                 Tgyrmfid,
                 Tgyrcatid,
                 Tgyrppcode,
                 Tgyrstartdate,
                 Tgyrenddate,
                 Tgyrstarttime,
                 Tgyrendtime,
                 --
                 Tgyrrate,
                 --invalid
                 Tgyrbillno,
                 --
                 Tgyrtcode,
                 --invalid
                 Tgyrtype,
                 --收券方式 6按金额
                 Tgyrqtype,
                 --券种
                 Tgyrtjje,
                 --条件金额 满金额
                 Tgyraq,
                 --收金额
                 Tgyrbq,
                 --invalid
                 Tgyrbillid,
                 Tgyrmanaunit,
                 Tgyrspsx,
                 Tgyzkmk)
              Values
                (Seq_Tktgoodsyqrate.Nextval,
                 --tgyrseqno
                 Row_h.Tbfhpid,
                 --tgyrpid
                 --row_h.tbfhmode, --tgyrmode
                 '1',
                 --设置模式 从3供应商 转换为1商品
                 --row_d.tbfdgdid, --tgyrbarcode
                 r_Sgs.Sgsgdid,
                 Decode(Row_h.Tbfhmode, '1', 'ALL', Row_d.Tbfdmfid),
                 --tgyrmfid
                 Decode(Row_h.Tbfhmode, '1', 'ALL', Row_d.Tbfdcatid),
                 --tbfdcatid
                 Decode(Row_h.Tbfhmode, '1', 'ALL', Row_d.Tbfdppcode),
                 --tbfdppcode
                 Row_h.Tbfhstartdate,
                 Row_h.Tbfhenddate,
                 '00:00',
                 '23:59',
                 0,
                 Row_h.Tbfhbillno,
                 '',
                 '6',
                 r_Tbyd.Tbyqtype,
                 --券种
                 r_Tbyd.Tbyqmje,
                 --条件金额 满金额
                 r_Tbyd.Tbyqsje,
                 --收金额
                 r_Tbyd.Tbyqnum2,
                 --最大可收券金额
                 p_Billid,
                 Row_h.Tbfhmanaunit,
                 Row_d.Tbfdspsx,
                 Row_h.Tbfhrate);
            End Loop;
          Elsif Row_h.Tbfhmode = Modecomb Then
            --组合
            For r_Mf In ( --柜组 与 商品组 通过manaframe.mfcatgroup 对应
                         Select Mfcode --3种组合（1.指定柜组指定商品组；2.全部商品组，指定柜组；3.指定商品组，全部柜组）
                           From Manaframe
                          Where (Mfcatgroup = Row_d.Tbfdrcode Or
                                (Nvl(Row_d.Tbfdrcode, 'ALL') = 'ALL' And
                                Row_d.Tbfdmfid <> 'ALL'))
                            And Mfmanaunit = Row_h.Tbfhmanaunit
                            And (Row_d.Tbfdmfid = Mfcode Or
                                Row_d.Tbfdmfid = 'ALL')
                            And Mfstatus = 'Y'
                         Union
                         Select 'ALL' Mfcode
                           From Dual --全部商品组，全部柜组
                          Where Nvl(Row_d.Tbfdrcode, 'ALL') = 'ALL'
                            And Row_d.Tbfdmfid = 'ALL') Loop
              Insert Into Tktgoodsyqrate
                (Tgyrseqno,
                 Tgyrpid,
                 --档期
                 Tgyrmode,
                 --设置模式 1商品 2组合 3供应商（转换为商品）
                 Tgyrbarcode,
                 --
                 Tgyrmfid,
                 Tgyrcatid,
                 Tgyrppcode,
                 Tgyrstartdate,
                 Tgyrenddate,
                 Tgyrstarttime,
                 Tgyrendtime,
                 --
                 Tgyrrate,
                 --invalid
                 Tgyrbillno,
                 --
                 Tgyrtcode,
                 --invalid
                 Tgyrtype,
                 --收券方式 6按金额
                 Tgyrqtype,
                 --券种
                 Tgyrtjje,
                 --条件金额 满金额
                 Tgyraq,
                 --收金额
                 Tgyrbq,
                 --invalid
                 Tgyrbillid,
                 Tgyrmanaunit,
                 Tgyrspsx,
                 Tgyzkmk)
              Values
                (Seq_Tktgoodsyqrate.Nextval,
                 --tgyrseqno
                 Row_h.Tbfhpid,
                 --tgyrpid
                 Row_h.Tbfhmode,
                 Row_d.Tbfdgdid,
                 r_Mf.Mfcode,
                 --tgyrmfid
                 Row_d.Tbfdcatid,
                 --tbfdcatid
                 Row_d.Tbfdppcode,
                 --tbfdppcode
                 Row_h.Tbfhstartdate,
                 Row_h.Tbfhenddate,
                 '00:00',
                 '23:59',
                 0,
                 Row_h.Tbfhbillno,
                 '',
                 '6',
                 r_Tbyd.Tbyqtype,
                 --券种
                 r_Tbyd.Tbyqmje,
                 --条件金额 满金额
                 r_Tbyd.Tbyqsje,
                 --收金额
                 r_Tbyd.Tbyqnum2,
                 --最大可收券金额
                 p_Billid,
                 Row_h.Tbfhmanaunit,
                 Row_d.Tbfdspsx,
                 Row_h.Tbfhrate);
            End Loop;
          Elsif Row_h.Tbfhmode = Modegoods Then
            --商品
            Insert Into Tktgoodsyqrate
              (Tgyrseqno,
               Tgyrpid,
               --档期
               Tgyrmode,
               --设置模式 1商品 2组合 3供应商（转换为商品）
               Tgyrbarcode,
               --
               Tgyrmfid,
               Tgyrcatid,
               Tgyrppcode,
               Tgyrstartdate,
               Tgyrenddate,
               Tgyrstarttime,
               Tgyrendtime,
               --
               Tgyrrate,
               --invalid
               Tgyrbillno,
               --
               Tgyrtcode,
               --invalid
               Tgyrtype,
               --收券方式 6按金额
               Tgyrqtype,
               --券种
               Tgyrtjje,
               --条件金额 满金额
               Tgyraq,
               --收金额
               Tgyrbq,
               --invalid
               Tgyrbillid,
               Tgyrmanaunit,
               tgyrspsx,
               Tgyzkmk)
            Values
              (Seq_Tktgoodsyqrate.Nextval,
               --tgyrseqno
               Row_h.Tbfhpid,
               --tgyrpid
               Row_h.Tbfhmode,
               --tgyrmode
               Row_d.Tbfdgdid,
               --tgyrbarcode
               Decode(Row_h.Tbfhmode, '1', 'ALL', Row_d.Tbfdmfid),
               --tgyrmfid
               Decode(Row_h.Tbfhmode, '1', 'ALL', Row_d.Tbfdcatid),
               --tbfdcatid
               Decode(Row_h.Tbfhmode, '1', 'ALL', Row_d.Tbfdppcode),
               --tbfdppcode
               Row_h.Tbfhstartdate,
               Row_h.Tbfhenddate,
               '00:00',
               '23:59',
               0,
               Row_h.Tbfhbillno,
               '',
               '6',
               r_Tbyd.Tbyqtype,
               --券种
               r_Tbyd.Tbyqmje,
               --条件金额 满金额
               r_Tbyd.Tbyqsje,
               --收金额
               r_Tbyd.Tbyqnum2,

               Row_h.Tbfhmanaunit,

                              p_Billid,
                              '''' || p_Billno || '''' || ',' || '''' ||
                              p_Billid || '''');
    End If;
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End;

    Row_d     Tktbgoodsfqdetail%Rowtype;

           --tgffstartdate

           --tgffstarttime

  Procedure Bapprove_Adj(p_Billno In Char,
                         p_Person In Char,
                         p_Billid In Char,
                         p_Djlb   In Char) Is
    Tch       Tktchangehead%Rowtype;
    Row_Fq    Tktcardfqtotal%Rowtype;
    Type_Area Tktpopinfo.Tpcertarea%Type;
  
    v_Count    Number;
    l_Vipseqno Number;
    l_qno      varchar2(20);
  Begin
    Begin
      Select * Into Tch From Tktchangehead Where Tchbillno = p_Billno;
    
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, p_Billno || '->1 单据不存在');
    End;
  
    --券转移
    If Tch.Tchoplb = Opflagtrans Then
      If Tch.Tchtrscdmno Is Null Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->Trans-1 转出卡号不能为空');
      End If;
    
      Select Count(*)
        Into v_Count
        From Cardmain
       Where Cdmno = Tch.Tchtrscdmno;
      If v_Count <= 0 Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->Trans-2  转出卡无效，请核实卡号是否正确!');
      End If;
    
      For Vr In (Select *
                   From Tktchangedetail
                  Where Tcdchecked = 'Y'
                    And Tcdbillno = p_Billno) Loop
      
        Begin
          Select *
            Into Row_Fq
            From Tktcardfqtotal
           Where Tktcardfqtotal.Tcftseqno = Vr.Tcdseqno;
        Exception
          When No_Data_Found Then
            Raise_Application_Error(Errcode,
                                    Vr.Tcdseqno || 'Trans>3 批号券不存在');
        End;
      
        Begin
          Select Tpcertarea
            Into Type_Area
            From Tktpopinfo
           Where Tpiid = Vr.Tcdytpid;
          If Type_Area Is Null And Vr.Tcdytpid <> '0' Then
            Raise_Application_Error(Errcode,
                                    p_Billno || '->Trans-4  活动档期定义错误!');
          End If;
        Exception
          When Others Then
            If Vr.Tcdytpid = '0' Then
              Type_Area := '0';
            Else
              Raise_Application_Error(Errcode,
                                      p_Billno || '->Trans-4  活动档期定义错误!');
            End If;
        End;

                              p_Billno || '->1.1  该小票类别为冲红销售，不能进行返券处理!');

      Raise_Application_Error(Errcode,
                              p_Billno || '->1.2  该小票类别不属于销售，不能进行返券处理!');

      Into v_Count
      From Tktsellrtndet_Ex
     Where Tsrdbillno = p_Billno;

      Raise_Application_Error(Errcode, p_Billno || '->2  单据无有效记录!');

                                p_Billno || '第<' || To_Char(Vr.Tsderow) ||

         Tcflmkt,

         Tcflsource,
         Tcfltype,
         Tcflchr3,
         Tcflnum1,
         Tcflywdh)
      Values
        (Seq_Tktcardfqlog.Nextval,
         l_Vipseqno,
         Tcft.Tcftvipno,
         Vr.Tsdestartdate,
         Vr.Tsdeenddate,
         'B', --券补录
         Tsrh.Tsrhtpid,
         Null,
         Sysdate,
         Vr.Tsdeqmoney,
         Vr.Tsdeqmoney,
         p_Person,
         Null,
         Tsrh.Tsrhsyjid,
         Tsrh.Tsrhinvno,
         Null,
         Vr.Tsdeqtype,
         Vr.Tsdeyfqmkt,
         Vr.Tsdeyjygs,
         '1', --1:销售返券

      /*           SELECT TSRDGDID INTO v_gdid

      WHERE TSRDBILLNO = vr.TSRDBILLNO AND
            TSRDROWNO = vr.TSDEROW;
      
      SELECT TSRDBILLROW INTO v_billrow
      FROM TKTSELLRTNDETAIL
      WHERE TSRDBILLNO = p_billno AND
            TSRDROWNO = vr.TSDEROW;*/
    
      i := i + 1;
      Insert Into Tktsellfqamt
        (Tsfabillno,
         Tsfarowno,
         Tsfapggdid,
         Tsfaseq,
         Tsfatransno,

         Set Tsdeseqno = l_Vipseqno

  Procedure Bapprove_Adjbatch(p_Billno In Char,
                              p_Person In Char,
                              p_Billid In Char,
                              p_Djlb   In Char) Is
    Hd         Tktchangehead%Rowtype;
    Det        Tktchangedetail%Rowtype;
    Row_Fq     Tktcardfqtotal%Rowtype;
    Type_Area  Tktpopinfo.Tpcertarea%Type;
    v_Count    Number;
    l_Vipseqno Number;
    l_Curqje   Number;
    l_Totje    Number;
    l_Oriqje   Number;
    Ls_Market  Varchar2(20);
  Begin
    Begin
      Select * Into Hd From Tktchangehead Where Tchbillno = p_Billno;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, p_Billno || '->1 单据不存在');
    End;
    If Hd.Tchflag <> 'N' Then
      Raise_Application_Error(Errcode, p_Billno || '->1.1 单据已审核');
    End If;
    If Hd.Tchcdmno Is Null Or Hd.Tchtrscdmno Is Null Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->1.2 开始卡号和结束卡号必须输入！');
    End If;
    If Length(Nvl(Hd.Tchcdmno, '#')) <> Length(Nvl(Hd.Tchtrscdmno, '#')) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->1.3 开始卡号和结束卡号位数不一致，不能审核！');
    End If;
    Ls_Market := Hd.Tchyfqmkt;
    Begin
      Select Count(*)
        Into v_Count
        From Cardmain
       Where Cdmtype Like
             Decode(Hd.Tchperson1, 'ALL', '%', '@1', '%', Hd.Tchperson1)
         And Cdmstatus = 'Y'
         And (Cdmmkt = Ls_Market Or hd.Tchoplb = Opflagadj) --新增券时才考虑卡所属门店
         And Cdmno >= Nvl(Hd.Tchcdmno, '0')
         And Cdmno <= Nvl(Hd.Tchtrscdmno, '99999999999999999999');
      If v_Count <= 0 Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->2 没有符合条件的会员卡!');
      End If;
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->2 查找会员卡失败：' || Sqlerrm);
    End;
    --券调整
    If Hd.Tchoplb = Opflagadj Then
      For Cc In (Select *
                   From Cardmain
                  Where Cdmtype Like Decode(Hd.Tchperson1,
                                            'ALL',
                                            '%',
                                            '@1',
                                            '%',
                                            Hd.Tchperson1)
                    And Cdmstatus = 'Y'
                       /*And CdmMkt = ls_Market*/
                    And Cdmno >= Nvl(Hd.Tchcdmno, '0')
                    And Cdmno <= Nvl(Hd.Tchtrscdmno, '99999999999999999999')) Loop
        l_Totje := 0;
        For Det In (Select * From Tktchangedetail Where Tcdbillno = p_Billno) Loop
          l_Totje := 0;
          If Det.Tcdytpid = 'ALL' Then
            Type_Area := '0';
          Else
            Begin
              Select Tpcertarea
                Into Type_Area
                From Tktpopinfo
               Where Tpiid = Det.Tcdytpid;
              If Type_Area Is Null Then
                Raise_Application_Error(Errcode,
                                        p_Billno || '->Adj-1  活动档期定义错误!');
              End If;
            Exception
              When Others Then
                If Det.Tcdytpid = '0' Then
                  Type_Area := '0';
                Else
                  Raise_Application_Error(Errcode,
                                          p_Billno ||
                                          '->Trans-4  活动档期定义错误!');
                End If;
            End;
          End If;
          Select Sum(Tcftfqje)
            Into l_Oriqje
            From Tktcardfqtotal
           Where Tktcardfqtotal.Tcftvipno = Cc.Cdmno
             And Tktcardfqtotal.Tcftqtype Like
                 Decode(Det.Tcdqtype, '@', '%', Det.Tcdqtype)
             And Tktcardfqtotal.Tcftjygs = Hd.Tchyjygs
             And Tktcardfqtotal.Tcftfqmkt = Hd.Tchyfqmkt
             And Decode(Det.Tcdytpid,
                        'ALL',
                        'ALL',
                        Tktcardfqtotal.Tcftfstpid) = Det.Tcdytpid
             And Tcftstartdate >= Det.Tcdstartdate
             And Tcftenddate <= Det.Tcdenddate
             And Tktcardfqtotal.Tcftfqrule Like Nvl(Det.Tcdyfqrule, '%');
          /*
          --忽略无券批次的卡
          If l_curqje Is Null Then
             raise_application_error(errcode,
                                     p_billno || '->Adj-1 卡[' || cc.cdmno ||
                                     ']没有符合条件的券批次，不能通过审核!');
          End If;
          */
          --tchperson2   1-增减券金额            2-增减到券金额
          If Hd.Tchperson2 = '1' Then
            If Det.Tcdadjmoney + Nvl(l_Oriqje, 0) < 0 Then
              Raise_Application_Error(Errcode,
                                      p_Billno ||
                                      '->Adj-1 调整后券金额为负数，不能通过审核!');
            End If;
            l_Oriqje := Det.Tcdadjmoney;
          Else
            l_Oriqje := Det.Tcdadjmoney - l_Oriqje;
            If Det.Tcdadjmoney < 0 Then
              Raise_Application_Error(Errcode,
                                      p_Billno ||
                                      '->Adj-1 调整后券金额为负数，不能通过审核!');
            End If;
          End If;
          For Row_Fq In (Select *
                           From Tktcardfqtotal
                          Where Tktcardfqtotal.Tcftvipno = Cc.Cdmno
                            And Tktcardfqtotal.Tcftqtype Like
                                Decode(Det.Tcdqtype, '@', '%', Det.Tcdqtype)
                            And Tktcardfqtotal.Tcftjygs = Hd.Tchyjygs
                            And Tktcardfqtotal.Tcftfqmkt = Hd.Tchyfqmkt
                            And Decode(Det.Tcdytpid,
                                       'ALL',
                                       'ALL',
                                       Tktcardfqtotal.Tcftfstpid) =
                                Det.Tcdytpid
                            And Tcftstartdate >= Det.Tcdstartdate
                            And Tcftenddate <= Det.Tcdenddate
                            And Tktcardfqtotal.Tcftfqrule Like
                                Nvl(Det.Tcdyfqrule, '%')
                          Order By Tcftenddate Desc, tcftseqno Desc) Loop
            --tchperson2   1-增减券金额            2-增减到券金额
            /*If Hd.tchperson2 = '1' Then
               If Det.Tcdadjmoney = 0 Then
                  Exit;
               End If;
               If Det.Tcdadjmoney > 0 Then
                  l_curqje        := det.tcdadjmoney;
                  Det.Tcdadjmoney := 0;
               Else
                  l_curqje        := -1 * Least(-1 * Det.Tcdadjmoney, row_fq.tcftfqje);
                  Det.Tcdadjmoney := Det.Tcdadjmoney - l_curqje;
               End If;
            Else
               If l_oriqje = 0 Then
                  Exit;
               End If;
               \*
               l_curqje        := Det.Tcdadjmoney - row_fq.tcftfqje;
               Det.Tcdadjmoney := 0;
               *\
            End If;*/
            If l_Oriqje > 0 Then
              l_Curqje := l_Oriqje;
              l_Oriqje := 0;
            Elsif l_Oriqje = 0 Then
              Exit;
            Else
              l_Curqje := -1 * Least(-1 * l_Oriqje, Row_Fq.Tcftfqje);
              l_Oriqje := l_Oriqje - l_Curqje;
            End If;
            Update Tktcardfqtotal
               Set Tcftfqje = Tcftfqje + l_Curqje,
                   Tcftflag = Decode(Sign(Tcftfqje + l_Curqje), 1, 'Y', 'N')
             Where Tcftseqno = Row_Fq.Tcftseqno;
            --记录日志
            Insert Into Tktcardfqlog
              (Tcflseqno,
               Tcflvipseq,
               Tcflvipno,
               Tcflstartdate,
               Tcflenddate,
               Tcflzy,
               Tcflpopid,
               Tcflpoprule,
               Tcfldate,
               Tcflmoney,
               Tcflye,
               Tcfloper,
               Tcflmemo,
               Tcflsyjid,
               Tcflinvno,
               Tcflsyjtrace,
               Tcfljetype,
               Tcflmkt,
               Tcfljygs,
               Tcflsource,
               Tcfltype,
               Tcflywdh)
            Values
              (Seq_Tktcardfqlog.Nextval,
               Row_Fq.Tcftseqno,
               Row_Fq.Tcftvipno,
               Row_Fq.Tcftstartdate,
               Row_Fq.Tcftenddate,
               'A',
               Det.Tcdytpid,
               Det.Tcdyfqrule,
               Sysdate,
               l_Curqje,
               Row_Fq.Tcftfqje + l_Curqje,
               p_Person,
               Null,
               Null,
               Null,
               Null,
               Row_Fq.Tcftqtype,
               Row_Fq.Tcftfqmkt,
               Row_Fq.Tcftjygs,
               '7',
               Row_Fq.Tcfttype,
               p_Billno);
            l_Totje := l_Totje + l_Curqje;
          End Loop;
          Update Tktchangedetail
             Set Tcdqmoney = Nvl(Tcdqmoney, 0) + Nvl(l_Totje, 0)
           Where Tcdbillno = Det.Tcdbillno
             And Tcdrowno = Det.Tcdrowno;
        End Loop;
      End Loop;
    End If;
    --券新增
    If Hd.Tchoplb = Opflagnew Then
      If Hd.Tchytpid Is Null Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->New-1  请输入活动档期!');
      End If;
      --获取活动档期券使用范围
      --如果是无档期的券新增，则为通用券。
      If Hd.Tchytpid <> '0' And Hd.Tchytpid <> 'ALL' Then
        Select Tpcertarea
          Into Type_Area
          From Tktpopinfo
         Where Tpiid = Hd.Tchytpid;
        If Type_Area Is Null Then
          Raise_Application_Error(Errcode,
                                  p_Billno || '->New-2  活动档期定义错误!');
        End If;
      Else
        Type_Area := '0';
      End If;
      Select Count(*)
        Into v_Count
        From Tktchangedetail
       Where Tcdbillno = p_Billno;
      If v_Count <= 0 Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->New-3  单据无有效记录!');
      End If;
      For Cc In (Select *
                   From Cardmain
                  Where Cdmtype Like Decode(Hd.Tchperson1,
                                            'ALL',
                                            '%',
                                            '@1',
                                            '%',
                                            Hd.Tchperson1)
                    And Cdmstatus = 'Y'
                    And Cdmmkt = Ls_Market
                    And Cdmno >= Nvl(Hd.Tchcdmno, '0')
                    And Cdmno <= Nvl(Hd.Tchtrscdmno, '99999999999999999999')) Loop
        For Vr In (Select * From Tktchangedetail Where Tcdbillno = p_Billno) Loop
          If Vr.Tcdadjmoney <= 0 Then
            Raise_Application_Error(Errcode,
                                    p_Billno || '->New-3  券金额必须大于0!');
          End If;
          Select Seq_Tktcardfqtotal.Nextval Into l_Vipseqno From Dual;
          --处理返券表
          Insert Into Tktcardfqtotal
            (Tcftseqno,
             Tcftvipno,
             Tcftstartdate,
             Tcftenddate,
             Tcftqtype,
             Tcftfqje,
             Tcftstatus,
             Tcftvipexno,
             Tcftfstpid,
             Tcftcltpid,
             Tcftflag,
             Tcftransno,
             Tcfttype,
             Tcftmkt,
             Tcftjygs,
             Tcftfqmkt,
             Tcftfqrule,
             Tcftsource)
          Values
            (l_Vipseqno,
             Cc.Cdmno,
             Vr.Tcdstartdate,
             Vr.Tcdenddate,
             Vr.Tcdqtype,
             Vr.Tcdadjmoney,
             'Y',
             Cc.Cdmno,
             Vr.Tcdytpid,
             Vr.Tcdytpid,
             'Y',
             0,
             1,
             Type_Area,
             Hd.Tchyjygs,
             Hd.Tchyfqmkt,
             Nvl(Vr.Tcdyfqrule, '#'),
             '7'); --.后台手工新增
          --记录日志
          Insert Into Tktcardfqlog
            (Tcflseqno,
             Tcflvipseq,
             Tcflvipno,
             Tcflstartdate,
             Tcflenddate,
             Tcflzy,
             Tcflpopid,
             Tcflpoprule,
             Tcfldate,
             Tcflmoney,
             Tcflye,
             Tcfloper,
             Tcflmemo,
             Tcflsyjid,
             Tcflinvno,
             Tcflsyjtrace,
             Tcfljetype,
             Tcflmkt,
             Tcfljygs,
             Tcflsource,
             Tcfltype,
             Tcflywdh)
          Values
            (Seq_Tktcardfqlog.Nextval,
             l_Vipseqno,
             Cc.Cdmno,
             Vr.Tcdstartdate,
             Vr.Tcdenddate,
             'I',
             --后台券新增
             Vr.Tcdytpid,
             Nvl(Vr.Tcdyfqrule, '#'),
             Sysdate,
             Vr.Tcdadjmoney,
             Vr.Tcdadjmoney,
             p_Person,
             Null,
             Null,
             Null,
             Null,
             Vr.Tcdqtype,
             Hd.Tchyfqmkt,
             Hd.Tchyjygs,
             --'1',--1:销售返券
             '7',
             --后台手工新增
             '1',
             --电子券
             p_Billno);

          Update Tktchangedetail
             Set Tcdqmoney = Nvl(Tcdqmoney, 0) + Nvl(Vr.Tcdadjmoney, 0)
           Where Tcdbillno = Vr.Tcdbillno
             And Tcdrowno = Vr.Tcdrowno;
        End Loop;
      End Loop;
    End If;
    --更新返卷调整单主表
    Update Tktchangehead
       Set Tchflag = 'Y', Auditor = p_Person, Auditdate = Sysdate
     Where Tchbillno = p_Billno;
  Exception
    When Others Then
      Rollback;
      Raise;
  End Bapprove_Adjbatch;
  Procedure Approve(p_Billno In Char,
                    p_Person In Char,
                    p_Billid In Char,
                    p_Djlb   In Char) Is
  Begin
    --审核单据  档期活动券分摊单,设置商品收券规则 C2605 C2103
    If p_Djlb In (Billtypeft, Billtypesq, BilltypeFKzkft, BilltypeFQzkft) Then
      Bapprove_Ft_Sq(p_Billno, p_Person, p_Billid, p_Djlb);
    End If;
  
    If p_Djlb In (Billtypeadj) Then
      Bapprove_Adj(p_Billno, p_Person, p_Billid, p_Djlb);
    End If;
  
    If p_Djlb In (Billtypebl) Then
      Bapprove_Bl(p_Billno, p_Person, p_Billid, p_Djlb);
    End If;
  
    If p_Djlb In (Billtypebz) Then
      Bapprove_Bz(p_Billno, p_Person, p_Billid, p_Djlb);
    End If;
  
    If p_Djlb In (Billtypezf) Then
      Bapprove_Zf(p_Billno, p_Person, p_Billid, p_Djlb);
    End If;
  
    If p_Djlb In (Billtypexz) Then
      Bapprove_Sgxz(p_Billno, p_Person, p_Billid, p_Djlb);
    End If;
    If p_Djlb = Billtypepark Then

  --取消单据  档期活动券分摊单,设置商品收券规则 C2605 C2103
  Procedure Cancel(p_Billno In Char,
                   p_Person In Char,
                   p_Billid In Char,
                   p_Djlb   In Char) Is
    Row_h Tktbgoodsfqhead%Rowtype;
  Begin
    --查找单据头
    Begin
      Select * Into Row_h From Tktbgoodsfqhead Where Tbfhbillno = p_Billno;
    Exception
      When Others Then
        Raise_Application_Error(Errcode, p_Billno || '->1 未找到单据');
    End;
    --检查单据类型
    If p_Djlb Not In (Billtypeft, Billtypesq, Billtypepark, BilltypeFKzkft, BilltypeFQzkft) Or
       p_Djlb <> Row_h.Tbfhdjlb Or
       p_Billid Not In (Billidft, Billidsq, Billidpark, BillidFKzkft, BillidFQzkft) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->2 单据类别错误,不能审核!');
    End If;
    --检查是否能取消
    If Row_h.Tbfhflag = Flagediting Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->3 单据未审核不能取消,请使用删除功能！');
    End If;
    If Row_h.Tbfhflag <> Flagapproved Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->4 单据已取消,不能修改!');
    End If;
    --取消单据
    If p_Djlb = Billtypesq Then
      Delete From Tktgoodsyqrate Where Tgyrbillno = p_Billno;
      Cmdsend.Sp_Cmdsend_Bill('TKTGOODSYQRATECAN',
                              p_Billno,
                              p_Billid,
                              '''' || p_Billno || '''' || ',' || '''' ||
                              p_Person || '''' || ',' || '''' || p_Billid || '''' || ',' || '''' ||
                              p_Djlb || '''');
    Elsif p_Djlb In ( Billtypeft , BilltypeFKzkft , BilltypeFQzkft) Then
      Delete From Tktgoodsfqft Where Tgffbillno = p_Billno;
      Insert Into Mailtoerp_Cancelbill
      Values
        (p_Billno, p_Billid, Trunc(Sysdate), p_Person, '', '', 0, 0);
    Elsif p_Djlb = Billtypepark Then
      Delete From Tktgoodsparkft Where Tgffbillno = p_Billno;
    End If;
    --修改取消标志
    Update Tktbgoodsfqhead
       Set Tbfhflag = Flagcanceled, Tbhqxr = p_Person, Tbhqxrq = Sysdate
     Where Tbfhbillno = p_Billno;
    Update Tktbgoodsfqdetail
       Set Tbfdflag = Flagcanceled
     Where Tbfdbillno = p_Billno;
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End;
  --券批次可用性调整
  Procedure Dayadjust(v_Date In Date Default Trunc(Sysdate)) Is
  Begin
    Update Tktcardfqtotal Set Tcftflag = 'Y' Where Tcftflag = 'N';

                 From Tktpopinfo
                Where Tpistartdate <= v_Date
                  And Tpienddate > = v_Date

      End Loop;
    End Loop;
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End;

  --自动返券金额转移 C4202
  Procedure Moneytransfer(Vcardseq  In Number,
                          Vincardno In Varchar2,
                          p_Person  In Char,
                          Iscommit  In Char Default 'Y',
                          p_Billno  In Varchar2 Default '') Is
    Row_Fq     Tktcardfqtotal%Rowtype;
    l_Viptype  Varchar(4);
    l_Vipexno  Varchar2(20);
    l_Vipseqno Number;
    Faqye      Number;
    Fbqye      Number;
    Flqye      Number;
  Begin
    --查找返券批次
    Begin
      Select *
        Into Row_Fq
        From Tktcardfqtotal
       Where Tcftseqno = Vcardseq
         For Update;
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                To_Char(Vcardseq) || '->1 未找到返券批次');
    End;
    --
    If Row_Fq.Tcftvipno = Vincardno Then
      Raise_Application_Error(Errcode,
                              Vincardno || '->2 批次和转入的卡号相同');
    End If;
    --检查是否有效批次
    If Row_Fq.Tcftstatus <> 'Y' Then
      Raise_Application_Error(Errcode,
                              To_Char(Vcardseq) || '->2 不是有效的返券批次');
    End If;
    --检查转出的卡号是否返券卡
    Begin
      Select Cdmtype Into l_Viptype From Cardmain Where Cdmno = Vincardno;
      l_Vipexno := Vincardno;
    Exception
      When Others Then
        Raise_Application_Error(Errcode, Vincardno || '->3 找不到该VIP卡!');
    End;
  
    --转移返券金额
    l_Vipseqno := 0;
    Begin
      Select Tcftseqno, Tcftfqje
        Into l_Vipseqno, Faqye
        From Tktcardfqtotal
       Where Tcftvipno = Vincardno
         And Tcftstartdate = Row_Fq.Tcftstartdate
         And Tcftenddate = Row_Fq.Tcftenddate
         And Tcftstatus = 'Y'
         And Tcfttype = '1'
         And Tcftjygs = Row_Fq.Tcftjygs
         And Tcftfqmkt = Row_Fq.Tcftfstpid
         And Tcftfqrule = Row_Fq.Tcftfqrule
         And Tcftqtype = Row_Fq.Tcftqtype
         And Rownum = 1;
    Exception
      When Others Then
        Null;
    End;
  
    If l_Vipseqno > 0 Then
      Update Tktcardfqtotal
         Set Tcftfqje = Tcftfqje + Row_Fq.Tcftfqje
       Where Tcftseqno = l_Vipseqno;
    Else
      Select Seq_Tktcardfqtotal.Nextval Into l_Vipseqno From Dual;
      Insert Into Tktcardfqtotal
        (Tcftseqno,
         Tcftvipno,
         Tcftstartdate,
         Tcftenddate,
         Tcftqtype,
         Tcftfqje,
         Tcftstatus,
         Tcftvipexno,
         Tcftfstpid,
         Tcftcltpid,
         Tcftflag,
         Tcftransno,
