  ---如果返券规则中返券金额为0 那么规则单号填0 by zhouwd FOR SB 2009-3-6
  Function f_Getfqrule(Vfqrule In Varchar2) Return Varchar2 Is
    l_Fqrule Varchar2(20);
    l_Fqje   Number;
  Begin
    l_Fqrule := Vfqrule;
    If l_Fqrule Is Not Null Then
      Begin
        Select Nvl(Trfaqje, 0)
          Into l_Fqje
          From Tktrule
         Where Trcode = l_Fqrule
           And Trrulelb = '1';
      Exception
        When Others Then
          l_Fqrule := '';
      End;
      If l_Fqje = 0 Then
        l_Fqrule := '0';
      End If;
    End If;
    Return l_Fqrule;
  End;

  Function f_getpmzklist(p_billno in varchar2) return Varchar2 is
    l_zklist varchar2(40);
  Begin
    for vr in (select *
                 from tktrulefqdet
                where trdcode = p_billno
                order by trdseq) loop
      l_zklist := l_zklist || to_char(vr.trdnum1) || ':' ||
                  to_char(vr.trdnum2) || '|';
    end loop;
    return l_zklist;
  end;

  --检查券种是否可以删除
  Procedure Qtypedelcheck(i_Qtype  In Varchar2,
                          o_Ret    Out Number, --0可以删除 1不可删除
                          o_Retmsg Out Varchar2) Is
    l_Exist Number;
  Begin
    If i_Qtype In ('A', 'B', 'C', 'D', 'E', 'F', 'G') Then
      o_Ret    := 1;
      o_Retmsg := '预定义券种' || i_Qtype || '不能删除!';
      Return;
    End If;
    --券规则已参与定义（返券，用券，买券，银行追送）
    Begin
      Select 1
        Into l_Exist
        From Tktrulefqdet
       Where Trdqtype = i_Qtype
         And Rownum = 1;
      o_Ret    := 1;
      o_Retmsg := '券种' || i_Qtype || '已参与定义规则不能删除!';
      Return;
    Exception
      When Others Then
        Null;
    End;
    --有券余额
    Begin
      Select 1
        Into l_Exist
        From Tktcardfqtotal
       Where Trunc(Sysdate) Between Tcftstartdate And Tcftenddate
         And Tcftqtype = i_Qtype
         And Rownum = 1;
      o_Ret    := 1;
      o_Retmsg := '券种' || i_Qtype || '已有券余额不能删除!';
      Return;
    Exception
      When Others Then
        Null;
    End;
    o_Ret    := 0;
    o_Retmsg := '可以删除';
  End;

  --组合促销单审核 C3001
  Procedure Approve(p_Billno In Char,
                    p_Person In Char,
                    p_Billid In Char,
                    p_Djlb   In Char) Is
  Begin
    If p_Djlb In (Typecomppop) Then
      Bapprove_Combpop(p_Billno, p_Person, p_Billid, p_Djlb);
    Elsif p_Djlb In (Typestrrule) Then
      Approve_Rule(p_Billno, p_Person, p_Billid, p_Djlb);
    Elsif p_Djlb In (Typestrsl,
                     Typestampgd,
                     Typestampcat,
                     Typestamppp,
                     Typestampcatpp) Then
      Approve_Sl(p_Billno, p_Person, p_Billid, p_Djlb);
    Elsif p_djlb = Typepm THEN  --价随量变
      Bapprove_pm(p_Billno, p_Person, p_Billid, p_Djlb);
    End If;
  End;

  --满减，满赠规则定义审核 C2202 C2203
  Procedure Auditpoprule(p_Billno In Varchar2,
                         p_Person In Char,
                         p_Billid In Char,
                         p_Djlb   In Char) Is
    l_Rulecnt   Integer;
    Hd          Poprule%Rowtype;
    l_Grouprule Varchar2(400);
    l_Onerule   Varchar2(40); --满,减
    l_Maxmjtj   Number; --最大满减条件
    l_Maxmjje   Number; --最大满减结果
    l_Thismjtj  Number; --当前满减条件
    l_Thismjje  Number; --当前满减结果
    l_Row       Number; --行号
    l_Mjmk      Number := 0;
    l_Mjmk_Last Number := 0;
    l_Mkfw      Varchar2(40);
    l_Rowcount  Number;
    l_Lx        Varchar2(10);
  Begin
    If p_Djlb Not In (Typerulemj, Typerulemd, Typerulemz, Typerulemh,Typerulebz) Or
       p_Billid Not In
       (Billidrulemj, Billidrulemz, Billidrulemd, Billidrulemh,Billidrulebz) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->RULE01 单据类型错误,请检查!');
    End If;
    If p_Djlb = Typerulemj Or p_Djlb = Typerulemz or p_Djlb = Typerulebz Then
      l_Lx := '1';
    Elsif p_Djlb = Typerulemd Then
      l_Lx := '3';
    Elsif p_Djlb = Typerulemh Then
      l_Lx := '5';
    End If;
    If p_djlb = Typerulesq Then
      l_Lx := '4';
    End If;
   --
    Select Count(*) Into l_Rulecnt From Popruledet Where Prdid = p_Billno;
    If l_Rulecnt = 0 Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->RULE02 促销规则无任何明细,请检查!');
    End If;
    --降序排列满减规则
    If p_Djlb = Typerulemz or p_Djlb = Typerulebz Then
      Update Popruledet
         Set Prdnum1 =
             (Select Prlxfje
                From Poprulelevel
               Where Prdid = p_Billno
                 And Prdrowno = Prdlevel),
             Prdnum5 =
             (Select Prlmaxsl
                From Poprulelevel
               Where Prdid = p_Billno
                 And Prdrowno = Prdlevel),
             Prdflag1 = Hd.Prflag1
       Where Prdid = p_Billno;
    Else
      l_Row := 0;
      Select Count(*)
        Into l_Rowcount
        From Popruledet
       Where Prdid = p_Billno;
      For Vr In (Select *
                   From Popruledet
                  Where Prdid = p_Billno
                  Order By Prdrowno) Loop
        l_Row := l_Row + 1;
        --当前行
        l_Thismjtj := Vr.Prdnum1;
        l_Thismjje := Vr.Prdnum2;
        If l_Row = l_Rowcount Then
          l_Maxmjtj   := Vr.Prdnum1;
          l_Maxmjje   := Vr.Prdnum2;
          l_Mjmk_Last := '9999999';
        Else
          Select Prdmjmk
            Into l_Mjmk_Last
            From Popruledet
           Where Prdid = p_Billno
             And Prdrowno = l_Row + 1;
        End If;
        l_Mjmk      := Vr.Prdmjmk;
        l_Mkfw      := To_Char(l_Mjmk) || ',' || To_Char(l_Mjmk_Last);
        l_Onerule   := l_Mkfw || ',' || To_Char(l_Thismjtj) || ',' ||
                       To_Char(l_Thismjje);
        l_Grouprule := l_Grouprule || ';' || l_Onerule;
      End Loop;
      If Length(l_Grouprule) > 400 Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->RULE02 规则超过级次限制,请检查!');
      End If;
      Select * Into Hd From Poprule Where Prid = p_Billno For Update;
      If Hd.Prstatus <> Flagediting Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->RULE02 促销规则已经审核,请检查!');
      End If;
      If p_Djlb In (Typerulemj, Typerulemd, Typerulemh) Then
        -- 规则促销
        Insert Into Popcurinfo
          Select Seq_Popcurinfo.Nextval,
                 --POPSEQUECE
                 Poptitle,
                 Popmode,
                 Popbillno,
                 Popgdid,
                 Popbarcode,
                 Popuid,
                 Popsupid,
                 Popmarket,
                 Popmfid,
                 Popcatcode,
                 Popppcode,
                 Popsj,
                 Popsjrate,
                 Pophyj,
                 Pophyjrate,
                 Popksrq,
                 Popjsrq,
                 Popkssj,
                 Popjssj,
                 Popspace,
                 Popsl,
                 Popzkfd,
                 Popnewkl,
                 Popzpbarcode,
                 Popzpspecunit,
                 Popzsjs,
                 Popzpsl,
                 Popzpxl,
                 Popzpsj,
                 Poppfj,
                 Poppfjrate,
                 Poppfjzkfd,
                 Pophyjzkfd,
                 Popmanaunit,
                 '' Poprule1,
                 '' Poprule2,
                 '' Poprule3,
                 '' Poprule4,
                 '' Poprule5,
                 Popmemo,
                 '',
                 0,
                 popspsx
            From (Select Prid Poptitle,
                         --POPTITLE
                         '1' Popmode,
                         --POPMODE
                         p_Billno Popbillno,
                         --POPBILLNO
                         '1' Popgdid,
                         --POPGDID
                         '1' Popbarcode,
                         --POPBARCODE
                         '00' Popuid,
                         --POPUID
                         '' Popsupid,
                         --POPSUPID
                         Currentmkt Popmarket,
                         --POPMARKET
                         l_Lx Popmfid,
                         --POPMFID 促销方式:1-满减,2-减现
                         Prflag2 Popcatcode,
                         --POPCATCODE  是否先剔除a券
                         Prtype Popppcode,
                         -- POPPPCODE  是否跨柜组促销
                         l_Maxmjtj Popsj,
                         --POPSJ 满减条件金额
                         0 Popsjrate,
                         --POPSJRATE 减现比例(无用)
                         l_Maxmjje Pophyj,
                         --POPHYJ 满减折扣金额
                         0 Pophyjrate,
                         --POPHYJRATE
                         To_Date('2000-01-01', 'yyyy-mm-dd') Popksrq,
                         --POPKSRQ
                         To_Date('3000-01-01', 'yyyy-mm-dd') Popjsrq,
                         --POPJSRQ
                         '00:00' Popkssj,
                         --POPKSSJ
                         '23:59' Popjssj,
                         --POPJSSJ
                         1 Popspace,
                         -- 十位表示黄码或白码，个位表示折上折（1可以折上折，默认为1）
                         Nvl(Pnum1, -1) Popsl,
                         --POPSL
                         1 Popzkfd,
                         --POPZKFD
                         Null Popnewkl,
                         --POPNEWKL
                         'DMJ' Popzpbarcode,
                         --POPZPBARCODE
                         Null Popzpspecunit,
                         --POPZPSPECUNIT
                         Null Popzsjs,
                         --POPZSJS
                         Null Popzpsl,
                         --POPZPSL
                         Null Popzpxl,
                         --POPZPXL
                         Null Popzpsj,
                         --POPZPSJ
                         0 Poppfj,
                         --POPPFJ
                         0 Poppfjrate,
                         --POPPFJRATE
                         1 Poppfjzkfd,
                         --POPPFJZKFD
                         1 Pophyjzkfd,
                         --POPHYJZKFD
                         '0' Popmanaunit,
                         --0 表示全部经营公司
                         l_Grouprule Popmemo,
                         '0' popspsx
                    From Poprule a
                   Where Prid = p_Billno);
      End If;
    End If;
    --
    Update Poprule
       Set Prstatus = Flagapproved, Auditor = p_Person, Auditdate = Sysdate
     Where Prid = p_Billno;
    If Sql%Rowcount <> 1 Or Sql%Rowcount Is Null Then
      Raise_Application_Error(Errcode,
                              p_Billno || '-->RULE01 无效的促销规则ID号!');
    End If;
    Cmdsend.Sp_Cmdsend_Allmkt(Cmdpopruleupt, '''' || p_Billno || '''');
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End;

                           p_Billid In Char,
                           p_Djlb   In Char) Is

    Lxfje Number;
    Lqje  Number;
  Begin
    Begin
      Select * Into Hd From Tktrule Where Trcode = p_Billno For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, p_Billno || '->1 单据不存在');
    End;

       p_Djlb Not In (Typerulefq, Typerulesa, Typerulemq,Typerulepm) Or

                              p_Billno || '->3 单据已取消或已审核，不能审核！');

        Into Lxfje
        From Tktrulefqlever
       Where Trdcode = p_Billno
         And Trdseq = 1;
      Select Sum(Trdqje)
        Into Lqje
        From Tktrulefqdet_Ex
       Where Trdcode = p_Billno
         And Trdnum1 = 1;
      Update Tktrule
         Set Trxfje = Lxfje, Trfaqje = Lqje
       Where Trcode = p_Billno;

    --修改本地标志
    Update Tktrule
       Set Trflag    = Flagapproved,
           Auditdate = Sysdate,
           Auditor   = p_Person,

  Procedure p_Autocreatecode(In_Controlid In Varchar2,
                             Out_Code     Out Varchar2) Is
    Vcodeid Varchar2(3);
    r_Rule  Coderule%Rowtype;
    Vseqno  Number;
  Begin

    If In_Controlid = Controlid_Dp Then
      Vcodeid := 'D01';
    Elsif In_Controlid = Controlid_Cat Then
      Vcodeid := 'C01';
    Else
      Vcodeid := 'P01';
    End If;
    Begin
      Select * Into r_Rule From Coderule Where Codeid = Vcodeid For Update;
    Exception
      When Others Then
        Raise_Application_Error(Errcode, '没有找到对应的值');
    End;
    Vseqno := r_Rule.Seqno + 1;

    Out_Code := r_Rule.Pixcode ||
                Lpad(To_Char(Vseqno), r_Rule.Codelen, r_Rule.Bccode);
    Update Coderule Set Seqno = Vseqno Where Codeid = Vcodeid;
  End;
  Procedure p_01(p_Billno In Char,
                 p_Person In Char,
                 p_Billid In Char,
                 p_Djlb   In Char) Is
    r_Hd    Pop_Xlhead%Rowtype;
    r_Det   Pop_Xldetail%Rowtype;
    Vxlcode Varchar2(20);
    vgdid Varchar2(13);
    Vcount  Number;
    Vcount1 Number;
    Cursor Cur_Det Is
      Select * From Pop_Xldetail Where Sheetid = p_Billno Order By Rowno;
  Begin
    Begin
      Select *
        Into r_Hd
        From Pop_Xlhead
       Where Sheetid = p_Billno
         For Update;
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                '单据号[' || p_Billno || ']不存在，请检查!');
    End;

    If r_Hd.Flag Not In (Flagediting) Then
      Raise_Application_Error(Errcode, '单据状态不正确，请检查单据状态!');
    End If;

    If r_Hd.Controlid Is Null Or r_Hd.Xlname Is Null Then
      Raise_Application_Error(Errcode, '单据头存在非空项没有输入，请检查!');
    End If;

    p_Autocreatecode(r_Hd.Controlid, Vxlcode);
    --检查单据明细
    Select Count(Distinct Codeid)

  --发送促销规则
  Procedure Sendpopgift(p_Billno In Varchar2,
                        p_Person In Char,
                        p_Billid In Char,
                        p_Djlb   In Char) Is
    l_Rulecnt Integer;
    l_Ruleall Integer;
    r_Gift    Popgift%Rowtype;
    Vtmpvalue Number;
    Vcnt      Number;
    Vmaxjc    Number;
  Begin
    If p_Billid = Typeidrule Then
      --判断规则是否从小到大
      Begin
        Select *
          Into r_Gift
          From Popgift
         Where Ppgid = p_Billno
           For Update;
      Exception
        When Others Then
          Raise_Application_Error(Errcode, p_Billno || '-> 不存在');
      End;
      If Nvl(r_Gift.Ppgvalue, 0) > 0 Then
        Select Count(*)
          Into Vcnt
          From Popgiftrulevalue
         Where Pprgid = p_Billno;
        If Vcnt <= 0 Then
          Raise_Application_Error(Errcode, '级次不能为空，请检查');
        End If;
        Select Max(Pprjc)
          Into Vmaxjc
          From Popgiftrulevalue
         Where Pprgid = p_Billno;
        If Vmaxjc <> r_Gift.Ppgvalue Then
          Raise_Application_Error(Errcode, '明细级次与头级次不等，请检查');
        End If;
      End If;
      --大于1 的情况下，不允许条件类型不一致
      If Nvl(r_Gift.Ppgvalue, 0) > 1 Then
        Select Count(Distinct Pprstyle)
          Into Vcnt
          From Popgiftrule
         Where Pprgid = p_Billno;

        If Vcnt > 1 Then
          Raise_Application_Error(Errcode,
                                  '规则明细中存在条件类型不一致的情况，不允许多级');
        End If;
      End If;
      If r_Gift.Ppgtype = 'A' Then
        Vtmpvalue := 0;
        For r In (Select *
                    From Popgiftrulevalue
                   Where Pprgid = p_Billno
                   Order By Pprno) Loop
          If r.Pprvalue < Vtmpvalue Then
            Raise_Application_Error(Errcode,
                                    p_Billno || '-> 第[' || To_Char(r.Pprno) ||
                                    '小于上一行');
          End If;
          Vtmpvalue := r.Pprvalue;

          If r.Pprrowno <> '99' Then
            Raise_Application_Error(Errcode, '部分满足行号必须为99');
          End If;
        End Loop;
      Else
        For r In (Select *
                    From Popgiftrule
                   Where Pprgid = p_Billno
                   Order By Pprno) Loop
          Vtmpvalue := 0;
          Select Count(*)
            Into Vcnt
            From Popgiftrulevalue
           Where Pprgid = p_Billno
             And Pprrowno = 99;
          If Vcnt > 0 Then
            Raise_Application_Error(Errcode, '明细规则行号必须不为99');
          End If;
          For R1 In (Select *
                       From Popgiftrulevalue
                      Where Pprgid = p_Billno
                        And Pprrowno = r.Pprno
                      Order By Pprno) Loop
            If R1.Pprvalue < Vtmpvalue Then
              Raise_Application_Error(Errcode,
                                      p_Billno || '-> 第[' ||
                                      To_Char(R1.Pprno) || '小于上一行');
            End If;
            Vtmpvalue := R1.Pprvalue;
            If R1.Pprrowno = '99' Then
              Raise_Application_Error(Errcode, '部分满足行号必须不为99');
            End If;
          End Loop;
        End Loop;
      End If;
      Update Popgift
         Set Ppgstatus = 'Y',
             Ppgtrack2 = p_Person,
             Auditor   = p_Person,
             Auditdate = Sysdate
       Where Ppgid = p_Billno;
      If Sql%Rowcount <> 1 Or Sql%Rowcount Is Null Then
        Raise_Application_Error(Errcode,
                                p_Billno || '-->GIFT01 无效的促销规则ID号!');
      End If;
      Select Count(*), Sum(Decode(Pprtype, '0', 1, 0))
        Into l_Rulecnt, l_Ruleall
        From Popgiftrule
       Where Pprgid = p_Billno;
      If l_Rulecnt = 0 Then
        Raise_Application_Error(Errcode,
                                p_Billno || '->GIFT02 促销规则无任何明细,请检查!');
      Elsif l_Rulecnt > 1 And l_Ruleall > 0 Then
        Raise_Application_Error(Errcode,
                                p_Billno ||
                                '-->GIFT03 部分满足必须单独使用,且只能有一条部分满足规则!');
      End If;
      /*lErpKey := '''' || p_BillNo || '''';
      lPosKey := NULL;*/
      /*IF CurrentMktType = Stock.MF_HQ THEN --总部的规则发送到门店，门店的规则只能自己使用，不发送到总部
         GoodsSendCmd('POPGIFT', 'I', CurrentMkt, '%');
      ELSE --其他地方做规则发送本地管辖的地点
         FOR R IN ( SELECT DISTINCT DBRTMFID
                      FROM DBRELATION, V_MFRAME_AREA
                     WHERE DBRTMFID = MFFCODE
                       AND DBROMFID = CurrentMkt
                       AND DBRTMFID <> CurrentMkt
                       AND DBRTTYPE IN ('2', '3')
                       AND DBRSTATUS = 'Y'
                       AND MFSTATUS = 'Y'
                  )
         LOOP
            GoodsSendCmd('POPGIFT', 'I', CurrentMkt, R.DBRTMFID);
         END LOOP;
         GoodsSendCmd('POPGIFT$HQ', 'I', CurrentMkt, u_Constant.HQCode); --总是发送到总部
      END IF;*/
    End If;
  End Sendpopgift;

  Procedure Approve_Rule(p_Billno In Char,
                         p_Person In Char,
                         p_Billid In Char,
                         p_Djlb   In Char) Is
    b_Hd  Prcpophead%Rowtype;
    b_Det Prcpopdetail%Rowtype;
    i     Number;
    l_Cnt Number;
  Begin
    Begin
      Select *
        Into b_Hd
        From Prcpophead
       Where Pphseqno = p_Billno
         For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, p_Billno || '->1 单据不存在');
    End;
    If p_Djlb Is Not Null And b_Hd.Pphtype <> p_Djlb Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->2 传入单据类别[' || p_Djlb ||
                              ']与实际单据类别[' || b_Hd.Pphtype || ']不一致');
    End If;
    If b_Hd.Pphflag Not In (Flagediting, Flagsended) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->CHKH03 单据已经审核或单据状态不一致!');
    End If;
    If b_Hd.Pphflag In (Flagapproved, Flagstopped, Flagcanceled) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->CHKH04 单据已经审核或已经结束或已取消!');
    End If;
    If Ltrim(b_Hd.Pphtitle) Is Null Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->CHKH08 规则促销必须输入促销规则!');
    End If;
    Begin
      Select * Into r_Ppg From Popgift Where Ppgid = b_Hd.Pphtitle;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode,
                                b_Hd.Pphseqno ||
                                '->CHKH09 规则促销必须输入有效的促销规则!--' ||
                                b_Hd.Pphtitle);
    End;
    If r_Ppg.Ppgstatus <> Flagapproved Then
      Raise_Application_Error(Errcode,
                              b_Hd.Pphseqno || '->CHKR10 促销规则[' ||
                              r_Ppg.Ppgid || ']状态不正确,请检查!');
    End If;
    --add by clyyu 20100823
    --判断级次是否为空
    If Nvl(r_Ppg.Ppgvalue, 0) < 1 Then
      Raise_Application_Error(Errcode,
                              b_Hd.Pphseqno || '-> CHKR11 促销规则[' ||
                              r_Ppg.Ppgid || ']中规则级次不能为空，请检查!');
    End If;
    --end add
    --判断是否存在排除条件
    Select Count(*)
      Into l_Cnt
      From Popgiftrulepctj
     Where Pprgid = b_Hd.Pphtitle;

    If l_Cnt > 0 Then
      b_Haspctj := 'Y';
    Else
      b_Haspctj := 'N';
    End If;
    Select Count(Distinct Ppdrtype)
      Into i
      From Prcpopdetail
     Where Ppdseqno = p_Billno
       And Ppdmarket = '%';
    If i > 1 Then
      Raise_Application_Error(Errcode, '结果不能是多种类型');
    End If;
    For r_Det In (Select *
                    From Prcpopdetail
                   Where Ppdseqno = p_Billno
                   Order By Ppdrowno) Loop
      Submit_Rule(b_Hd, r_Det);
      Submit_Curpop(b_Hd, r_Det);
    End Loop;

    Insert_Timefb(p_Billno);
    Updateflag(p_Billno, p_Djlb, Flagapproved, p_Person);
    Cmdsend.Sp_Cmdsend_Bill('POPCURINFOUPTR',
                            p_Billno,
                            p_Billid,
                            '''' || p_Billno || '''' || ',' || '''' ||
                            p_Billid || '''');
  End Approve_Rule;

  --处理促销规则(条件)
  Procedure Submit_Rule(b_Hd  In Out Prcpophead%Rowtype,
                        b_Det In Out Prcpopdetail%Rowtype) Is
    i         Integer; --规则行数
    l_Hasall  Boolean; --有部分满足
    Lpop      Popcurinfo_r%Rowtype;
    Lday_Cnt  Number;
    Ldayweek  Number;
    Ltempdate Date;
    --r_PopTime  POPTIME%Rowtype;
    j   Number;
    Ljc Number;

    Cursor Cur_Pctj Is
      Select *
        From (Select Pprgid,
                     Pprno,
                     Pprtype,
                     Pprstatus,
                     Pprgroup,
                     Pprstyle,
                     Pprgdid,
                     Pprbarcode,
                     Ppruid,
                     Pprunit,
                     Pprcatcode,
                     Pprppcode,
                     Pprdesc,
                     Pprmanaunit,
                     Pprmarket,
                     Pprmfid,
                     Pprstime,
                     Ppretime,
                     Pprsalesl,
                     Pprgiftsl,
                     Pprgiftbl,
                     Pprmemo
                From Popgiftrulepctj
               Where Pprgid = r_Ppg.Ppgid
                 And Pprstyle <> '5'
              Union All
              Select Pprgid,
                     Pprno,
                     '1',
                     Pprstatus,
                     Pprgroup,
                     Pprstyle,
                     c.Gbid,
                     c.Gbbarcode,
                     '00',
                     c.Gbunit,
                     c.Gbcatcode,
                     c.Gbppcode,
                     Pprdesc,
                     Pprmanaunit,
                     Pprmarket,
                     Pprmfid,
                     Pprstime,
                     Ppretime,
                     Pprsalesl,
                     Pprgiftsl,
                     Pprgiftbl,
                     Pprmemo
                From Popgiftrulepctj a, Pop_Xllist b, Goodsbase c
               Where Pprgid = r_Ppg.Ppgid
                 And Pprgdid = b.Xlcode
                 And b.Codeid = c.Gbid
                 And c.Gbmanaunit = B_hd.Pphmanaunit
                 And Pprstyle = '5')
       Order By Pprno;
    r_Pctj Popgiftrulepctj%Rowtype;
  Begin
    If r_Ppg.Ppgid Is Null Then
      Return;
    End If;
    --同一单据,在同一门店一条规则只生成一次
    Select Count(*)
      Into i
      From Popcurinfo
     Where Popbillno = b_Hd.Pphseqno
       And Nvl(Popzpspecunit, Modegoods) = Moderule --一个单据只能有一个规则
       And Popmfid = b_Det.Ppdmfid;
    If i > 0 Then
      Return;

                       a.Ppruid,
                       a.Pprunit,

                       a.Pprmfid,
                       a.Pprstime,
                       a.Ppretime,

                       a.Pprgiftsl,
                       a.Pprgiftbl,
                       a.Pprmemo,
                       b.Pprjc
                  From Popgiftrule a, Popgiftrulevalue b
                 Where a.Pprgid = b.Pprgid
                   And a.Pprno =
                       Decode(r_Ppg.Ppgtype, 'A', a.Pprno, b.Pprrowno)
                   And a.Pprgid = r_Ppg.Ppgid
                   And b.Pprjc = Ljc
                 Order By b.Pprrowno) Loop
        If l_Hasall Then
          Raise_Application_Error(Errcode,
                                  r_Ppg.Ppgid ||
                                  '-->CHKR01 部分满足必须单独使用,且只能有一条部分满足规则!');
        End If;
        Lpop.Poptitle := r_Ppg.Ppgppcid; --促销主题
        --处理单据类别
        If r.Pprtype = '0' Then
          --部分满足
          Lpop.Popmode := Moderule; --部分满足
          Lpop.Popzsjs := 1; --金额条件
          l_Hasall     := True; --有整单条件
        Else
          --明细规则
          If r.Pprtype = '1' Then
            Lpop.Popzsjs := 0; --数量条件
          Else
            Lpop.Popzsjs := 1; --金额条件
          End If;
          If r.Pprstyle = '1' Or r.Pprstyle = '5' Then
            --单品,系类商品
            Lpop.Popmode := Modegoods;
          Elsif r.Pprstyle = '2' Then
            --品类
            Lpop.Popmode := Modecatpp;
          Elsif r.Pprstyle = '3' Then
            --类别
            Lpop.Popmode := Modecat;
          Elsif r.Pprstyle = '4' Then
            --品牌
            Lpop.Popmode := Modepp;
          End If;
        End If;
        Lpop.Popzpspecunit := Moderule; --单据类别
        Lpop.Popbillno     := b_Hd.Pphseqno; --单号
        Lpop.Popcusttype   := b_Hd.Custtype;--会员类别
        Lpop.Poptitle      := r_Ppg.Ppgppcid; --促销主题
        Lpop.Popgdid       := r.Pprgdid; --促销商品代码
        Lpop.Popbarcode    := r.Pprbarcode; --促销商品条码
        Lpop.Popuid        := r.Ppruid; --单位
        Lpop.Popsupid      := b_Hd.Pphsupid; --促销供应商代码

        Lpop.Popnewkl      := Null; --将抽扣率
        Lpop.Popzpbarcode  := Null; --借用为印花条码
        Lpop.Popsj         := 0; --售价
        Lpop.Popsjrate     := 1; --促销售价折扣率
        Lpop.Pophyjrate    := 1; --促销会员比率
        Lpop.Popzpsl       := 0; --赠品数量
        Lpop.Popzpxl       := 0; --赠品限量
        Lpop.Popzpsj       := r.Pprsalesl; --赠送标准
        Lpop.Poppl         := Nvl(b_Det.Ppdpl, 1); --批量促销

        Lpop.Popmarket     := b_Hd.Pphmarket;
        Lpop.Popmanaunit   := b_Hd.Pphmanaunit;

        --增加添加的字段对其初始化
        If r.Pprstyle In ('5') Then
          --系类商品
          Lpop.Popisxl := '1'; --0-> 正常商品代码1-> 系列商品某个子码
        Elsif r.Pprstyle In ('1') Then

          Lpop.Popisxl := '0';
        Else

        End If;

        Lpop.Popsjcxlx  := Null; --1->全量2->超量3->第N件4->整箱

        Lpop.Popisgzyx  := r_Ppg.Ppgisgzyx; --是否规则优先
        Lpop.Popispctj  := 'N'; --是否排除条件 ‘N' (只用于规则）
        Lpop.Pophaspctj := b_Haspctj; --是否含有排除条件 'N' (只用于规则）
        Select Decode(r_Ppg.Ppgtype, 'A', 'Y', 'N')
          Into Lpop.Popiszdgz
          From Dual; --是否部分满足
        Lpop.Popistjn   := Nvl(r_Ppg.Ppgstr1, '0'); --是否在条件内(先用于,排除条件类型）
        Lpop.Popalljc   := r_Ppg.Ppgvalue; --总的级次

        Lpop.Popnum3    := Null; --备用字段
        Lpop.Popbillno1 := Lpop.Popbillno || ',' || To_Char(r.Pprjc);

                              Codename,
                              b.Gbbarcode,
                              b.Gbcatcode,
                              b.Gbppcode
                         From Pop_Xllist a, Goodsbase b

            Lpop.Popiszdgz  := 'Y'; --是OR 条件
            Lpop.Popstr2    := r.Pprgdid; --洗了吗
            Insert_Ruletime(b_Hd.Pphseqno, Lpop);
          End Loop;
        Else
          Insert_Ruletime(b_Hd.Pphseqno, Lpop);
        End If;
        i := i + 1;
      End Loop;
      If i = 0 Then
        Raise_Application_Error(Errcode,
                                b_Hd.Pphseqno || '->CHKR02 促销规则[' ||
                                r_Ppg.Ppgid || ']无任何明细,请检查!');
      Elsif i > 1 And l_Hasall Then
        Raise_Application_Error(Errcode,
                                r_Ppg.Ppgid ||
                                '-->CHKR03 部分满足必须单独使用,且只能有一条部分满足规则!');
      End If;
      --如果存在排除条件则写入排除条件
      If b_Haspctj = 'Y' Then
        Open Cur_Pctj;
        Loop
          Fetch Cur_Pctj
            Into r_Pctj;
          Exit When Cur_Pctj%Notfound Or Cur_Pctj%Notfound Is Null;
          Submit_Pctj(Lpop, Ljc, r_Pctj, b_Hd);
        End Loop;
        Close Cur_Pctj;
      End If;
    End Loop; --jC
  End Submit_Rule;

  --修改单据头审核标记、审核人和审核时间
  Procedure Updateflag(In_Billno In Char,
                       In_Djlb   In Char,
                       In_Flag   In Char,
                       In_Person In Char,
                       In_Oper   In Char Default 'S') Is
  Begin
    If In_Oper = 'S' Then
      --审核
      Update Prcpophead
         Set Pphflag = In_Flag, Auditor = In_Person, Auditdate = Sysdate
       Where Pphseqno = In_Billno
         And Pphtype = In_Djlb;
      --印花促销时单据审核后对应单据印花码不能重复使用
      If Sql%Rowcount > 0 Then
        If In_Djlb In
           (Typestampgd, Typestampcat, Typestamppp, Typestampcatpp) Then
          Update Printcode
             Set Pcflag = 'N',
                 Pcmemo = Substr(To_Char(Trunc(Sysdate), 'YYYY-MM-DD') ||
                                 '->||' || In_Billno || '->已使用->' || Pcmemo,
                                 1,
                                 254)
           Where Pccode In (Select Distinct Ppdv4
                              From Prcpopdetail
                             Where Ppdseqno = In_Billno
                               And Ppdv4 Is Not Null);
        End If;
      Else
        Raise_Application_Error(Errcode,
                                In_Billno || '->3 修改审核标志失败！');
      End If;
    Else
      --取消
      Update Prcpophead
         Set Pphflag = In_Flag, Qxr = In_Person, Qxrq = Sysdate
       Where Pphseqno = In_Billno
         And Pphtype = In_Djlb;
      If Sql%Rowcount <> 1 Or Sql%Rowcount Is Null Then
        Raise_Application_Error(Errcode,
                                In_Billno || '->3 修改审核标志失败！');
      End If;
    End If;
  End Updateflag;

  Procedure Approve_Sl(p_Billno In Char,
                       p_Person In Char,
                       p_Billid In Char,
                       p_Djlb   In Char) Is
    b_Hd        Prcpophead%Rowtype;
    b_Det       Prcpopdetail%Rowtype;
    b_Gb        Goodsbase%Rowtype;
    i           Number;
    l_Cnt       Number;
    Vcnt        Number;
    l_Stampcode Varchar2(13);
  Begin
    Begin
      Select *
        Into b_Hd
        From Prcpophead
       Where Pphseqno = p_Billno
         For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, p_Billno || '->1 单据不存在');
    End;
    If p_Djlb Is Not Null And b_Hd.Pphtype <> p_Djlb Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->2 传入单据类别[' || p_Djlb ||
                              ']与实际单据类别[' || b_Hd.Pphtype || ']不一致');
    End If;
    If b_Hd.Pphflag Not In (Flagediting, Flagsended) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->CHKH03 单据已经审核或单据状态不一致!');
    End If;
    If b_Hd.Pphflag In (Flagapproved, Flagstopped, Flagcanceled) Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->CHKH04 单据已经审核或已经结束或已取消!');
    End If;

    --end add
    --判断是否存在排除条件
    Select Count(*)
      Into l_Cnt
      From Popgiftrulepctj
     Where Pprgid = b_Hd.Pphtitle;

    If l_Cnt > 0 Then
      b_Haspctj := 'Y';
    Else
      b_Haspctj := 'N';
    End If;
    Select Count(Distinct Ppdrtype)
      Into i
      From Prcpopdetail
     Where Ppdseqno = p_Billno
       And Ppdmarket = '%';
    If i > 1 Then
      Raise_Application_Error(Errcode, '结果不能是多种类型');
    End If;
    If b_Hd.Pphtype In
       (Typestampgd, Typestampcat, Typestamppp, Typestampcatpp) Then
      For r In (Select Ppdbarcode, Ppdcatcode, Ppdppcode, Ppdv4
                  From Prcpopdetail Ta
                 Where Ppdseqno = p_Billno
                      /* AND PPDMFID <> '%'
                      AND PPDMFID LIKE l_Mkt*/
                   And (Ppdv4 Is Null Or Substr(Ppdv4, 1, 2) <> '18')
                 Order By Ppdmfid, Ppdcatcode, Ppdppcode, Ppdgdid) Loop
        --如果没有录入印花码则自动生成
        If r.Ppdv4 Is Null Then
          Select '18' ||
                 Rtrim(Ltrim(To_Char(Seq_Stampcode.Nextval, '0000000000')))
            Into l_Stampcode
            From Dual;

          l_Stampcode := Rtrim(To_Char(Feancode12to13(To_Number(l_Stampcode))));
          Update Prcpopdetail
             Set Ppdv4 = l_Stampcode
           Where Ppdseqno = p_Billno
             And Nvl(Ppdbarcode, 'NO') = Nvl(r.Ppdbarcode, 'NO')
             And Nvl(Ppdcatcode, 'NO') = Nvl(r.Ppdcatcode, 'NO')
             And Nvl(Ppdppcode, 'NO') = Nvl(r.Ppdppcode, 'NO');
        End If;
      End Loop;
    End If;
    For r_Det In (Select *
                    From Prcpopdetail
                   Where Ppdseqno = p_Billno
                   Order By Ppdrowno) Loop
      --Submit_Rule(b_hd,r_det);
      b_Det := r_Det;
      Checkbilldetail(b_Hd, b_Det);
      Submit_Curpop(b_Hd, r_Det);
    End Loop;
    Insert_Timefb(p_Billno);
    Updateflag(p_Billno, p_Djlb, Flagapproved, p_Person);
    Cmdsend.Sp_Cmdsend_Bill('POPCURINFOUPTR',
                            p_Billno,
                            p_Billid,
                            '''' || p_Billno || '''' || ',' || '''' ||
                            p_Billid || '''');
  End Approve_Sl;
  --处理售价促销信息
  Procedure Submit_Curpop(b_Hd  In Out Prcpophead%Rowtype,
                          b_Det In Out Prcpopdetail%Rowtype) Is
    Lmode  Char(1);
    l_Bzhl Goodsunits.Gubzhl%Type;
    Lpop   Popcurinfo_r%Rowtype;
    --Add By Cl 20110326
    l_Cmpsj Number;
    Vrowno  Number;
    Vcnt    Number;
    b_Gb    Goodsbase%Rowtype;
  Begin
    --售价优惠信息中不记录任何进价相关信息
    Lpop.Popsjcxlx := Null; --1->全量2->超量3->第N件4->整箱
    Lpop.Popzsjs   := Null; --赠送基数
    Lpop.Popzpsl   := Null; --赠品数量
    --lPop.Popzpxl      := NULL;               --赠品限量
    Lpop.Popzpsj := Null; --赠品售价
    Lpop.Popisxl := '0'; --商品属性
    Lpop.Popqjyz := '0'; --0->取价格1->取折扣2->取扣额 3->总扣额
    Select Decode(r_Ppg.Ppgtype, 'A', 'Y', 'N')
      Into Lpop.Popiszdgz
      From Dual; --是否部分满足
    --单品促销情况
    Lpop.Popzpspecunit := '1'; --一般情况下,单据类别为1

    --印花促销
    If b_Hd.Pphtype = Typestampgd Then
      Lmode              := Modegoods;
      Lpop.Popzpspecunit := Modestamp;
      --类别印花促销
    Elsif b_Hd.Pphtype = Typestampcat Then
      Lmode              := Modecat;
      Lpop.Popzpspecunit := Modestamp;
      Lpop.Popqjyz       := '1';
      --品牌印花促销
    Elsif b_Hd.Pphtype = Typestamppp Then
      Lmode              := Modepp;
      Lpop.Popzpspecunit := Modestamp;
      Lpop.Popqjyz       := '1';
      --印花类别品牌促销
    Elsif b_Hd.Pphtype = Typestampcatpp Then
      Lmode              := Modecatpp;
      Lpop.Popzpspecunit := Modestamp;
      Lpop.Popqjyz       := '1';
      --数量促销
    Elsif b_Hd.Pphtype = Typestrsl Then
      Lmode              := Modegoods;
      Lpop.Popzpspecunit := Moderulesl;
      Lpop.Popsjcxlx     := b_Det.Ppdrule;
      --规则促销、满减促销
    Elsif b_Hd.Pphtype = Typestrrule Then
      Lmode := Modegoods;
      --'P';  --规则促销
      If b_Det.Ppdrtype = Ruletype_Pop Then
        Lpop.Popzpspecunit := Moderulepop;
        --'G';  --赠品促销
      Elsif b_Det.Ppdrtype = Ruletype_Gift Then
        Lpop.Popzpspecunit := Moderulegift;
        --'F';  --返现金
      Elsif b_Det.Ppdrtype = Ruletype_Ret Then
        Lpop.Popzpspecunit := Moderuleret;
        --'D';  --整单满减
      Elsif b_Det.Ppdrtype = Ruletype_Minus Then
        Lpop.Popzpspecunit := Moderuleminus;
        --'A'; --系列优惠
      Elsif b_Det.Ppdrtype = Ruletype_Popxl Then
        Lpop.Popzpspecunit := Moderulepopxl;
        Lpop.Popqjyz       := 1;
        Lpop.Popiszdgz     := 'Y'; --OR 条件
        --'B'; --系列商品赠送
      Elsif b_Det.Ppdrtype = Ruletype_Giftxl Then
        Lpop.Popzpspecunit := Moderulegiftxl;
        Lpop.Popqjyz       := 1;
        Lpop.Popiszdgz     := 'Y'; --OR 条件
        --'C'; --满减
      Elsif b_Det.Ppdrtype = Ruletype_Nmj Then
        Lpop.Popzpspecunit := Moderulenmj;
        --'T'; --固定金额
      Elsif b_Det.Ppdrtype = Ruletype_Gdje Then
        Lpop.Popzpspecunit := Moderuleje;
        --'Z'; --固定金额(单品)
      Elsif b_Det.Ppdrtype = Ruletype_Gdje1 Then
        Lpop.Popzpspecunit := Moderuleje1;
        --'Y'; --条件折扣（条件只能是部分满足）
      Elsif b_Det.Ppdrtype = Ruletype_Tjzk Then
        If Lpop.Popiszdgz <> 'Y' Then
          Raise_Application_Error(Errcode, '条件折扣必须是部分满足');
        End If;
        Lpop.Popzpspecunit := Moderuletjzk;
        Lpop.Popqjyz       := '1'; --取价原则
      Elsif b_Det.Ppdrtype = Ruletype_v Then
        Lpop.Popzpspecunit := Moderule_v;
        /*If b_Det.Ppdv5 Not In (Qj_Rate,Qj_ALLKE) Then
           Raise_Application_Error(ErrCode,'取价类型错误');
        End If;*/
        If b_Det.Ppdv7 Not In ('2', '3', '4') Then
          Raise_Application_Error(Errcode, '控制点错误');
        End If;
        If b_Det.Ppdv7 = '2' Then
          Lmode := Modecatpp;
        Elsif b_Det.Ppdv7 = '3' Then
          --类别
          Lmode := Modecat;
        Elsif b_Det.Ppdv7 = '4' Then
          --品牌
          Lmode := Modepp;
        End If;
        Lpop.Popqjyz := b_Det.Ppdv5;
      End If;
    Else
      Return;
    End If;
    If b_Hd.Pphtype = Typestrrule Then
      Lpop.Poptitle := r_Ppg.Ppgppcid; --规则促销的促销主题是促销规则的主题,促销单的主题字段关联到促销规则
    Else
      Lpop.Poptitle := b_Hd.Pphtitle; --促销主题
    End If;

    Lpop.Popmode    := Lmode; --促销方法
    Lpop.Popbillno  := b_Hd.Pphseqno; --单号
    Lpop.Popcusttype := b_Hd.Custtype; --会员类别
    Lpop.Popgdid    := b_Det.Ppdgdid; --促销商品代码
    Lpop.Popbarcode := b_Det.Ppdbarcode; --促销商品条码
    Lpop.Popuid     := b_Det.Ppduid; --单位
    Lpop.Popsupid   := b_Hd.Pphsupid; --促销供应商代码
    Lpop.Popcatcode := b_Det.Ppdcatcode; --商品类别代码
    Lpop.Popppcode  := b_Det.Ppdppcode; --品牌编码
    Lpop.Popspsx    := b_Det.Ppdspsx;--商品属性码
    --商品印花促销的限量默认为1
    If b_Hd.Pphtype = Typestampgd Then
      Lpop.Popsl := 1;
    Else
      Lpop.Popsl := b_Det.Ppdmaxsl; --促销限量
    End If;
    Lpop.Popzpxl := Null; --赠品限量
    Lpop.Popfdfs := b_Det.Ppdfdfs; --分担方式
    Lpop.Popzkfd := b_Det.Ppdzkfd; --商家折扣分担
    If b_Det.Ppdfdfs = Zkfdfs_Fallkl Then
      Lpop.Popnewkl := b_Det.Ppdxhsjj; --将抽
      Lpop.Popnewjj := 0; --倒扣商品，进价、进价差额或扣额无效
    Else
      Lpop.Popnewjj := b_Det.Ppdxhsjj; --进价、进价差额或扣额
      Lpop.Popnewkl := 0; --顺加商品扣率意义
    End If;
    Lpop.Poppl    := 1; --批量促销
    Lpop.Popyxfw  := b_Det.Ppdmode; --影响范围(默认为销售)
    Lpop.Popnewkl := Null; --将抽扣率
    --lPop.Popzpbarcode := NULL;               --赠品条码
    If b_Hd.Pphtype In
       (Typestampgd, Typestampcat, Typestamppp, Typestampcatpp) Then
      Lpop.Popzpbarcode := b_Det.Ppdv4; --借用为印花条码
    Else
      Lpop.Popzpbarcode := Null;
    End If;
    Lpop.Popzsjs     := Null; --赠送基数
    Lpop.Popzpsl     := Null; --赠品数量
    Lpop.Popzpsj     := Null; --赠品售价
    Lpop.Popmfid     := b_Det.Ppdmfid; --促销柜组代码
    Lpop.Popmarket   := b_Hd.Pphmarket;
    Lpop.Popmanaunit := b_Hd.Pphmanaunit;
    --lPop.Popmarket    := fGetMFrameFCode(lPop.Popmfid);--促销地点
    /*中百促销平台需求   将下面功能屏蔽掉
    --赠品促销和返现金特殊处理(规则促销和一般促销的记录方式一样)
    IF lPop.POPZPSPECUNIT = ModeRuleGift THEN
       lPop.PopZPSL   := b_DET.ppdrSL;       --赠品数量
       lPop.PopZPXL   := b_DET.ppdrMaxSL;    --单张小票限量
       lPop.PopZPSJ   := b_DET.ppdrAddJE;    --附加金额
    ELSIF lPop.POPZPSPECUNIT = ModeRuleRet THEN
       lPop.PopZPSL   := b_DET.ppdrSL;       --暂时无效
       lPop.PopZPXL   := b_DET.ppdrMaxSL;    --单张小票限量
       lPop.PopZPSJ   := b_DET.ppdrAddJE;    --返现金金额
    ELSIF lPop.POPZPSPECUNIT = ModeRulePop THEN
       lPop.PopZPSL   := b_DET.ppdrSL;       --优惠数量
       lPop.PopZPXL   := b_DET.ppdrMaxSL;    --单张小票限量
    ELSE
       lPop.PopPL     := NVL(b_DET.ppdPL, 1);  --批量促销
    END IF;
    */
    --促销类别记录到POPSPACE列
    If b_Det.Ppdc4 = 'A' Then
      Lpop.Popspace := 1;
    Elsif b_Det.Ppdc4 = 'B' Then
      Lpop.Popspace := 2;
    Else
      Lpop.Popspace := Null;
    End If;

    Lpop.Popjc      := Null; --级次
    Lpop.Popiszsz   := Nvl(b_Hd.Pphiszsz, 'Y'); --是否可以折上折
    Lpop.Popisgzyx  := Nvl(b_Hd.Pphispt, 'N'); --是否规则优先
    Lpop.Popispctj  := 'N'; --是否排除条件 ‘N' (只用于规则）
    Lpop.Pophaspctj := b_Haspctj; --是否含有排除条件 'N' (只用于规则）

    Lpop.Popistjn   := Nvl(r_Ppg.Ppgstr1, '0'); --0->排除条件，1->排除结果，2->条件结果都排除
    Lpop.Popalljc   := Nvl(To_Number(b_Hd.Pphstr4), 1); --总的级次
    Lpop.Popstr1    := Nvl(b_Hd.Pphstr3, 'N'); --是否会员专用规则
    Lpop.Popstr2    := Null; --
    Lpop.Popstr3    := Null; --赠品类型
    Lpop.Popnum1    := Null; --备用字段
    Lpop.Popnum2    := Null; --备用字段--用于是否封顶0->不封顶 1->封顶
    Lpop.Popnum3    := Null; --备用字段
    Lpop.Popbillno1 := Lpop.Popbillno;
    --去掉是否在条件内
    Lpop.Popistjn := Nvl(r_Ppg.Ppgstr1, '0'); --排除条件
    --赠品促销和返现金特殊处理(规则促销和一般促销的记录方式一样)
    If b_Hd.Pphtype = Typestrrule Then
      --的到行号
      Select Ppdrowno
        Into Vrowno
        From Prcpopdetail
       Where Ppdseqno = b_Det.Ppdseqno
         And Ppdmfid = '%'
         And Nvl(Ppdgdid, '#') = Nvl(b_Det.Ppdgdid, '#')
         And Nvl(Ppdcatcode, '#') = Nvl(b_Det.Ppdcatcode, '#')
         And Nvl(Ppdppcode, '#') = Nvl(b_Det.Ppdppcode, '#');
      If b_Det.Ppdrtype In ('G', 'B') Then
        Lpop.Popstr3 := Nvl(b_Det.Ppdv6, '4');
      Else

                       Where Nvl(b_Hd.Pphstr4, '1') = '1') Loop

          b_Det.Ppdrmaxsl := r_Detex.Ppdrmaxsl;

          b_Det.Ppdsj     := r_Detex.Ppdnum1;

        Lpop.Popsjrate  := b_Det.Ppdsjzkl; --促销售价折扣率

        Lpop.Pophyjrate := b_Det.Ppdhyjzkl; --促销会员比率
        Lpop.Popjc      := r_Detex.Ppdjc; --级次
        Lpop.Popbillno1 := Lpop.Popbillno || ',' || To_Char(r_Detex.Ppdjc);

          Lpop.Popzpxl := r_Detex.Ppdrmaxsl; --单张小票限量
          Lpop.Popzpsj := r_Detex.Ppdraddje; --附加金额
          If Nvl(Lpop.Popzpsl, 0) < 1 Then

          End If;
        Elsif Lpop.Popzpspecunit = Moderuleret Then
          Lpop.Popzpsl := r_Detex.Ppdrsl; --暂时无效
          Lpop.Popzpxl := r_Detex.Ppdrmaxsl; --单张小票限量
          Lpop.Popzpsj := r_Detex.Ppdraddje; --返现金金额
        Elsif Lpop.Popzpspecunit = Moderulepop Or
              Lpop.Popzpspecunit = Moderuletjzk Then
          Lpop.Popzpsl := r_Detex.Ppdrsl; --优惠数量
          Lpop.Popzpxl := r_Detex.Ppdrmaxsl; --单张小票限量
        Elsif Lpop.Popzpspecunit = Moderulepopxl Then
          Lpop.Popzpsl := r_Detex.Ppdrsl; --优惠数量
          Lpop.Popzpxl := r_Detex.Ppdrmaxsl; --单张小票限量
          Lpop.Popisxl := '1';
        Elsif Lpop.Popzpspecunit = Moderulegiftxl Then
          Lpop.Popzpsl := r_Detex.Ppdrsl; --赠品数量
          Lpop.Popzpxl := r_Detex.Ppdrmaxsl; --单张小票限量
          Lpop.Popzpsj := r_Detex.Ppdraddje; --附加金额
          Lpop.Popisxl := '1';
          If Nvl(Lpop.Popzpsl, 0) < 1 Then
            Raise_Application_Error(Errcode, '赠送的数量必须大于1');
          End If;
        Elsif Lpop.Popzpspecunit = Moderuleje Then
          Lpop.Popzpsj := r_Detex.Ppdraddje;
        Elsif Lpop.Popzpspecunit = Moderuleje1 Then
          Lpop.Popzpsj := r_Detex.Ppdraddje;
          --检查条件中价格是否大于固定金额的价格
          --Of_CheckGdJe1(r_DetEx.PPDJC,r_DetEx.Ppdraddje,lPop.Popmarket);
        Elsif Lpop.Popzpspecunit = Moderulenmj Then
          Lpop.Popzpsj := r_Detex.Ppdraddje;
        Elsif Lpop.Popzpspecunit = Moderule_v Then
          --优惠
          If Lpop.Popmode = Modecat Or Lpop.Popmode = Modecatpp Then
            If Lpop.Popcatcode Is Null Or Lpop.Popcatcode = '0' Then
              Raise_Application_Error(Errcode,
                                      '—> v 类别不能为空，请检查');
            End If;
          End If;

          If Lpop.Popmode = Modepp Or Lpop.Popmode = Modecatpp Then
            If Lpop.Popppcode Is Null Or Lpop.Popppcode = '0' Then
              Raise_Application_Error(Errcode,
                                      '—> v 品牌不能为空，请检查');
            End If;
          End If;

          If Nvl(r_Detex.Ppdv2, '#') = '#' Then
            Raise_Application_Error(Errcode, '->V 物价控制点没有输入');
          End If;

          If r_Detex.Ppdv2 = '1' And
             (r_Detex.Ppdraddje >= 1 Or r_Detex.Ppdraddje <= 0) Then
            Raise_Application_Error(Errcode,
                                    '->V 物价折扣率必须[0 - 1]之间!');
          End If;
          If r_Detex.Ppdv2 = '1' Then
            Lpop.Popsj      := r_Detex.Ppdraddje;
            Lpop.Pophyj     := r_Detex.Ppdraddje;
            Lpop.Popsjrate  := r_Detex.Ppdraddje;
            Lpop.Pophyjrate := r_Detex.Ppdraddje;
          End If;
          Lpop.Popzpsl := r_Detex.Ppdrsl; --优惠数量
          Lpop.Popzpsj := r_Detex.Ppdraddje; --固定金额
        Else
          Raise_Application_Error(Errcode, '没有符合的规则信息');
        End If;

        If Lpop.Popisxl = '1' Then
          For r_Xl In (Select Codeid,
                              Codename,
                              b.Gbbarcode,
                              b.Gbcatcode,
                              b.Gbppcode
                         From Pop_Xllist a, Goodsbase b
                        Where a.Codeid = b.Gbid
                          And a.Xlcode = b_Det.Ppdgdid
                          And b.gbmanaunit = b_hd.pphmanaunit) Loop
            Lpop.Popgdid    := r_Xl.Codeid; --促销商品代码

        Else

           Min(Ppdsj)
            Into l_Cmpsj
            From Prcpopdetail
           Where Ppdseqno = b_Det.Ppdseqno

            End If;
            /*ELSE
            l_BZHL := f_GetBZHL(b_Det.ppdGdId, b_Det.ppdUId, 'Y');

  Procedure Checkbilldetail(b_Hd  In Out Prcpophead%Rowtype,
                            b_Det In Out Prcpopdetail%Rowtype) Is

    Lprcgrp Varchar2(20);
    Lcount  Integer;
    Lsdate  Prcpoptime.Ppteffdate%Type;
    Vcnt    Number;
    Lflag   Boolean := False;
    b_Gb    Goodsbase%Rowtype;
  Begin
    b_Det.Ppdsupcode := Nvl(b_Det.Ppdsupcode, Nvl(b_Hd.Pphsupid, '%')); --不指定供应商表示所有供应商都促销
    b_Det.Ppdsj      := Nvl(b_Det.Ppdsj, 0);
    b_Det.Ppdhyj     := Nvl(b_Det.Ppdhyj, b_Det.Ppdsj);
    b_Det.Ppdsjzkl   := Nvl(b_Det.Ppdsjzkl, 0);
    b_Det.Ppdhyjzkl  := Nvl(b_Det.Ppdhyjzkl, b_Det.Ppdsjzkl);
    b_Det.Ppdzkfd    := Nvl(b_Det.Ppdzkfd, 1);
    b_Det.Ppdmaxsj   := Nvl(b_Det.Ppdmaxsj, b_Det.Ppdsj);
    b_Det.Ppdminsj   := Nvl(b_Det.Ppdmaxsj, b_Det.Ppdsj);
    b_Det.Ppdmaxsl   := Nvl(b_Det.Ppdmaxsl, -1);
    If b_Det.Ppdmaxsl = 0 Then
      --零,没有意义,认为是不限量
      b_Det.Ppdmaxsl := -1;
    End If;
    b_Det.Ppdxhsjj := Nvl(b_Det.Ppdxhsjj, 0);
    b_Det.Ppdyhsjj := Nvl(b_Det.Ppdyhsjj, 0);
    b_Det.Ppdfdfs  := Nvl(b_Det.Ppdfdfs, Zkfdfs_Kl);
    b_Det.Ppdmode  := Nvl(b_Det.Ppdmode, '2');

    If b_Hd.Pphtype In (Typestampgd, Typestrrule, Typestrsl) Then
      If b_Det.Ppdrtype In ('F', 'C', 'T', 'Z', 'Y', 'V') And
         b_Hd.Pphtype = Typestrrule Then
        If b_Det.Ppdrtype In ('F', 'C', 'T', 'Z', 'V') Then
          If b_Det.Ppdraddje <= 0 Then
            Raise_Application_Error(Errcode, '没有输入金额，请检查!');
          End If;
          If b_Det.Ppdrtype In ('V') And b_Det.Ppdv5 = '1' And
             b_Det.Ppdraddje >= 1 Then
            Raise_Application_Error(Errcode,
                                    '满额优惠，金额想请输入折扣率!');
          End If;
        Else
          If b_Det.Ppdsj <= 0 Or b_Det.Ppdsj >= 1 Then
            Raise_Application_Error(Errcode, '条件折扣售价为售价比率');
          End If;
          If b_Hd.Pphstr4 <> '1' Then
            Raise_Application_Error(Errcode, '条件折扣必须只能一级');
          End If;

          Select Count(*)
            Into Vcnt
            From Popgiftrule
           Where Pprtype = 0
             And Pprgid = r_Ppg.Ppgid;
          If Vcnt > 0 Then
            Raise_Application_Error(Errcode, '条件折扣不允许整单金额');
          End If;
          If r_Ppg.Ppgtype Not In ('A') Then
            Raise_Application_Error(Errcode, '条件折扣必须是部分满足');
          End If;
        End If;
      Else
        If Ltrim(b_Det.Ppdgdid) Is Null Or Ltrim(b_Det.Ppdbarcode) Is Null Then
          Raise_Application_Error(Errcode,
                                  '->CHD04 商品代码、条码必须输入!');
        End If;
        If b_Det.Ppdrtype In ('A', 'B') Then
          If b_Det.Ppdsj >= 1 And b_Det.Ppdrtype = 'A' And b_Det.Ppdsj <= 0 Then
            Raise_Application_Error(Errcode, 'xlCode 系列优惠必须输入售价');
          End If;
          Goto Skip1;
        End If;
        Begin
          Select * Into b_Gb From Goodsbase Where Gbid = b_Det.Ppdgdid And GbManaUnit = b_Hd.Pphmanaunit;
        Exception
          When No_Data_Found Then
            Raise_Application_Error(Errcode,
                                    b_Det.Ppdgdid || '->CHKD06 商品不存在!');
        End;
        If b_Gb.Gbstatus <> 'Y' Then
          Raise_Application_Error(Errcode,
                                  b_Det.Ppdgdid || '->CHKD07 商品状态无效!');
        End If;
        /*IF b_DET.ppdfdfs <> ZkfdFs_KL AND b_gb.gbtypeid = u_Constant.GtZB THEN
           Raise_Application_Error(ErrCode, b_DET.Ppdgdid || '->CHKD08 组包商品' || b_DET.PPDGDID || '折扣分担方式必须为比率!');
        END IF;*/
        b_Det.Ppdcatcode := b_Gb.Gbcatcode;
        b_Det.Ppdppcode  := b_Gb.Gbppcode;
        <<skip1>>
        Null;
      End If;

      If b_Det.Ppdrtype In ('F', 'C', 'T', 'Z', 'Y', 'V') And
         b_Hd.Pphtype = Typestrrule Then
        Null;
      Else
        --取商品原售价
        If b_Det.Ppdrtype In ('A', 'B') Then
          --检查系列商品是否存在
          Select Count(*)
            Into Vcnt
            From Pop_Xllist
           Where Xlcode = b_Det.Ppdgdid;
          If Vcnt <= 0 Then
            Raise_Application_Error(Errcode,
                                    b_Det.Ppdgdid || '->xlchk ' ||
                                    b_Det.Ppdmfid || '系列商品不存在!');
          End If;
        End If;
      End If;
      If b_Hd.Pphtype In (Typestrrule) Then
        Select Count(*)
          Into Vcnt
          From Popgiftrule
         Where Pprtype = '0'
           And Pprgid = r_Ppg.Ppgid;
        /*If vCnt<>1  And R_PPG.PPGTYPE<>'A' Then
           Raise_Application_Error(ErrCode,'只有部分满足才能满减');
        End If;*/
      End If;
    Else
      --IF b_HD.pphtype IN (TypeGrpCat, TypeStrCat, TypeGrpCatPp, TypeStrCatPp) THEN
      If b_Hd.Pphtype In (Typestampcat, Typestampcatpp) Then
        If Trim(b_Det.Ppdcatcode) Is Null Then
          Raise_Application_Error(Errcode, '->CHKD16 商品分类必须输入!');
        End If;
      End If;
      If b_Hd.Pphtype In (Typestamppp, Typestampcatpp) Then
        If Trim(b_Det.Ppdppcode) Is Null Then
          Raise_Application_Error(Errcode, '->CHKD17 商品品牌必须输入!');
        End If;
      End If;

      If b_Det.Ppdfdfs In (Zkfdfs_Jj, Zkfdfs_Ke, Zkfdfs_Kljj) Then
        Raise_Application_Error(Errcode,
                                '->CHKD18 非单品促销，不能指定进价或进价差!');
      End If;
      If Nvl(b_Det.Ppdrtype, 'P') = 'P' Or
         Nvl(b_Det.Ppdrtype, Ruletype_Popxl) = Ruletype_Popxl Then
        If b_Det.Ppdsjzkl <= 0 Or b_Det.Ppdsjzkl > 1 Then
          Raise_Application_Error(Errcode,
                                  b_Det.Ppdgdid || '->CHKD19 促销折扣率输入错误!');
        End If;
        If b_Det.Ppdhyjzkl <= 0 Or b_Det.Ppdhyjzkl > 1 Then
          Raise_Application_Error(Errcode,
                                  b_Det.Ppdgdid || '->CHKD20 会员促销折扣率输入错误!');
        End If;
        If b_Det.Ppdsjzkl = 1 And b_Det.Ppdhyjzkl = 1 Then
          Raise_Application_Error(Errcode,
                                  b_Det.Ppdgdid ||
                                  '->CHKD22 售价折扣率、会员价折扣率和批发价折扣率至少要有一个小于1!');
        End If;
      End If;
    End If;
  End Checkbilldetail;
