CREATE OR REPLACE PACKAGE BODY JAVA_GETYHINFOLIST
AS
  -- 返回结果集为
  -- 序号,优惠单号,优惠类型  ,商品编码,柜组,多单位代码,商品品类  ,商品品牌,销售数量,电子秤商品的优惠的时间间隔(秒)
  -- 开始日期,结束日期,开始时间,结束时间,优惠零售价,优惠会员价,优惠批发价,优惠零售价折扣率,优惠会员价折扣率
  --优惠批发价折扣率,优惠零售价折扣分担,优惠会员价折扣分担,优惠批发价折扣分担

  PROCEDURE GETYHINFOLIST
    (
            vcode in varchar2,
            vgz in varchar2,
            vtype in varchar2,
            vpp in varchar2,
            vpec in varchar2,

      rc OUT RETCURSOR
    )
  AS
  BEGIN
    OPEN rc FOR
       select ppiseq,ppibillno,ppimode,ppibarcode,ppimfid,ppispec,ppicatid,ppippcode,ppimaxnum,ppispace,to_char(ppistartdate,'YYYY/MM/DD')
       ,to_char(ppienddate,'YYYY/MM/DD'),ppistarttime,ppiendtime,ppinewsj,ppinewhyj,ppinewpfj,ppinewrate,ppinewhyrate,ppinewpfrate,ppizkfd,ppihyzkfd,ppipfzkfd from pospopinfo
            where (ppibarcode = vcode AND (ppimfid = vgz OR ppimfid = '0') AND NVL(LTRIM(ppispec),'00') = NVL(LTRIM(vpec),'00')
       AND ppimaxnum <> 0 AND (ppimode = '1' OR ppimode = '7')) OR (ppibarcode = vgz AND ppimode = '2') OR
       (ppibarcode = vgz AND ppippcode = vpp AND ppimode = '4') OR
       (ppibarcode = vtype AND ppimode = '3') OR
       (ppibarcode = vtype AND ppippcode = vpp AND ppimode = '5') OR
       (ppibarcode = vpp AND ppimode = '6') order by ppiseq;

  END GETYHINFOLIST;

END JAVA_GETYHINFOLIST;
