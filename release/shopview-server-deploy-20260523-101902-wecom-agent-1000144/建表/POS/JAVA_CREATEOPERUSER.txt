CREATE OR REPLACE PACKAGE BODY JAVA_CREATEOPERUSER
AS
  -- 返回结果集为
  -- 工号,姓名,密码,员工类别,是否允许登录,有效期,角色代码,营业员营业柜组,备注
  PROCEDURE CREATEOPERUSER
    (
      rc OUT RETCURSOR
    )
  AS
  BEGIN
    OPEN rc FOR
      select poagh,substr(poaname,1,20),java_password(poapwd),poapertype,poaislogin,to_char(poamaxdate,'YYYY/MM/DD'),poaposrole,poasmcounter,'' from posoperaccnt;
  END CREATEOPERUSER;

END JAVA_CREATEOPERUSER;
