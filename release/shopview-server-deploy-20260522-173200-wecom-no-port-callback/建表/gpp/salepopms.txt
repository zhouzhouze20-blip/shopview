  --得到返券等级
  FUNCTION fgetrulelev
  (
    vrule IN VARCHAR2,
    vxfje IN NUMBER
  ) RETURN NUMBER IS
    l_lev NUMBER;
  BEGIN
    l_lev := 0;
    FOR vr IN (SELECT *
                 FROM tktrulefqlever
                WHERE trdcode = vrule
                  AND trdxfmk <= vxfje
                ORDER BY trdseq DESC)
    LOOP
      l_lev := vr.trdseq;
      EXIT;
    END LOOP;
    RETURN l_lev;
  EXCEPTION
    WHEN OTHERS THEN
      RETURN 0;
  END;
  FUNCTION fgetrulemk(vrule IN VARCHAR2) RETURN NUMBER IS
    l_lev NUMBER;
  BEGIN
    SELECT trdxfmk
      INTO l_lev
      FROM tktrulefqlever
     WHERE trdcode = vrule
       AND trdseq = 1;
    RETURN l_lev;
  EXCEPTION
    WHEN OTHERS THEN
      RETURN 0;
  END;

  FUNCTION fgetrulemje
  (
    vrule IN VARCHAR2,
    vdj   IN NUMBER
  ) RETURN NUMBER IS
    l_lev NUMBER;
  BEGIN
    SELECT trdzje
      INTO l_lev
      FROM tktrulefqlever
     WHERE trdcode = vrule
       AND trdseq = vdj;
    RETURN l_lev;
  EXCEPTION
    WHEN OTHERS THEN
      RETURN 99999999999;
  END;
  FUNCTION fgetrulefqje
  (
    vrule IN VARCHAR2,
    vdj   IN NUMBER
  ) RETURN NUMBER IS
    l_lev NUMBER;
  BEGIN
    l_lev := 0;
    SELECT SUM(nvl(trdqje, 0))
      INTO l_lev
      FROM tktrulefqdet_ex
     WHERE trdcode = vrule
       AND trdnum1 = vdj;
    RETURN nvl(l_lev, 0);
  EXCEPTION
    WHEN OTHERS THEN
      RETURN 0;
  END;

    RETURN l_lev;
  EXCEPTION
    WHEN OTHERS THEN
      RETURN 0;
  END;

  FUNCTION fgetticketno RETURN VARCHAR2 IS
    vseqno        NUMBER;
    vticketnobase VARCHAR2(20);
    vticketno     VARCHAR2(20);
    --手工券号增加校验位及随机位
  BEGIN
    SELECT seq_tktseq.NEXTVAL INTO vseqno FROM dual;
    --填充到TktLen长度
    IF length(to_char(vseqno)) < tktlen THEN
      SELECT lpad(to_char(vseqno), tktlen, '0')
        INTO vticketnobase
        FROM dual;
    ELSE
      SELECT lpad(substr(to_char(vseqno), -tktlen, tktlen), tktlen, '0')
        INTO vticketnobase
        FROM dual;
    END IF;
    SELECT vticketnobase || f_makecheckcode(vticketnobase, 1) ||
           --lpad(trunc(dbms_random.VALUE(0, 999999)), 6, '0')
            lpad(trunc(dbms_random.VALUE(0, power(10, tl_random) - 1)),
                 tl_random, '0')
      INTO vticketno
      FROM dual;
    RETURN vticketno;
  END;

  --是否促销付款方式
  FUNCTION fgetispoppay
  (
    vjygs    IN VARCHAR2,
    vcrmmkt  IN VARCHAR2,
    vpaymode IN VARCHAR2,
    vpara    IN VARCHAR2 --sellpayment.sppayerid 标识券种
  ) RETURN VARCHAR2 IS
    ls_ispop CHAR(1);
    vjetype  CHAR(1);
    l_pmtype CHAR(1);
  BEGIN
    SELECT pmtype
      INTO l_pmtype
      FROM paymode
     WHERE pmcode = vpaymode
       AND pmmanaunit = vjygs;
    SELECT pnpispop
      INTO ls_ispop
      FROM paymktnotpop
     WHERE pnpcode = vpaymode
       AND pnpmanaunit = vjygs
       AND pnpmkt = vcrmmkt;
    --如果券的付款方式设置的是促销，那么就参与促销
    --如果券的付款方式设置的是不促销，那么找券种的定义是否促销TEMP
    IF ls_ispop = 'N' AND l_pmtype = '5' THEN
      vjetype := substr(vpara, 1, 1);
      IF vjetype = 'a' THEN
        --返利券默认参与
        ls_ispop := nvl(fgetmanapara('4D'), 'Y');
      ELSIF vjetype BETWEEN 'A' AND 'Z' THEN
        BEGIN
          SELECT decode(nvl(tqisfq, '0'), '0', 'N', 'Y')
            INTO ls_ispop
            FROM tktqtype
           WHERE tqcode = vjetype;
        EXCEPTION
          WHEN OTHERS THEN
            --活动券默认不参与
            ls_ispop := 'N';
        END;
      END IF;
    END IF;
    RETURN ls_ispop;
  EXCEPTION
    WHEN OTHERS THEN
      BEGIN
        SELECT pnpispop
          INTO ls_ispop
          FROM paynotpop
         WHERE pnpcode = vpaymode
           AND pnpmanaunit = vjygs;
        --如果券的付款方式设置的是促销，那么就参与促销
        --如果券的付款方式设置的是不促销，那么找券种的定义是否促销
        IF ls_ispop = 'N' AND l_pmtype = '5' THEN
          vjetype := substr(vpara, 1, 1);
          IF vjetype = 'a' THEN
            --返利券默认参与
            ls_ispop := nvl(fgetmanapara('4D'), 'Y');
          ELSIF vjetype BETWEEN 'A' AND 'Z' THEN
            BEGIN
              SELECT decode(nvl(tqisfq, '0'), '0', 'N', 'Y')
                INTO ls_ispop
                FROM tktqtype
               WHERE tqcode = vjetype;
            EXCEPTION
              WHEN OTHERS THEN
                --活动券默认不参与
                ls_ispop := 'N';
            END;
          END IF;
        END IF;
        RETURN ls_ispop;
      EXCEPTION
        WHEN OTHERS THEN
          ls_ispop := 'N';
          RETURN ls_ispop;
      END;
  END;

  --档期 券开始结束日期
  PROCEDURE gettktpopinfo
  (
    p_tpid      IN VARCHAR2,
    p_cardno    IN VARCHAR2,
    p_action    IN VARCHAR2 DEFAULT 'ALL',
    o_retcode   OUT NUMBER,
    o_retmsg    OUT VARCHAR2,
    o_startdate OUT DATE, --券开始日期
    o_enddate   OUT DATE, --券结束日期
    o_qno       OUT VARCHAR2, -- 券号（持卡VIP，手工券号）
    o_a_m       OUT VARCHAR2, --q_auto or q_manu 电子券 or 打印券
    o_pj        OUT VARCHAR2, -- 拼券 invalid
    o_area      OUT VARCHAR2, -- 券使用范围 指定门店，本门店，通用
    o_ismz      OUT VARCHAR2, -- 是否固定面值
    o_prnsite   OUT VARCHAR2 -- 券打印地点 invalid
  ) IS
    row_fqtitle tktpopinfo%ROWTYPE;
    l_isvip     CHAR(1);
    --vcurtime    CHAR(10);
  BEGIN
    --查找返券档期

    BEGIN
      SELECT * INTO row_fqtitle FROM tktpopinfo WHERE tpiid = p_tpid;
    EXCEPTION
      WHEN OTHERS THEN
        o_retcode := 1;
        o_retmsg  := '返券档期找不到!返券档期:' || p_tpid;
        RETURN;
    END;
    IF p_action = 'ALL' THEN
      l_isvip := 'Y';
      IF TRIM(p_cardno) IS NULL THEN
        l_isvip := 'N';
      END IF;
      --row_fqtitle.TPHYFQMODE   会员返券方式 0电子券,1指定手工券号,2生成手工券号
      --row_fqtitle.TPFHYFQMODE  非会员返券方式 1指定手工券号,2生成手工券号
      --row_fqtitle.TPMZMODE     面值方式：0 规则合并(一个80),1固定面值(两个40)
      --row_fqtitle.TPPRINTPLACE 手工券打印地点 0前台,1后台 ,2 不用打券
      IF l_isvip = 'Y' THEN
        IF row_fqtitle.tphyfqmode = '0' THEN
          o_qno := p_cardno;
          o_a_m := q_auto;
        ELSIF row_fqtitle.tphyfqmode = '1' THEN
          SELECT 'MANU' || to_char(seq_tktcust.NEXTVAL)
            INTO o_qno
            FROM dual;
          o_a_m := q_manu;
        ELSIF row_fqtitle.tphyfqmode = '2' THEN
          SELECT fgetticketno INTO o_qno FROM dual;
          o_a_m := q_manu;
        END IF;
      ELSE
        o_a_m := q_manu;
        IF row_fqtitle.tpfhyfqmode = '1' THEN
          SELECT 'MANU' || to_char(seq_tktcust.NEXTVAL)
            INTO o_qno
            FROM dual;
        ELSIF row_fqtitle.tpfhyfqmode = '2' THEN
          SELECT fgetticketno INTO o_qno FROM dual;
        ELSIF row_fqtitle.tpfhyfqmode = 'N' Then
           o_retcode := 1;
           o_retmsg  := '非持卡用户不返券,返券档期:' || p_tpid ;
           RETURN;
        END IF;
      END IF;
    END IF;
    --券有效期
    o_startdate := row_fqtitle.tpiyqstartdate;
    o_enddate   := row_fqtitle.tpiyqenddate;
    /*----返券有效期
    SELECT substr(to_char(SYSDATE, 'yyyy-mm-dd HH24:mi:ss'), 12, 10)
      INTO vcurtime
      FROM dual;
    IF nvl(row_fqtitle.tpisoneday, 'N') <> 'N' THEN
      ----如果是当日有效的
      IF vcurtime < row_fqtitle.tptime THEN
        o_startdate := trunc(SYSDATE);
        o_enddate   := trunc(SYSDATE);
      ELSE
        --大于等于延期时间点的，有效期顺延到第二天
        o_startdate := trunc(SYSDATE);
        o_enddate   := trunc(SYSDATE);
        o_enddate   := o_enddate + 1; --延期一天
      END IF;
    END IF;*/
    o_pj   := row_fqtitle.tprulepj; --0不区分规则 1同规则用券
    o_area := row_fqtitle.tpcertarea; --0通用 1本门店 2指定门店
    --是否区分面值
    IF TRIM(p_cardno) IS NULL THEN
      o_ismz    := row_fqtitle.tpmzmode;
      o_prnsite := row_fqtitle.tpprintplace;
    ELSE
      IF row_fqtitle.tphyfqmode = '0' THEN
        --电子券模式总是用规则合并
        o_ismz    := q_gzhb;
        o_prnsite := '2';
      ELSE
        o_ismz    := row_fqtitle.tpmzmode;
        o_prnsite := row_fqtitle.tpprintplace;
      END IF;
    END IF;
    --
    o_retcode := 0;
    o_retmsg  := '成功';
  EXCEPTION
    WHEN OTHERS THEN
      o_retcode := -1;
      o_retmsg  := 'gettktpopinfo失败!' || SQLCODE || ',' || SQLERRM;
      raise_application_error(errcode, 'gettktpopinfo失败!' || SQLERRM);
  END;
  --后台返券计算  重置商品返券规则  用于前台查找商品返券规则失败的情况
  PROCEDURE refreshfqrule
  (
    p_transno IN NUMBER,
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) IS
    cardtype    VARCHAR2(4);
    retcode     NUMBER; -- 返回值
    rseqno      NUMBER; -- 序号
    rdjbh       VARCHAR2(60); --  促销活动单号
    rtype       CHAR; --
    rcode       VARCHAR2(60); --
    rgz         VARCHAR2(60); -- 促销的方式(1-满减/2-减现)
    rdzxl       VARCHAR2(60); -- 除开券付款算满条件金额标志
    rpp         VARCHAR2(60); -- 跨柜组计算满条件标志
    rspec       VARCHAR2(60); --
    rksrq       VARCHAR2(60); --
    rjsrq       VARCHAR2(60); --
    rkssj       VARCHAR2(60); --
    rjssj       VARCHAR2(60); --
    rsl         NUMBER; -- 满减上限金额
    ryhspace    NUMBER; --组合码（1210,打折，满减，返券，返礼）
    rpoplsj     NUMBER; -- 满的条件金额： 满金额n
    rpophyj     NUMBER; -- 减现的金额：   减金额n
    rpoppfj     NUMBER; --  促销档期
    rpoplsjzkl  NUMBER; --BHLS:规则促销的打折比例
    rpophyjzkl  NUMBER; -- BHLS:促销折上折标志
    rpoppfjzkl  NUMBER; -- 忽略其他积分优惠（1是，0否）
    rpoplsjzkfd NUMBER; -- 折扣分担
    rpophyjzkfd NUMBER; --  规则促销折扣控制
    rpoppfjzkfd NUMBER; -- 积分倍率
    rrule       VARCHAR2(60); -- 商品属性码
    rmode       VARCHAR2(60); -- 无用
    rpopbillno  VARCHAR2(60); --
    rmemo       VARCHAR2(60); -- 满减规则,返券规则,返礼规则(逗号分格)
    rstr1       VARCHAR2(60); -- ALL:会员与非会员/@1:全部会员/@2:非会员/首字符为#:卡分组）
    rstr2       VARCHAR2(60); -- 活动单号,收券规则编码,收券规则单据编号,B收券模式,B条件金额 , 收券模式a,条件金额a
    rstr3       VARCHAR2(200); -- 多级满减(由大到小)：满金额n-1,减金额n-1;满金额n-2,减金额n-2;满金额n-3,减金额3-2; ...
    rnum1       NUMBER; -- 收A券比率： 0 不收，1 收
    rnum2       NUMBER; -- 收B券比率或满收金额
    ls_grouppop VARCHAR2(10);
    l_exists    NUMBER; --是否已返券
    l_sdfqrule  VARCHAR2(60);
    l_sdflrule  VARCHAR2(60);
    v_enablerule CHAR(1) := 'N';
  BEGIN
    v_enablerule := fterp_context.get_isenablerule;
    fterp_searchrule.enablerule(FALSE);
    FOR r_slt IN (SELECT sltbillno
                    FROM selllisttemp
                   WHERE slttransno = p_transno)
    LOOP
      --是否已返券
      BEGIN
        SELECT 1
          INTO l_exists
          FROM tktcardfqlog
         WHERE tcflsyjtrace = r_slt.sltbillno
           AND rownum = 1;
        ROLLBACK;
        o_retcode := 1;
        o_retmsg  := '重置商品返券规则失败,小票序号:' || to_char(r_slt.sltbillno) ||
                     '已返券!';
        RETURN;
      EXCEPTION
        WHEN OTHERS THEN
          NULL;
      END;
      --
      FOR r_sd IN (SELECT shmanaunit || ',' || shmarket mktcode, shdate,
                          shsyjid, shinvno,
                          to_char(shdate, 'YYYY/MM/DD HH24:MI:SS') yhsj,
                          TRIM(shcustno) shcustno, t1.*
                     FROM selldetail t1, sellhead t2
                    WHERE sdbillno = shbillno
                      AND sdbillno = r_slt.sltbillno)
      LOOP
        IF r_sd.shcustno IS NULL THEN
          cardtype := NULL;
        ELSE
          BEGIN
            SELECT cdmtype
              INTO cardtype
              FROM cardmain
             WHERE cdmno = r_sd.shcustno;
          EXCEPTION
            WHEN OTHERS THEN
              cardtype := NULL;
          END;
        END IF;
        sp_findrulecrm(r_sd.mktcode, r_sd.sdgdid, r_sd.sdmfid, r_sd.sdmsuid,
                       r_sd.sdspsx, r_sd.sdcatid, r_sd.sdppcode, r_sd.yhsj,
                       r_sd.shcustno, cardtype, retcode, rseqno, rdjbh,
                       rtype, rcode, rgz, rdzxl, rpp, rspec, rksrq, rjsrq,
                       rkssj, rjssj, rsl, ryhspace, rpoplsj, rpophyj,
                       rpoppfj, rpoplsjzkl, rpophyjzkl, rpoppfjzkl,
                       rpoplsjzkfd, rpophyjzkfd, rpoppfjzkfd, rrule, rmode,
                       rpopbillno, rmemo, rstr1, rstr2, rstr3, rnum1, rnum2);
        IF retcode = 1 THEN
          o_retcode := 0;
          o_retmsg  := '重置商品返券规则成功';
        ELSE
          o_retcode := 1;
          o_retmsg  := '重置商品返券规则失败:' || SQLERRM;
          RETURN;
        END IF;
        IF dbusrpop.fgetmanapara('T1') = 'Y' THEN
          ls_grouppop := rpad(to_char(ryhspace), 5, '0');
        ELSE
          ls_grouppop := rpad(to_char(ryhspace), 4, '0');
        END IF;
        l_sdfqrule := nvl(fgetpara(',' || rmemo || ',', ',', 2), '');
        l_sdflrule := nvl(fgetpara(',' || rmemo || ',', ',', 3), '');
        UPDATE selldetail
           SET sdpoprule = ls_grouppop, sdfqrule = l_sdfqrule,
               sdflrule = l_sdflrule
         WHERE sdbillno = r_sd.sdbillno
           AND sdrowno = r_sd.sdrowno;
        IF SQL%ROWCOUNT = 0 OR SQL%ROWCOUNT IS NULL THEN
          o_retcode := 0;
          o_retmsg  := '重置商品返券规则失败:' || SQLERRM;
          RETURN;
        END IF;
      END LOOP;
      --删除返券中间表
      DELETE FROM tktsellfqdetail WHERE tsfdcbillno = r_slt.sltbillno;
    END LOOP;
    IF v_enablerule = 'Y' THEN
      fterp_searchrule.enablerule;
    END IF;
    COMMIT;
  EXCEPTION
    WHEN OTHERS THEN
      IF v_enablerule = 'Y' THEN
        fterp_searchrule.enablerule;
      END IF;
      ROLLBACK;
      o_retcode := -1;
      o_retmsg  := '重置商品返券规则失败,交易号:' || to_char(p_transno) || ',' ||
                   to_char(SQLCODE) || ',' || SQLERRM;
  END;

  PROCEDURE uptshfqflag
  (
    p_transno IN NUMBER,
    p_billno  IN NUMBER,
    p_fqflag  IN VARCHAR2,
    p_saflag  IN VARCHAR2
  ) IS
  BEGIN
    FOR r_slt IN (SELECT *
                    FROM selllisttemp
                   WHERE slttransno = p_transno
                     AND (sltbillno = p_billno OR p_billno IS NULL))
    LOOP
      UPDATE sellhead
         SET shfqflag = decode(shfqflag, flagnofq, shfqflag, p_fqflag),
             shsaflag = decode(shsaflag, flagnofq, shsaflag, p_saflag)
       WHERE shbillno = r_slt.sltbillno;
      --TEMP
      IF p_fqflag = flagfinish THEN
        UPDATE crm_sellmail
           SET shfqflag = flagfinish
         WHERE shbillno = r_slt.sltbillno;
        /*INSERT INTO sellmail
        VALUES
          (SYSDATE, r_slt.sltbillno, r_slt.sltsyjid, r_slt.sltinvno, NULL);*/
      END IF;
    END LOOP;
    UPDATE selllisttemp
       SET sltstatus = p_fqflag
     WHERE slttransno = p_transno;
  EXCEPTION
    WHEN OTHERS THEN
      raise_application_error(errcode,
                              '小票状态修改失败!交易号:' || to_char(p_transno));
  END;

  --1.规则数据,  --小票满送规则
  -- insert tktsellfqdetail
  PROCEDURE sp_salemsrule
  (
    p_billno  IN NUMBER,
    p_transno IN NUMBER,
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) AS
    lispop       CHAR(1);
    l_crmmkt     VARCHAR2(20); --CRM MKT
    l_xssj       CHAR(5); --销售时间
    l_totfqje    NUMBER; --总可返券金额 根据行递减
    l_valid_fqje NUMBER; --行可返券金额
    l_totwxje    NUMBER; --总无效金额 根据行递减
    l_totlimitje NUMBER; --合计 受限券金额
    l_rowlimitje NUMBER; --行 受限券金额
    l_jesign     NUMBER;
    row_fqrule   tktrule%ROWTYPE; --小票明细的返券规则
    row_flrue    poprule%ROWTYPE; --满赠规则
    row_h        sellhead%ROWTYPE;
    row_d        selldetail%ROWTYPE;
    l_i          NUMBER; --规则促销小票行数
    l_exists     NUMBER;
    --lisgk        NUMBER;
    llever NUMBER;
    lmje   NUMBER;
    lfqje  NUMBER;
    lmk    NUMBER;
    ----
    --lrownum          NUMBER;
    l_act_invalid_je NUMBER; --实际无效金额
    l_temp1          NUMBER; --
    --l_rowkc   NUMBER; --行扣除
    l_flflag2 CHAR(1); --满赠 付款方式控制参数
    --返券有效金额分摊方法
    --1返券力度降序分摊有效金额 2行号降序分摊无效金额(卓展需求)
    l_fqvalidjeft CHAR(1) := TRIM(fgetmanapara('FQ_VALIDJEFT'));
    l_overnight   DATE; --通宵营业时间
    CURSOR cur_d IS
      SELECT *
        FROM selldetail
       WHERE sdbillno = p_billno
       ORDER BY decode(l_fqvalidjeft, '2', sdrowno, 1) DESC,
                fgetrulepriority(sdfqrule) DESC, sdrowno;
    --
    row_tsfd tktsellfqdetail%ROWTYPE;
    --
  BEGIN
    DELETE FROM tktsellfqdetail
     WHERE tsfdcal = 'Y'
       AND tsfdcbillno = p_billno;
    BEGIN
      --是否已计算
      SELECT 1
        INTO l_exists
        FROM tktsellfqdetail
       WHERE tsfdcbillno = p_billno
         AND rownum = 1;
      GOTO sp_salemsruleskip;
    EXCEPTION
      WHEN no_data_found THEN
        NULL;
    END;
    --查找小票信息
    BEGIN
      SELECT * INTO row_h FROM sellhead WHERE shbillno = p_billno;
    EXCEPTION
      WHEN no_data_found THEN
        o_retcode := 1;
        o_retmsg  := '未找到该小票!||小票序列号:' || to_char(p_billno);
        RETURN;
    END;
    IF row_h.shfqflag = 'Y' THEN
      o_retcode := 1;
      o_retmsg  := '小票已经返券!||小票序列号:' || to_char(p_billno);
      RETURN;
    END IF;
    --销售、红冲销售才返券
    -- 红冲销售走扣回 salethhc 过程 20090417
    IF fvirtualsaletype(row_h.shdjlb) <> '1' THEN
      o_retcode := 3;
      o_retmsg  := '不支持的小票类型!小票序列号:' || to_char(p_billno) || ', 小票类别:' ||
                   fvirtualsaletype(row_h.shdjlb);
      RETURN;
    END IF;
    --换货小票
    IF row_h.shishh = 'Y' THEN
      o_retcode := 3;
      o_retmsg  := '此为换货小票!小票序列号:' || to_char(p_billno) || ', 小票类别:' ||
                  fvirtualsaletype( row_h.shdjlb);
      RETURN;
    END IF;
    -- 查找小票 是否已经红冲 如果已红冲则不计算返券
    -- 销售小票A成时，上传了POS库，但没有成功上传GPP（断网） 没有返券，顾客有投诉
    -- 此时款台红冲小票A成功 再做同样一张小票B 成功 返券了
    -- 款台再联网操作 上传GPP时，不应再返券
    BEGIN
      SELECT shbillno
        INTO l_exists
        FROM sellhead
       WHERE shbillno = (SELECT sdbillno
                           FROM selldetail, sellhead
                          WHERE sdbillno = shbillno
                            AND shmarket = row_h.shmarket
                            AND shmanaunit = row_h.shmanaunit
                            AND sdysyjid = row_h.shsyjid
                            AND sdyinvno = row_h.shinvno
                            AND rownum = 1)
         AND shmarket = row_h.shmarket
         AND shmanaunit = row_h.shmanaunit
         AND fvirtualsaletype(shdjlb) = '2';
      o_retcode := 3;
      o_retmsg  := '小票'||to_char(p_billno)||'已经红冲!||红冲小票序列号:' || to_char(l_exists);
      RETURN;
    EXCEPTION
      WHEN no_data_found THEN
        NULL;
    END;
    -- 取消检查,前台能刷卡就可以返券
    --init 返券金额
    UPDATE sellhead SET shfqje = 0 WHERE shbillno = p_billno;
    --
    l_crmmkt := fgetcrmmkt(row_h.shmanaunit, row_h.shmarket);
    l_jesign := 1;
    FOR vr IN (SELECT ROWID, t.*
                 FROM sellpayment t
                WHERE spbillno = p_billno)
    LOOP
      lispop := least(fgetispoppay(row_h.shmanaunit, l_crmmkt, vr.sppmcode,
                                   vr.sppayerid),
                      fgetisczkpop(row_h.shmanaunit, vr.sppmcode, vr.spno));
      UPDATE sellpayment SET spispop = lispop WHERE ROWID = vr.ROWID;
    END LOOP;
    UPDATE sellpaygoods t1
       SET spgispop = (SELECT spispop
                          FROM sellpayment
                         WHERE spbillno = t1.spgbillno
                           AND sprowno = t1.spgrowno)
     WHERE spgbillno = p_billno;
    --计算小票中不允许返券的付款金额
    IF l_fqvalidjeft = '2' THEN
      SELECT round(nvl(SUM(spggdmoney * spgerate), 0), 2)
        INTO l_totlimitje --合计 受限付款金额
        FROM sellpaygoods
       WHERE spgbillno = p_billno
         AND spgispop = 'N';
    END IF;
    --
    SELECT round(nvl(SUM(spmoney * sperate), 0), 2)
      INTO l_temp1 --无效金额
      FROM sellpayment
     WHERE spbillno = p_billno
       AND spispop = 'N';
    --
    l_act_invalid_je := l_temp1 - row_h.shsysy;
    IF l_act_invalid_je < 0 THEN
      l_act_invalid_je := 0;
    END IF;
    l_totfqje := row_h.shoughtpay - l_act_invalid_je; --总可返券金额
    l_totwxje := l_act_invalid_je;
    --扣除 指定行受限
    l_totwxje := l_totwxje - l_totlimitje;
    IF l_totwxje < 0 THEN
      l_totwxje := 0;
    END IF;
    -- 考虑余额累计 就低 此时不RETURN 2009.04.16
    -- 小票当天返券余额累计
    l_overnight := fgetbusinessdate(row_h.shdate);
    IF yemode = 'Y' AND row_h.shcustno IS NOT NULL THEN
      BEGIN
        INSERT INTO tktsellfqdetail
          (tsfdbillno, tsfdrowno, tsfdpgmfid, tsfdpggdid, tsfdcjje, tsfdpid,
           tsfdfqrule, tsfdfqxfje, tsfdfqje, tsfdpayje, tsfdisjd, tsfdsyjid,
           tsfdinvno, tsfdno, tsfdtransno, tsfdyfqje, tsfdrfqye1, tsfdrfqye2,
           tsfdmaxsbqje, tsfdftqtfkkc, tsfdflrule, tsfdflskje, tsfdflyxje,
           tsfdflflag1, tsfdflflag2, tsfdmarket, tsfdmu, tsfdflflag,
           tsfdnum1, tsfdnum2, tsfdnum3, tsfdchr1, tsfdchr2, tsfdchr3,
           tsfdbcje, tsfdfqye, tsfdcbillno, tsfdcrowno, tsfddate,
           tsfdyefqrule, tsfdpriority, tsfdmfid, tsfdcal, tsfdptransno)
          SELECT tsfdbillno, tsfdrowno, tsfdpgmfid, tsfdpggdid, tsfdcjje,
                 tsfdpid, tsfdyefqrule, fqrule.trxfje, fqrule.trfaqje,
                 tsfdfqye, fqrule.isjd, row_h.shsyjid, row_h.shinvno, tsfdno,
                 p_transno, tsfdyfqje, tsfdfqye, tsfdfqye, tsfdmaxsbqje,
                 tsfdftqtfkkc, tsfdflrule, tsfdflskje, tsfdflyxje,
                 tsfdflflag1, tsfdflflag2, row_h.shmarket, row_h.shmanaunit,
                 tsfdflflag, tsfdnum1, tsfdnum2, tsfdnum3, fqmode, yemode,
                 tsfdchr3, tsfdfqye, tsfdfqye, p_billno, -1 * rownum,
                 tsfddate, tsfdyefqrule, 0, tsfdmfid, b_precal, tsfdtransno
            FROM tktsellfqdetail,
                 (SELECT trcode, trxfje, trfaqje, trstr2 isjd
                     FROM tktrule
                    WHERE trrulelb = '1'
                      AND trflag = 'Y') fqrule
           WHERE tsfddate = l_overnight--trunc(row_h.shdate)
             AND tsfdno = row_h.shcustno
             AND tsfdfqye > 0
             AND tsfdcal = 'N'
             AND tsfdcbillno < p_billno
             AND tsfdyefqrule = fqrule.trcode;
        l_exists := SQL%ROWCOUNT;
        --
        IF b_precal = 'N' THEN
          UPDATE tktsellfqdetail
             SET tsfdfqye = 0
           WHERE tsfddate = l_overnight--trunc(row_h.shdate)
             AND tsfdno = row_h.shcustno
             AND tsfdfqye > 0
             AND tsfdcbillno < p_billno;
        END IF;
        l_exists := SQL%ROWCOUNT;
      EXCEPTION
        WHEN OTHERS THEN
          o_retcode := -1;
          o_retmsg  := '小票余额累计返券执行失败1,小票序列号:' || to_char(p_billno) || ',' ||
                       to_char(SQLCODE) || ',' || SQLERRM;
          RETURN;
      END;
    END IF;
    --销售时间
    SELECT to_char(row_h.shdate, 'HH24:MI') INTO l_xssj FROM dual;
    --找出小票中所有可以返券的商品
    l_i := 0;
    OPEN cur_d;
    LOOP
      FETCH cur_d
        INTO row_d;
      EXIT WHEN cur_d%NOTFOUND OR cur_d%NOTFOUND IS NULL;
      --档期表 tktpopinfo
      --档期规则 tktruledet trrulelb->规则类型1:返券 3:追券 4:买券 B:满赠 C:满减
      --确认商品是否参与返券满赠,前台已获取
      --
      --行受限付款金额
      l_rowlimitje := fgetrowlimitje(row_d.sdbillno, row_d.sdrowno);
      --0000 打折||满减||返券||满赠
      IF substr(TRIM(row_d.sdpoprule), 3, 2) <> '00' THEN
        --根据成交金额分解受限金额
        --SELECT decode(row_h.shoughtpay,
        --              0,
        --              0,
        --              round(l_act_invalid_je * row_d.sdcjje / row_h.shoughtpay, 2))
        --  INTO l_rowkc
        --  FROM dual;
        --row_d.sdfqrule
        -- tktrule trrulelb='1'返券规则 trcode 返券规则
        row_fqrule := NULL;
        IF substr(TRIM(row_d.sdpoprule), 3, 1) <> '0' AND
           row_d.sdfqrule <> '0' THEN
          BEGIN
            SELECT *
              INTO row_fqrule
              FROM tktrule
             WHERE trcode = row_d.sdfqrule
               AND trrulelb = '1';
            IF row_fqrule.trfaqje = 0 OR row_fqrule.trxfje = 0 THEN
              o_retcode := 1;
              o_retmsg  := '返券规则[' || row_d.sdfqrule || ']满金额或返券总金额为零!';
              RETURN;
            END IF;
          EXCEPTION
            WHEN OTHERS THEN
              o_retcode := 1;
              o_retmsg  := '返券规则[' || row_d.sdfqrule || ']不存在!小票序列号:' ||
                           to_char(p_billno);
              RETURN;
          END;
          l_i := l_i + 1; --找到返券促销信息
        END IF;
        --row_d.sdflrule
        --popcurinfo popzpbarcode='DMJ' 满赠规则 popbillno
        l_flflag2 := '#';
        row_flrue := NULL;
        IF substr(TRIM(row_d.sdpoprule), 4, 1) <> '0' THEN
          BEGIN
            SELECT *
              INTO row_flrue
              FROM poprule
             WHERE prid = row_d.sdflrule
               AND prmode = 'B';
            l_flflag2 := row_flrue.prflag2;
          EXCEPTION
            WHEN OTHERS THEN
              o_retcode := 1;
              o_retmsg  := '满赠规则[' || row_d.sdflrule || ']不存在!小票序列号:' ||
                           to_char(p_billno);
              RETURN;
          END;
          --row_flrue.prflag1
          --相同赠送条件的赠品是全送或任选其一
          --row_flrue.prflag2
          --有限制的付款方式或参加了满减活动就整单都不能送
          --1.按整笔销售金额赠送
          --2.按受限付款方式外的金额赠送
          --3.有受限付款方式则整笔不赠送
        END IF;
        --返券有效金额分摊方法
        --1返券力度降序分摊有效金额 2行号降序分摊无效金额(卓展需求)
        --
        IF l_fqvalidjeft = '1' THEN
          --
          l_valid_fqje := least(l_totfqje, row_d.sdcjje - l_rowlimitje);
          l_totfqje    := l_totfqje - l_valid_fqje;
          IF l_totfqje < 0 THEN
            l_totfqje := 0;
          END IF;
        ELSIF l_fqvalidjeft = '2' THEN
          l_temp1      := least(l_totwxje, row_d.sdcjje - l_rowlimitje);
          l_valid_fqje := row_d.sdcjje - l_temp1 - l_rowlimitje;
          l_totwxje    := l_totwxje - l_temp1;
          IF l_totwxje < 0 THEN
            l_totwxje := 0;
          END IF;
        END IF;
        row_tsfd              := NULL;
        row_tsfd.tsfdbillno   := row_d.sdbillno; --小票billno
        row_tsfd.tsfdrowno    := row_d.sdrowno; --小票商品行号
        row_tsfd.tsfdpgmfid   := row_d.sdmfid; --商品柜组
        row_tsfd.tsfdpggdid   := row_d.sdgdid; --商品编码
        row_tsfd.tsfdcjje     := row_d.sdcjje * l_jesign; --成交金额
        row_tsfd.tsfdpid      := row_d.sdnum1; --档期
        row_tsfd.tsfdfqrule   := nvl(row_fqrule.trcode, '0'); --返券规则
        row_tsfd.tsfdfqxfje   := nvl(row_fqrule.trxfje, 0); --返券规则满金额
        row_tsfd.tsfdfqje     := nvl(row_fqrule.trfaqje, 0); --返券规则返总金额
        row_tsfd.tsfdisjd     := nvl(TRIM(row_fqrule.trstr2), '1'); --返券规则是否就低 1/0
        row_tsfd.tsfdpayje    := l_valid_fqje * l_jesign; --行可返券金额
        row_tsfd.tsfdsyjid    := row_h.shsyjid; --收银机
        row_tsfd.tsfdinvno    := row_h.shinvno; --发票号
        row_tsfd.tsfdno       := row_h.shcustno; --sellhead.shcustno 会员卡号
        row_tsfd.tsfdtransno  := p_transno; --返券事务号
        row_tsfd.tsfdyfqje    := 0; --应返券总金额
        row_tsfd.tsfdrfqye1   := l_valid_fqje * l_jesign; --行余额1  当前行计算
        row_tsfd.tsfdrfqye2   := l_valid_fqje * l_jesign; --行余额2  实际余额
        row_tsfd.tsfdmaxsbqje := 0;--row_d.sdn2; --最大收券额
        row_tsfd.tsfdftqtfkkc := 0; --分摊其他付款扣除
        row_tsfd.tsfdcal      := b_precal;
        row_tsfd.tsfdptransno := 0;
        SELECT decode(row_fqrule.trnum1, 0, row_tsfd.tsfdpgmfid, '0')
          INTO row_tsfd.tsfdmfid
          FROM dual;
        -- 满赠用------------> start
        --row_flrue.prflag1
        --相同赠送条件的赠品是全送或任选其一
        --row_flrue.prflag2
        --有限制的付款方式或参加了满减活动就整单都不能送
        --1.按整笔销售金额赠送
        --2.按受限付款方式外的金额赠送
        --3.有受限付款方式则整笔不赠送
        SELECT decode(l_flflag2, '#', NULL, row_d.sdflrule),
               --满赠规则(返礼)
               --l_jesign * decode(l_flflag2, '#', 0, '1', 0, l_rowkc), --满赠_实扣金额
               l_jesign *
                decode(l_flflag2, '#', 0, '1', row_d.sdcjje, '2',
                       l_valid_fqje, '3',
                       decode(sign(l_act_invalid_je), 0, row_d.sdcjje, 0)) --满赠有效金额
          INTO row_tsfd.tsfdflrule,
               --满赠规则(返礼)
               --row_tsfd.tsfdflskje, --满赠_实扣金额
               row_tsfd.tsfdflyxje --满赠有效金额
          FROM dual;
        row_tsfd.tsfdflskje := row_d.sdcjje - row_tsfd.tsfdflyxje;
        --
        row_tsfd.tsfdflflag1 := nvl(row_flrue.prflag1, 'N'); --相同赠送条件的赠品
        row_tsfd.tsfdflflag2 := nvl(row_flrue.prflag2, 'N'); --受限付款方式控制
        -- 满赠用<----------- end
        row_tsfd.tsfdmarket   := row_h.shmarket; --门店
        row_tsfd.tsfdmu       := row_h.shmanaunit; --经营公司
        row_tsfd.tsfdflflag   := 'N'; --是否已满赠
        row_tsfd.tsfdnum1     := 0;
        row_tsfd.tsfdnum2     := 0;
        row_tsfd.tsfdnum3     := 0;
        row_tsfd.tsfdchr1     := fqmode; --返券模式,对应manapara[q2]参数
        row_tsfd.tsfdchr2     := '';
        row_tsfd.tsfdchr3     := '';
        row_tsfd.tsfdbcje     := l_valid_fqje * l_jesign; --本次可返券金额
        row_tsfd.tsfdfqye     := l_valid_fqje * l_jesign; --小票返券余额
        row_tsfd.tsfdcbillno  := row_d.sdbillno; --当前小票号
        row_tsfd.tsfdcrowno   := row_d.sdrowno; --当前行号
        row_tsfd.tsfddate     := l_overnight; --小票日期 trunc(sellhead.shdate)
        row_tsfd.tsfdyefqrule := nvl(row_fqrule.trcode, '0'); --余额返券规则
        row_tsfd.tsfdpriority := 0; -- 返券力度优先级 降序
        --插入到小票返券明细表
        INSERT INTO tktsellfqdetail VALUES row_tsfd;
        IF l_fqvalidjeft = '1' AND l_totfqje = 0 THEN
          EXIT; --有效付款金额已分摊完
        END IF;
      ELSE
        IF l_fqvalidjeft = '1' THEN
          --
          l_valid_fqje := least(l_totfqje, row_d.sdcjje - l_rowlimitje);
          l_totfqje    := l_totfqje - row_d.sdcjje;
          IF l_totfqje < 0 THEN
            l_totfqje := 0;
          END IF;
        ELSIF l_fqvalidjeft = '2' THEN
          --l_temp1 := least(l_totwxje, row_d.sdcjje - l_rowlimitje);
          l_temp1      := row_d.sdcjje - l_rowlimitje;
          l_valid_fqje := row_d.sdcjje - l_temp1 - l_rowlimitje;
          l_totwxje    := l_totwxje - l_temp1;
          IF l_totwxje < 0 THEN
            l_totwxje := 0;
          END IF;
        END IF;
        --
      END IF;
    END LOOP;
    CLOSE cur_d;
    FOR vr IN (SELECT tsfdfqrule, tsfdmfid, SUM(tsfdrfqye2) xfje
                 FROM tktsellfqdetail
                WHERE tsfdcbillno = p_billno
                GROUP BY tsfdfqrule, tsfdmfid)
    LOOP

      llever := fgetrulelev(vr.tsfdfqrule, vr.xfje);

      lfqje := fgetrulefqje(vr.tsfdfqrule, llever);
      lmje  := fgetrulemje(vr.tsfdfqrule, llever);
      IF llever = 0 THEN
        lfqje := fgetrulefqje(vr.tsfdfqrule, 1);
        lmje  := fgetrulemje(vr.tsfdfqrule, 1);
      END IF;
      lmk := fgetrulemje(vr.tsfdfqrule, 1);
      UPDATE tktsellfqdetail
         SET tsfdfqje = lfqje, tsfdfqxfje = lmje, tsfdnum2 = llever
       WHERE tsfdfqrule = vr.tsfdfqrule
         AND tsfdmfid = vr.tsfdmfid
         AND tsfdcbillno = p_billno;
    END LOOP;
    -- 考虑余额累计 就低 此时不RETURN 2009.04.16
    --IF l_i = 0 THEN
    --  o_retcode := 2;
    --  o_retmsg  := '该小票没有参与促销的商品!小票序列号:' || to_char(p_billno);
    --  RETURN;
    --END IF;
    -- 更新返券优先级 降序

    UPDATE tktsellfqdetail a
       SET tsfdpriority = (SELECT r
                              FROM (SELECT rownum r, tsfdfqrule, frate, tsfdmfid
                                       FROM (SELECT tsfdfqrule,
                                                     tsfdfqje / tsfdfqxfje frate,
                                                     tsfdmfid
                                                FROM tktsellfqdetail b
                                               WHERE tsfdtransno = p_transno
                                                 AND tsfdfqrule <> '0'
                                               GROUP BY tsfdfqrule,
                                                        tsfdfqje / tsfdfqxfje,
                                                        tsfdmfid
                                               ORDER BY frate DESC, tsfdfqrule DESC,
                                                        tsfdmfid DESC)) b
                             WHERE b.tsfdfqrule = a.tsfdfqrule
                               AND b.tsfdmfid = a.tsfdmfid)
     WHERE tsfdtransno = p_transno
       AND tsfdfqrule <> '0';
    FOR vr IN (SELECT tsfdfqrule, tsfdmfid, SUM(tsfdrfqye2) xfje
                 FROM tktsellfqdetail
                WHERE tsfdcbillno = p_billno
                  AND tsfdnum2 = 0
                GROUP BY tsfdfqrule, tsfdmfid)
    LOOP
      lfqje := fgetrulefqje(vr.tsfdfqrule, 1);
      lmje  := fgetrulemje(vr.tsfdfqrule, 1);

      UPDATE tktsellfqdetail
         SET tsfdfqje = lfqje, tsfdfqxfje = lmje
       WHERE tsfdfqrule = vr.tsfdfqrule
         AND tsfdmfid = vr.tsfdmfid
         AND tsfdcbillno = p_billno
         AND tsfdnum2 = 0;
    END LOOP;
    /*   lrownum :=0;
    for vr in (select rowid id  from  tktsellfqdetail WHERE tsfdtransno = p_transno AND tsfdfqrule <> '0' order by
       tsfdfqje / tsfdfqxfje desc , tsfdmfid asc )
      loop
      lrownum := lrownum + 1;
      update tktsellfqdetail
        set  tsfdpriority = lrownum
        where  rowid = vr.id;
       end loop;*/
    l_exists := SQL%ROWCOUNT;
    dbms_output.put_line(l_exists);
    --
    <<sp_salemsruleskip>>
    o_retcode := 0;
    o_retmsg  := '执行成功';
  EXCEPTION
    WHEN OTHERS THEN
      IF cur_d%ISOPEN THEN
        CLOSE cur_d;
      END IF;
      ROLLBACK;
      o_retcode := -1;
      o_retmsg  := '执行失败,小票序列号:' || to_char(p_billno) || ',' ||
                   to_char(SQLCODE) || ',' || SQLERRM;
  END;
  PROCEDURE upttkt
  (
    i_r_tcft  IN tktcardfqtotal%ROWTYPE,
    i_cardno  IN VARCHAR2,
    i_billno  IN NUMBER,
    i_person  IN VARCHAR2,
    i_syjid   IN VARCHAR2,
    i_invno   IN VARCHAR2,
    i_source  IN VARCHAR2,
    i_uptins  IN VARCHAR2,
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) IS
    l_r_tcft tktcardfqtotal%ROWTYPE;
    l_r_tcfl tktcardfqlog%ROWTYPE;
    l_seq    NUMBER;
    l_ye     NUMBER;
    l_ins    NUMBER := 0;
    l_ismz   CHAR(1); --是否固定面值
    l_topicstartdate DATE;
    l_topicenddate   DATE;
  BEGIN
    IF debug = 'Y' THEN
      GOTO upttktskip;
    END IF;
    IF i_uptins NOT IN ('U', 'I') THEN
      o_retcode := -1;
      o_retmsg  := 'i_uptins:' || i_uptins || '错误！';
      RETURN;
    END IF;
    l_r_tcft := i_r_tcft;
    IF l_r_tcft.tcftfqje = 0 THEN
      o_retcode := -1;
      o_retmsg  := '券金额:' || l_r_tcft.tcftfqje || '为零！';
      RETURN;
    END IF;
    gettktpopinfo(l_r_tcft.tcftfstpid, i_cardno, 'ALL', o_retcode, o_retmsg,
                  --l_r_tcft.tcftstartdate, l_r_tcft.tcftenddate,
                  l_topicstartdate, l_topicenddate,
                  l_r_tcft.tcftvipno, l_r_tcft.tcfttype, l_r_tcft.tcftrulepj,
                  l_r_tcft.tcftmkt, l_ismz, l_r_tcfl.tcflchr3);
    IF o_retcode <> 0 THEN
      RETURN;
    END IF;
    IF l_r_tcft.tcftstartdate IS NULL THEN
      l_r_tcft.tcftstartdate := l_topicstartdate;
    END IF;
    IF l_r_tcft.tcftenddate IS NULL THEN
      l_r_tcft.tcftenddate := l_topicenddate;
    END IF;
    l_r_tcft.tcftstatus  := 'Y';
    l_r_tcft.tcftflag    := 'Y';
    l_r_tcft.tcftcltpid  := l_r_tcft.tcftfstpid;
    l_r_tcft.tcftvipexno := l_r_tcft.tcftvipno;
    l_r_tcft.tcftsource  := i_source;
    IF b_precal <> 'Y' THEN
      IF i_uptins = 'U' THEN
        --VIP卡返券金额
        UPDATE tktcardfqtotal
           SET tcftfqje = tcftfqje + l_r_tcft.tcftfqje
         WHERE tcftvipno = l_r_tcft.tcftvipno
           AND tcftstartdate = l_r_tcft.tcftstartdate
           AND tcftenddate = l_r_tcft.tcftenddate
           AND tcftqtype = l_r_tcft.tcftqtype --券种A～Z
           AND tcftfstpid = l_r_tcft.tcftfstpid
              --AND tcftfqrule = l_r_tcft.tcftfqrule
              --AND tcftrulepj = l_r_tcft.tcftrulepj
           AND tcftstatus = l_r_tcft.tcftstatus
           AND tcfttype = l_r_tcft.tcfttype --电子券，手工券
           AND tcftmkt = l_r_tcft.tcftmkt --券使用范围，--0通用 1本门店 2指定门店
           AND tcftjygs = l_r_tcft.tcftjygs
           AND tcftfqmkt = l_r_tcft.tcftfqmkt
           AND tcftsource = l_r_tcft.tcftsource
           AND tcftflag = l_r_tcft.tcftflag
           AND rownum = 1
        RETURNING tcftseqno, tcftfqje INTO l_seq, l_ye;
        IF SQL%ROWCOUNT = 0 OR SQL%ROWCOUNT IS NULL THEN
          l_ins := 1;
        END IF;
      END IF;
      IF i_uptins = 'I' OR l_ins = 1 THEN
        SELECT seq_tktcardfqtotal.NEXTVAL INTO l_seq FROM dual;
        l_r_tcft.tcftseqno := l_seq;
        l_ye               := l_r_tcft.tcftfqje;
        INSERT INTO tktcardfqtotal VALUES l_r_tcft;
      END IF;
    END IF;

    l_r_tcfl.tcflsource := i_source;
    IF l_r_tcfl.tcflsource IN ('1', '3', '9') THEN
      l_r_tcfl.tcflzy := 'F';
    ELSIF l_r_tcfl.tcflsource = '2' THEN
      l_r_tcfl.tcflzy := 'm';
    ELSIF l_r_tcfl.tcflsource = '4' THEN
      l_r_tcfl.tcflzy := 'P';
    END IF;
    --插入日志
    IF b_precal <> 'Y' THEN
      INSERT INTO tktcardfqlog
        (tcflseqno, tcflvipno, tcflstartdate, tcflenddate, tcfldate, tcflzy,
         tcflpopid, tcflpoprule, tcflmoney, tcflye, tcfloper, tcflsyjid,
         tcflinvno, tcflmemo, tcflsyjtrace, tcfljetype, tcflvipseq, tcflmkt,
         tcfljygs, tcflsource, tcfltransno, tcfltype, tcflchr1, tcflchr2,
         tcflchr3, tcflnum1)
      VALUES
        (seq_tktcardfqlog.NEXTVAL, l_r_tcft.tcftvipno,
         l_r_tcft.tcftstartdate, l_r_tcft.tcftenddate, SYSDATE,
         l_r_tcfl.tcflzy, l_r_tcft.tcftfstpid, l_r_tcft.tcftfqrule,
         l_r_tcft.tcftfqje, l_ye, i_person, i_syjid, i_invno, '', i_billno,
         l_r_tcft.tcftqtype, l_seq, l_r_tcft.tcftfqmkt, l_r_tcft.tcftjygs,
         l_r_tcfl.tcflsource, l_r_tcft.tcftransno, l_r_tcft.tcfttype,
         bankcode, sppayno, l_r_tcfl.tcflchr3, 0);
    ELSE
      INSERT INTO tktcardfqcal
        (tcflseqno, tcflvipno, tcflstartdate, tcflenddate, tcfldate, tcflzy,
         tcflpopid, tcflpoprule, tcflmoney, tcflye, tcfloper, tcflsyjid,
         tcflinvno, tcflmemo, tcflsyjtrace, tcfljetype, tcflvipseq, tcflmkt,
         tcfljygs, tcflsource, tcfltransno, tcfltype, tcflchr1, tcflchr2,
         tcflchr3, tcflnum1)
      VALUES
        (seq_tktcardfqlog.NEXTVAL, l_r_tcft.tcftvipno,
         l_r_tcft.tcftstartdate, l_r_tcft.tcftenddate, SYSDATE,
         l_r_tcfl.tcflzy, l_r_tcft.tcftfstpid, l_r_tcft.tcftfqrule,
         l_r_tcft.tcftfqje, l_ye, i_person, i_syjid, i_invno, '', i_billno,
         l_r_tcft.tcftqtype, 1, l_r_tcft.tcftfqmkt, l_r_tcft.tcftjygs,
         l_r_tcfl.tcflsource, l_r_tcft.tcftransno, l_r_tcft.tcfttype,
         bankcode, sppayno, l_r_tcfl.tcflchr3, 0);
    END IF;
    <<upttktskip>>
    o_retcode := 0;
    o_retmsg  := '成功';
  EXCEPTION
    WHEN OTHERS THEN
      o_retcode := 1;
      o_retmsg  := '失败:' || to_char(SQLCODE) || ',' || SQLERRM;
  END;
  PROCEDURE calctkt_sa
  (
    p_transno   IN NUMBER,
    p_mu        IN VARCHAR2,
    p_mkt       IN VARCHAR2,
    p_ismz      IN VARCHAR2,
    p_cardno    IN VARCHAR2,
    p_billno    IN NUMBER,
    p_syjid     IN VARCHAR2,
    p_invno     IN VARCHAR2,
    p_rule      IN VARCHAR2, --返券规则或者 银行追送规则
    p_fqrulemje IN NUMBER,
    p_fqxspay   IN NUMBER,
    p_max       IN NUMBER, --最大可返券金额
    p_tpid      IN VARCHAR2,
    p_person    IN VARCHAR2,
    p_source    IN VARCHAR2, --券来源1:销售返券 2:买券 3:银行追送 TCFLSOURCE
    o_retcode   OUT NUMBER,
    o_retmsg    OUT VARCHAR2,
    o_totfqje   OUT NUMBER
  ) AS
    CURSOR cur_fqdet(vrule IN VARCHAR2) IS
      SELECT * FROM tktrulefqdet WHERE trdcode = vrule;
    row_fqdet tktrulefqdet%ROWTYPE;
    l_exists  NUMBER;
    l_fqje    NUMBER;
    CURSOR cur_mz(vqtype IN VARCHAR2) IS
      SELECT *
        FROM tktcardmzrec
       WHERE tcmtransno = p_transno
         AND tcmqtype = vqtype
       ORDER BY tcmqtype, tcmmz;
    row_mz tktcardmzrec%ROWTYPE;
    --
    l_r_tcft  tktcardfqtotal%ROWTYPE;
    l_totfqje NUMBER := 0;
    --l_dj      NUMBER;
    --l_mje     NUMBER;
    l_maxfqje NUMBER;
    --
    row_tktexp    tktexpiredef%ROWTYPE;
    vcurtime      CHAR(20);
    l_delaytime   CHAR(20);
    l_topenddate  DATE;
    lyqstartdate  DATE;
    lyqenddate    DATE;
  BEGIN
    -- init r_rulefq
    l_maxfqje := p_max;
    r_rulefq  := tktqjelist();
    -- l_dj      := fgetrulelev(p_rule, p_fqxspay);
    -- l_mje     := fgetrulemje(p_rule, l_dj);
    OPEN cur_fqdet(p_rule); --返券规则 返明细券种
    LOOP
      FETCH cur_fqdet
        INTO row_fqdet;
      EXIT WHEN cur_fqdet%NOTFOUND OR cur_fqdet%NOTFOUND IS NULL;
      -- row_fqdet.trdqje 返券规则返券明细金额
      --计算行返券金额
      IF l_maxfqje = 0 THEN
        GOTO sucend;
      END IF;
      --按券种有效期
      BEGIN
        SELECT * INTO row_tktexp FROM tktexpiredef
         WHERE tedcode = p_tpid AND tedqtype = row_fqdet.trdqtype;
      EXCEPTION
        WHEN OTHERS THEN
          o_retcode := -1;
          o_retmsg  := '档期[' || p_tpid || ']没有定义券种[' || row_fqdet.trdqtype || ']的有效期!';
          RETURN;
      END;
      --SELECT substr(to_char(SYSDATE, 'yyyy-mm-dd HH24:mi:ss'), 12, 10) INTO vcurtime FROM dual;
      IF Nvl(row_tktexp.tedisoneday,'N') <> 'N' THEN
        SELECT to_char(SYSDATE, 'yyyy-mm-dd HH24:mi:ss')
          INTO vcurtime
          FROM dual;
        /*SELECT trunc(tpiyqenddate)
          INTO l_topenddate
          FROM tktpopinfo
         WHERE tpiid = p_tpid;*/
        SELECT to_char(Sysdate, 'yyyy-mm-dd') || ' ' || row_tktexp.tedtime
          INTO l_delaytime
          FROM dual;
        --IF vcurtime < row_tktexp.tedtime THEN  --不用延期，当日有效
        IF vcurtime < l_delaytime THEN --不用延期，当日有效
          lyqstartdate := fGetbusinessdate(SYSDATE);
          lyqenddate   := fGetbusinessdate(SYSDATE);
        ELSE  --大于等于延期时间点的，有效期顺延到第二天
          lyqstartdate := fGetbusinessdate(SYSDATE);
          lyqenddate   := fGetbusinessdate(SYSDATE);
          lyqenddate  := lyqenddate + 1;
        END IF;
      ELSE
        lyqstartdate := row_tktexp.tedyqstartdate;
        lyqenddate   := row_tktexp.tedyqenddate;
      END IF;
      --
      l_fqje := trunc(p_fqxspay / p_fqrulemje, 0) * row_fqdet.trdqje;
      IF p_ismz = q_gdmz THEN
        --固定面值
        IF row_fqdet.trdqje > 0 THEN
          --面值记录到列表
          INSERT INTO tktcardmzrec
            (tcmtransno, tcmmz, tcmnum, tcmqtype)
          VALUES
            (p_transno, row_fqdet.trdqje, trunc(p_fqxspay / p_fqrulemje, 0),
             row_fqdet.trdqtype);
        END IF;
      END IF;
      --返券按面值记录多张
      IF p_ismz = q_gdmz THEN
        OPEN cur_mz(row_fqdet.trdqtype);
        LOOP
          FETCH cur_mz
            INTO row_mz;
          EXIT WHEN cur_mz%NOTFOUND OR cur_mz%NOTFOUND IS NULL;
          FOR i IN 1 .. row_mz.tcmnum
          LOOP
            l_r_tcft            := NULL;
            l_r_tcft.tcftfstpid := p_tpid;
            l_r_tcft.tcftfqrule := p_rule;
            l_r_tcft.tcftfqje   := row_mz.tcmmz;
            l_r_tcft.tcftqtype  := row_fqdet.trdqtype;
            l_r_tcft.tcftjygs   := p_mu;
            l_r_tcft.tcftfqmkt  := p_mkt;
            l_r_tcft.tcftransno := p_transno;
            l_r_tcft.tcftstartdate := lyqstartdate;
            l_r_tcft.tcftenddate   := lyqenddate;
            IF l_r_tcft.tcftfqje <= l_maxfqje THEN
              l_maxfqje := l_maxfqje - l_r_tcft.tcftfqje;
            ELSE
              l_r_tcft.tcftfqje := l_maxfqje;
              l_maxfqje         := 0;
            END IF;
            --记录券update tktcardfqtotal,tktcardfqlog
            upttkt(i_r_tcft => l_r_tcft, i_cardno => p_cardno,
                   i_billno => p_billno, i_person => p_person,
                   i_syjid => p_syjid, i_invno => p_invno,
                   i_source => p_source, i_uptins => 'I',
                   o_retcode => o_retcode, o_retmsg => o_retmsg);
            IF o_retcode <> 0 THEN
              RETURN;
            END IF;
            -- 记录返券券种及余额
            l_exists := 0;
            FOR i IN 1 .. r_rulefq.COUNT
            LOOP
              IF r_rulefq(i).qtype = l_r_tcft.tcftqtype THEN
                l_exists := 1;
                r_rulefq(i).qje := r_rulefq(i).qje + l_r_tcft.tcftfqje;
              END IF;
            END LOOP;
            IF l_exists = 0 THEN
              r_rulefq.EXTEND;
              r_rulefq(r_rulefq.LAST).qtype := l_r_tcft.tcftqtype;
              r_rulefq(r_rulefq.LAST).qje := l_r_tcft.tcftfqje;
            END IF;
            --
          END LOOP; --面值张数 FOR i IN 1 .. row_mz.tcmnum LOOP
        END LOOP; --面值 OPEN cur_mz; tktcardmzrec
        CLOSE cur_mz;
      ELSE
        --非固定面值
        l_r_tcft            := NULL;
        l_r_tcft.tcftfstpid := p_tpid;
        l_r_tcft.tcftfqrule := p_rule;
        l_r_tcft.tcftfqje   := l_fqje;
        l_r_tcft.tcftqtype  := row_fqdet.trdqtype;
        l_r_tcft.tcftjygs   := p_mu;
        l_r_tcft.tcftfqmkt  := p_mkt;
        l_r_tcft.tcftransno := p_transno;
        l_r_tcft.tcftstartdate := lyqstartdate;
        l_r_tcft.tcftenddate   := lyqenddate;
        IF l_r_tcft.tcftfqje <= l_maxfqje THEN
          l_maxfqje := l_maxfqje - l_r_tcft.tcftfqje;
        ELSE
          l_r_tcft.tcftfqje := l_maxfqje;
          l_maxfqje         := 0;
        END IF;
        --记录券update tktcardfqtotal,tktcardfqlog
        upttkt(i_r_tcft => l_r_tcft, i_cardno => p_cardno,
               i_billno => p_billno, i_person => p_person,
               i_syjid => p_syjid, i_invno => p_invno, i_source => p_source,
               i_uptins => 'U', o_retcode => o_retcode, o_retmsg => o_retmsg);
        IF o_retcode <> 0 THEN
          RETURN;
        END IF;
        --
        -- 记录返券券种及余额
        r_rulefq.EXTEND;
        r_rulefq(r_rulefq.LAST).qtype := l_r_tcft.tcftqtype;
        r_rulefq(r_rulefq.LAST).qje := l_r_tcft.tcftfqje;
        --
      END IF;
      l_totfqje := l_totfqje + l_fqje;
    END LOOP; -- open cur_fqdet 返券规则 返明细券种
    CLOSE cur_fqdet;
    --
    <<sucend>>
    o_retcode := 0;
    o_retmsg  := '成功';
    o_totfqje := l_totfqje;
  EXCEPTION
    WHEN OTHERS THEN
      o_retcode := -1;
      o_retmsg  := 'calctkt失败,' || to_char(SQLCODE) || ',' || SQLERRM;
      o_totfqje := 0;
  END;
  PROCEDURE calctkt_fq
  (
    p_transno   IN NUMBER,
    p_mu        IN VARCHAR2,
    p_mkt       IN VARCHAR2,
    p_ismz      IN VARCHAR2,
    p_cardno    IN VARCHAR2,
    p_billno    IN NUMBER,
    p_syjid     IN VARCHAR2,
    p_invno     IN VARCHAR2,
    p_rule      IN VARCHAR2, --返券规则或者 银行追送规则
    p_fqrulemje IN NUMBER,
    p_fqxspay   IN NUMBER,
    p_tpid      IN VARCHAR2,
    p_person    IN VARCHAR2,
    p_source    IN VARCHAR2, --券来源1:销售返券 2:买券 3:银行追送 TCFLSOURCE
    o_retcode   OUT NUMBER,
    o_retmsg    OUT VARCHAR2,
    o_totfqje   OUT NUMBER,
    o_xfje      OUT NUMBER
  ) AS
    CURSOR cur_fqdet(vrule IN VARCHAR2, vdj IN NUMBER) IS
      SELECT *
        FROM tktrulefqdet_ex
       WHERE trdcode = vrule
         AND trdnum1 = vdj;
    row_fqdet tktrulefqdet_ex%ROWTYPE;
    l_exists  NUMBER;
    l_fqje    NUMBER;
    CURSOR cur_mz(vqtype IN VARCHAR2) IS
      SELECT *
        FROM tktcardmzrec
       WHERE tcmtransno = p_transno
         AND tcmqtype = vqtype
       ORDER BY tcmqtype, tcmmz;
    row_mz tktcardmzrec%ROWTYPE;
    --
    l_r_tcft  tktcardfqtotal%ROWTYPE;
    l_totfqje NUMBER := 0;
    l_dj      NUMBER;
    l_mje     NUMBER;
    l_fqjeavg NUMBER;
    l_maxfqje NUMBER;
    l_fqtemp  NUMBER;
    --
    row_tktexp    tktexpiredef%ROWTYPE;
    vcurtime      CHAR(20);
    l_delaytime   CHAR(20);
    l_topenddate  DATE;
    lyqstartdate  DATE;
    lyqenddate    DATE;
  BEGIN
    -- init r_rulefq
    l_fqtemp  := 0;
    r_rulefq  := tktqjelist();
    l_dj      := fgetrulelev(p_rule, p_fqxspay);
    l_mje     := fgetrulemje(p_rule, l_dj);
    l_fqjeavg := fgetrulefqje(p_rule, l_dj);
    l_maxfqje := fgetrulemax(p_rule);
    OPEN cur_fqdet(p_rule, l_dj); --返券规则 返明细券种
    LOOP
      FETCH cur_fqdet
        INTO row_fqdet;
      EXIT WHEN cur_fqdet%NOTFOUND OR cur_fqdet%NOTFOUND IS NULL;
      -- row_fqdet.trdqje 返券规则返券明细金额
      --计算行返券金额

      l_fqje := trunc(p_fqxspay / l_mje, 0) * row_fqdet.trdqje;
      IF l_maxfqje - l_fqtemp < l_fqje THEN
        l_fqje   := l_maxfqje - l_fqtemp;
        l_fqtemp := l_maxfqje;
      ELSE
        l_fqtemp := l_fqtemp + l_fqje;
      END IF;
      --按券种有效期
      BEGIN
        SELECT * INTO row_tktexp FROM tktexpiredef
         WHERE tedcode = p_tpid AND tedqtype = row_fqdet.trdqtype;
      EXCEPTION
        WHEN OTHERS THEN
          o_retcode := -1;
          o_retmsg  := '档期[' || p_tpid || ']没有定义券种[' || row_fqdet.trdqtype || ']的有效期!';
          RETURN;
      END;
      --SELECT substr(to_char(SYSDATE, 'yyyy-mm-dd HH24:mi:ss'), 12, 10) INTO vcurtime FROM dual;
      IF Nvl(row_tktexp.tedisoneday,'N') <> 'N' THEN
        SELECT to_char(SYSDATE, 'yyyy-mm-dd HH24:mi:ss')
          INTO vcurtime
          FROM dual;
        /*SELECT trunc(tpiyqenddate)
          INTO l_topenddate
          FROM tktpopinfo
         WHERE tpiid = p_tpid;*/
        SELECT to_char(Sysdate, 'yyyy-mm-dd') || ' ' || row_tktexp.tedtime
          INTO l_delaytime
          FROM dual;
        --IF vcurtime < row_tktexp.tedtime THEN  --不用延期，当日有效
        IF vcurtime < l_delaytime THEN --不用延期，当日有效
          lyqstartdate := fGetbusinessdate(SYSDATE);
          lyqenddate   := fGetbusinessdate(SYSDATE);
        ELSE  --大于等于延期时间点的，有效期顺延到第二天
          lyqstartdate := fGetbusinessdate(SYSDATE);
          lyqenddate   := fGetbusinessdate(SYSDATE);
          lyqenddate  := lyqenddate + 1;
        END IF;
      ELSE
        lyqstartdate := row_tktexp.tedyqstartdate;
        lyqenddate   := row_tktexp.tedyqenddate;
      END IF;

      IF p_ismz = q_gdmz THEN
        --固定面值
        IF row_fqdet.trdqje > 0 THEN
          --面值记录到列表
          INSERT INTO tktcardmzrec
            (tcmtransno, tcmmz, tcmnum, tcmqtype)
          VALUES
            (p_transno, row_fqdet.trdqje, trunc(p_fqxspay / p_fqrulemje, 0),
             row_fqdet.trdqtype);
        END IF;
      END IF;
      --返券按面值记录多张
      IF p_ismz = q_gdmz THEN
        OPEN cur_mz(row_fqdet.trdqtype);
        LOOP
          FETCH cur_mz
            INTO row_mz;
          EXIT WHEN cur_mz%NOTFOUND OR cur_mz%NOTFOUND IS NULL;
          FOR i IN 1 .. row_mz.tcmnum
          LOOP
            l_r_tcft            := NULL;
            l_r_tcft.tcftfstpid := p_tpid;
            l_r_tcft.tcftfqrule := p_rule;
            l_r_tcft.tcftfqje   := row_mz.tcmmz;
            l_r_tcft.tcftqtype  := row_fqdet.trdqtype;
            l_r_tcft.tcftjygs   := p_mu;
            l_r_tcft.tcftfqmkt  := p_mkt;
            l_r_tcft.tcftransno := p_transno;
            l_r_tcft.tcftstartdate := lyqstartdate;
            l_r_tcft.tcftenddate   := lyqenddate;
            --记录券update tktcardfqtotal,tktcardfqlog
            upttkt(i_r_tcft => l_r_tcft, i_cardno => p_cardno,
                   i_billno => p_billno, i_person => p_person,
                   i_syjid => p_syjid, i_invno => p_invno,
                   i_source => p_source, i_uptins => 'I',
                   o_retcode => o_retcode, o_retmsg => o_retmsg);
            IF o_retcode <> 0 THEN
              RETURN;
            END IF;
            -- 记录返券券种及余额
            l_exists := 0;
            FOR i IN 1 .. r_rulefq.COUNT
            LOOP
              IF r_rulefq(i).qtype = l_r_tcft.tcftqtype THEN
                l_exists := 1;
                r_rulefq(i).qje := r_rulefq(i).qje + l_r_tcft.tcftfqje;
              END IF;
            END LOOP;
            IF l_exists = 0 THEN
              r_rulefq.EXTEND;
              r_rulefq(r_rulefq.LAST).qtype := l_r_tcft.tcftqtype;
              r_rulefq(r_rulefq.LAST).qje := l_r_tcft.tcftfqje;
            END IF;
            --
          END LOOP; --面值张数 FOR i IN 1 .. row_mz.tcmnum LOOP
        END LOOP; --面值 OPEN cur_mz; tktcardmzrec
        CLOSE cur_mz;
      ELSE
        --非固定面值
        l_r_tcft            := NULL;
        l_r_tcft.tcftfstpid := p_tpid;
        l_r_tcft.tcftfqrule := p_rule;
        l_r_tcft.tcftfqje   := l_fqje;
        l_r_tcft.tcftqtype  := row_fqdet.trdqtype;
        l_r_tcft.tcftjygs   := p_mu;
        l_r_tcft.tcftfqmkt  := p_mkt;
        l_r_tcft.tcftransno := p_transno;
        l_r_tcft.tcftstartdate := lyqstartdate;
        l_r_tcft.tcftenddate   := lyqenddate;
        --记录券update tktcardfqtotal,tktcardfqlog
        upttkt(i_r_tcft => l_r_tcft, i_cardno => p_cardno,
               i_billno => p_billno, i_person => p_person,
               i_syjid => p_syjid, i_invno => p_invno, i_source => p_source,
               i_uptins => 'U', o_retcode => o_retcode, o_retmsg => o_retmsg);
        IF o_retcode <> 0 THEN
          RETURN;
        END IF;
        --
        -- 记录返券券种及余额
        r_rulefq.EXTEND;
        r_rulefq(r_rulefq.LAST).qtype := l_r_tcft.tcftqtype;
        r_rulefq(r_rulefq.LAST).qje := l_r_tcft.tcftfqje;
        --
      END IF;
      l_totfqje := l_totfqje + l_fqje;
    END LOOP; -- open cur_fqdet 返券规则 返明细券种
    CLOSE cur_fqdet;
    --
    o_retcode := 0;
    o_retmsg  := '成功';
    o_totfqje := l_totfqje;
  EXCEPTION
    WHEN OTHERS THEN
      o_retcode := -1;
      o_retmsg  := 'calctkt失败,' || to_char(SQLCODE) || ',' || SQLERRM;
      o_totfqje := 0;
  END;
  PROCEDURE calctkt
  (
    p_transno   IN NUMBER,
    p_mu        IN VARCHAR2,
    p_mkt       IN VARCHAR2,
    p_ismz      IN VARCHAR2,
    p_cardno    IN VARCHAR2,
    p_billno    IN NUMBER,
    p_syjid     IN VARCHAR2,
    p_invno     IN VARCHAR2,
    p_rule      IN VARCHAR2, --返券规则或者 银行追送规则
    p_fqrulemje IN NUMBER,
    p_fqxspay   IN NUMBER,
    p_tpid      IN VARCHAR2,
    p_person    IN VARCHAR2,
    p_source    IN VARCHAR2, --券来源1:销售返券 2:买券 3:银行追送 TCFLSOURCE
    o_retcode   OUT NUMBER,
    o_retmsg    OUT VARCHAR2,
    o_totfqje   OUT NUMBER
  ) AS
    CURSOR cur_fqdet(vrule IN VARCHAR2) IS
      SELECT * FROM tktrulefqdet WHERE trdcode = vrule;
    row_fqdet tktrulefqdet%ROWTYPE;
    l_exists  NUMBER;
    l_fqje    NUMBER;
    CURSOR cur_mz(vqtype IN VARCHAR2) IS
      SELECT *
        FROM tktcardmzrec
       WHERE tcmtransno = p_transno
         AND tcmqtype = vqtype
       ORDER BY tcmqtype, tcmmz;
    row_mz tktcardmzrec%ROWTYPE;
    --
    l_r_tcft  tktcardfqtotal%ROWTYPE;
    l_totfqje NUMBER := 0;
    --
    row_tktexp    tktexpiredef%ROWTYPE;
    vcurtime      CHAR(20);
    l_delaytime   CHAR(20);
    l_topenddate  DATE;
    lyqstartdate  DATE;
    lyqenddate    DATE;
  BEGIN
    -- init r_rulefq
    r_rulefq := tktqjelist();
    OPEN cur_fqdet(p_rule); --返券规则 返明细券种
    LOOP
      FETCH cur_fqdet
        INTO row_fqdet;
      EXIT WHEN cur_fqdet%NOTFOUND OR cur_fqdet%NOTFOUND IS NULL;
      -- row_fqdet.trdqje 返券规则返券明细金额
      --计算行返券金额
      l_fqje := trunc(p_fqxspay / p_fqrulemje, 0) * row_fqdet.trdqje;
      --按券种有效期
      BEGIN
        SELECT * INTO row_tktexp FROM tktexpiredef
         WHERE tedcode = p_tpid AND tedqtype = row_fqdet.trdqtype;
      EXCEPTION
        WHEN OTHERS THEN
          o_retcode := -1;
          o_retmsg  := '档期[' || p_tpid || ']没有定义券种[' || row_fqdet.trdqtype || ']的有效期!';
          RETURN;
      END;
      --SELECT substr(to_char(SYSDATE, 'yyyy-mm-dd HH24:mi:ss'), 12, 10) INTO vcurtime FROM dual;
      IF Nvl(row_tktexp.tedisoneday,'N') <> 'N' THEN
        SELECT to_char(SYSDATE, 'yyyy-mm-dd HH24:mi:ss')
          INTO vcurtime
          FROM dual;
        /*SELECT trunc(tpiyqenddate)
          INTO l_topenddate
          FROM tktpopinfo
         WHERE tpiid = p_tpid;*/
        SELECT to_char(Sysdate, 'yyyy-mm-dd') || ' ' || row_tktexp.tedtime
          INTO l_delaytime
          FROM dual;
        --  IF vcurtime < row_expire.tedtime THEN --不用延期，当日有效
        IF vcurtime < l_delaytime THEN --不用延期，当日有效
          lyqstartdate := fGetbusinessdate(SYSDATE);
          lyqenddate   := fGetbusinessdate(SYSDATE);
        ELSE  --大于等于延期时间点的，有效期顺延到第二天
          lyqstartdate := fGetbusinessdate(SYSDATE);
          lyqenddate   := fGetbusinessdate(SYSDATE);
          lyqenddate  := lyqenddate + 1;
        END IF;
      ELSE
        lyqstartdate := row_tktexp.tedyqstartdate;
        lyqenddate   := row_tktexp.tedyqenddate;
      END IF;
      IF p_ismz = q_gdmz THEN
        --固定面值
        IF row_fqdet.trdqje > 0 THEN
          --面值记录到列表
          INSERT INTO tktcardmzrec
            (tcmtransno, tcmmz, tcmnum, tcmqtype)
          VALUES
            (p_transno, row_fqdet.trdqje, trunc(p_fqxspay / p_fqrulemje, 0),
             row_fqdet.trdqtype);
        END IF;
      END IF;
      --返券按面值记录多张
      IF p_ismz = q_gdmz THEN
        OPEN cur_mz(row_fqdet.trdqtype);
        LOOP
          FETCH cur_mz
            INTO row_mz;
          EXIT WHEN cur_mz%NOTFOUND OR cur_mz%NOTFOUND IS NULL;
          FOR i IN 1 .. row_mz.tcmnum
          LOOP
            l_r_tcft            := NULL;
            l_r_tcft.tcftfstpid := p_tpid;
            l_r_tcft.tcftfqrule := p_rule;
            l_r_tcft.tcftfqje   := row_mz.tcmmz;
            l_r_tcft.tcftqtype  := row_fqdet.trdqtype;
            l_r_tcft.tcftjygs   := p_mu;
            l_r_tcft.tcftfqmkt  := p_mkt;
            l_r_tcft.tcftransno := p_transno;
            l_r_tcft.tcftstartdate := lyqstartdate;
            l_r_tcft.tcftenddate   := lyqenddate;
            --记录券update tktcardfqtotal,tktcardfqlog
            upttkt(i_r_tcft => l_r_tcft, i_cardno => p_cardno,
                   i_billno => p_billno, i_person => p_person,
                   i_syjid => p_syjid, i_invno => p_invno,
                   i_source => p_source, i_uptins => 'I',
                   o_retcode => o_retcode, o_retmsg => o_retmsg);
            IF o_retcode <> 0 THEN
              RETURN;
            END IF;
            -- 记录返券券种及余额
            l_exists := 0;
            FOR i IN 1 .. r_rulefq.COUNT
            LOOP
              IF r_rulefq(i).qtype = l_r_tcft.tcftqtype THEN
                l_exists := 1;
                r_rulefq(i).qje := r_rulefq(i).qje + l_r_tcft.tcftfqje;
              END IF;
            END LOOP;
            IF l_exists = 0 THEN
              r_rulefq.EXTEND;
              r_rulefq(r_rulefq.LAST).qtype := l_r_tcft.tcftqtype;
              r_rulefq(r_rulefq.LAST).qje := l_r_tcft.tcftfqje;
            END IF;
            --
          END LOOP; --面值张数 FOR i IN 1 .. row_mz.tcmnum LOOP
        END LOOP; --面值 OPEN cur_mz; tktcardmzrec
        CLOSE cur_mz;
      ELSE
        --非固定面值
        l_r_tcft            := NULL;
        l_r_tcft.tcftfstpid := p_tpid;
        l_r_tcft.tcftfqrule := p_rule;
        l_r_tcft.tcftfqje   := l_fqje;
        l_r_tcft.tcftqtype  := row_fqdet.trdqtype;
        l_r_tcft.tcftjygs   := p_mu;
        l_r_tcft.tcftfqmkt  := p_mkt;
        l_r_tcft.tcftransno := p_transno;
        l_r_tcft.tcftstartdate := lyqstartdate;
        l_r_tcft.tcftenddate   := lyqenddate;
        --记录券update tktcardfqtotal,tktcardfqlog
        upttkt(i_r_tcft => l_r_tcft, i_cardno => p_cardno,
               i_billno => p_billno, i_person => p_person,
               i_syjid => p_syjid, i_invno => p_invno, i_source => p_source,
               i_uptins => 'U', o_retcode => o_retcode, o_retmsg => o_retmsg);
        IF o_retcode <> 0 THEN
          RETURN;
        END IF;
        --
        -- 记录返券券种及余额
        r_rulefq.EXTEND;
        r_rulefq(r_rulefq.LAST).qtype := l_r_tcft.tcftqtype;
        r_rulefq(r_rulefq.LAST).qje := l_r_tcft.tcftfqje;
        --
      END IF;
      l_totfqje := l_totfqje + l_fqje;
    END LOOP; -- open cur_fqdet 返券规则 返明细券种
    CLOSE cur_fqdet;
    --
    o_retcode := 0;
    o_retmsg  := '成功';
    o_totfqje := l_totfqje;
  EXCEPTION
    WHEN OTHERS THEN
      o_retcode := -1;
      o_retmsg  := 'calctkt失败,' || to_char(SQLCODE) || ',' || SQLERRM;
      o_totfqje := 0;
  END;
  --2 计算返券, 分摊返券金额到小票行
  -- insert tktcardfqtotal（返券余额）, tktcardfqlog（返券日志），tktsellfqamt（返券分摊）
  PROCEDURE sp_salefq
  (
    p_transno IN NUMBER,
    p_person  IN VARCHAR2,
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) IS
    l_fqpid      VARCHAR2(20); --档期
    l_fqrule     VARCHAR2(20); --返券规则号
    l_valid_fqje NUMBER; --有效返券金额
    l_fqje       NUMBER; --行返券金额
    l_ftqje      NUMBER; --券分摊
    l_ftpay      NUMBER; --分摊付款金额
    i            NUMBER;
    j            NUMBER;
    l_bcje       NUMBER;
    l_totfqpay   NUMBER;
    l_tmp        NUMBER; --
    l_dj         NUMBER;
    --档期信息
    vstartdate DATE;
    venddate   DATE;
    l_pj       CHAR(1);
    l_area     VARCHAR2(20);
    l_ismz     CHAR(1);
    l_prtsite  CHAR(1);
    l_a_m      CHAR(1); --1:VIP电子券 q_auto, 2:手工券 q_manu
    l_qno      VARCHAR2(20); --券号
    --
    row_ft      tktsellfqdetail%ROWTYPE;
    l_fqrulemje NUMBER; --返券规则满金额
    l_fqrulefje NUMBER; --返券规则返总金额
    l_fqrulejd  CHAR(1); --返券规则是否就低 1/0
    l_priority  NUMBER; --优先级
    l_fqxspay   NUMBER;
    l_cardno    VARCHAR2(20); --vip卡号
    --
    l_billno NUMBER;
    l_syjid  VARCHAR2(20);
    l_invno  VARCHAR2(20);
    l_mkt    VARCHAR2(20);
    l_mu     VARCHAR2(20);
    l_fqmkt  VARCHAR2(20); --返券门店CRM
    l_exists NUMBER;
    l_debit  NUMBER;
    --
    cur_fqmx       cursorms;
    cur_fqft       cursorms;
    l_prioritylist VARCHAR2(500);
    l_totfqje      NUMBER;
    l_custno       VARCHAR2(20);
    l_custtype     VARCHAR2(10);
    l_mfid         VARCHAR2(20);
  BEGIN
    BEGIN
      SELECT 1
        INTO l_exists
        FROM tktcardfqlog
       WHERE tcfltransno = p_transno
         AND tcflsource = '1'
         AND rownum = 1;
      GOTO sp_salefqskip; --已计算
    EXCEPTION
      WHEN no_data_found THEN
        NULL;
    END;
    --
    j := 0;
    --按返券规则分组统计小票返券金额
    IF bone = 1 THEN
      OPEN cur_fqmx FOR
        SELECT MAX(tsfdpid), tsfdfqrule, tsfdno, tsfdcbillno, tsfdsyjid,
               tsfdinvno, tsfdmarket, tsfdmu, tsfdmfid,
               SUM(nvl(tsfdbcje, 0)) valid_fqje, MAX(tsfdfqxfje) rulemje,
               MAX(tsfdfqje) rulefje, MAX(nvl(tsfdisjd, '1')) rulejd,
               MAX(nvl(tsfdpriority, 0)) rulepriority
          FROM tktsellfqdetail
         WHERE tsfdtransno = p_transno
           AND tsfdfqrule <> '0'
         GROUP BY tsfdfqrule, tsfdno, tsfdcbillno, tsfdsyjid, tsfdinvno,
                  tsfdmarket, tsfdmu, tsfdmfid
         ORDER BY rulepriority, tsfdfqrule;
    ELSE
      OPEN cur_fqmx FOR
        SELECT MAX(tsfdpid), tsfdfqrule, tsfdno, NULL tsfdcbillno,
               '' tsfdsyjid, '' tsfdinvno, tsfdmarket, tsfdmu, tsfdmfid,
               SUM(nvl(tsfdbcje, 0)) valid_fqje, MAX(tsfdfqxfje) rulemje,
               MAX(tsfdfqje) rulefje, MAX(nvl(tsfdisjd, '1')) rulejd,
               MAX(nvl(tsfdpriority, 0)) rulepriority
          FROM tktsellfqdetail
         WHERE tsfdtransno = p_transno
           AND tsfdfqrule <> '0'
         GROUP BY tsfdfqrule, tsfdno, tsfdmarket, tsfdmu, tsfdmfid
         ORDER BY rulepriority, tsfdfqrule;
    END IF;
    LOOP
      FETCH cur_fqmx
        INTO l_fqpid, l_fqrule, l_cardno, l_billno, l_syjid, l_invno, l_mkt, l_mu, l_mfid, l_valid_fqje, l_fqrulemje, l_fqrulefje, l_fqrulejd, l_priority;
      EXIT WHEN cur_fqmx%NOTFOUND OR cur_fqmx%NOTFOUND IS NULL;
      --得到是否固定面值l_ismz 不生成券号
      gettktpopinfo(l_fqpid, l_cardno, 'GDMZ', o_retcode, o_retmsg,
                    vstartdate, venddate, l_qno, l_a_m, l_pj, l_area, l_ismz,
                    l_prtsite);
      IF o_retcode <> 0 THEN
        RETURN;
      END IF;
      --消费基数至少一分钱
      IF l_fqrulemje <= 0 THEN
        l_fqrulemje := 0.01;
      END IF;
      l_fqmkt := fgetcrmmkt(l_mu, l_mkt);
      DELETE FROM tktcardmzrec;
      --
      --如果全局允许就低 AND 返券规则允许就低
      l_debit        := 0;
      l_prioritylist := '';
      IF fqmode = '1' AND l_fqrulejd = '1' THEN
        FOR vr IN (SELECT tsfdpriority, tsfdmfid,
                          nvl(SUM(tsfdrfqye2), 0) rye2
                     FROM tktsellfqdetail
                    WHERE tsfdtransno = p_transno
                      AND tsfdrfqye2 > 0
                      AND tsfdpriority < l_priority
                      AND tsfdpriority <> 0
                      AND tsfdisjd = '1'
                      AND tsfdmfid = l_mfid
                    GROUP BY tsfdpriority, tsfdmfid)
        LOOP
          IF l_debit = 0 THEN
            l_debit := 1;
          END IF;
          l_valid_fqje   := l_valid_fqje + vr.rye2;
          l_prioritylist := l_prioritylist || ',' ||
                            to_char(vr.tsfdpriority);

        END LOOP;
      END IF;
      --销售金额
      l_fqxspay := l_valid_fqje;
      --
      bankcode := NULL;
      sppayno  := NULL;
      --有效的付款金额足够返券
      l_dj := fgetrulelev(l_fqrule, l_fqxspay);
      IF l_fqxspay >= l_fqrulemje AND l_dj > 0 THEN
        -- 计算返券
        calctkt_fq(p_transno => p_transno, p_mu => l_mu, p_mkt => l_fqmkt,
                   p_ismz => l_ismz, p_cardno => l_cardno,
                   p_billno => l_billno, p_syjid => l_syjid,
                   p_invno => l_invno, p_rule => l_fqrule,
                   --返券规则或者 银行追送规则
                   p_fqrulemje => l_fqrulemje, p_fqxspay => l_fqxspay,
                   p_tpid => l_fqpid, p_person => p_person, p_source => '1',
                   --
                   o_retcode => o_retcode, o_retmsg => o_retmsg,
                   o_totfqje => l_totfqje, o_xfje => l_totfqje);
      ELSE
        -- init r_rulefq
        r_rulefq := tktqjelist();
      END IF;
      --
      -- 返券余额递减
      --100 := SELECT 1000 - trunc(1000 / 300, 0) * 300 FROM dual;
      IF l_dj > 0 THEN
        l_fqxspay := l_fqxspay -
                     trunc(l_fqxspay / l_fqrulemje, 0) * l_fqrulemje;
        --参与分摊的金额
        --900 := SELECT 1000 - 100 FROM dual;
        l_totfqpay := l_valid_fqje - l_fqxspay;
      ELSE
        l_fqxspay  := 0;
        l_totfqpay := 0;
      END IF;
      --
      --            行  规则     销售  行余额1  行余额2 小票余额
      --第一次计算  1   100,50   199   99       99      99
      --第2行分摊后                    99       99      99
      --第3行分摊后                    99       25      25

      --第一次计算  2   150,50   158   107      107     107
      --第2行分摊后                    107      8       8
      --第3行分摊后                    107      0       0

      --第一次计算  3   200,50   318   25       25      25
      --第3行分摊后                    25       0       0
      --
      --
      --1  以下为返券分摊
      --计算小票明细商品返券
      --
      ---分摊券金额 tktsellfqamt
      IF r_rulefq.COUNT > 0 THEN
        l_custno := l_cardno;
        BEGIN
          SELECT cdmtype
            INTO l_custtype
            FROM cardmain
           WHERE cdmno = l_cardno;
        EXCEPTION
          WHEN OTHERS THEN
            l_custtype := '';
        END;
        l_tmp  := l_totfqpay;
        row_ft := NULL;
        --'Q4'参数取消
        /*
        IF TRIM(fgetmanapara('Q4')) = '1' THEN
          --1 全部分摊到当前小票
          OPEN cur_fqft FOR
            SELECT *
              FROM tktsellfqdetail
             WHERE tsfdtransno = p_transno
               AND tsfdrfqye2 > 0
               AND (tsfdfqrule = l_fqrule OR
                   (l_debit = 1 AND
                   instr(l_prioritylist, to_char(tsfdpriority)) > 0))
               AND tsfdcrowno > 0
             ORDER BY tsfdcbillno,
                      tsfdpriority DESC,
                      tsfdbillno,
                      tsfdcrowno;
        ELSE
        */
        -- 2 余额小票参与分摊
        r_rulefq_tmp := r_rulefq;
        OPEN cur_fqft FOR
          SELECT *
            FROM tktsellfqdetail
           WHERE tsfdtransno = p_transno
             AND tsfdrfqye2 > 0
             AND (tsfdfqrule = l_fqrule OR
                 (l_debit = 1 AND
                 instr(l_prioritylist, to_char(tsfdpriority)) > 0))
             AND tsfdmfid = l_mfid
           ORDER BY tsfdcbillno, tsfdpriority DESC, tsfdbillno, tsfdcrowno;
        --END IF;
        LOOP
          FETCH cur_fqft
            INTO row_ft;
          EXIT WHEN cur_fqft%NOTFOUND OR cur_fqft%NOTFOUND IS NULL;
          EXIT WHEN l_tmp <= 0;
          l_ftpay := least(row_ft.tsfdrfqye2, l_tmp);
          l_tmp   := l_tmp - l_ftpay;
          FOR i IN 1 .. r_rulefq.COUNT
          LOOP
            j      := j + 1;
            l_fqje := r_rulefq(i).qje;
            IF l_totfqpay = 0 THEN
              l_ftqje := 0;
            ELSE
              -- 行参与返券金额 / 总参与返券金额 * 返金额
              IF l_tmp <= 0 THEN
                l_ftqje := r_rulefq_tmp(i).qje;
              ELSE
                l_ftqje := round(l_ftpay / l_totfqpay * l_fqje, 2);
              END IF;
            END IF;
            r_rulefq_tmp(i).qje := r_rulefq_tmp(i).qje - l_ftqje;
            IF l_ftqje <> 0 AND b_precal = 'N' THEN
              INSERT INTO tktsellfqamt
                (tsfabillno, tsfarowno, tsfapggdid, tsfaseq, tsfatransno,
                 tsfacbillno, tsfacrowno, tsfaqtype, tsfaqje, tsfavalidje,
                 tsfatotvalidje, tsfatotqje, tsfacjje, tsfaiskh)
              VALUES
                (row_ft.tsfdbillno, row_ft.tsfdrowno, row_ft.tsfdpggdid, j,
                 p_transno, row_ft.tsfdcbillno, row_ft.tsfdcrowno,
                 r_rulefq(i).qtype, l_ftqje, l_ftpay, l_totfqpay, l_fqje,
                 row_ft.tsfdcjje, 'Y');
              --更新小票返券金额
              UPDATE sellhead
                 SET shfqje = nvl(shfqje, 0) + l_ftqje
               WHERE shbillno = row_ft.tsfdcbillno;
              IF row_ft.tsfdcrowno < 0 THEN
                INSERT INTO salebatchtask
                  (sbtseq, sbtbillno, sbtrowno, sbttype, sbtfqje, sbtcustno,
                   sbtcusttype, sbtjygs, sbtmarket)
                VALUES
                  (seq_salebatchtask.NEXTVAL, row_ft.tsfdbillno,
                   row_ft.tsfdrowno, 'F', l_ftqje, l_custno, l_custtype,
                   row_ft.tsfdmu, row_ft.tsfdmarket);
              END IF;
            END IF;
          END LOOP;

        END LOOP;
        CLOSE cur_fqft;
        --
      END IF; -- IF r_rulefq.COUNT > 0 THEN
      --
      --2 当前返券规则 按小票明细分摊返券后的 行余额
      --DD把余额全部借下来DD  多余的金额还回到各行
      l_tmp := l_totfqpay;
      OPEN cur_fqft FOR
        SELECT *
          FROM tktsellfqdetail
         WHERE tsfdtransno = p_transno
           AND tsfdrfqye2 > 0
           AND (tsfdfqrule = l_fqrule OR
               (l_debit = 1 AND
               instr(l_prioritylist, to_char(tsfdpriority)) > 0))
           AND tsfdmfid = l_mfid
         ORDER BY tsfdcbillno, tsfdpriority DESC, tsfdbillno, tsfdcrowno;
      LOOP
        FETCH cur_fqft
          INTO row_ft;
        EXIT WHEN cur_fqft%NOTFOUND OR cur_fqft%NOTFOUND IS NULL;
        --
        l_bcje := least(row_ft.tsfdrfqye2, l_tmp);
        --
        UPDATE tktsellfqdetail
           SET tsfdrfqye1 = decode(tsfdfqrule, l_fqrule, tsfdrfqye1 - l_bcje,
                                    tsfdrfqye1),
               --计算行余额1
               tsfdrfqye2 = tsfdrfqye2 - l_bcje,
               --计算行余额2
               tsfdfqye = tsfdfqye - l_bcje --计算小票余额
         WHERE tsfdbillno = row_ft.tsfdbillno
           AND tsfdrowno = row_ft.tsfdrowno
           AND tsfdcbillno = row_ft.tsfdcbillno
           AND tsfdcrowno = row_ft.tsfdcrowno;
        --
        l_tmp := l_tmp - l_bcje;

      END LOOP;
      CLOSE cur_fqft;
      NULL;
    END LOOP;
    CLOSE cur_fqmx;
    --
    <<sp_salefqskip>>
    o_retcode := 0;
    o_retmsg  := '执行成功';
  EXCEPTION
    WHEN OTHERS THEN
      IF cur_fqft%ISOPEN THEN
        CLOSE cur_fqft;
      END IF;
      IF cur_fqmx%ISOPEN THEN
        CLOSE cur_fqmx;
      END IF;
      o_retcode := -1;
      o_retmsg  := '执行失败,' || to_char(SQLCODE) || ',' || SQLERRM;
  END;

  FUNCTION fgetsarule
  (
    i_date IN DATE,
    jygs   IN VARCHAR2,
    mkt    IN VARCHAR2,
    code   IN VARCHAR2,
    gz     IN VARCHAR2,
    catid  IN VARCHAR2,
    pp     IN VARCHAR2,
    spsx   IN VARCHAR2,
    payno  IN VARCHAR2,
    p_type IN CHAR --追送类型 3-银行追送 5-商场追送
  ) RETURN VARCHAR2 AS
    l_ret    NUMBER;
    l_retmsg VARCHAR2(255);
    --l_sarule saruleinfo%ROWTYPE;
    l_seq    NUMBER;
    l_gbid   VARCHAR2(13);
    l_retstr VARCHAR2(200);
  BEGIN
    --子母码商品
    /*
    --暂不考虑子母码商品 性能为先
    BEGIN
      SELECT nvl(gbanalcode, gbid)
        INTO l_gbid
        FROM goodsbase
       WHERE gbid = code
         AND gbmanaunit = jygs;
    EXCEPTION
      WHEN no_data_found THEN
        l_gbid := code;
    END;
    */
    l_gbid := code;
    --
    --l_sarule := NULL;
    SELECT nvl(MAX(sriseqno), 0)
      INTO l_seq
      FROM saruleinfo
     WHERE sriksrq <= i_date
       AND srijsrq >= i_date
       AND srtype = p_type
       AND (p_type = '2' AND ((payno LIKE REPLACE(srino1, '*', '%') AND
           instr(srino1, '*') > 0) OR
           (srino1 <= payno AND srino2 >= payno AND
           instr(srino1, '*') = 0)) OR p_type = '5')
       AND srijygs = jygs --经营公司
       AND srimkt = mkt
       AND ((srigdid = l_gbid AND (srimfid = gz OR srimfid = 'ALL') AND
           srimode = '1') OR ((srimfid = gz OR srimfid = 'ALL') AND
           (srippcode = pp OR srippcode = 'ALL') AND
           (sricatid = catid OR sricatid = 'ALL') AND
           (srispsx = spsx OR srispsx = '0' OR
           srispsx IS NULL) AND srimode = '2'));
    IF l_seq = 0 THEN
      l_ret    := 0;
      l_retmsg := 'no data found';
      RETURN '0';
    END IF;
    -- 追券规则,档期,银行编号-单卡限额
    SELECT ',' || sriruleid || ',' || sritopid || ',' ||
            nvl(sribankcode, '') || ',' || srdkxe || ',' || sribillno || ',' str
      INTO l_retstr
      FROM saruleinfo
     WHERE sriseqno = l_seq;
    --
    RETURN l_retstr;
    l_ret    := 0;
    l_retmsg := '成功';
  EXCEPTION
    WHEN OTHERS THEN
      l_ret    := -1;
      l_retmsg := substr(to_char(SQLCODE) || ',' || SQLERRM, 1, 255);
      dbms_output.put_line(to_char(l_ret) || ',' || l_retmsg);
      RETURN '0';
  END;

    PROCEDURE sp_salesa_sc --追送设置
    (
      p_transno IN NUMBER,
      p_person  IN VARCHAR2,
      o_retcode OUT NUMBER,
      o_retmsg  OUT VARCHAR2
    ) AS
      vstartdate     DATE;
      venddate       DATE;
      l_qno          VARCHAR2(20);
      l_a_m          CHAR(1); --1:VIP电子券,2:手工券
      l_pj           CHAR(1);
      l_area         VARCHAR2(20);
      l_ismz         CHAR(1);
      l_prtsite      CHAR(1);
      r_tktrule      tktrule%ROWTYPE;
      l_mkje         NUMBER;
      l_maxfqje      NUMBER;
      l_exists       NUMBER;
      l_billno       selllisttemp.sltbillno%TYPE;
      l_totfqje      NUMBER;
      l_fqje         NUMBER;
      l_ftpay        NUMBER;
      l_totfqpay     NUMBER;
      i              NUMBER := 0;
      j              NUMBER := 0;
      l_ftqje        NUMBER;
      l_rowlimitje   NUMBER; --行 受限券金额
      l_rowyftkfje   NUMBER; --行 前台已分摊可返券金额

      l_rowvalidfqje NUMBER;

      --1返券力度降序分摊有效金额 2行号降序分摊无效金额(卓展需求)
      l_fqvalidjeft CHAR(1) := TRIM(fgetmanapara('FQ_VALIDJEFT'));
    BEGIN
      BEGIN
        SELECT 1
          INTO l_exists
          FROM tktcardfqlog
         WHERE tcfltransno = p_transno
           AND tcflsource = '9'
           AND rownum = 1;
        GOTO sp_salesaskip;
      EXCEPTION
        WHEN no_data_found THEN
          NULL;
      END;
      UPDATE sellhead
         SET shsaflag = flagnofq
       WHERE shbillno IN
             (SELECT sltbillno FROM selllisttemp WHERE slttransno = p_transno);
      --
      BEGIN
        SELECT 1
          INTO l_exists
          FROM sellhead
         WHERE shbillno IN (SELECT sltbillno
                              FROM selllisttemp
                             WHERE slttransno = p_transno)
           AND (fvirtualsaletype(shdjlb) <> '1' OR (fvirtualsaletype(shdjlb) = '1' AND shishh = 'Y'))
           AND rownum = 1;
        o_retcode := 2;
        o_retmsg  := '不是销售不票,或者是换货小票,不需要计算追送!';
        RETURN;
      EXCEPTION
        WHEN no_data_found THEN
          NULL;
      END;
      FOR v_slt IN (SELECT sltbillno
                      FROM selllisttemp
                     WHERE slttransno = p_transno)
      LOOP
        l_billno := v_slt.sltbillno;
        sp_salesa_scrule(l_billno, p_transno, l_totvalidje, o_retcode,
                         o_retmsg);
        IF o_retcode <> 0 THEN
          IF o_retcode = 2 THEN
            GOTO sp_salesaskip;
          ELSE
            RETURN;
          END IF;
        END IF;
        --
        IF l_totvalidje <= 0 THEN
          GOTO sp_salesaskip;
        END IF;
        FOR vr IN (SELECT *
                     FROM (SELECT tsfdmu, tsfdmarket, tsfdbillno, tsfdsyjid,
                                   tsfdinvno, tsfdpid, tsfdfqrule, tsfdno,
                                   tsfdfqxfje, tsfdfqje, SUM(tsfdcjje) cjje,
                                   SUM(tsfdrowlimit) tsfdrowlimit,
                                   SUM(tsfdrowlimit) tsfdrowyftjf
                              FROM tktsellscsadetail
                             WHERE tsfdtransno = p_transno
                               AND tsfdfqrule <> 0
                               AND tsfdfqxfje <> 0

                                      tsfdinvno, tsfdpid, tsfdfqrule, tsfdno,
                                      tsfdfqxfje, tsfdfqje
                            HAVING SUM(tsfdcjje) - SUM(tsfdrowlimit) <> 0)
                    ORDER BY trunc((cjje - tsfdrowlimit) / tsfdfqxfje) *
                              tsfdfqje DESC)
        LOOP
          --检查追送规则
          BEGIN
            SELECT *
              INTO r_tktrule
              FROM tktrule
             WHERE trcode = vr.tsfdfqrule;
          EXCEPTION
            WHEN OTHERS THEN
              o_retcode := 1;
              o_retmsg  := '商场追送规则:' || vr.tsfdfqrule || '不存在！';
              RETURN;
          END;
          l_mkje    := r_tktrule.trmkje;
          l_maxfqje := r_tktrule.trmkaqje;
          IF l_fqvalidjeft = '2' THEN
            l_valid_fqje := vr.cjje - vr.tsfdrowlimit;
          ELSE
            --已分摊不能返券金额
            l_rowlimitje := vr.tsfdrowlimit;
            --已分摊可返券金额
            l_rowyftkfje := vr.tsfdrowyftjf;
            l_totvalidje := l_totvalidje + l_rowyftkfje;
            l_valid_fqje := least(l_totvalidje, vr.cjje - l_rowlimitje);
            --有效金额为满金额的整数倍
            l_valid_fqje := trunc(l_valid_fqje / r_tktrule.trxfje) *
                            r_tktrule.trxfje;
            --有效金额不能小于 已分摊可返券金额
            IF l_valid_fqje < l_rowyftkfje THEN
              l_rowyftkfje := l_rowyftkfje;
            END IF;
            l_totvalidje := l_totvalidje - l_valid_fqje;
            ---待摊到每行的有效金额
            l_rowvalidfqje := l_valid_fqje - l_rowyftkfje;
          END IF;
          IF l_valid_fqje < l_mkje THEN
            GOTO nextrow;
          END IF;
          IF l_valid_fqje <= 0 THEN
            GOTO nextrow;
          END IF;
          IF l_valid_fqje < r_tktrule.trxfje THEN
            GOTO nextrow;
          END IF;
          IF l_maxfqje <= 0 OR l_maxfqje IS NULL THEN
            l_maxfqje := 99999999999;
          END IF;
          gettktpopinfo(vr.tsfdpid, vr.tsfdno, 'ALL', o_retcode, o_retmsg,
                        vstartdate, venddate, l_qno, l_a_m, l_pj, l_area,
                        l_ismz, l_prtsite);
          IF o_retcode <> 0 THEN
            RETURN;
          END IF;
          DELETE FROM tktcardmzrec;
          calctkt_sa(p_transno => p_transno, p_mu => vr.tsfdmu,
                     p_mkt => vr.tsfdmarket, p_ismz => l_ismz,
                     p_cardno => vr.tsfdno, p_billno => vr.tsfdbillno,
                     p_syjid => vr.tsfdsyjid, p_invno => vr.tsfdinvno,
                     p_rule => r_tktrule.trcode,
                     --返券规则或者 银行追送规则
                     p_fqrulemje => r_tktrule.trxfje,
                     p_fqxspay => l_valid_fqje, p_max => l_maxfqje,
                     p_tpid => vr.tsfdpid, p_person => p_person,
                     p_source => '9', o_retcode => o_retcode,
                     o_retmsg => o_retmsg, o_totfqje => l_totfqje);
          IF o_retcode <> 0 THEN
            RETURN;
          END IF;
          IF b_precal <> 'Y' THEN
            ---分摊券
            FOR row_ft IN (SELECT tktsellscsadetail.*, ROWID
                             FROM tktsellscsadetail
                            WHERE tsfdmu = vr.tsfdmu
                              AND tsfdmarket = vr.tsfdmarket
                              AND tsfdbillno = vr.tsfdbillno
                              AND tsfdsyjid = vr.tsfdsyjid
                              AND tsfdinvno = vr.tsfdinvno
                              AND tsfdpid = vr.tsfdpid
                              AND tsfdfqrule = vr.tsfdfqrule
                              AND (tsfdno = vr.tsfdno OR vr.tsfdno IS NULL)
                              AND tsfdfqxfje = vr.tsfdfqxfje
                              AND tsfdfqje = vr.tsfdfqje)
            LOOP
              FOR i IN 1 .. r_rulefq.COUNT
              LOOP
                j      := j + 1;
                l_fqje := r_rulefq(i).qje;
                -- l_ftpay    := row_ft.tsfdpayje;
                IF l_fqvalidjeft = '2' THEN
                  l_ftpay := row_ft.tsfdcjje - row_ft.tsfdrowlimit;
                  IF l_ftpay <= 0 THEN
                    GOTO nextftrow;
                  END IF;
                ELSE
                  l_rowvalidfqje := l_rowvalidfqje + row_ft.tsfdrowyftjf;
                  IF l_rowvalidfqje <= 0 THEN
                    GOTO nextftrow;
                  END IF;
                  l_ftpay        := least(l_rowvalidfqje,
                                          row_ft.tsfdcjje -
                                           row_ft.tsfdrowlimit);
                  l_rowvalidfqje := l_rowvalidfqje - l_ftpay;
                END IF;
                l_totfqpay := l_valid_fqje;
                l_ftqje    := round((l_ftpay / l_totfqpay) * l_fqje, 2);
                IF l_ftqje <> 0 THEN
                  INSERT INTO tktsellfqamt
                    (tsfabillno, tsfarowno, tsfapggdid, tsfaseq, tsfatransno,
                     tsfacbillno, tsfacrowno, tsfaqtype, tsfaqje, tsfavalidje,
                     tsfatotvalidje, tsfatotqje, tsfacjje, tsftype)
                  VALUES
                    (row_ft.tsfdbillno, row_ft.tsfdrowno, row_ft.tsfdpggdid,
                     j, p_transno, row_ft.tsfdbillno, row_ft.tsfdrowno,
                     r_rulefq(i).qtype, l_ftqje, l_ftpay, l_totfqpay, l_fqje,
                     row_ft.tsfdcjje, 'S');
                END IF;
                UPDATE tktsellscsadetail
                   SET tsfdpayje = l_ftpay
                 WHERE ROWID = row_ft.ROWID;
                <<nextftrow>>
                NULL;
              END LOOP;
            END LOOP;
          END IF;
          <<nextrow>>
          NULL;
        END LOOP;
      END LOOP;
      <<sp_salesaskip>>
      o_retcode := 0;
      o_retmsg  := '执行成功';
    EXCEPTION
      WHEN OTHERS THEN
        o_retcode := -1;
        o_retmsg  := '执行失败,' || to_char(SQLCODE) || ',' || SQLERRM;
    END;
  */
  --追送规则 优先级
  FUNCTION fgetsarulepriority
  (
    p_transno IN NUMBER,
    p_sarule  IN VARCHAR2
  ) RETURN NUMBER IS
    l_num    NUMBER;
    l_sarule VARCHAR2(15); --追送规则
    l_type   CHAR(1); --排序依据 1:银行卡号 2:商品录入顺序
  BEGIN
    l_sarule := nvl(fgetpara(',' || p_sarule || ',', ',', 1), '');
    l_type   := nvl(fgetpara(',' || p_sarule || ',', ',', 2), '');
    l_num    := 0;
    IF l_type = '1' THEN
      SELECT COUNT(*)
        INTO l_num
        FROM tktsellsadetail
       WHERE tssdtransno = p_transno
         AND tssdspno IN (SELECT tssdspno
                            FROM tktsellsadetail
                           WHERE tssdtransno = p_transno
                             AND tssdsarule = l_sarule);
    ELSIF l_type = '2' THEN
      SELECT MIN(tssdgbno) --小票行号
        INTO l_num
        FROM tktsellsadetail
       WHERE tssdtransno = p_transno
         AND tssdsarule = l_sarule;
    END IF;
    RETURN l_num;
  EXCEPTION
    WHEN OTHERS THEN
      RETURN 0;
  END;
  --3.银行卡追送计算
  PROCEDURE sp_salesa --追送设置
  (
    p_transno IN NUMBER,
    p_person  IN VARCHAR2,
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) AS
    --
    l_qno        VARCHAR2(20);
    l_fqxspay    NUMBER; --返券付款金额
    l_valid_fqje NUMBER; --有效返券金额
    l_a_m        CHAR(1); --1:VIP电子券,2:手工券
    vstartdate   DATE;
    venddate     DATE;
    l_exists     NUMBER;
    l_pj         CHAR(1);
    l_area       VARCHAR2(20);
    l_ismz       CHAR(1);
    l_prtsite    CHAR(1);
    r_tktrule    tktrule%ROWTYPE;
    l_crmmkt     VARCHAR2(20);
    row_h        sellhead%ROWTYPE;
    l_totfqje    NUMBER;
    -- 数组结构
    TYPE t_sa IS RECORD(
      no  VARCHAR2(20),
      je1 NUMBER,
      je2 NUMBER);
    --银行卡号
    TYPE t_splist IS TABLE OF t_sa;
    b_spno          t_splist := t_splist();
    -- 商品数据
    TYPE t_gdlist IS TABLE OF t_sa;
    b_gdid          t_gdlist := t_gdlist();
    --
    l_temp1         NUMBER;
    i               NUMBER;
    l_yfq           NUMBER; --已返券金额
    l_maxfqje       NUMBER; --单卡最大可返券金额
    l_rulemaxfqje   NUMBER; --规则最大返券金额
    l_rulepayje     NUMBER := 0; --同一规则的付款方式金额
    --l_rulepayje_tmp NUMBER := 0;
    l_rulegdje      NUMBER; --规则 商品金额
    l_rulegdje_tmp  NUMBER;
    j               NUMBER;
    l_tmp111        NUMBER;
    lrule           NUMBER := 0;
  BEGIN
    --
    BEGIN
      SELECT 1
        INTO l_exists
        FROM tktcardfqlog
       WHERE tcfltransno = p_transno
         AND tcflsource = '3'
         AND rownum = 1;
      GOTO sp_salesaskip;
    EXCEPTION
      WHEN no_data_found THEN
        NULL;
    END;
    BEGIN
      SELECT 1
        INTO l_exists
        FROM tktsellsadetail
       WHERE tssdtransno = p_transno
         AND rownum = 1;
      GOTO sp_salesaskip;
    EXCEPTION
      WHEN no_data_found THEN
        NULL;
    END;
    --
    UPDATE sellhead
       SET shsaflag = flagnofq
     WHERE shbillno IN
           (SELECT sltbillno FROM selllisttemp WHERE slttransno = p_transno);
    --
    BEGIN
      SELECT 1
        INTO l_exists
        FROM sellhead
       WHERE shbillno IN (SELECT sltbillno
                            FROM selllisttemp
                           WHERE slttransno = p_transno)
         AND (fvirtualsaletype(shdjlb) <> '1' OR (fvirtualsaletype(shdjlb) = '1' AND shishh = 'Y'))
         AND rownum = 1;
      o_retcode := 2;
      o_retmsg  := '不是销售不票,或者是换货小票,不需要计算追送!';
      RETURN;
    EXCEPTION
      WHEN no_data_found THEN
        NULL;
    END;
    --生成追送临时表
    FOR r_slt IN (SELECT sltbillno
                    FROM selllisttemp
                   WHERE slttransno = p_transno)
    LOOP
      SELECT * INTO row_h FROM sellhead WHERE shbillno = r_slt.sltbillno;
      l_crmmkt := fgetcrmmkt(row_h.shmanaunit, row_h.shmarket);
      FOR vr IN (SELECT p_transno, r_slt.sltbillno, sdrowno, spno, sprowno,
                        money, sdgdid, sdmfid, sdcatid, sdppcode, sdspsx,
                        sdcjje, fgetpara(sarule, ',', 1) tssdsarule,
                        fgetpara(sarule, ',', 2) tssdtpid,
                        fgetpara(sarule, ',', 3) tssdbankcode,
                        row_h.shmanaunit, row_h.shmarket,
                        fgetpara(sarule, ',', 4) tssdkzsxe
                   FROM (SELECT sdrowno, spno, sprowno,
                                 spmoney * sperate money, sdgdid, sdmfid,
                                 sdcatid, sdppcode, nvl(sdspsx, '0') sdspsx,
                                 sdcjje,
                                 fgetsarule(trunc(row_h.shdate),
                                             row_h.shmanaunit, l_crmmkt, sdgdid,
                                             sdmfid, sdcatid, sdppcode,
                                             nvl(sdspsx, '0'), spno, '2') sarule
                            FROM sellpayment, paymode, selldetail
                           WHERE spbillno = r_slt.sltbillno
                             AND spbillno = sdbillno
                             AND pmmanaunit = row_h.shmanaunit
                             AND pmcode = sppmcode
                             AND pmtype = '3') t1
                  WHERE sarule <> '0')
      LOOP
        INSERT INTO tktsellsadetail
          (tssdtransno, tssdbillno, tssdgbno, tssdspno, tssdpayrow,
           tssdmoney, tssdgdid, tssdmfid, tssdcatid, tssdppcode, tssdspsx,
           tssdcjje, tssdsarule, tssdtpid, tssdbankcode, tssdmu, tssdmkt,
           tssdkzsxe)
        VALUES
          (p_transno, vr.sltbillno, vr.sdrowno, vr.spno, vr.sprowno,
           vr.money, vr.sdgdid, vr.sdmfid, vr.sdcatid, vr.sdppcode,
           vr.sdspsx, vr.sdcjje, vr.tssdsarule, vr.tssdtpid, vr.tssdbankcode,
           vr.shmanaunit, vr.shmarket, vr.tssdkzsxe);
      END LOOP;
    END LOOP;
    --更新追送规则
    UPDATE tktsellsadetail a
       SET (tssdrulemje, tssdrulefje) = (SELECT trxfje, trfaqje
                                            FROM tktrule
                                           WHERE trcode = a.tssdsarule)
     WHERE tssdtransno = p_transno;
    --
    --将商品 商品成交金额放入数组
    SELECT tssdgdid, SUM(tssdcjje), SUM(tssdcjje) BULK COLLECT
      INTO b_gdid
      FROM (SELECT DISTINCT tssdgdid, tssdgbno, tssdcjje
               FROM tktsellsadetail
              WHERE tssdtransno = p_transno)
     GROUP BY tssdgdid;
    --没有追送规则
    IF b_gdid.COUNT = 0 THEN
      GOTO sp_salesaskip;
    END IF;
    SELECT tssdspno, SUM(tssdmoney), SUM(tssdmoney) BULK COLLECT
      INTO b_spno
     FROM (SELECT DISTINCT tssdspno, tssdpayrow, tssdmoney
              FROM tktsellsadetail
             WHERE tssdtransno = p_transno)
     GROUP BY tssdspno;
    --没有追送规则
    IF b_spno.COUNT = 0 THEN
      GOTO sp_salesaskip;
    END IF;
    --根据规则汇总
    FOR r_sarule IN (SELECT tssdmu, tssdmkt, tssdsarule, SUM(tssdcjje) cjje
                       FROM (SELECT DISTINCT tssdgdid, tssdgbno, tssdmfid, tssdsarule,
                                              tssdcjje, tssdmoney, tssdrulemje,
                                              tssdrulefje, tssdmu, tssdmkt
                                FROM tktsellsadetail
                               WHERE tssdtransno = p_transno)
                      GROUP BY tssdmu, tssdmkt, tssdsarule, tssdrulemje,
                               tssdrulefje
                      ORDER BY trunc(cjje / tssdrulemje) * tssdrulefje DESC,
                               fgetsarulepriority(p_transno, tssdsarule||',1'),
                               fgetsarulepriority(p_transno, tssdsarule||',2'))
    LOOP
      lrule := lrule + 1;
      --检查追送规则
      BEGIN
        SELECT *
          INTO r_tktrule
          FROM tktrule
         WHERE trcode = r_sarule.tssdsarule;
      EXCEPTION
        WHEN OTHERS THEN
          o_retcode := 1;
          o_retmsg  := '追送规则:' || r_sarule.tssdsarule || '不存在！';
          RETURN;
      END;
      --规则总付款金额
      SELECT nvl(SUM(tssdmoney), 0)
        INTO l_rulepayje
        FROM (SELECT DISTINCT tssdpayrow, tssdmoney
                 FROM tktsellsadetail
                WHERE tssdtransno = p_transno
                  AND tssdsarule = r_sarule.tssdsarule);
      --规则总付款金额 小于门槛金额 ,
      IF l_rulepayje < r_tktrule.trmkje THEN
        GOTO nextrule;
      END IF;
      -- 规则成交金额
      l_valid_fqje  := r_sarule.cjje;
      l_rulemaxfqje := r_tktrule.trmkaqje;
      IF l_rulemaxfqje = 0 THEN
        l_rulemaxfqje := 99999999999;
      END IF;
      FOR r_sarulelist IN (SELECT DISTINCT tssdspno, tssdpayrow, tssdmoney,
                                           tssdtpid, tssdbankcode, tssdkzsxe
                             FROM tktsellsadetail
                            WHERE tssdtransno = p_transno
                              AND tssdsarule = r_sarule.tssdsarule
                            ORDER BY tssdmoney DESC)
      LOOP
        -- 银行卡剩余 付款金额
        l_exists := 0;
        FOR i IN 1 .. b_spno.COUNT
        LOOP
          l_exists := l_exists + 1;
          IF b_spno(i).no = r_sarulelist.tssdspno THEN
            --单卡付款金额（剩余）
            l_temp1 := b_spno(i).je2;
            EXIT;
          END IF;
        END LOOP;
        i := l_exists;
        --
        --银行卡 追送规则 对应的商品
        l_rulegdje := 0;
        FOR r_spnoruegd IN (
          SELECT DISTINCT tssdgdid
            FROM tktsellsadetail
           WHERE tssdtransno = p_transno
             AND tssdsarule = r_sarule.tssdsarule
             AND tssdspno = r_sarulelist.tssdspno)
        LOOP
          FOR j IN 1 .. b_gdid.COUNT
          LOOP
            IF b_gdid(j).no = r_spnoruegd.tssdgdid THEN
              --剩余金额 je2
              l_rulegdje := l_rulegdje + b_gdid(j).je2;
              EXIT;
            END IF;
          END LOOP;
        END LOOP;
        --检查有效金额
        --l_fqxspay := least(l_temp1, r_sarule.cjje, l_valid_fqje);
        l_fqxspay := least(l_temp1, l_rulegdje, l_valid_fqje);
        IF l_fqxspay < r_tktrule.trxfje THEN
          GOTO nextrow;
        END IF;
        -- 实际用于返券金额 (可返金额 / 满金额) * 满金额
        l_temp1 := trunc(l_fqxspay / r_tktrule.trxfje, 0) * r_tktrule.trxfje;
        -- 扣除银行卡号对应的 已计算金额
        b_spno(i).je2 := b_spno(i).je2 - l_temp1;
        -- 扣除商品对应的 已计算金额
        l_rulegdje_tmp := l_temp1;
        FOR r_spnoruegd IN (
          SELECT DISTINCT tssdgdid
            FROM tktsellsadetail
           WHERE tssdtransno = p_transno
             AND tssdsarule = r_sarule.tssdsarule
             AND tssdspno = r_sarulelist.tssdspno)
        LOOP
          FOR j IN 1 .. b_gdid.COUNT
          LOOP
            EXIT WHEN l_rulegdje_tmp <= 0;
            IF b_gdid(j).no = r_spnoruegd.tssdgdid THEN
              l_tmp111       := least(l_rulegdje_tmp, b_gdid(j).je2);
              b_gdid(j).je2  := b_gdid(j).je2 - l_tmp111;
              l_rulegdje_tmp := l_rulegdje_tmp - l_tmp111;
            END IF;
          END LOOP;
        END LOOP;
        --
        gettktpopinfo(r_sarulelist.tssdtpid, row_h.shcustno, 'ALL',
                      o_retcode, o_retmsg, vstartdate, venddate, l_qno,
                      l_a_m, l_pj, l_area, l_ismz, l_prtsite);
        IF o_retcode <> 0 THEN
          RETURN;
        END IF;
        --
        DELETE FROM tktcardmzrec;
        bankcode := r_sarulelist.tssdbankcode;
        sppayno  := r_sarulelist.tssdspno;
        -- 得到已经返券的金额
        SELECT nvl(SUM(tcflmoney), 0)
          INTO l_yfq
          FROM tktcardfqlog
         WHERE tcflpoprule = r_sarule.tssdsarule
           AND tcflchr2 = sppayno
           AND tcflchr1 = bankcode
           AND tcflsource = '3'
           AND tcfldate >= trunc(SYSDATE)
           AND tcfldate < trunc(SYSDATE) + 1;
        IF r_sarulelist.tssdkzsxe = 0 THEN
          l_maxfqje := 99999999999;
        ELSE
          l_maxfqje := r_sarulelist.tssdkzsxe - l_yfq;
          IF l_maxfqje < 0 THEN
            l_maxfqje := 0;
          END IF;
          IF l_maxfqje > r_sarulelist.tssdkzsxe THEN
            l_maxfqje := r_sarulelist.tssdkzsxe;
          END IF;
        END IF;
        --单卡最大可返券金额，不能大于规则上的返券总金额
        IF l_maxfqje > l_rulemaxfqje THEN
          l_maxfqje := l_rulemaxfqje;
        END IF;
        l_crmmkt := fgetcrmmkt(r_sarule.tssdmu, r_sarule.tssdmkt);
        -- 计算返券
        calctkt_sa(p_transno => p_transno, p_mu => r_sarule.tssdmu,
                   p_mkt => l_crmmkt, p_ismz => l_ismz,
                   p_cardno => row_h.shcustno, p_billno => row_h.shbillno,
                   p_syjid => row_h.shsyjid, p_invno => row_h.shinvno,
                   p_rule => r_tktrule.trcode,
                   --返券规则或者 银行追送规则
                   p_fqrulemje => r_tktrule.trxfje, p_fqxspay => l_temp1,
                   p_max => l_maxfqje, p_tpid => r_sarulelist.tssdtpid,
                   p_person => p_person, p_source => '3',
                   --
                   o_retcode => o_retcode, o_retmsg => o_retmsg,
                   o_totfqje => l_totfqje);
        IF o_retcode <> 0 THEN
          RETURN;
        END IF;
        l_rulemaxfqje := l_rulemaxfqje - l_totfqje;
        l_valid_fqje  := l_valid_fqje - l_temp1;
        IF l_valid_fqje <= 0 THEN
          EXIT;
        END IF;
        <<nextrow>>
        NULL;
      END LOOP; --r_sarulelist 规则卡号分组
      <<nextrule>>
      NULL;
    END LOOP; --r_sarule
    --
    UPDATE sellhead
       SET shsaflag = flagfinish
     WHERE shbillno IN
           (SELECT sltbillno FROM selllisttemp WHERE slttransno = p_transno);
    <<sp_salesaskip>>
    o_retcode := 0;
    o_retmsg  := '执行成功';
  EXCEPTION
    WHEN OTHERS THEN
      o_retcode := -1;
      o_retmsg  := '执行失败,' || to_char(SQLCODE) || ',' || SQLERRM;
  END;
  -- 满送计算 返券+满赠礼品+银行追送
  PROCEDURE b_salems
  (
    p_transno IN NUMBER,
    p_cmd     IN VARCHAR2,
    p_person  IN VARCHAR2,
    p_debug   IN VARCHAR2 DEFAULT 'N',
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) AS
    v_enablerule CHAR(1) := 'N';
    --l_fqflag     CHAR(1) := flagedit;
    l_saflag CHAR(1) := flagnofq;
    l_shmu   VARCHAR2(20);
    l_shmkt  VARCHAR2(20);
  BEGIN
    debug := p_debug;
    IF p_cmd = 'PRECAL' THEN
      b_precal := 'Y';
    ELSE
      b_precal := 'N';
    END IF;

    --释放权限控制
    v_enablerule := fterp_context.get_isenablerule;
    fterp_searchrule.enablerule(FALSE);
    tktlen    := to_number(fgetmanapara('TL'));
    tl_random := nvl(to_number(fgetmanapara('TL_random')), 4);
    IF tktlen + tl_random > 19 OR tktlen < 6 THEN
      o_retmsg := '返券参数(TL+TL_random)大于19或者TL小于6';
      sp_opererrors('满送计算-券号', to_char(p_transno), o_retmsg);
      GOTO release;
    END IF;
    --一笔小票两种返券方法 1返券就低,3不就低
    fqmode := fgetmanapara('Q2');
    -- 是否启用余额累计 Y/N
    yemode := fgetmanapara('RD');
    bone   := 0;
    --将小票与规则对应
    FOR r_slt IN (SELECT * FROM selllisttemp WHERE slttransno = p_transno)
    LOOP
      BEGIN
        SELECT shmanaunit, shmarket
          INTO l_shmu, l_shmkt
          FROM sellhead
         WHERE shbillno = r_slt.sltbillno;
        DELETE FROM tktcardfqcal
         WHERE tcflsyjid = r_slt.sltsyjid
           AND tcflinvno = r_slt.sltinvno
           AND tcflmkt = l_shmkt
           AND tcfljygs = l_shmu;
        DELETE FROM tktsellfqdetail
         WHERE tsfdsyjid = r_slt.sltsyjid
           AND tsfdinvno = r_slt.sltinvno
           AND tsfdmarket = l_shmkt
           AND tsfdmu = l_shmu
           AND tsfdcal = 'Y';
        DELETE FROM tktsellscsadetail
         WHERE tsfdsyjid = r_slt.sltsyjid
           AND tsfdinvno = r_slt.sltinvno;
        sp_salemsrule(r_slt.sltbillno, p_transno, o_retcode, o_retmsg);
      EXCEPTION
        WHEN OTHERS THEN
          raise_application_error(errcode, 'err2!' || SQLERRM);
      END;
      --o_retcode 0 成功 ，1 失败，- 1 意外错误， 2 无返券
      IF o_retcode <> 0 THEN
        IF o_retcode < 0 THEN
          --errors
          sp_opererrors('满送计算-商品', to_char(p_transno), o_retmsg);
          GOTO release;
        ELSIF o_retcode = 2 THEN
          uptshfqflag(p_transno, r_slt.sltbillno, flagnofq, l_saflag);
        ELSIF o_retcode = 3 THEN
          -- 退货小票,换销小票,不处理返券,视为正常已处理
          uptshfqflag(p_transno, r_slt.sltbillno, flagfinish, flagfinish);
          -- 当前小票已红冲
          INSERT INTO dbusrsys.opererrors
          VALUES
            (SYSDATE, '满送计算-商品', to_char(p_transno),
             substr(o_retmsg, 1, 255));
          GOTO release;
        ELSE
          sp_opererrors('满送计算-商品', to_char(p_transno), o_retmsg);
        END IF;
      END IF;
      bone := bone + 1;
    END LOOP;
    IF o_retcode <> 0 AND bone = 1 THEN
      --没有销售返券，直接计算追送
      IF o_retcode = 2 THEN
        o_retcode := 0;
      END IF;
      GOTO sa;
    END IF;
    --销售返券
    BEGIN
      sp_salefq(p_transno, p_person, o_retcode, o_retmsg);
    EXCEPTION
      WHEN OTHERS THEN
        raise_application_error(errcode, '失败3!' || SQLERRM);
    END;
    IF o_retcode <> 0 THEN
      sp_opererrors('满送计算-返券', to_char(p_transno), o_retmsg);
      IF o_retcode < 0 THEN
        --errors
        GOTO release;
      ELSE
        GOTO sa;
      END IF;
    END IF;
    <<sa>>
  /*
    --商场追送
    sp_salesa_sc(p_transno, p_person, o_retcode, o_retmsg);
    IF o_retcode <> 0 THEN
      sp_opererrors('满送计算-商场追送', to_char(p_transno), o_retmsg);
      GOTO release;
    END IF;
    */
    --银行追送
    sp_salesa(p_transno, p_person, o_retcode, o_retmsg);
    IF o_retcode = 2 THEN
      l_saflag := flagnofq;
    ELSIF o_retcode = 0 THEN
      l_saflag := flagfinish;
    END IF;
    IF o_retcode <> 0 THEN
      sp_opererrors('满送计算-追送', to_char(p_transno), o_retmsg);
      GOTO release;
    END IF;
    IF p_cmd = 'ALL' THEN
      NULL;
    END IF;
    --计算券日汇总
    --此处计算销售djlb=1
    --红冲、退货（2、4、5）在dbusrpos.sp_sendsaleok中计算
    FOR vr IN (SELECT sltbillno FROM selllisttemp WHERE slttransno = p_transno) LOOP
      uptdaytotal(vr.sltbillno);
    END LOOP;
    --修改标志
    IF b_precal <> 'Y' THEN
      uptshfqflag(p_transno, NULL, flagfinish, l_saflag);
    END IF;
    o_retcode := 0;
    o_retmsg  := '执行成功';
    <<release>>
    IF v_enablerule = 'Y' THEN
      fterp_searchrule.enablerule;
    END IF;
  EXCEPTION
    WHEN OTHERS THEN
      ROLLBACK;
      IF v_enablerule = 'Y' THEN
        fterp_searchrule.enablerule;
      END IF;
  END;

  PROCEDURE th_rollback
  (
    vinvno  IN VARCHAR2,
    vsyjid  IN VARCHAR2,
    vjygs   IN VARCHAR2,
    vcrmmkt IN VARCHAR2,
    vperson IN VARCHAR2
  ) IS
    ld_ye    NUMBER;
    ld_curje NUMBER;
  BEGIN
    FOR vr IN (SELECT *
                 FROM tktcardfqlog
                WHERE tcfljygs = vjygs
                  AND tcflmkt = vcrmmkt
                  AND tcflsyjid = vsyjid
                  AND tcflinvno = vinvno
                  AND tcflzy IN ('K', 'L'))
    LOOP
      SELECT tcftfqje
        INTO ld_curje
        FROM tktcardfqtotal
       WHERE tcftseqno = vr.tcflvipseq;
      ld_ye := ld_curje - vr.tcflmoney;
      -- 增加券
      UPDATE tktcardfqtotal
         SET tcftfqje = ld_ye, tcftstatus = decode(ld_ye, 0, 'N', 'Y')
       WHERE tcftseqno = vr.tcflvipseq;
      --
      INSERT INTO tktcardfqlog
        (tcflseqno, tcflvipno, tcflstartdate, tcflenddate, tcfldate, tcflzy,
         tcflpopid, tcflpoprule, tcflmoney, tcflye, tcfloper, tcflsyjid,
         tcflinvno, tcflmemo, tcflsyjtrace, tcfljetype, tcflvipseq, tcflmkt,
         tcfljygs, tcflsource, tcfltransno, tcfltype, tcflchr1, tcflchr2,
         tcflchr3, tcflnum1)
        SELECT seq_tktcardfqlog.NEXTVAL, vr.tcflvipno, vr.tcflstartdate,
               vr.tcflenddate, SYSDATE, vr.tcflzy, vr.tcflpopid,
               vr.tcflpoprule, vr.tcflmoney * -1, ld_ye, vperson,
               vr.tcflsyjid, vr.tcflinvno, vr.tcflmemo, vr.tcflsyjtrace,
               vr.tcfljetype, vr.tcflvipseq, vr.tcflmkt, vr.tcfljygs,
               vr.tcflsource, vr.tcfltransno, vr.tcfltype, vr.tcflchr1,
               vr.tcflchr2, vr.tcflchr3, vr.tcflnum1
          FROM dual;
    END LOOP;
  END th_rollback;

  PROCEDURE salethhctkt
  (
    vbillno  IN NUMBER,
    vcrmmkt  IN VARCHAR2,
    vsyjh    IN VARCHAR2,
    vfphm    IN VARCHAR2,
    vcardno  IN VARCHAR2,
    vperson  IN VARCHAR2,
    vqje     IN NUMBER,
    vqtype   IN VARCHAR2,
    vupt     IN VARCHAR2 DEFAULT '1', -- 1:upt 0:only calc
    vfqsa    IN VARCHAR2 DEFAULT 'FQ',
    vcallsrc IN VARCHAR2, --调用来源 sp_sendsaleok,
    --salepopms.salehh, htth_getsalekhje, java_getsalekhje
    p_htthno IN NUMBER,
    vdjlb    IN CHAR,
    oqye     OUT NUMBER
  ) IS
    cur_tkt cursorms;
    row_tkt tktcardfqtotal%ROWTYPE;
    v_btye  NUMBER;
    vcurqje NUMBER;
    --log
    vseqno NUMBER;
    vzy    CHAR(1);
    vxfseq NUMBER;
    vsrc   CHAR(1);
    --l_cardflag    CHAR(2);
    --l_cardownerid VARCHAR2(20);
    l_sign    NUMBER;
    r_lasttkt tktcardfqtotal%ROWTYPE;
    l_bankno  VARCHAR2(20);
    l_payno   VARCHAR2(20);
    lcardno   VARCHAR2(20);
  BEGIN
    vxfseq := vbillno;
    oqye := vqje;
    IF vfqsa = 'SA' THEN
      vsrc := '3';
    ELSE
      vsrc := '1';
    END IF;
    --
    IF vdjlb = '2' THEN
      --红冲销售
      vzy    := 'F'; --返券 -
      l_sign := fgettktzysign(vzy) * -1;
    ELSIF vdjlb = '4' THEN
      --退货
      vzy    := 'K'; --券扣回
      l_sign := fgettktzysign(vzy);
    ELSE
      --红冲退货
      vzy    := 'K'; --券扣回
      l_sign := fgettktzysign(vzy) * -1;
    END IF;
    --后台退货 在客户端扣除TKTCARDFQTOTAL
    IF vcallsrc = 'htth_getsalekhje' AND vupt = '1' THEN
      goto htth;
    END IF;
    --后台退货 在客户端扣除TKTCARDFQTOTAL
    IF vcardno IS NOT NULL THEN
      OPEN cur_tkt FOR
        SELECT *
          FROM (SELECT tcftseqno, tcftvipno, tcftstartdate, tcftenddate,
                        tcftfqje, tcftfstpid, tcftfqrule, tcftjygs, tcftsource,
                        tcfttype
                   FROM tktcardfqtotal t1
                  WHERE tcftvipno = vcardno
                    AND tcftqtype = vqtype
                    AND (khhis = 'Y' OR (tcftstartdate <= trunc(SYSDATE) AND
                        tcftenddate >= trunc(SYSDATE)))
                    AND (((khhis = 'Y' OR tcftstatus = 'Y') AND
                        tcftflag = 'Y' AND nvl(tcftfqje, 0) > 0) OR
                        vdjlb = '5')
                    AND (tcftmkt = '0' OR
                        (tcftmkt = '1' AND tcftfqmkt = vcrmmkt))
                 UNION ALL
                 SELECT tcftseqno, tcftvipno, tcftstartdate, tcftenddate,
                        tcftfqje, tcftfstpid, tcftfqrule, tcftjygs, tcftsource,
                        tcfttype
                   FROM tktcardfqtotal t1, billmktdetail
                  WHERE tcftvipno = vcardno
                    AND tcftqtype = vqtype
                    AND (khhis = 'Y' OR (tcftstartdate <= trunc(SYSDATE) AND
                        tcftenddate >= trunc(SYSDATE)))
                    AND (((khhis = 'Y' OR tcftstatus = 'Y') AND
                        tcftflag = 'Y' AND nvl(tcftfqje, 0) > 0) OR
                        vdjlb = '5')
                    AND (tcftmkt = '2' OR tcftmkt IS NULL)
                    AND tcftfstpid = bmdbillno
                    AND bmdbillid = 'RRR'
                    AND bmdmkt = vcrmmkt)
         ORDER BY tcftenddate, tcftseqno;
    ELSE
      OPEN cur_tkt FOR
        SELECT *
          FROM (SELECT tcftseqno, tcftvipno, tcftstartdate, tcftenddate,
                        tcftfqje, tcftfstpid, tcftfqrule, tcftjygs, tcftsource,
                        tcfttype
                   FROM tktcardfqtotal t1, tktseqlist t2
                  WHERE tcftseqno = t2.seq
                    AND t2.htthno = p_htthno
                    AND tcftqtype = vqtype
                    AND (khhis = 'Y' OR (tcftstartdate <= trunc(SYSDATE) AND
                        tcftenddate >= trunc(SYSDATE)))
                    AND (((khhis = 'Y' OR tcftstatus = 'Y') AND
                        tcftflag = 'Y' AND nvl(tcftfqje, 0) > 0) OR
                        vdjlb = '5')
                    AND (tcftmkt = '0' OR
                        (tcftmkt = '1' AND tcftfqmkt = vcrmmkt))
                 UNION ALL
                 SELECT tcftseqno, tcftvipno, tcftstartdate, tcftenddate,
                        tcftfqje, tcftfstpid, tcftfqrule, tcftjygs, tcftsource,
                        tcfttype
                   FROM tktcardfqtotal t1, billmktdetail, tktseqlist t2
                  WHERE tcftseqno = t2.seq
                    AND tcftqtype = vqtype
                    AND t2.htthno = p_htthno
                    AND (khhis = 'Y' OR (tcftstartdate <= trunc(SYSDATE) AND
                        tcftenddate >= trunc(SYSDATE)))
                    AND (((khhis = 'Y' OR tcftstatus = 'Y') AND
                        tcftflag = 'Y' AND nvl(tcftfqje, 0) > 0) OR
                        vdjlb = '5')
                    AND (tcftmkt = '2' OR tcftmkt IS NULL)
                    AND tcftfstpid = bmdbillno
                    AND bmdbillid = 'RRR'
                    AND bmdmkt = vcrmmkt)
         ORDER BY tcftenddate, tcftseqno;
    END IF;
    LOOP
      FETCH cur_tkt
        INTO row_tkt.tcftseqno, row_tkt.tcftvipno, row_tkt.tcftstartdate, row_tkt.tcftenddate, row_tkt.tcftfqje, row_tkt.tcftfstpid, row_tkt.tcftfqrule, row_tkt.tcftjygs, row_tkt.tcftsource, row_tkt.tcfttype;
      EXIT WHEN cur_tkt%NOTFOUND OR cur_tkt%NOTFOUND IS NULL;
      IF row_tkt.tcftsource = '3' THEN
        BEGIN
          SELECT tcflchr1, tcflchr2
            INTO l_bankno, l_payno
            FROM tktcardfqlog
           WHERE tcflvipseq = row_tkt.tcftseqno
             AND tcflvipno = row_tkt.tcftvipno
             AND tcflsource = row_tkt.tcftsource
             AND tcflzy = 'F'
             AND tcflchr1 IS NOT NULL
             AND rownum = 1;
        EXCEPTION
          WHEN OTHERS THEN
            l_bankno := '-';
            l_payno  := '-';
        END;
      ELSE
        l_bankno := NULL;
        l_payno  := NULL;
      END IF;
      IF vdjlb = '4' THEN
        IF vupt = '1' THEN
          SELECT seq_tktcardfqlog.NEXTVAL INTO vseqno FROM dual;
          INSERT INTO tktcardfqlog
            (tcflseqno, tcflvipseq, tcflvipno, tcflstartdate, tcflenddate,
             tcflzy, tcflpopid, tcflpoprule, tcfldate, tcflmoney, tcflye,
             tcfloper, tcflmemo, tcflsyjid, tcflinvno, tcflsyjtrace,
             tcfljetype, tcflmkt, tcfljygs, tcflsource, tcfltype, tcflchr1,
             tcflchr2)
          VALUES
            (vseqno, row_tkt.tcftseqno, row_tkt.tcftvipno,
             row_tkt.tcftstartdate, row_tkt.tcftenddate, 'L',
             row_tkt.tcftfstpid, row_tkt.tcftfqrule, SYSDATE, vqje,
             row_tkt.tcftfqje + vqje, vperson, vcallsrc, vsyjh, vfphm, vxfseq,
             vqtype, vcrmmkt, row_tkt.tcftjygs, row_tkt.tcftsource,
             row_tkt.tcfttype, l_bankno, l_payno);
          SELECT seq_tktcardfqlog.NEXTVAL INTO vseqno FROM dual;
          INSERT INTO tktcardfqlog
            (tcflseqno, tcflvipseq, tcflvipno, tcflstartdate, tcflenddate,
             tcflzy, tcflpopid, tcflpoprule, tcfldate, tcflmoney, tcflye,
             tcfloper, tcflmemo, tcflsyjid, tcflinvno, tcflsyjtrace,
             tcfljetype, tcflmkt, tcfljygs, tcflsource, tcfltype, tcflchr1,
             tcflchr2)
          VALUES
            (vseqno, row_tkt.tcftseqno, row_tkt.tcftvipno,
             row_tkt.tcftstartdate, row_tkt.tcftenddate, vzy,
             row_tkt.tcftfstpid, row_tkt.tcftfqrule, SYSDATE, vqje * l_sign,
             row_tkt.tcftfqje, vperson, vcallsrc, vsyjh, vfphm, vxfseq, vqtype,
             vcrmmkt, row_tkt.tcftjygs, row_tkt.tcftsource, row_tkt.tcfttype,
             l_bankno, l_payno);
        END IF;
        oqye      := 0;
        r_lasttkt := row_tkt;
        EXIT;
      END IF;
      IF l_sign < 0 THEN
        vcurqje := least(oqye, row_tkt.tcftfqje);
      ELSE
        vcurqje := oqye;
      END IF;
      IF vcurqje <= 0.01 THEN
        EXIT;
      END IF;
      --批次余额
      v_btye := row_tkt.tcftfqje + vcurqje * l_sign;
      IF vupt = '1' THEN
        UPDATE tktcardfqtotal
           SET tcftfqje = tcftfqje + l_sign * vcurqje
         WHERE tcftseqno = row_tkt.tcftseqno;
        UPDATE tktcardfqtotal
           SET tcftstatus = decode(tcftfqje, 0, 'N', 'Y')
         WHERE tcftseqno = row_tkt.tcftseqno;
        --
        SELECT seq_tktcardfqlog.NEXTVAL INTO vseqno FROM dual;
        INSERT INTO tktcardfqlog
          (tcflseqno, tcflvipseq, tcflvipno, tcflstartdate, tcflenddate,
           tcflzy, tcflpopid, tcflpoprule, tcfldate, tcflmoney, tcflye,
           tcfloper, tcflmemo, tcflsyjid, tcflinvno, tcflsyjtrace, tcfljetype,
           tcflmkt, tcfljygs, tcflsource, tcfltype, tcflchr1, tcflchr2)
        VALUES
          (vseqno, row_tkt.tcftseqno, row_tkt.tcftvipno,
           row_tkt.tcftstartdate, row_tkt.tcftenddate, vzy, row_tkt.tcftfstpid,
           row_tkt.tcftfqrule, SYSDATE, vcurqje * l_sign, v_btye, vperson,
           vcallsrc, vsyjh, vfphm, vxfseq, vqtype, vcrmmkt, row_tkt.tcftjygs,
           row_tkt.tcftsource, row_tkt.tcfttype, l_bankno, l_payno);
      END IF;
      r_lasttkt := row_tkt;
      --计算余额
      oqye := oqye - vcurqje;
      IF oqye <= 0.01 THEN
        oqye := 0;
        EXIT;
      END IF;
    END LOOP;
    CLOSE cur_tkt;
    --
    --有券余额时记录一正一负两笔日志
    IF oqye > 0 AND vupt = '1' THEN
      IF r_lasttkt.tcftseqno IS NULL OR r_lasttkt.tcftseqno = 0 THEN
        IF vdjlb = '4' AND vcardno IS NULL THEN
          SELECT fgetticketno INTO lcardno FROM dual;
        ELSIF vdjlb = '2' THEN
          RETURN;
        ELSE
          lcardno := vcardno;
        END IF;
        --新增一个券批次
        SELECT seq_tktcardfqtotal.NEXTVAL INTO vseqno FROM dual;
        INSERT INTO tktcardfqtotal
          (tcftseqno, tcftvipno, tcftstartdate, tcftenddate, tcftqtype,
           tcftfqje, tcftstatus, tcftvipexno, tcftmemo, tcftzpno, tcftfstpid,
           tcftcltpid, tcftflag, tcftransno, tcfttype, tcftqmz, tcftmkt,
           tcftjygs, tcftfqmkt, tcftfqrule, tcftrulepj, tcftsource)
        VALUES
          (vseqno, lcardno, trunc(SYSDATE),
           trunc(SYSDATE) + to_number(fgetmanapara('Q1')), vqtype, 0, 'N',
           lcardno, NULL, NULL, 0, 0, 'N', 0, '1', 0, '0', fgetjygs(vcrmmkt),
           vcrmmkt, '0', '0', '9');
        --
        SELECT *
          INTO r_lasttkt
          FROM tktcardfqtotal
         WHERE tcftseqno = vseqno;
      END IF;
      --
      SELECT seq_tktcardfqlog.NEXTVAL INTO vseqno FROM dual;
      INSERT INTO tktcardfqlog
        (tcflseqno, tcflvipseq, tcflvipno, tcflstartdate, tcflenddate,
         tcflzy, tcflpopid, tcflpoprule, tcfldate, tcflmoney, tcflye,
         tcfloper, tcflmemo, tcflsyjid, tcflinvno, tcflsyjtrace, tcfljetype,
         tcflmkt, tcfljygs, tcflsource, tcfltype)
      VALUES
        (vseqno, r_lasttkt.tcftseqno, r_lasttkt.tcftvipno,
         r_lasttkt.tcftstartdate, r_lasttkt.tcftenddate, 'L',
         --扣券补现
         r_lasttkt.tcftfstpid, r_lasttkt.tcftfqrule, SYSDATE,
         oqye * l_sign * -1, oqye * l_sign * -1, vperson, vcallsrc, vsyjh,
         vfphm, vxfseq, vqtype, vcrmmkt, r_lasttkt.tcftjygs,
         r_lasttkt.tcftsource, r_lasttkt.tcfttype);
      --
      SELECT seq_tktcardfqlog.NEXTVAL INTO vseqno FROM dual;
      INSERT INTO tktcardfqlog
        (tcflseqno, tcflvipseq, tcflvipno, tcflstartdate, tcflenddate,
         tcflzy, tcflpopid, tcflpoprule, tcfldate, tcflmoney, tcflye,
         tcfloper, tcflmemo, tcflsyjid, tcflinvno, tcflsyjtrace, tcfljetype,
         tcflmkt, tcfljygs, tcflsource, tcfltype)
      VALUES
        (vseqno, r_lasttkt.tcftseqno, r_lasttkt.tcftvipno,
         r_lasttkt.tcftstartdate, r_lasttkt.tcftenddate, vzy,
         r_lasttkt.tcftfstpid, r_lasttkt.tcftfqrule, SYSDATE, oqye * l_sign,
         0, vperson, vcallsrc, vsyjh, vfphm, vxfseq, vqtype, vcrmmkt,
         r_lasttkt.tcftjygs, r_lasttkt.tcftsource, r_lasttkt.tcfttype);
    END IF;
    --后台退货 在客户端扣除TKTCARDFQTOTAL
    <<htth>>
    IF vfqsa = 'FQ' AND vupt = '1' THEN
      UPDATE sellhead
         SET shfqje = nvl(shfqje, 0) + vqje * l_sign
       WHERE shbillno = vbillno;
    END IF;
  EXCEPTION
    WHEN OTHERS THEN
      IF cur_tkt%ISOPEN THEN
        CLOSE cur_tkt;
      END IF;
      RAISE;
  END;
  PROCEDURE tktseq
  (
    p_htthno  IN VARCHAR2,
    p_crmmkt  IN VARCHAR2,
    p_osyjid  IN VARCHAR2,
    p_oinvno  IN VARCHAR2,
    p_obillno IN VARCHAR2,
    o_ret     OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) IS
    PRAGMA AUTONOMOUS_TRANSACTION;
  BEGIN
    SAVEPOINT seq;
    DELETE FROM tktseqlist
     WHERE htthno = p_htthno
       AND crmmkt = p_crmmkt;
    INSERT INTO tktseqlist
      SELECT DISTINCT tcflvipseq, p_htthno, p_crmmkt, tcflvipno
        FROM tktcardfqlog
       WHERE tcflsyjid = p_osyjid
         AND tcflinvno = p_oinvno
         AND tcflmkt = p_crmmkt
         AND tcflsyjtrace = p_obillno
         AND tcflzy = 'F';
    COMMIT;
    o_ret    := 1;
    o_retmsg := '成功';
  EXCEPTION
    WHEN OTHERS THEN
      ROLLBACK TO seq;
      o_ret    := 0;
      o_retmsg := 'tktseq执行失败->' || SQLERRM;
  END;
  PROCEDURE uptdaytotal(p_billno IN NUMBER) IS
    row_sh sellhead%ROWTYPE;
    l_sign NUMBER;
    l_date DATE;
  BEGIN
    BEGIN
      SELECT * INTO row_sh FROM sellhead WHERE shbillno = p_billno;
    EXCEPTION
      WHEN OTHERS THEN
        sp_opererrors('计算券日汇总', to_char(p_billno), '小票不存在！');
        RETURN;
    END;
    --
    IF fvirtualsaletype(row_sh.shdjlb) IN ('1', '5') THEN
      l_sign := 1;
    ELSIF fvirtualsaletype(row_sh.shdjlb) IN ('2', '4') THEN
      l_sign := -1;
    ELSE
      --sp_opererrors('计算券日汇总', to_char(p_billno), '小票类型错误！');
      RETURN;
    END IF;
    l_date := trunc(row_sh.shdate);
    -- 返券
    FOR vr IN (SELECT *
                 FROM tktsellfqamt, selldetail
                WHERE tsfacbillno = p_billno
                  AND tsfabillno = sdbillno
                  AND tsfarowno = sdrowno)
    LOOP
      UPDATE tktdaytotal
         SET tdtfqje = nvl(tdtfqje, 0) + l_sign * vr.tsfaqje
       WHERE tdtdate = l_date
         AND tdtmu = row_sh.shmanaunit
         AND tdtmkt = row_sh.shmarket
         AND tdtmfid = vr.sdmfid
         AND tdtgdid = vr.tsfapggdid
         AND tdtqtype = vr.tsfaqtype
         AND tdtpmcode = '0';
      IF SQL%ROWCOUNT = 0 OR SQL%ROWCOUNT IS NULL THEN
        INSERT INTO tktdaytotal
          (tdtdate, tdtmu, tdtmkt, tdtmfid, tdtgdid, tdtqtype, tdtfqje,
           tdtsqje, tdtpmcode)
        VALUES
          (l_date, row_sh.shmanaunit, row_sh.shmarket, vr.sdmfid,
           vr.tsfapggdid, vr.tsfaqtype, l_sign * vr.tsfaqje, 0, '0');
      END IF;
      dbusrpopint.debug.debug_new('TKTDAYTOTAL_FQ',
                                  to_char(p_billno) || ';' ||
                                   to_char(vr.tsfarowno) || ';' ||
                                   to_char(vr.sdrowno) || ';' ||
                                   to_char(vr.tsfaqje));
    END LOOP;
    --
    FOR vr IN (SELECT sellpaygoods.*, selldetail.*,
                      substr(spgpayerid, 1, 1) qtype
                 FROM sellpaygoods, selldetail
                WHERE spgbillno = p_billno
                  --AND spgpmcode = '0505'
                  AND spgpmtype = '5'
                  AND spgbillno = sdbillno
                  AND spggdrow = sdrowno
                  AND substr(spgpayerid, 1, 1) IS NOT NULL)
    LOOP
      UPDATE tktdaytotal
         SET tdtsqje = nvl(tdtsqje, 0) + l_sign * vr.spggdmoney
       WHERE tdtdate = l_date
         AND tdtmu = row_sh.shmanaunit
         AND tdtmkt = row_sh.shmarket
         AND tdtmfid = vr.sdmfid
         AND tdtgdid = vr.spggdid
         AND tdtqtype = vr.qtype
         AND tdtpmcode = vr.spgpmcode;
      IF SQL%ROWCOUNT = 0 OR SQL%ROWCOUNT IS NULL THEN
        INSERT INTO tktdaytotal
          (tdtdate, tdtmu, tdtmkt, tdtmfid, tdtgdid, tdtqtype, tdtfqje,
           tdtsqje, tdtpmcode)
        VALUES
          (l_date, row_sh.shmanaunit, row_sh.shmarket, vr.sdmfid,
           vr.spggdid, vr.qtype, 0, l_sign * vr.spggdmoney, vr.spgpmcode);
      END IF;
      dbusrpopint.debug.debug_new('TKTDAYTOTAL_SQ',
                                  to_char(p_billno) || ';' ||
                                   to_char(vr.spgrowno) || ';' ||
                                   to_char(vr.spggdrow) || ';' ||
                                   to_char(vr.spggdmoney));
    END LOOP;
  EXCEPTION
    WHEN OTHERS THEN
      sp_opererrors('计算券日汇总', to_char(p_billno),
                    '失败:' || to_char(SQLCODE) || ',' || SQLERRM);
  END;
  --
  --1.返券扣回
  --2.用券退回
  --
  --退货返券退回  dbusrpos.sp_sendsaleok调用
  --针对于 小票类别2（红冲销售），4（退货）
  -- 卓展走后台退货流程
  --先扣券现有券余额（包括过期的也可以参与扣减），不够的按比率计算返多少现金
  PROCEDURE salethhc
  (
    p_billno  IN NUMBER,
    p_upt     IN VARCHAR2, --1:upt 0:only calc
    p_callsrc IN VARCHAR2,
    --调用来源
    --sp_sendsaleok, 红冲
    --salepopms.salehh, 换货
    --htth_getsalekhje, 后台退货
    --java_getsalekhje 红冲退货时扣回判断
    p_htthno  IN NUMBER DEFAULT 0, --后台退货billno
    o_khje    OUT NUMBER,
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2,
    o_fqmemo  OUT VARCHAR2 --当前小票返券信息
  ) AS
    row_h sellhead%ROWTYPE;
    CURSOR cur_d IS
      SELECT * FROM selldetail WHERE sdbillno = p_billno;
    row_d selldetail%ROWTYPE;
    --
    volddate   DATE;
    voldbillno NUMBER;
    voldsyjid  VARCHAR2(20);
    voldinvno  VARCHAR2(20);
    volddjlb   CHAR(1);
    voldsaflag CHAR(1);
    vyetype    CHAR(1);
    --
    voldxsdate   DATE;
    voldxsbillno NUMBER;
    voldxsdjlb   CHAR(1);
    l_yxs        CHAR(1);
    l_htthno     NUMBER;
    l_cjje       NUMBER;
    --
    CURSOR cur_tsfa IS
      SELECT * FROM tktsellfqamt WHERE tsfacbillno = voldbillno;
    CURSOR cur_tsfa4(vrowno IN NUMBER, vgdid IN VARCHAR2) IS
      SELECT tsfabillno, vrowno tsfarowno, tsfapggdid, 1 tsfaseq,
             1 tsfatransno, tsfabillno tsfacbillno, vrowno tsfacrowno,
             tsfaqtype, SUM(tsfaqje) tsfaqje, SUM(tsfavalidje) tsfavalidje,
             SUM(tsfatotvalidje) tsfatotvalidje, SUM(tsfatotqje) tsfatotqje,
             SUM(tsfacjje) tsfacjje, tsftype, tsfaiskh,tsfmu,tsfmkt,trunc(tsfdate)
        FROM tktsellfqamt
       WHERE tsfabillno = voldbillno
         AND --扣回本笔所返的券 + 本笔余额在下笔所返的券
             tsfapggdid = vgdid
         AND tsfaiskh = 'Y'
       GROUP BY tsfabillno, tsfapggdid,
                -- TSFATRANSNO,
                --TSFACBILLNO,
                tsfaqtype, tsftype, tsfaiskh,tsfmu,tsfmkt,trunc(tsfdate);
    /*SELECT *
     FROM tktsellfqamt
    WHERE tsfabillno = voldbillno
      AND --扣回本笔所返的券 + 本笔余额在下笔所返的券
          tsfapggdid = vgdid
      AND tsfaiskh = 'Y';*/
    row_tsfa   tktsellfqamt%ROWTYPE;
    row_tsfa_i tktsellfqamt%ROWTYPE;
    l_rate     NUMBER;
    l_ftqje    NUMBER;
    l_crmmkt   VARCHAR2(20);
    o_ye       NUMBER; --大于零 券不够扣, 等于零券够扣
    r_khlist   tktqjelist; --记录返券券种及金额
    l_exists   NUMBER; --
    l_khje     NUMBER;
    l_qinfo    VARCHAR2(1000);
    l_transno  NUMBER;
    --换货
    row_n_xs sellhead%ROWTYPE;
    l_qe     NUMBER;
    l_totq   NUMBER := 0;
    --
    l_oricustno VARCHAR2(20);
    --l_manamode  CHAR(1);
    l_custtype  VARCHAR2(10);
    b_htth      BOOLEAN := FALSE; --是否后台退货
    l_htthno_1  NUMBER;
    l_oughtpay  NUMBER;
  BEGIN
    SELECT * INTO row_h FROM sellhead WHERE shbillno = p_billno;

    IF fvirtualsaletype(row_h.shdjlb) NOT IN ('2', '4', '5') THEN
      raise_application_error(errcode, '单据类型错误,必须为退货或者红冲销售小票!');
    END IF;
    IF row_h.shishh = 'Y' AND fvirtualsaletype(row_h.shdjlb) = '4' THEN
      -- 换货流程>>>>>>>>>>>>
      -- 原销售<换货>--->新退货小票[当前节点]--->新销售小票
      --
      --查找 换货产生的新销售小票是否存在
      BEGIN
        SELECT *
          INTO row_n_xs
          FROM sellhead
         WHERE shbillno = (SELECT sdbillno
                             FROM selldetail, sellhead
                            WHERE sdbillno = shbillno
                              AND shmarket = row_h.shmarket
                              AND shmanaunit = row_h.shmanaunit
                              AND sdysyjid = row_h.shsyjid
                              AND sdyinvno = row_h.shinvno
                              AND rownum = 1)
           AND shmarket = row_h.shmarket
           AND shmanaunit = row_h.shmanaunit
           AND fvirtualsaletype(shdjlb) = '1';
      EXCEPTION
        WHEN no_data_found THEN
          o_retcode := 2;
          o_retmsg  := '换货查找新销售小票失败!||小票序列号:' || to_char(p_billno);
          RETURN;
      END;
      --
    END IF;
    --vyetype := nvl(fgetmanapara('Q3'), '1'); --1 ：四舍五入到角，2：截断到角   2016-6-12  delete
    khhis   := nvl(fgetmanapara('Q6'), 'N'); --退货时是否允许扣回过期的券
    --一笔小票两种返券方法 1返券就低,3不就低
    fqmode := fgetmanapara('Q2');
    -- 是否启用余额累计 Y/N
    yemode := fgetmanapara('RD');
    --
    BEGIN
      SELECT shbillno, trunc(shdate), fvirtualsaletype(shdjlb), shsaflag, shsyjid, shinvno,
             shcustno, shoughtpay
        INTO voldbillno, volddate, volddjlb, voldsaflag, voldsyjid,
             voldinvno, l_oricustno, l_oughtpay
        FROM sellhead
       WHERE shmanaunit = row_h.shmanaunit
         AND shmarket = row_h.shmarket
         AND (shsyjid, shinvno) IN
             (SELECT sdysyjid, sdyinvno
                FROM selldetail
               WHERE sdbillno = p_billno
                 AND rownum = 1);
      IF fvirtualsaletype(row_h.shdjlb) = '5' THEN
        IF volddjlb <> '4' THEN
          raise_application_error(errcode, '单据类型错误,原小票单据类型必须为退货小票!');
        END IF;
        BEGIN
          SELECT shbillno, trunc(shdate), fvirtualsaletype(shdjlb)
            INTO voldxsbillno, voldxsdate, voldxsdjlb
            FROM sellhead
           WHERE shmanaunit = row_h.shmanaunit
             AND shmarket = row_h.shmarket
             AND (shsyjid, shinvno) IN
                 (SELECT sdysyjid, sdyinvno
                    FROM selldetail
                   WHERE sdbillno = voldbillno
                     AND rownum = 1);
          l_yxs := 'Y';
        EXCEPTION
          WHEN OTHERS THEN
            l_yxs := 'N';
        END;
      ELSE
        IF volddjlb <> '1' THEN
          raise_application_error(errcode, '单据类型错误,原小票单据类型必须为销售小票!');
        END IF;
      END IF;
      --
      l_crmmkt := fgetcrmmkt(row_h.shmanaunit, row_h.shmarket);
    EXCEPTION
      WHEN no_data_found THEN
        -- UPDATE shfqflag
        UPDATE crm_sellmail
           SET shfqflag = flagfinish
         WHERE shbillno = p_billno;
        UPDATE sellhead
           SET shfqflag = flagfinish
         WHERE shbillno = p_billno;
        o_retcode := 2;
        o_retmsg  := '不是指定小票交易';
        RETURN;
    END;
    IF l_oricustno IS NOT NULL THEN
      BEGIN
        SELECT cdmtype
          INTO l_custtype
          FROM cardmain
         WHERE cdmno = l_oricustno;
      EXCEPTION
        WHEN OTHERS THEN
          l_custtype := '';
      END;
    ELSE
      l_custtype := '';
    END IF;
    IF fvirtualsaletype(row_h.shdjlb) = '4' AND p_upt = '1' THEN
      --退货时对原销售的TKTSELLFQDETAIL记录设置标记
      UPDATE tktsellfqdetail
         SET tsfdisth = 'Y', tsfdthbillno = p_billno
       WHERE tsfdbillno = voldbillno
         AND tsfdpggdid IN
             (SELECT sdgdid FROM selldetail WHERE sdbillno = p_billno);
    ELSIF fvirtualsaletype(row_h.shdjlb) = '5' AND l_yxs = 'Y' AND p_upt = '1' THEN
      --红冲退货时执行反向操作
      UPDATE tktsellfqdetail
         SET tsfdisth = 'N', tsfdthbillno = ''
       WHERE tsfdbillno = voldxsbillno
         AND tsfdisth = 'Y'
         AND tsfdthbillno = voldbillno;
    END IF;
    ----启用余额累积
    IF yemode = 'Y' AND p_upt = '1' THEN
      --退货后余额清0
      IF fvirtualsaletype(row_h.shdjlb) = '4' THEN
        UPDATE tktsellfqdetail
           SET tsfdnum1 = tsfdfqye,
               --备份tsfdfqye 为了后台退货取消
               tsfdfqye = 0
         WHERE tsfdbillno = voldbillno --原小票对应的余额清0
           AND tsfddate = volddate
           AND tsfdfqye > 0
           AND tsfdpggdid IN
               (SELECT sdgdid FROM selldetail WHERE sdbillno = p_billno);
      END IF;
      IF fvirtualsaletype(row_h.shdjlb) = '5' AND l_yxs = 'Y' THEN
        UPDATE tktsellfqdetail
           SET tsfdfqye = tsfdnum1,
               --将退货的原小票余额加回来
               tsfdnum1 = 0
         WHERE tsfdbillno = voldxsbillno
           AND tsfddate = voldxsdate
           AND tsfdnum1 > 0;
      END IF;
      --
      IF fvirtualsaletype( row_h.shdjlb) = '2' THEN
        FOR vr IN (SELECT *
                     FROM tktsellfqdetail
                    WHERE tsfdcbillno = voldbillno
                      AND tsfddate = volddate
                      AND tsfdcrowno < 0)
        LOOP
          UPDATE tktsellfqdetail
             SET tsfdfqye = vr.tsfdbcje, tsfdrfqye2 = vr.tsfdbcje
           WHERE tsfdtransno = vr.tsfdptransno
             AND tsfdbillno = vr.tsfdbillno
             AND tsfdrowno = vr.tsfdrowno;
        END LOOP;
        --
        DELETE FROM tktsellfqdetail
         WHERE tsfdcbillno = voldbillno
           AND tsfddate = volddate;
      END IF;
    END IF;
    -- init r_rulefq
    r_khlist := tktqjelist();
    --
    --多笔小票余额累计返券后，扣回处理原则：
    --退货4时，扣回本笔所返的券 + 余额所返的券；退货后余额清0
    --红冲2时，原则上只允许红冲同卡号最后一笔（客户管理流程控制）；
    --         要求系统能扣回本笔返券 + 上笔余额所返的券，同时加回上笔余额
    -- init r_rulefq
    r_rulefq := tktqjelist(); --记录当前小票返券信息
    IF nvl(p_htthno, '0') = '0' THEN
      l_htthno := p_billno;
    END IF;
    --IF row_h.shcustno IS NULL THEN --打印券
    IF l_oricustno IS NULL THEN
      tktseq(p_htthno => l_htthno, p_crmmkt => l_crmmkt,
             p_osyjid => voldsyjid, p_oinvno => voldinvno,
             p_obillno => voldbillno, o_ret => o_retcode,
             o_retmsg => o_retmsg);
      IF o_retcode <> 1 THEN
        RETURN;
      END IF;
    END IF;
    --
    IF fvirtualsaletype(row_h.shdjlb) = '5' THEN
      th_rollback(voldinvno, voldsyjid, row_h.shmanaunit, l_crmmkt,
                  row_h.shchecker);
      OPEN cur_tsfa;
      LOOP
        FETCH cur_tsfa
          INTO row_tsfa;
        EXIT WHEN cur_tsfa%NOTFOUND OR cur_tsfa%NOTFOUND IS NULL;
        l_ftqje                := row_tsfa.tsfaqje;
        row_tsfa_i             := row_tsfa;
        row_tsfa_i.tsfabillno  := p_billno;
        row_tsfa_i.tsfacbillno := p_billno;
        row_tsfa_i.tsfaseq     := -1 * row_tsfa_i.tsfaseq;
        row_tsfa_i.tsfatransno := -1 * row_tsfa_i.tsfatransno;
        row_tsfa_i.tsfaqje     := l_ftqje;
        row_tsfa_i.tsfaiskh    := 'N';
        IF p_upt = '1' THEN
          INSERT INTO tktsellfqamt VALUES row_tsfa_i;
        END IF;
      END LOOP;
      CLOSE cur_tsfa;
    ELSE
      IF fvirtualsaletype(row_h.shdjlb) = '2' THEN
        OPEN cur_tsfa;
        LOOP
          FETCH cur_tsfa
            INTO row_tsfa;
          EXIT WHEN cur_tsfa%NOTFOUND OR cur_tsfa%NOTFOUND IS NULL;
          l_ftqje := row_tsfa.tsfaqje;
          --扣
          --...
          row_tsfa_i := row_tsfa;
          --只有当笔销售的记录才更改为红冲单号，余额累计的记录不修改
          IF row_tsfa_i.tsfabillno = row_tsfa_i.tsfacbillno THEN
            row_tsfa_i.tsfabillno := p_billno;
          END IF;
          row_tsfa_i.tsfacbillno := p_billno;
          --row_tsfa_i.tsfarowno   := row_d.sdrowno;
          row_tsfa_i.tsfaseq     := -1 * row_tsfa_i.tsfaseq;
          row_tsfa_i.tsfatransno := -1 * row_tsfa_i.tsfatransno;
          row_tsfa_i.tsfaqje     := l_ftqje;
          row_tsfa_i.tsfaiskh    := 'N';
          --l_ftqje := l_ftqje * -1; --红冲记事务的负数

          l_totq := l_totq + l_ftqje;
          salethhctkt(p_billno, l_crmmkt, row_h.shsyjid, row_h.shinvno,
                      row_h.shcustno, row_h.shchecker, l_ftqje,
                      row_tsfa_i.tsfaqtype, p_upt,
                      --针对后台退 扣回试算,小票上传扣券
                      'FQ', p_callsrc, l_htthno, fvirtualsaletype(row_h.shdjlb), o_ye);
          IF p_upt = '1' THEN
            INSERT INTO tktsellfqamt VALUES row_tsfa_i;
          END IF;
          --记录未扣回券种及券金额
          IF o_ye <> 0 THEN
            l_exists := 0;
            FOR i IN 1 .. r_khlist.COUNT
            LOOP
              IF r_khlist(i).qtype = row_tsfa_i.tsfaqtype THEN
                l_exists := 1;
                r_khlist(i).qje := r_khlist(i).qje + o_ye;
              END IF;
            END LOOP;
            IF l_exists = 0 THEN
              r_khlist.EXTEND;
              r_khlist(r_khlist.LAST).qtype := row_tsfa_i.tsfaqtype;
              r_khlist(r_khlist.LAST).qje := o_ye;
            END IF;
          END IF;
          --记录当前小票返券信息
          l_exists := 0;
          FOR i IN 1 .. r_rulefq.COUNT
          LOOP
            IF r_rulefq(i).qtype = row_tsfa_i.tsfaqtype THEN
              l_exists := 1;
              r_rulefq(i).qje := r_rulefq(i).qje + l_ftqje;
            END IF;
          END LOOP;
          IF l_exists = 0 THEN
            r_rulefq.EXTEND;
            r_rulefq(r_rulefq.LAST).qtype := row_tsfa_i.tsfaqtype;
            r_rulefq(r_rulefq.LAST).qje := l_ftqje;
          END IF;
          --
        END LOOP;
        CLOSE cur_tsfa;
        --红冲销售的记录不再参与退货扣回
        UPDATE tktsellfqamt
           SET tsfaiskh = 'N'
         WHERE tsfacbillno = voldbillno;
      ELSIF fvirtualsaletype(row_h.shdjlb) = '4' THEN
        --前台退货 还是后台退货
        BEGIN
          SELECT to_number(nvl(sdfhdd,'0'))
            INTO l_htthno_1
            FROM selldetail
           WHERE sdbillno = p_billno
             AND rownum = 1;
          IF l_htthno_1 > 0 THEN
            b_htth := TRUE;
          END IF;
        EXCEPTION
          WHEN no_data_found THEN
            b_htth := FALSE;
        END;
        --全额退货 不根据商品金额分摊
        IF l_oughtpay = row_h.shoughtpay THEN
          OPEN cur_tsfa;
          LOOP
            FETCH cur_tsfa
              INTO row_tsfa;
            EXIT WHEN cur_tsfa%NOTFOUND OR cur_tsfa%NOTFOUND IS NULL;
            l_ftqje := row_tsfa.tsfaqje;
            IF l_ftqje > 0 THEN
              --扣
              --...
              row_tsfa_i             := row_tsfa;
              row_tsfa_i.tsfabillno  := p_billno;
              row_tsfa_i.tsfacbillno := p_billno;
              row_tsfa_i.tsfarowno   := row_d.sdrowno;
              row_tsfa_i.tsfaseq     := -1 * row_tsfa_i.tsfaseq;
              row_tsfa_i.tsfatransno := -1 * row_tsfa.tsfatransno;
              row_tsfa_i.tsfaqje     := l_ftqje;
              row_tsfa_i.tsfacjje    := l_cjje;
              row_tsfa_i.tsfaiskh    := 'Y';
              l_totq                 := l_totq + l_ftqje;
              ----------------------------------------------------
              --前台退货直接返回扣回金额
              o_ye := row_tsfa_i.tsfaqje;
              IF p_callsrc <> 'salepopms.salehh' THEN
                salethhctkt(p_billno,
                            l_crmmkt,
                            row_h.shsyjid,
                            row_h.shinvno,
                            row_h.shcustno,
                            row_h.shchecker,
                            row_tsfa_i.tsfaqje,
                            row_tsfa_i.tsfaqtype,
                            p_upt, --针对后台退 扣回试算,小票上传扣券
                            'FQ',
                            p_callsrc,
                            l_htthno,
                            fvirtualsaletype(row_h.shdjlb),
                            o_ye);
              END IF;
              ----------------------------------------------------
              IF p_upt = '1' THEN
                INSERT INTO tktsellfqamt VALUES row_tsfa_i;
              END IF;
              --记录未扣回券种及券金额
              IF o_ye <> 0 THEN
                l_exists := 0;
                FOR i IN 1 .. r_khlist.COUNT
                LOOP
                  IF r_khlist(i).qtype = row_tsfa_i.tsfaqtype THEN
                    l_exists := 1;
                    r_khlist(i).qje := r_khlist(i).qje + o_ye;
                  END IF;
                END LOOP;
                IF l_exists = 0 THEN
                  r_khlist.EXTEND;
                  r_khlist(r_khlist.LAST).qtype := row_tsfa_i.tsfaqtype;
                  r_khlist(r_khlist.LAST).qje := o_ye;
                END IF;
              END IF; --IF o_ye <> 0 THEN
              --记录当前小票返券信息
              l_exists := 0;
              FOR i IN 1 .. r_rulefq.COUNT
              LOOP
                IF r_rulefq(i).qtype = row_tsfa_i.tsfaqtype THEN
                  l_exists := 1;
                  r_rulefq(i).qje := r_rulefq(i).qje + l_ftqje;
                END IF;
              END LOOP;
              IF l_exists = 0 THEN
                r_rulefq.EXTEND;
                r_rulefq(r_rulefq.LAST).qtype := row_tsfa_i.tsfaqtype;
                r_rulefq(r_rulefq.LAST).qje := l_ftqje;
              END IF;
            END IF; --IF l_ftqje > 0 THEN
          END LOOP;
          CLOSE cur_tsfa;
        ELSE
          --
          OPEN cur_d;
          LOOP
            FETCH cur_d
              INTO row_d;
            EXIT WHEN cur_d%NOTFOUND OR cur_d%NOTFOUND IS NULL;
            --
            OPEN cur_tsfa4(row_d.sdnum2, row_d.sdgdid);
            LOOP
              FETCH cur_tsfa4
                INTO row_tsfa;
              EXIT WHEN cur_tsfa4%NOTFOUND OR cur_tsfa4%NOTFOUND IS NULL;
              IF row_tsfa.tsfavalidje = 0 THEN
                GOTO nextrow;
              END IF;
              --
              BEGIN
                SELECT SUM(sdcjje)
                  INTO l_cjje
                  FROM selldetail
                 WHERE sdbillno = row_tsfa.tsfabillno
                   AND sdgdid = row_d.sdgdid;
                l_rate := row_d.sdcjje / l_cjje;
              EXCEPTION
                WHEN OTHERS THEN
                  l_rate := row_d.sdcjje / row_tsfa.tsfacjje;
              END;
              l_ftqje := round(row_tsfa.tsfaqje * l_rate, 4);
              IF l_ftqje > 0 THEN
                --扣
                --...
                row_tsfa_i             := row_tsfa;
                row_tsfa_i.tsfabillno  := p_billno;
                row_tsfa_i.tsfacbillno := p_billno;
                row_tsfa_i.tsfarowno   := row_d.sdrowno;
                row_tsfa_i.tsfaseq     := -1 * row_tsfa_i.tsfaseq;
                row_tsfa_i.tsfatransno := -1 * row_tsfa.tsfatransno;
                row_tsfa_i.tsfaqje     := l_ftqje;
                row_tsfa_i.tsfacjje    := l_cjje;
                row_tsfa_i.tsfaiskh    := 'Y';
                l_totq                 := l_totq + l_ftqje;
                ----------------------------------------------------
                --前台退货直接返回扣回金额
                o_ye := row_tsfa_i.tsfaqje;
                IF p_callsrc <> 'salepopms.salehh' THEN
                  salethhctkt(p_billno,
                              l_crmmkt,
                              row_h.shsyjid,
                              row_h.shinvno,
                              row_h.shcustno,
                              row_h.shchecker,
                              row_tsfa_i.tsfaqje,
                              row_tsfa_i.tsfaqtype,
                              p_upt, --针对后台退 扣回试算,小票上传扣券
                              'FQ',
                              p_callsrc,
                              l_htthno,
                              fvirtualsaletype(row_h.shdjlb),
                              o_ye);
                END IF;
                ----------------------------------------------------
                IF p_upt = '1' THEN
                  INSERT INTO tktsellfqamt VALUES row_tsfa_i;
                END IF;
                --记录未扣回券种及券金额
                IF o_ye <> 0 THEN
                  l_exists := 0;
                  FOR i IN 1 .. r_khlist.COUNT
                  LOOP
                    IF r_khlist(i).qtype = row_tsfa_i.tsfaqtype THEN
                      l_exists := 1;
                      r_khlist(i).qje := r_khlist(i).qje + o_ye;
                    END IF;
                  END LOOP;
                  IF l_exists = 0 THEN
                    r_khlist.EXTEND;
                    r_khlist(r_khlist.LAST).qtype := row_tsfa_i.tsfaqtype;
                    r_khlist(r_khlist.LAST).qje := o_ye;
                  END IF;
                END IF; --IF o_ye <> 0 THEN
                --记录当前小票返券信息
                l_exists := 0;
                FOR i IN 1 .. r_rulefq.COUNT
                LOOP
                  IF r_rulefq(i).qtype = row_tsfa_i.tsfaqtype THEN
                    l_exists := 1;
                    r_rulefq(i).qje := r_rulefq(i).qje + l_ftqje;
                  END IF;
                END LOOP;
                IF l_exists = 0 THEN
                  r_rulefq.EXTEND;
                  r_rulefq(r_rulefq.LAST).qtype := row_tsfa_i.tsfaqtype;
                  r_rulefq(r_rulefq.LAST).qje := l_ftqje;
                END IF;
              END IF; --IF l_ftqje > 0 THEN
              --
              <<nextrow>>
              NULL;
            END LOOP;
            CLOSE cur_tsfa4;
          END LOOP;
          CLOSE cur_d;
        END IF;
      END IF;
      --检查银行追送
      IF p_callsrc <> 'salepopms.salehh' THEN
        IF voldsaflag = flagfinish THEN
          SELECT slttransno
            INTO l_transno
            FROM selllisttemp
           WHERE sltbillno = voldbillno;
          FOR row_tcfl IN (SELECT tcfljetype, SUM(tcflmoney) tcflmoney
                             FROM tktcardfqlog
                            WHERE /*tcfltransno = l_transno                                                                                                                                                                                                                                                                    AND */
                            tcflsyjtrace = voldbillno
                        AND tcflsource = '3'
                            GROUP BY tcfljetype)
          LOOP
            l_ftqje              := row_tcfl.tcflmoney;
            row_tsfa_i.tsfaqtype := row_tcfl.tcfljetype;
            --
            l_totq := l_totq + l_ftqje;
            salethhctkt(p_billno, l_crmmkt, row_h.shsyjid, row_h.shinvno,
                        row_h.shcustno, row_h.shchecker, l_ftqje,
                        row_tsfa_i.tsfaqtype, p_upt,
                        --针对后台退 扣回试算,小票上传扣券
                        'SA', p_callsrc, l_htthno, fvirtualsaletype(row_h.shdjlb), o_ye);
            --记录未扣回券种及券金额
            IF o_ye <> 0 THEN
              l_exists := 0;
              FOR i IN 1 .. r_khlist.COUNT
              LOOP
                IF r_khlist(i).qtype = row_tsfa_i.tsfaqtype THEN
                  l_exists := 1;
                  r_khlist(i).qje := r_khlist(i).qje + o_ye;
                END IF;
              END LOOP;
              IF l_exists = 0 THEN
                r_khlist.EXTEND;
                r_khlist(r_khlist.LAST).qtype := row_tsfa_i.tsfaqtype;
                r_khlist(r_khlist.LAST).qje := o_ye;
              END IF;
            END IF; --IF o_ye <> 0 THEN
            --记录当前小票返券信息
            l_exists := 0;
            FOR i IN 1 .. r_rulefq.COUNT
            LOOP
              IF r_rulefq(i).qtype = row_tsfa_i.tsfaqtype THEN
                l_exists := 1;
                r_rulefq(i).qje := r_rulefq(i).qje + l_ftqje;
              END IF;
            END LOOP;
            IF l_exists = 0 THEN
              r_rulefq.EXTEND;
              r_rulefq(r_rulefq.LAST).qtype := row_tsfa_i.tsfaqtype;
              r_rulefq(r_rulefq.LAST).qje := l_ftqje;
            END IF;
          END LOOP;
          UPDATE sellhead
             SET shsaflag = flagfinish
           WHERE shbillno = p_billno;
        END IF; --IF voldsaflag = flagfinish THEN
      END IF; --IF p_callsrc <> 'salepopms.salehh' THEN
    END IF;
    --计算扣回金额
    l_khje  := 0;
    l_qinfo := '';
    l_qe    := 0;
    FOR i IN 1 .. r_khlist.COUNT
    LOOP
      l_qe    := l_qe + r_khlist(i).qje;
      l_rate  := fgetthkhzxl(r_khlist(i).qtype);
      l_khje  := l_khje + r_khlist(i).qje * l_rate;
      l_qinfo := l_qinfo || r_khlist(i)
                .qtype || '券扣除金额:' || to_char(r_khlist(i).qje) || '元,';
    END LOOP;
    /*IF vyetype = '1' THEN
      l_khje := round(l_khje, 1);
    ELSE
      l_khje := trunc(l_khje, 1);
    END IF;*/
    --前台退货预算扣回时已经记录，不需要再扣记录多余的部分
    --后台退货
    IF fvirtualsaletype(row_h.shdjlb) = '4' AND b_htth THEN
      UPDATE sellheadth
         SET shqje = l_khje, shqmemo = l_qinfo, shqe = l_qe,
             shdate1 = SYSDATE, shtotqe = l_totq, shchr1 = voldsaflag
       WHERE shsyjid = row_h.shsyjid
         AND shinvno = row_h.shinvno
         AND shmarket = row_h.shmarket
         AND shmanaunit = row_h.shmanaunit;
      IF SQL%ROWCOUNT = 0 OR SQL%ROWCOUNT IS NULL THEN
        INSERT INTO sellheadth
          (shbillno, shstatus, shsyjid, shinvno, shmarket, shmanaunit,
           shjfje, shqje, shqmemo, shnum1, shnum2, shnum3, shchr1, shchr2,
           shchr3, shqe, shdate1, shtotqe, shdjlb)
        VALUES
          (row_h.shbillno, row_h.shstatus, row_h.shsyjid, row_h.shinvno,
           row_h.shmarket, row_h.shmanaunit, 0, l_khje, l_qinfo, 0, p_htthno,
           0, voldsaflag, NULL, NULL, l_qe, SYSDATE, l_totq, fvirtualsaletype(row_h.shdjlb));
      END IF;
    ELSIF fvirtualsaletype(row_h.shdjlb) = '5' THEN
      --红冲后台退货
      SELECT to_number(sdfhdd)
        INTO l_htthno
        FROM selldetail
       WHERE sdbillno = voldbillno
         AND rownum = 1;
      UPDATE sellheadth
         SET shstatus = 'C', shqmemo = shqmemo || '[红冲]', shdate1 = SYSDATE
       WHERE shsyjid = voldsyjid
         AND shinvno = voldinvno
         AND shmarket = row_h.shmarket
         AND shmanaunit = row_h.shmanaunit;
      /*更新赠礼退货记录 by zhouwd 2009-8-9*/
      --UPDATE mzgiftselldetailhtth SET msdstatus = 'C' WHERE msdhtthno = l_htthno;
    ELSIF fvirtualsaletype(row_h.shdjlb) = '2' THEN
      FOR vr IN (SELECT tsfabillno, tsfarowno, SUM(tsfaqje) qje
                   FROM tktsellfqamt
                  WHERE tsfacbillno = p_billno
                    AND tsfabillno <> tsfacbillno
                  GROUP BY tsfabillno, tsfarowno)
      LOOP
        --红冲销售需要更新salegoodslist
        INSERT INTO salebatchtask
          (sbtseq, sbtbillno, sbtrowno, sbttype, sbtfqje, sbtcustno,
           sbtcusttype, sbtjygs, sbtmarket)
        VALUES
          (seq_salebatchtask.NEXTVAL, vr.tsfabillno, vr.tsfarowno, 'F',
           -1 * vr.qje, l_oricustno, l_custtype, row_h.shmanaunit,
           row_h.shmarket);
      END LOOP;
    END IF;
    --A券扣除金额:10元,B券扣除金额:20元,券总折现金额:30元
    IF l_khje <> 0 THEN
      o_retcode := 1;
      o_retmsg  := l_qinfo || '券总折现金额:' || to_char(l_khje) || '元';
    ELSE
      o_retcode := 2;
      o_retmsg  := '';
    END IF;
    o_khje := l_khje;
    --返回返券信息
    o_fqmemo := '';
    FOR i IN 1 .. r_rulefq.COUNT
    LOOP
      o_fqmemo := o_fqmemo || r_rulefq(i)
                 .qtype || '券:' || to_char(r_rulefq(i).qje) || '元';
      IF i < r_rulefq.COUNT THEN
        o_fqmemo := o_fqmemo || ',';
      END IF;
    END LOOP;
    UPDATE sellhead
       SET shfqflag = flagfinish
     WHERE shbillno = p_billno;
    UPDATE crm_sellmail
       SET shfqflag = flagfinish
     WHERE shbillno = p_billno;
  EXCEPTION
    WHEN OTHERS THEN
      IF cur_d%ISOPEN THEN
        CLOSE cur_d;
      END IF;
      IF cur_tsfa%ISOPEN THEN
        CLOSE cur_tsfa;
      END IF;
      IF cur_tsfa4%ISOPEN THEN
        CLOSE cur_tsfa4;
      END IF;
      ROLLBACK;
      o_retcode := 0;
      o_retmsg  := to_char(p_billno) || '执行失败,' || to_char(SQLCODE) || ',' ||
                   SQLERRM;
      sp_opererrors('salepopms.salethhc', '1111', o_retmsg);
  END;
  --后台退货取消
  PROCEDURE htthcancel
  (
    p_billno  IN NUMBER,
    p_person  IN VARCHAR2,
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) AS
    row_h sellhead%ROWTYPE;
    --
    volddate   DATE;
    voldbillno NUMBER;
    volddjlb   CHAR(1);
    --
  BEGIN
    SELECT * INTO row_h FROM sellhead WHERE shbillno = p_billno;
    IF fvirtualsaletype(row_h.shdjlb) <> '4' THEN
      raise_application_error(errcode, '单据类型错误,必须为退货或者红冲销售小票!');
    END IF;
    BEGIN
      SELECT shbillno, trunc(shdate), fvirtualsaletype(shdjlb)
        INTO voldbillno, volddate, volddjlb
        FROM sellhead
       WHERE shmanaunit = row_h.shmanaunit
         AND shmarket = row_h.shmarket
         AND (shsyjid, shinvno) IN
             (SELECT sdysyjid, sdyinvno
                FROM selldetail
               WHERE sdbillno = p_billno
                 AND rownum = 1);
      IF volddjlb <> '1' THEN
        raise_application_error(errcode, '单据类型错误,原小票单据类型必须为销售小票!');
      END IF;
      --
    EXCEPTION
      WHEN no_data_found THEN
        --o_retcode := 2;
        --o_retmsg  := '不是指定小票交易';
        --RETURN;
        -- 考虑促销切换前的小票 此处不return 09.11.20 zhangsy
        NULL;
    END;
    -- 是否启用余额累计 Y/N
    yemode := fgetmanapara('RD');
    IF yemode = 'Y' THEN
      --退货后余额清0
      UPDATE tktsellfqdetail
         SET tsfdfqye = tsfdnum1 --恢复余额
       WHERE tsfdbillno = voldbillno
         AND tsfddate = volddate
         AND tsfdnum1 > 0;
    END IF;
    th_rollback(row_h.shinvno,--row_h.shbillno,
                row_h.shsyjid,
                row_h.shmanaunit,
                fgetcrmmkt(row_h.shmanaunit, row_h.shmarket),
                p_person);
    --
    DELETE FROM tktsellfqamt
     WHERE tsfabillno = p_billno
       AND tsfacbillno = p_billno
       AND tsfaseq < 0;
    DELETE FROM mzgiftselldetailhtth WHERE msdnum5 = p_billno;
    DELETE FROM sellhead WHERE shbillno = p_billno;
    --DELETE FROM sellhead_htth WHERE shbillno = p_billno;
    o_retcode := 1;
    o_retmsg  := 'salepopms.htthcancel执行成功';
  EXCEPTION
    WHEN OTHERS THEN
      --ROLLBACK;
      o_retcode := 0;
      o_retmsg  := to_char(p_billno) || '执行失败,' || to_char(SQLCODE) || ',' ||
                   SQLERRM;
      --sp_opererrors('salepopms.htthcancel', '1234', o_retmsg);
  END;
  PROCEDURE htthsendok(p_billno  IN NUMBER,
                       o_retcode OUT NUMBER,
                       o_retmsg  OUT VARCHAR2) AS
    row_h  sellhead%ROWTYPE;
    row_th sellheadth%ROWTYPE;
    --
    v_htthno NUMBER;
    i        NUMBER;
    l_crmmkt VARCHAR2(20);
    l_maxthno NUMBER;
  BEGIN
    --
    SELECT to_number(sdfhdd)
      INTO v_htthno
      FROM selldetail
     WHERE sdbillno = p_billno
       AND rownum = 1;
    --
    IF v_htthno > 0 THEN
      SELECT * INTO row_h FROM sellhead WHERE shbillno = p_billno;
      SELECT nvl(MAX(shbillno), 0)
        INTO l_maxthno
        FROM sellheadth
       WHERE shnum2 = v_htthno
         AND shmarket = row_h.shmarket
         AND shmanaunit = row_h.shmanaunit;
      -- 后台退货退 促销切换之前的小票
      IF l_maxthno > 0 THEN
        SELECT *
          INTO row_th
          FROM sellheadth
         WHERE shbillno = l_maxthno;
        l_crmmkt := fgetcrmmkt(row_h.shmanaunit, row_h.shmarket);
        --tktcardfqlog
        UPDATE tktcardfqlog
           SET tcflsyjid    = row_h.shsyjid,
               tcflinvno    = row_h.shinvno,
               tcflsyjtrace = p_billno
         WHERE tcflsyjid = row_th.shsyjid
           AND tcflinvno = row_th.shinvno
           AND tcflmkt = l_crmmkt
           AND tcflzy IN ('K', 'L');
        i := SQL%ROWCOUNT;
        dbms_output.put_line('tktcardfqlog:' || to_char(i));
        --tktsellfqamt
        UPDATE tktsellfqamt
           SET tsfabillno  = p_billno,
               tsfacbillno = p_billno
         WHERE tsfabillno = row_th.shbillno
           AND tsfacbillno = row_th.shbillno;
        i := SQL%ROWCOUNT;
        dbms_output.put_line('tktsellfqamt:' || to_char(i));
        -- sellpaygoods
        UPDATE sellpaygoods SET spgbillno = p_billno WHERE spgbillno = row_th.shbillno;
        --
        i := SQL%ROWCOUNT;
        dbms_output.put_line('sellpaygoods:' || to_char(i));
        -- 删除扣回审核时的小票
        DELETE FROM sellhead WHERE shbillno = row_th.shbillno AND shbillno <> p_billno;
        /*-- 删除扣回时小票 扣回时不update sellheadth没有相关记录
        -- 客服后台退货 小票号为 .1格式
        -- row_th.shinvno = l_invno := to_char(p_sh.shinvno + 0.1);
        IF INSTR(row_th.shinvno,'.1', 1) > 0 THEN
          DELETE FROM sellhead
           WHERE shsyjid = row_th.shsyjid
             AND shinvno = row_th.shinvno
             AND shmarket = row_th.shmarket
             AND shmanaunit = row_th.shmanaunit;
        END IF;*/
      END IF;
      ----------------------------------
      -- 更新退货小票的返券追送标志 shfqflag
      UPDATE sellhead
         SET shfqflag = flagfinish, shsaflag = nvl(row_th.shchr1, '2')
       WHERE shbillno = p_billno;
      UPDATE crm_sellmail
         SET shfqflag = flagfinish
       WHERE shbillno = p_billno;
      /*更新赠礼退货记录状态 by zhouwd 2009-8-9*/
      UPDATE mzgiftselldetailhtth
         SET msdstatus = 'Y',
             msdcrmno  = p_billno
       WHERE msdhtthno = v_htthno;
    END IF;
    o_retcode := 1;
    o_retmsg  := 'salepopms.htthsendok执行成功';
  EXCEPTION
    WHEN no_data_found THEN
      o_retcode := 2;
      o_retmsg  := 'no_data_found';
      --sp_opererrors('salepopms.htthsendok', '2345', o_retmsg);
    WHEN OTHERS THEN
      ROLLBACK;
      o_retcode := 0;
      o_retmsg  := to_char(p_billno) || '执行失败,' || to_char(SQLCODE) || ',' || SQLERRM;
      sp_opererrors('salepopms.htthsendok', '2345', o_retmsg);
  END;
  --换货处理
  PROCEDURE salehh
  (
    p_billno  IN NUMBER,
    o_retcode OUT NUMBER,
    o_retmsg  OUT VARCHAR2
  ) AS
    row_n_xs sellhead%ROWTYPE; --换货新销售小票
    row_o_xs sellhead%ROWTYPE; --换货原销售小票
    row_n_th sellhead%ROWTYPE; --换货新退货小票
    -- 原销售<换货>--->新退货小票--->新销售小票
    --
    l_khje    NUMBER;
    l_totcjje NUMBER;
    CURSOR cur_d IS
      SELECT * FROM selldetail WHERE sdbillno = p_billno;
    row_d    selldetail%ROWTYPE;
    l_rate   NUMBER;
    l_ftqje  NUMBER;
    j        NUMBER; --递加
    o_fqmemo VARCHAR2(500);
  BEGIN
    --查找换货新销售小票
    BEGIN
      SELECT *
        INTO row_n_xs
        FROM sellhead
       WHERE shbillno = p_billno
         AND fvirtualsaletype(shdjlb) = '1'
         AND shishh = 'Y';
    EXCEPTION
      WHEN no_data_found THEN
        o_retcode := 0;
        o_retmsg  := '未找到该小票!||小票序列号:' || to_char(p_billno);
        RETURN;
    END;
    --查找新退货小票
    BEGIN
      SELECT *
        INTO row_n_th
        FROM sellhead
       WHERE shmarket = row_n_xs.shmarket
         AND shmanaunit = row_n_xs.shmanaunit
         AND (shsyjid, shinvno) = (SELECT sdysyjid, sdyinvno
                                     FROM selldetail
                                    WHERE sdbillno = p_billno
                                      AND rownum = 1)
         AND fvirtualsaletype(shdjlb) = '4';
    EXCEPTION
      WHEN no_data_found THEN
        o_retcode := 0;
        o_retmsg  := '查找新退货小票失败!||小票序列号:' || to_char(p_billno);
        RETURN;
    END;
    --
    BEGIN
      --查找原销售小票
      SELECT *
        INTO row_o_xs
        FROM sellhead
       WHERE shmarket = row_n_xs.shmarket
         AND shmanaunit = row_n_xs.shmanaunit
         AND (shsyjid, shinvno) = (SELECT sdysyjid, sdyinvno
                                     FROM selldetail
                                    WHERE sdbillno = row_n_th.shbillno
                                      AND rownum = 1)
         AND fvirtualsaletype(shdjlb) = '1';
    EXCEPTION
      WHEN no_data_found THEN
        o_retcode := 0;
        o_retmsg  := '查找原销售小票失败!||小票序列号:' || to_char(p_billno);
        RETURN;
    END;
    --计算新退货小票的券
    salethhc(p_billno => row_n_th.shbillno, p_upt => '1',
             --1:upt 0:only calc
             p_callsrc => 'salepopms.salehh', o_khje => l_khje,
             o_retcode => o_retcode, o_retmsg => o_retmsg,
             o_fqmemo => o_fqmemo);
    --计算失败
    IF o_retcode = 0 THEN
      RETURN;
    END IF;
    --券不够扣,--不处理???
    IF l_khje > 0 THEN
      NULL;
    END IF;
    -- 将券分摊到新销售小票各个商品
    SELECT nvl(SUM(sdcjje), 0)
      INTO l_totcjje
      FROM selldetail
     WHERE sdbillno = p_billno; --row_n_th.shbillno;
    IF l_totcjje > 0 THEN
      j := 0;
      OPEN cur_d;
      LOOP
        FETCH cur_d
          INTO row_d;
        EXIT WHEN cur_d%NOTFOUND OR cur_d%NOTFOUND IS NULL;
        --
        --j := j + 1;
        l_rate := row_d.sdcjje / l_totcjje;
        FOR row_tsfa IN (SELECT tsfaqtype, SUM(tsfaqje) qje
                           FROM tktsellfqamt
                          WHERE tsfabillno = row_n_th.shbillno --退货小票分摊
                          GROUP BY tsfaqtype)
        LOOP
          l_ftqje := round(row_tsfa.qje * l_rate, 2);
          j       := j + 1;
          INSERT INTO tktsellfqamt
            (tsfabillno, tsfarowno, tsfapggdid, tsfaseq, tsfatransno,
             tsfacbillno, tsfacrowno, tsfaqtype, tsfaqje, tsfavalidje,
             tsfatotvalidje, tsfatotqje, tsfacjje)
          VALUES
            (p_billno, row_d.sdrowno, row_d.sdgdid, j, 0, p_billno,
             row_d.sdrowno, row_tsfa.tsfaqtype, l_ftqje, row_d.sdcjje,
             l_totcjje, row_tsfa.qje, row_d.sdcjje);
        END LOOP;
      END LOOP;
    END IF;
    --删除换货退货小票的分摊
    DELETE FROM tktsellfqamt WHERE tsfabillno = row_n_th.shbillno;
    o_retcode := 1;
    o_retmsg  := '成功';
  EXCEPTION
    WHEN OTHERS THEN
      o_retcode := 0;
      o_retmsg  := '执行换货退券失败!||小票序列号:' || to_char(p_billno) || ',' ||
                   SQLERRM;
  END;

  PROCEDURE jobrun IS
    CURSOR cur_fq IS
      SELECT * FROM sellmailfq;
    r_fq      cur_fq%ROWTYPE;
    l_billno  NUMBER;
    l_transno NUMBER;
    l_retcode NUMBER;
    l_person  VARCHAR2(20);
    l_retmsg  VARCHAR2(255);
  BEGIN
    OPEN cur_fq;
    LOOP
      FETCH cur_fq
        INTO r_fq;
      EXIT WHEN cur_fq%NOTFOUND OR cur_fq%NOTFOUND IS NULL;
      BEGIN
        DELETE FROM sellmailfq WHERE smkey = r_fq.smkey;
        --
        l_billno := to_number(r_fq.smkey);
        SELECT seq_popms.NEXTVAL INTO l_transno FROM dual;
        salepopms.appendsh(l_billno, r_fq.smsyjid, r_fq.sminvno, l_transno);
        salepopms.salems(l_transno, 'Q', l_person, l_retcode, l_retmsg);
        IF l_retcode <> 0 THEN
          sp_opererrors('满送计算-JOB', '1111',
                        'salepopms.JobRun执行失败,事务号:' || to_char(l_transno) ||
                         ',billno:' || to_char(l_billno) || ',' || l_retmsg);
        END IF;
      EXCEPTION
        WHEN OTHERS THEN
          sp_opererrors('满送计算-JOB', '1112',
                        'salepopms.JobRun执行失败,事务号:' || l_transno ||
                         ',billno:' || to_char(l_billno) || ',' || SQLERRM);
      END;
    END LOOP;
    CLOSE cur_fq;
  END;
  PROCEDURE job_submit(i_time IN NUMBER) AS
    jobid  BINARY_INTEGER;
    l_time NUMBER;
  BEGIN
    l_time := round(i_time / 24 / 60, 3);
    dbms_job.submit(job => jobid, what => 'salepopms.JobRUN;',
                    next_date => SYSDATE + l_time,
                    INTERVAL => 'SYSDATE + ' || to_char(l_time),
                    no_parse => NULL);
    COMMIT;
    dbms_output.put_line('JobID = ' || jobid);
  END job_submit;
