  Procedure Pktruleadjust(p_Billno In Char,
                          p_Person In Char,
                          p_Billid In Char,
                          p_Djlb   In Char) Is
    Hd Tkparkrule%Rowtype;
  Begin
    Begin
      Select * Into Hd From Tkparkrule Where Tkpdjbh = p_Billno For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, p_Billno || '->1 单据不存在');
    End;
    If Hd.Tkpflag Not In (Flagediting) Then
      Raise_Application_Error(Errcode, p_Billno || '->3 单据已经审核！');
    End If;
    --修改本地标志
    Update Tkparkrule
       Set Tkpflag = Flagapproved, Auditdate = Sysdate, Auditor = p_Person
     Where Tkpdjbh = p_Billno;
    Cmdsend.Sp_Cmdsend_Allmkt('TKPARKRULEUPT', '''' || p_Billno || '''');
    Commit;
  End;

  Procedure Pktjsadjust(p_Billno In Char,
                        p_Person In Char,
                        p_Billid In Char,
                        p_Djlb   In Char) Is
    Hd           Tkpjshead%Rowtype;
    Row_Detail   Tkpjsdetail%Rowtype;
    Row_Tkdetail Tkpark_Detail%Rowtype;
    Row_Tklist   Tkpark_List%Rowtype;
    Vsupid       Varchar2(20);
    Vwmid        Char(1);
    Vcount       Number;
    Vsum         Number;
    Vfdjesum     Number;
    Vfdje        Number;
    Vcontno      Varchar2(20);
    Vrate        Number;
    Lseq         Number;
    Cursor r_Detail(Vdjbh Varchar2) Is
      Select *
        From Tkpjsdetail Ta
       Where Ta.Tkpjdbillno = Vdjbh
         And Ta.Tkpjdsource = '1'
         For Update;
    Cursor r_List(Vseq Number, Vruleid Varchar2) Is
      Select *
        From Tkpark_List Ta
       Where Ta.Billno = Vseq
         And Ta.Tkruleid = Vruleid
         For Update;
    Cursor r_List_Fd(Vseq Number, Vruleid Varchar2) Is
      Select *
        From Tkpark_List
       Where Billno = Row_Tkdetail.Sellbillno
         And Tkruleid = Row_Tkdetail.Tkruleid
         And Decode(Jsgys, '', '@1', Null, '@1', Jsgys) <> '@1';
  
  Begin
    Begin
      Select *
        Into Hd
        From Tkpjshead
       Where Tkpjhbillno = p_Billno
         For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, p_Billno || '->1 单据不存在');
    End;
    If Hd.Tkpjhflag Not In (Flagediting) Then
      Raise_Application_Error(Errcode, p_Billno || '->3 单据已经审核！');
    End If;
    Open r_Detail(p_Billno); --单据明细
    Loop
      Fetch r_Detail
        Into Row_Detail;
      Exit When r_Detail%Notfound Or r_Detail%Notfound Is Null;
      Begin
        Select * --停车券状态
          Into Row_Tkdetail
          From Tkpark_Detail
         Where Tkbillno = Row_Detail.Tkpjdtpno
           And (Jspayoutid = 'N' Or Jspayoutid Is Null);
      Exception
        When Others Then
          Raise_Application_Error(Errcode,
                                  Row_Detail.Tkpjdtpno || '->4 停车券已经核销');
      End;
      Open r_List(Row_Tkdetail.Sellbillno, Row_Tkdetail.Tkruleid); --该规则,小票下的销售明细
      Loop
        Fetch r_List
          Into Row_Tklist;
        Exit When r_List%Notfound Or r_List%Notfound Is Null;
        Begin
          Select Gmfsupid, Gmfwmid
            Into Vsupid, Vwmid
            From Goodsmframe
           Where Gmfgdid = Row_Tklist.Code
             And Gmfmfid = Row_Tklist.Gz
             And Rownum = 1;
        Exception
          When Others Then
            Raise_Application_Error(Errcode,
                                    Row_Tklist.Code || '->5 未定义主营供应商');
        End;
        Select Count(*)
          Into Vcount
          From Tkp_Js_Nogys
         Where Supid = Vsupid
           And Mkt = Row_Tklist.Mkt;
        If Vcount <= 0 Then
          Update Tkpark_List
             Set Jsgys = Vsupid, Wmid = Vwmid
           Where Current Of r_List;
        Else
          Update Tkpark_List Set Jsgys = '' Where Current Of r_List;
        End If;
      End Loop;
      Close r_List;
      Select Sum(Hjje)
        Into Vsum
        From Tkpark_List
       Where Billno = Row_Tkdetail.Sellbillno
         And Tkruleid = Row_Tkdetail.Tkruleid
         And Decode(Jsgys, '', '@1', Null, '@1', Jsgys) <> '@1';
      If Vsum <= 0 Then
        Raise_Application_Error(Errcode,
                                Row_Tklist.Code || '->6 参加核销的供应商,有效余额不足');
      End If;
      Select Count(*)
        Into Vcount
        From Tkpark_List
       Where Billno = Row_Tkdetail.Sellbillno
         And Tkruleid = Row_Tkdetail.Tkruleid
         And Decode(Jsgys, '', '@1', Null, '@1', Jsgys) <> '@1';
      Vfdjesum := Row_Tkdetail.Tkparkrulejz;
    
      Open r_List_Fd(Row_Tkdetail.Sellbillno, Row_Tkdetail.Tkruleid); --该规则,小票下的销售明细
      Loop
        Fetch r_List_Fd
          Into Row_Tklist;
        Exit When r_List_Fd%Notfound Or r_List_Fd%Notfound Is Null;
        If Vcount = 1 Then
          Vfdje    := Vfdjesum;
          Vfdjesum := Vfdjesum - Vfdje;
        Else
          Vfdje    := Round(Row_Tklist.Hjje / Vsum *
                            Row_Tkdetail.Tkparkrulejz,
                            2);
          Vfdjesum := Vfdjesum - Vfdje;
        End If;
        Vcount := Vcount - 1;
        Select Fgetcontno(Row_Tklist.Jsgys,
                          Row_Tklist.Wmid,
                          Trunc(Sysdate),
                          Row_Tklist.Gz,
                          Row_Tklist.Ppcode,
                          Row_Tklist.Catid)
          Into Vcontno
          From Dual;
        If Vcontno Is Null Or Vcontno = '' Then
          Raise_Application_Error(Errcode,
                                  Row_Tklist.Jsgys ||
                                  '->7 参加核销的供应商,找不到有效合同');
        End If;
        Select Max(Tgffseqno)
          Into Lseq
          From Tktgoodsparkft, Billmktdetail
         Where ((Tgffmode = '3' And Tgffsupid = Row_Tklist.Jsgys And
               Tgffwmid = Row_Tklist.Wmid And
               Tgffmfid In (Row_Tklist.Gz, 'ALL')) Or
               (Tgffmode = '1' And Tgffbarcode = Row_Tklist.Code))
           And Tgffstartdate <= Trunc(Sysdate)
           And Tgffenddate >= Trunc(Sysdate)
           And Tgffbillno = Bmdbillno
           And Bmdmkt = Row_Tklist.Mkt
           And Bmdbillid = 'QS9';
        If nvl(Lseq,0) = 0 Then
          Vrate := 1;
        Else
          Select Tgffrate
            Into Vrate
            From Tktgoodsparkft
           Where Tgffseqno = Lseq;
        End If;
        Insert Into Tkpjslist
        --单据号,停车券号,小票号,供应商,经营方式,合同,商品编码,费用ID,分摊金额
          Select p_Billno,
                 Row_Tkdetail.Tkbillno,
                 Row_Tkdetail.Sellbillno,
                 Row_Tklist.Jsgys,
                 Row_Tklist.Wmid,
                 Vcontno,
                 Row_Tklist.Code,
                 Row_Detail.Tkpjdpayoutid,
                 Vfdje * Vrate,
                 Row_Tklist.Gz,
                 Row_Tklist.Catid,
                 Row_Tklist.Ppcode,
                 Trunc(Sysdate)
            From Dual;
        Update Tkpark_Detail Ta
           Set Ta.Jspayoutid = 'Y', Ta.Jsgys = Row_Tklist.Jsgys
         Where Ta.Tkbillno = Row_Detail.Tkpjdtpno
           And (Jspayoutid = 'N' Or Jspayoutid Is Null);
      
      End Loop;
      Close r_List_Fd;
    
    End Loop;
    Close r_Detail;
    --取有效分摊余额
    /*Update tkpark_list ta
      Set ta.jsyxje = ta.hjje
    Where ta.jsyxje Is Null
      And ta.billno = row_tkdetail.sellbillno;*/
    ------全部取主营供应商
  
    --修改本地标志
    Update Tkpjshead
       Set Tkpjhflag = Flagapproved,
           Auditdate = Sysdate,
           Auditor   = p_Person
     Where Tkpjhbillno = p_Billno;
    Commit;
  End;
