CREATE OR REPLACE PACKAGE BODY JAVA_GETBARCODEGIFT
AS
  -- 序号,优惠单号,优惠类型,优惠规则,优惠方式,商品编码,柜组,多单位代码,商品品类,商品品牌,销售数量,电子秤商品的优惠的时间间隔(秒),开始日期
  -- 结束日期,开始时间,结束时间,优惠零售价,优惠会员价,优惠批发价,优惠零售价折扣率,优惠会员价折扣率,优惠批发价折扣率,优惠零售价折扣分担  ,
  -- 优惠会员价折扣分担,优惠批发价折扣分担,备用,备用,备用,备用,备用

  PROCEDURE GETBARCODEGIFT
    (
            vcode in varchar2,

      rc OUT RETCURSOR
    )
  AS
  BEGIN
    OPEN rc FOR

    SELECT PPISEQ,PPIBILLNO,PPIMODE,PPIPRESENTCODE,PPIPRESENTUNIT,PPIBARCODE,PPIMFID,PPISPEC,PPICATID,PPIPPCODE,PPIMAXNUM,PPISPACE,
       TO_CHAR(PPISTARTDATE,'YYYY/MM/DD'),TO_CHAR(PPIENDDATE,'YYYY/MM/DD'),PPISTARTTIME,PPIENDTIME,
       PPINEWSJ,PPINEWHYJ,PPINEWPFJ,PPINEWPFRATE,PPINEWHYRATE,PPINEWPFRATE,PPIZKFD,PPIHYZKFD,PPIPFZKFD,
       '','','',0,0 FROM POSPOPINFO WHERE PPIBILLNO = vcode and PPIPRESENTCODE = 'RJG';

  END GETBARCODEGIFT;

END JAVA_GETBARCODEGIFT;
