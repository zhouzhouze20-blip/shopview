CREATE OR REPLACE PACKAGE BODY MOB_GETPAYMODE
AS
  -- 获取终端付款方式定义
  -- 返回结果集为[代码,上级代码,级次,名称,类型,是否找零,是否溢余,是否末级,是否面值卡(N（不是），Y（是），S（返券卡）),是否银联卡最小金额,最大金额,汇率,备用字符串,备用数字]
  PROCEDURE GETPAYMODE
  (
    vterm IN VARCHAR2,
    rc OUT RETCURSOR
  )
  AS
  BEGIN
    OPEN rc FOR
      SELECT PPTCODE,PPMPCODE,PPMCLASS,PPMNAME,PPMTYPE,PPMISZL,PPMISYY,PPMFLAG,
        DECODE(PPMTYPE,'4','Y','6','Y','7','Y','8','Y','5','S','N'),DECODE(PPMTYPE,'3','Y','N'),
        PPMMINJE,PPMMAXJE,PPMHL,'',0
      FROM POSPAYTEMPLET,POSPAYMODE,SYJMAIN
        WHERE PPTID = SYJPAYTEMPLET
          AND SYJID= vterm AND PPTCODE = PPMCODE
          AND PPMSTATUS='Y'
          ORDER BY PPMCLASS,PPMPCODE,PPTCODE;
  END GETPAYMODE;

END MOB_GETPAYMODE;
