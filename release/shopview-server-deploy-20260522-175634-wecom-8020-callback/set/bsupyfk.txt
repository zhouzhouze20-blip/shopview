CREATE OR REPLACE PACKAGE BODY "BSUPYFK" Is
  currentdate    Date;
  currentmkt     Varchar2(20);
  currentmkttype Char(1);

  -- 根据订单生成预付款单
  Procedure generate(p_orderbillno In Char) Is
    row_boh   borderhead%Rowtype;
    ltotje    Number;
    ltotbhsje Number;
    lrate     Number;
    lyfk      Number;
    lyfkbhs   Number;
    ljxtax    Number;
    lbillno   supsetyfk.ssybillno%Type;
    lbank     Varchar2(100);
    laccount  varchar2(100);
    ltaxno    Varchar2(100);
  Begin
    -- 查找原始订单
    Begin
      Select * Into row_boh From borderhead Where bohbillno = p_orderbillno;
    Exception
      When Others Then
        Return;
    End;
    If row_boh.bohwmid <> '1' Then
      Return;
    End If;
    If currentmkttype = stock.mf_hq Then
      If row_boh.bohflag <> border.flagsended Then
        Return;
      End If;
    Elsif currentmkttype = stock.mf_gfc Then
      Return;
    Else
      If row_boh.bohflag Not In (border.flagapproved, bOrder.FlagSended) Then
        Return;
      End If;
    End If;

    --订单税率
    Begin
      Select Distinct bodjxtax
        Into ljxtax
        From borderdetail
       Where bodbillno = p_orderbillno
         And bodiszp = 'N';
    Exception
      When no_data_found Then
        Null;
      When Others Then
        raise_application_error(errcode,
                                '订单【' || p_orderbillno || '】在生成预付款时税率取值错误！');
    End;
    --订单总含税进价金额
    Begin
      Select nvl(Sum(nvl(bodyshsjjje, 0)), 0), nvl(Sum(nvl(bodysbhsjjje, 0)), 0)
        Into ltotje, ltotbhsje
        From borderdetail
       Where bodbillno = p_orderbillno;
    Exception
      When Others Then
        Return;
    End;
    If ltotje <= 0 Then
      Return;
    End If;
    --预付款比例
    Begin
      Select nvl(bosrate, 0)
        Into lrate
        From bordersettle
       Where bosbillno = p_orderbillno
         And bostype = '1'
         And bossetid = '1'
         And rownum = 1;
    Exception
      When Others Then
        Return;
    End;
    --预付款
    lyfk    := round(ltotje * lrate, 2);
    lyfkbhs := round(ltotbhsje * lrate, 2);
    If lyfk <= 0 Or lyfk Is Null Then
      Return;
    End If;
    If lrate <= 0 Then
      Return;
    End If;
    --获得预付款单号
    If fgetsysplace Is Null Then
       fterp_oper.Set_Oper_Market(row_boh.bohshmkt);
    End If;
    sp_billseq(billid, lbillno, 'N');
    begin
         Select nvl(CMCHAR1,SBBANK), nvl(cmchar2,SBACCNTNO),sbtaxno
         into lbank,laccount,ltaxno
         from contmain ,supplierbase
         where CMCONTNO = row_boh.bohhtbh and sbid = row_boh.bohsupid
               and cmsupid = sbid;
    exception when others then
              lbank :='';
              laccount :='';
              ltaxno := '';
    end ;
    Insert Into supsetyfk
      (ssybillno,
       ssydjbh,
       ssyorderno,
       ssysupid,
       ssywmid,
       ssyflag,
       ssymarket,
       ssymfid,
       ssycontno,
       ssydhje,
       ssyrate,
       ssyyfje,
       ssysfje,
       inputor,
       inputdate,
       ssyjxtax,
       ssysfbhsje,
       ssybank,
       ssyaccount,
       ssytaxno)
    Values
      (lbillno,
       lbillno,
       row_boh.bohbillno,
       row_boh.bohsupid,
       row_boh.bohwmid,
       flagautogen,
       row_boh.bohshmkt,
       row_boh.bohshmfid,
       nvl(row_boh.bohhtbh, '0'),
       ltotje,
       lrate,
       lyfk,
       lyfk,
       row_boh.auditor,
       Sysdate,
       ljxtax,
       lyfkbhs,
       lbank,
       laccount,
       ltaxno);
    --本门店结算，故不再发送
    --IF currentmkttype = stock.mf_mkt THEN
    --  cmdsend.sp_cmdsend_target(cmdcopytohq,
    --                            currentmkt,
    --                            stock.hqcode, /*接收逻辑点*/
    --                            '''' || lbillno || '''', /*ERP参数，可能包括多个参数的串*/
    --                            NULL /*POS参数，可能包括多个参数的串*/);
    --END IF;
  Exception
    When Others Then
      Raise;
      Rollback;
  End generate;

  -- 审核
  Procedure approve(p_billno In Char,
                    p_person In Char,
                    p_billid In Char,
                    p_djlb   In Char) Is
    row_ssy supsetyfk%Rowtype;
    row_boh borderhead%Rowtype;
    row_bih binstrhead%Rowtype;
  Begin
    --IF currentmkttype <> stock.mf_hq THEN
    --  raise_application_error(errcode, '预付款单【' || p_billno || '】只能在总部审核！');
    --END IF;

    Begin
      Select * Into row_ssy From supsetyfk Where ssybillno = p_billno;
    Exception
      When no_data_found Then
        raise_application_error(errcode, '预付款单【' || p_billno || '】不存在，可能已被删除！');
      When Others Then
        raise_application_error(errcode, '预付款单【' || p_billno || '】打开失败！');
    End;

    -- 检查单据状态
    If row_ssy.ssyflag = flagapproved Then
      raise_application_error(errcode, '预付款单【' || p_billno || '】已审核，不能重复审核！');
    End If;
    If row_ssy.ssyflag = flagcanceled Then
      raise_application_error(errcode, '预付款单【' || p_billno || '】已取消，不能审核！');
    End If;
    If currentmkttype <> stock.mf_hq Then
       If row_ssy.ssymarket <> fGetSysPlace Then
         raise_application_error(errcode, '预付款单【' || p_billno || '】只能在本门店审核！');
       End If;
    End If;
    -- 检查订单
    Begin
      Select * Into row_boh From borderhead Where bohbillno = row_ssy.ssyorderno;
    Exception
      When no_data_found Then
        raise_application_error(errcode,
                                '订单【' || row_ssy.ssyorderno || '】不存在，可能已被删除！');
      When Others Then
        raise_application_error(errcode, '订单【' || row_ssy.ssyorderno || '】打开失败！');
    End;

    If row_boh.bohsupid <> row_ssy.ssysupid Or row_boh.bohwmid <> row_ssy.ssywmid Then
      raise_application_error(errcode,
                              '预付款单【' || p_billno || '】供应商经营方式与订单不一致！');
    End If;

    -- 检查经营方式
    If row_ssy.ssywmid <> '1' Then
      raise_application_error(errcode, '预付款单【' || p_billno || '】经营方式无效！');
    End If;

    -- 检查入库单
    Begin
      Select * Into row_bih From binstrhead Where bihorderno = row_ssy.ssyorderno;
    Exception
      When Others Then
        row_bih.bihflag := 'N';
    End;

    If row_bih.bihflag = 'Y' Or row_bih.bihflag = 'Q' Then
      raise_application_error(errcode,
                              '预付款单【' || p_billno || '】对应的入库单【' || row_bih.bihbillno ||
                              '】已审核或取消，故预付款不能再审核！');
    End If;

    If row_ssy.ssysfje < 0 Then
      raise_application_error(errcode, '预付款单【' || p_billno || '】实付金额不能为负数！');
    End If;

    -- 插入结算批次
    /*DELETE FROM settlesupbatch WHERE ssbbillno = row_ssy.ssybillno AND ssbtype = '0';
    INSERT INTO settlesupbatch
      (ssbseq, ssbbillno, ssbsetno, ssbdjbh, ssbtran, ssbcontno, ssbsupid, ssbwmid, ssbmmkt,
       ssbmmfid, ssbjhdate, ssbday, ssbrate, ssbbhsje, ssbhsje, ssbstatus, ssbjsbillno, ssbjsdjbh,
       ssbjsdate, ssbisjs, ssbflag, ssbtype, ssbn0, ssbn1, ssbn2, ssbn3, ssbn4, ssbn5, ssbn6, ssbn7,
       ssbn8, ssbn9, ssbvc0, ssbvc1, ssbvc2, ssbvc3, ssbvc4, ssbvc5, ssbvc6, ssbvc7, ssbvc8, ssbvc9)
    VALUES
      (seq_settlesupbatch.NEXTVAL, row_ssy.ssybillno, 1, row_ssy.ssydjbh, stock.transid_jh,
       nvl(row_ssy.ssycontno, '0'), row_ssy.ssysupid, row_ssy.ssywmid, row_ssy.ssymarket,
       row_ssy.ssymfid, currentdate, 0, row_ssy.ssyrate, 0, row_ssy.ssysfje, 'N', NULL, NULL, NULL,
       'Y', 'Y', '0', row_ssy.ssydhje, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL,
       row_ssy.ssyorderno, row_bih.bihbillno, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL);
    */
    -- 更新标志
    Update supsetyfk
       Set ssyflag = flagapproved, auditor = p_person, auditdate = currentdate
     Where ssybillno = row_ssy.ssybillno;
    supcamana.yfkgen(p_billno, 'PAY');
    
    If CurrentMktType = Stock.MF_HQ Then
       CmdSend.sp_CmdSend_Target('SUPYFK', '0', row_ssy.ssymarket, ''''|| row_ssy.ssybillno || '''', null);
    End If;
    
    Commit;
  Exception
    When Others Then
      Raise;
      Rollback;
  End approve;

  -- 取消
  Procedure cancel(p_billno In Char,
                   p_person In Char,
                   p_billid In Char,
                   p_djlb   In Char) Is
    row_ssy supsetyfk%Rowtype;
    row_bih binstrhead%Rowtype;
    row_ssb settlesupbatch%Rowtype;
  Begin

    raise_application_error(errcode, '预付款单【' || p_billno || '】不能取消！');

    Return;

    If currentmkttype <> stock.mf_hq Then
      raise_application_error(errcode, '预付款单【' || p_billno || '】只能在总部取消！');
    End If;

    Begin
      Select * Into row_ssy From supsetyfk Where ssybillno = p_billno;
    Exception
      When no_data_found Then
        raise_application_error(errcode, '预付款单【' || p_billno || '】不存在，可能已被删除！');
      When Others Then
        raise_application_error(errcode, '预付款单【' || p_billno || '】打开失败！');
    End;

    -- 检查单据状态
    If row_ssy.ssyflag In (flagediting, flagautogen) Then
      raise_application_error(errcode,
                              '预付款单【' || p_billno || '】未审核，不需取消，请直接删除！');
    End If;
    If row_ssy.ssyflag = flagcanceled Then
      raise_application_error(errcode, '预付款单【' || p_billno || '】已取消，不能重复取消！');
    End If;

    -- 检查入库单
    Begin
      Select * Into row_bih From binstrhead Where bihorderno = row_ssy.ssyorderno;
      If row_bih.bihflag = 'Y' Then
        raise_application_error(errcode,
                                '预付款单【' || p_billno || '】对应的入库单【' + row_bih.bihbillno +
                                '】已审核，故预付款不能取消！');
      End If;
    Exception
      When Others Then
        Null;
    End;

    -- 检查结算批次
    Begin
      Select *
        Into row_ssb
        From settlesupbatch
       Where ssbbillno = row_ssy.ssybillno
         And ssbtype = '0';
      If row_ssb.ssbstatus <> 'N' Then
        raise_application_error(errcode, '预付款单【' || p_billno || '】已结算付款，不能取消！');
      End If;
    Exception
      When Others Then
        raise_application_error(errcode,
                                '预付款单【' || p_billno || '】结算批次信息不存在，可能已结算，无法取消！');
    End;

    -- 删除结算批次
    Delete From settlesupbatch Where ssbseq = row_ssb.ssbseq;

    -- 更新标志
    Update supsetyfk
       Set ssyflag = flagcanceled, auditor = p_person, auditdate = currentdate
     Where ssybillno = row_ssy.ssybillno;
     
    supcamana.yfkgen(p_billno, 'PAY_CANCEL');
    
    If CurrentMktType = Stock.MF_HQ Then
       CmdSend.sp_CmdSend_Target('SUPYFK', '0', row_ssy.ssymarket, ''''|| row_ssy.ssybillno || '''', null);
    End If;
     
    Commit;
  End cancel;
Begin
  /*currentdate    := fgeterpdate;
  currentmkt     := fgetsysplace;
  currentmkttype := fgetsystype(currentmkt);*/
  CurrentDate := fGetErpDate;
  CurrentMkt  := fGetPrivatePara('1N');

  --CurrentMktType := FGetSysType(CurrentMkt);
  If CurrentMkt = 'HQ' Then
    CurrentMktType := Stock.MF_HQ;
  Else
    CurrentMktType := Stock.MF_MKT;
  End If;
End bsupyfk;
