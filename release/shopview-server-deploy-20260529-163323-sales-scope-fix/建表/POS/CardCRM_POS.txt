CREATE OR REPLACE Package Body CardCRM_POS Is

	--CurrentMkt Varchar2(20) := fGetSysPlace();
	--CurrentMu  Varchar2(20) := fGetPrivatePara('1M');
	b_SysDate  Date := Trunc(Sysdate);

	Function f_GetPosPara
	(
		p_Market In Varchar2,
		p_Code   In Varchar2
	) Return Varchar2 Is
		Result Varchar2(80);
	Begin
		Select ParaValue
		  Into Result
		  From dbusrposnew.syspara
		 Where Mkt = p_Market And
				 Code = p_Code;
		Return Result;
	Exception
		When Others Then
			Return '';
	End;

	Function f_PayModeIsMzk
	(
		p_Market  In Varchar2,
		p_PayCode In Varchar2
	) Return Char Is
		l_Value Varchar2(100);
		l_sPos  Number;
		l_ePos  Number;
	Begin
		l_Value := f_GetPosPara(p_Market, 'OC');
	
		l_sPos := Instr(l_Value, 'PaymentMzk', 1);
		l_ePos := Instr(l_Value, ';', l_sPos);
		If l_ePos <= 0 Then
			l_ePos := Length(l_Value) + 1;
		End If;
		If Instr(SubStr(l_Value, l_sPos, l_ePos - l_sPos) || ',', ',' || p_PayCode || ',', 1) > 0 Then
			Return 'Y';
		End If;
	
		l_sPos := Instr(l_Value, 'PaymentMzkPaper', 1);
		l_ePos := Instr(l_Value, ';', l_sPos);
		If l_ePos <= 0 Then
			l_ePos := Length(l_Value) + 1;
		End If;
		If Instr(SubStr(l_Value, l_sPos, l_ePos - l_sPos) || ',', ',' || p_PayCode || ',', 1) > 0 Then
			Return 'Y';
		End If;
	
		Return 'N';
	End f_PayModeIsMzk;

	--==================================================================
	-- POS消费对帐
	--==================================================================
	Procedure Sp_Build_PosCardPay(p_Date In Date) As
	
		Cursor c_c(vdate In Date) Is
			Select trunc(ta.shdate) pcpdate,
					 shmarket  Market,
					 ta.shdjlb pcpdjlb,
					 ta.shbillno pcpbillno,
					 ta.shsyjid pcpsyjid,
					 ta.shinvno pcpinvno,
					 tb.spno pcpcdmno,
					 tc.ppmtype paytype,
					 fgetinvsign(ta.shdjlb) * tb.spmoney pcpmoney,
					 'N' pcpflag,
					 sprowno  pcpnum1,
					 Null pcpchr,
					 shmanaunit  pcpvchr
			  From sellhead    ta,
					 sellpayment tb,
					 PosPaymode  tc
			 Where ta.shbillno = tb.spbillno And
					 tb.sppmcode = tc.ppmcode And
					 ta.shdate >= trunc(vdate) And
					 ta.shdate < trunc(vdate) + 1 And
					 (tc.ppmtype In ('4') Or f_PayModeIsMzk(ShMarket, PpmCode) = 'Y') And
					 ta.shdjlb <> '3' And
					 tb.spno Is Not Null
			 Order By ta.shbillno;
		r_c poscardpay%Rowtype;
	Begin
		Delete From poscardpay Where pcpdate = p_Date;
		Open c_c(p_Date);
		Loop
			Fetch c_c
				Into r_c;
			Exit When c_c%Notfound Or c_c%Notfound Is Null;
		
			Insert Into poscardpay Values R_c;
		End Loop;
		Close c_c;
		Commit;
	Exception
		When Others Then
			Rollback;
			Raise;
	End Sp_Build_PosCardPay;

	Procedure Sp_Build_PosCardPay
	(
		p_Date1 In Date,
		p_Date2 In Date
	) As
		lddates Number;
		vdate   Date;
		lret    Number;
	Begin
		Select trunc(p_Date2) - trunc(p_Date1) + 1 Into lddates From dual;
	
		For lret In 1 .. lddates
		Loop
			vdate := p_Date1 + lret - 1;
			Delete From poscardpay Where pcpdate = vdate;
			Sp_Build_PosCardPay(vdate);
			Commit;
		End Loop;
		Commit;
	Exception
		When Others Then
			Rollback;
			raise_application_error(-20101, 'Sp_Build');
	End Sp_Build_PosCardPay;

  PROCEDURE sp_sendpay_day(p_date IN DATE) AS
  BEGIN
    FOR vr IN (SELECT DISTINCT pcpmkt, pcpmu 
                 FROM poscardpay
                WHERE pcpdate = p_date
                  AND pcpflag = 'N')
    LOOP
      DELETE FROM cardpayment
       WHERE cpmmkt = vr.pcpmkt
         AND cpmjygs = vr.pcpmu
         AND cpmdate = p_date;
    END LOOP;
  
    INSERT INTO cardpayment
      (cpmdate, cpmjygs, cpmmkt, cpmdjlb, cpmbillno, cpmsyjid, cpminvno,
       cpmcdmno, cpmpaytype, cpmmoney, cpmprowno, cpmchr, cpmvchr)
      SELECT pcpdate, pcpmu, pcpmkt, pcpdjlb, pcpbillno, pcpsyjid, pcpinvno,
             pcpcdmno, pcppaytype, pcpmoney, pcpprowno , pcpchr,  ''
        FROM poscardpay
       WHERE pcpdate = p_date
         AND pcpflag = 'N';
    UPDATE poscardpay
       SET pcpflag = 'Y'
     WHERE pcpdate = p_date
       AND pcpflag = 'N';
    COMMIT;
  EXCEPTION
    WHEN OTHERS THEN
      ROLLBACK;
      raise_application_error(-20101, 'sp_send_day');
  END sp_sendpay_day;

	Procedure SubmitPay(p_Date In Date) Is
	Begin
		Sp_Build_PosCardPay(p_Date);
		Commit;
		Sp_SendPay_Day(p_Date);
		Commit;
	End SubmitPay;

	Procedure SubmitPay Is
	Begin
		SubmitPay(b_SysDate - 1);
	End SubmitPay;

Begin
	Null;
End CardCRM_POS;
