  FUNCTION fIsDateType(vdate IN VARCHAR2) RETURN CHAR IS
    ldate    DATE;
  BEGIN
    SELECT to_date(vdate,'yyyymmdd') INTO ldate FROM dual;
    RETURN 'Y';
  EXCEPTION
    WHEN OTHERS THEN
      RETURN 'N';
  END;

  Function fcwzzhsje(row_ref cwzzref%Rowtype) Return Number Is
    lhsje Number;
  Begin
    If row_ref.vmode = '1' Then
      /*取值cwzz_value, para3  */
      lhsje := row_ref.bhsje * row_ref.ishz; /*核算金额*/
    Elsif row_ref.vmode = '2' Then
      lhsje := row_ref.zzse * row_ref.ishz;
    Elsif row_ref.vmode = '3' Then
      lhsje := row_ref.hsje * row_ref.ishz;
    Else
      lhsje := 0;
    End If;
    lhsje := nvl(lhsje, 0);
    Return(lhsje);
  End fcwzzhsje;

  Procedure ins_midvoucher(row_voucher cwzzvoucher%Rowtype) As
    b_gl    nc_temp_gl%rowtype;

    l_type  nvarchar2(100);
  Begin
    select seq_nc_temp_gl.nextval into l_id from dual;

    b_gl.voucher_type                        := b_type;
    b_gl.fiscal_year                         := b_year;
    b_gl.accounting_period                   := b_perd;
    b_gl.voucher_id                          := row_voucher.transno;
    b_gl.c_datetime                          := to_char(row_voucher.fdate, 'yyyy-mm-dd');

    b_gl.c_signature                         := 'N';

       b_gl.account_code                        := row_voucher.facctid;

       b_gl.account_code                        := row_voucher.fobjid3;

    b_gl.abstract                            := row_voucher.fexd;

    b_gl.dept_id                             := row_voucher.fobjid1;

  Procedure submit
  (
    vqsrq In Varchar2,
    vjsrq In Varchar2,
    vtype In Char,
    vname In Char,
    vzt   In CHAR,
    vtransno IN NUMBER
  ) Is
    row_type cwzztype%Rowtype;
    cur_det  Integer;
    row_det  Integer;
    billno   Varchar2(20);
  Begin
    If vqsrq Is Null Or vjsrq Is Null Or vqsrq > vjsrq Or ltrim(vtype) Is Null Then
      raise_application_error(-20535, '输入参数无效，请检查');
    End If;
    Begin
      Select * Into row_type From cwzztype Where code = vtype And flag = 'Y';
    Exception
      When Others Then
        raise_application_error(-20535,
                                '在 CWZZTYPE 事务参数中未定义或无效,事务为' ||
                                vtype);
    End;
    If instr(row_type.cur_qt, 'HSQSRQ') > 15 Then
      row_type.cur_qt := Replace(row_type.cur_qt, 'HSQSRQ', vqsrq);
    End If;
    If instr(row_type.cur_qt, 'HSJSRQ') > 15 Then
      row_type.cur_qt := Replace(row_type.cur_qt, 'HSJSRQ', vjsrq);
    End If;
    If row_type.procpara = '3' Then
      If dbms_sql.is_open(cur_det) Then
        dbms_sql.close_cursor(cur_det);
      End If;
      cur_det := dbms_sql.open_cursor;
      dbms_sql.parse(cur_det, row_type.cur_qt, dbms_sql.v7);
      dbms_sql.define_column(cur_det, 1, billno, 20);
      Begin
        row_det := dbms_sql.Execute(cur_det);
      Exception
        When Others Then
          raise_application_error(-20535, 'SQL参数错误' || vtype);
      End;
      Loop
        If dbms_sql.fetch_rows(cur_det) > 0 Then
          dbms_sql.column_value(cur_det, 1, billno);
          gen_voucher(vqsrq, vjsrq, billno, vtype, vname, vzt, vtransno);
        Else
          Exit;
        End If;
      End Loop;
    Else
      billno := Null;
      gen_voucher(vqsrq, vjsrq, billno, vtype, vname, vzt, vtransno);
    End If;
  Exception
    When Others Then
      If dbms_sql.is_open(cur_det) Then
        dbms_sql.close_cursor(cur_det);
      End If;
      Raise;
  End submit;

  (
    vqsrq    In Varchar2,

    vtype    In Char,

    vzt      In CHAR
  ) Is
  row_type cwzztype%Rowtype;

      raise_application_error(-20535, '输入凭证类型参数无效，请检查');

      Select * Into row_type From cwzztype Where code = vtype And flag = 'Y';
