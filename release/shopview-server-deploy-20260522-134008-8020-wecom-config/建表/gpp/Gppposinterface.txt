  Function Fgettktbzbs(Vjygs  In Varchar2,
                       Vmkt   In Varchar2,
                       Vqtype In Varchar2,
                       Vdjlb  In Varchar2,
                       Vsyjid In Varchar2,
                       Vinvno In Varchar2) Return Number Is
    l_Seq      Number;
    l_Exchange Number := 1;
    l_Xssj     Char(5);
    l_invno    Varchar2(20);
  Begin
    If Vqtype = Fgetflqtype() Then
      Return 1;
    End If;
    Select To_Char(Sysdate, 'HH24:MI') Into l_Xssj From Dual;
    /*消费*/
    If Vdjlb = '1' Or (Vsyjid = '' Or Vinvno = '' Or Vinvno = '0000000') Then
      --券变值
      Select Nvl(Max(Tptseqno), 0)
        Into l_Seq
        From Tktpopticket
       Where Tptstartdate <= Trunc(Sysdate)
         And Tptenddate >= Trunc(Sysdate)
         And Tptstarttime <= l_Xssj
         And Tptendtime >= l_Xssj
         And Tptqtype = Vqtype
         And Tptjygs = Vjygs
         And Tptmkt = Vmkt;
      If l_Seq > 0 Then
        Select Tptbl
          Into l_Exchange
          From Tktpopticket
         Where Tptseqno = l_Seq;
      Else
        l_Exchange := 1;
      End If;
    End If;
    If Vdjlb In ('2', '4', '5') Then
      l_invno := to_number(Vinvno);
      Begin
        Select Nvl(Tcflnum3, 1)
          Into l_Exchange
          From Tktcardfqlog
         Where Tcfljygs = Vjygs
           And Tcflmkt = Vmkt
           And Tcflsyjid = Vsyjid
           And Tcflinvno = l_invno
           And Tcfljetype = Vqtype
           And Tcflzy In ('O', 'P')
           And Rownum = 1;
      Exception
        When Others Then
          l_Exchange := 1;
      End;
    End If;
    Return Nvl(l_Exchange, 1);
  End;
  /*得到券号*/
  Function Fgetticketno Return Varchar2 Is
    Vseqno        Number;
    Vticketnobase Varchar2(20);
    Vticketno     Varchar2(20);
    --手工券号增加校验位及随机位
  Begin

    Select Seq_Tktseq.Nextval Into Vseqno From Dual;

    --填充到TktLen长度
    If Length(To_Char(Vseqno)) < Tktlen Then
      Select Lpad(To_Char(Vseqno), Tktlen, '0')
        Into Vticketnobase
        From Dual;
    Else
      Select Lpad(Substr(To_Char(Vseqno), -tktlen, Tktlen), Tktlen, '0')
        Into Vticketnobase
        From Dual;
    End If;
    Select Vticketnobase || f_Makecheckcode(Vticketnobase, 1) ||
           --lpad(trunc(dbms_random.VALUE(0, 999999)), 6, '0')
            Lpad(Trunc(Dbms_Random.Value(0, Power(10, Tl_Random) - 1)),
                 Tl_Random,
                 '0')
      Into Vticketno
      From Dual;
    If fgetmanapara('BARCODE') = 'EAN13' Then
      Vticketno := fcodeadd20ean13(Vticketno);
    End If;

    Return Vticketno;
  End;

  Procedure Gettktpopinfo(p_Tpid      In Varchar2,
                          p_Cardno    In Varchar2,
                          p_Action    In Varchar2 Default 'ALL',
                          o_Retcode   Out Number,
                          o_Retmsg    Out Varchar2,
                          o_Startdate Out Date, --券开始日期
                          o_Enddate   Out Date, --券结束日期
                          o_Qno       Out Varchar2, -- 券号（持卡VIP，手工券号）
                          o_a_m       Out Varchar2, --q_auto or q_manu 电子券 or 打印券
                          o_Pj        Out Varchar2, -- 拼券 invalid
                          o_Area      Out Varchar2, -- 券使用范围 指定门店，本门店，通用
                          o_Ismz      Out Varchar2, -- 是否固定面值
                          o_Prnsite   Out Varchar2 -- 券打印地点 invalid
                          ) Is
    Row_Fqtitle Tktpopinfo%Rowtype;
    l_Isvip     Char(1);
    --vcurtime    CHAR(10);
  Begin
    Debug.Debug_New('Gettktpopinfo',
                    p_Tpid || ',' || p_Cardno || ',' || p_Action);
    --查找返券档期
    Begin
      Select * Into Row_Fqtitle From Tktpopinfo Where Tpiid = p_Tpid;
    Exception
      When Others Then
        Errhandle.Errlog();
        Raise_Error(Rtn_Failure, '返券档期找不到!返券档期:' || p_Tpid);
        /*o_Retcode := Rtn_Failure;
        o_Retmsg  := '返券档期找不到!返券档期:' || p_Tpid;
        Return;*/
    End;
    If p_Action = 'ALL' Then
      l_Isvip := 'Y';
      If Trim(p_Cardno) Is Null Then
        l_Isvip := 'N';
      End If;
      --row_fqtitle.TPHYFQMODE   会员返券方式 0电子券,1指定手工券号,2生成手工券号
      --row_fqtitle.TPFHYFQMODE  非会员返券方式 1指定手工券号,2生成手工券号
      --row_fqtitle.TPMZMODE     面值方式：0 规则合并(一个80),1固定面值(两个40)
      --row_fqtitle.TPPRINTPLACE 手工券打印地点 0前台,1后台 ,2 不用打券
      If l_Isvip = 'Y' Then
        If Row_Fqtitle.Tphyfqmode = '0' Then
          o_Qno := p_Cardno;
          o_a_m := q_Auto;
        Elsif Row_Fqtitle.Tphyfqmode = '1' Then
          Select 'MANU' || To_Char(Seq_Tktcust.Nextval)
            Into o_Qno
            From Dual;
          o_a_m := q_Manu;
        Elsif Row_Fqtitle.Tphyfqmode = '2' Then
          Select Fgetticketno Into o_Qno From Dual;
          o_a_m := q_Manu;
        End If;
      Else
        o_a_m := q_Manu;
        -- 2010-12-10 纸券如果过了活动期退货 直接生成券号
        If Nvl(b_tcflsouce, '1') = '4' Then
          Select Fgetticketno Into o_Qno From Dual;
        Else
          If Row_Fqtitle.Tpfhyfqmode = '1' Then
            Select 'MANU' || To_Char(Seq_Tktcust.Nextval)
              Into o_Qno
              From Dual;
          Elsif Row_Fqtitle.Tpfhyfqmode = '2' Then
            Select Fgetticketno Into o_Qno From Dual;
          Elsif Row_Fqtitle.Tpfhyfqmode = 'N' Then
            Raise_Error(Rtn_Failure,
                        '非持卡用户不返券,返券档期:' || p_Tpid);
            /*o_Retcode := Rtn_Failure;
            o_Retmsg  := '非持卡用户不返券,返券档期:' || p_Tpid;
            Return;*/
          End If;
        End If;
      End If;
    End If;
    --券有效期
    o_Startdate := Row_Fqtitle.Tpiyqstartdate;
    o_Enddate   := Row_Fqtitle.Tpiyqenddate;
    /*----返券有效期
    SELECT substr(to_char(SYSDATE, 'yyyy-mm-dd HH24:mi:ss'), 12, 10)
      INTO vcurtime
      FROM dual;
    IF nvl(row_fqtitle.tpisoneday, 'N') <> 'N' THEN
      ----如果是当日有效的
      IF vcurtime < row_fqtitle.tptime THEN
        o_startdate := trunc(SYSDATE);
        o_enddate   := trunc(SYSDATE);
      ELSE
        --大于等于延期时间点的，有效期顺延到第二天
        o_startdate := trunc(SYSDATE);
        o_enddate   := trunc(SYSDATE);
        o_enddate   := o_enddate + 1; --延期一天
      END IF;
    END IF;*/
    o_Pj   := Row_Fqtitle.Tprulepj; --0不区分规则 1同规则用券
    o_Area := Row_Fqtitle.Tpcertarea; --0通用 1本门店 2指定门店
    --是否区分面值
    If Trim(p_Cardno) Is Null Then
      o_Ismz    := Row_Fqtitle.Tpmzmode;
      o_Prnsite := Row_Fqtitle.Tpprintplace;
    Else
      If Row_Fqtitle.Tphyfqmode = '0' Then
        --电子券模式总是用规则合并
        o_Ismz    := q_Gzhb;
        o_Prnsite := '2';
      Else
        o_Ismz    := Row_Fqtitle.Tpmzmode;
        o_Prnsite := Row_Fqtitle.Tpprintplace;
      End If;
    End If;
    --
    o_Retcode := Rtn_Success;
    o_Retmsg  := '成功';
  Exception
    When Others Then
      If Sqlcode <> Errcode Then
        Errhandle.Errlog();
      End If;
      o_Retcode := Nvl(b_Outret, Rtn_Failure);
      o_Retmsg  := 'gettktpopinfo失败!' || Nvl(b_Outmsg, Sqlerrm);
      Raise;
      /*o_Retcode := Rtn_Failure;
      o_Retmsg  := 'gettktpopinfo失败!' || Sqlcode || ',' || Sqlerrm;
      Raise_Application_Error(Errcode, 'gettktpopinfo失败!' || Sqlerrm);*/
  End;

  Procedure gettktsource(vjygs   In Varchar2,
                         vmkt    In Varchar2,
                         vsyjid  In Varchar2,
                         vinvno  In Varchar2,
                         vqtype  In Varchar2,
                         osource Out b_source) Is
    y_sumpay  Number;
    r_Sh      sellheadth%Rowtype;
    loldqtype Char(1);
  Begin
    --找退货原小票，原交易有对应券种，取券种来源，否则 取不区分券种找来源，
    --如果原交易没有券消费，取退货券来源
    osource := b_source();
    Begin
      Select *
        Into r_Sh
        From SELLHEADTH
       Where shsyjid = vsyjid
         And shinvno = vinvno
         And shmarket = vmkt
         And shmanaunit = vjygs;
    Exception
      When Others Then
        osource.Extend;
        osource(osource.last).qtype := vqtype;
        osource(osource.last).poprule := '0';
        osource(osource.last).qsource := '4';
        osource(osource.last).rate := 1;
        Return;
    End;
    Begin
      Select TCFLMONEY
        Into y_sumpay
        From tktcardfqlog
       Where tcfljygs = r_Sh.Shmanaunit
         And tcflmkt = r_Sh.Shmarket
         And tcflsyjid = r_sh.shchr1
         And tcflinvno = r_Sh.shchr2
         And tcflzy = 'O' --字母O 消费
         And tcfljetype = vqtype
         And rownum = 1;
      loldqtype := vqtype;
    Exception
      When Others Then
        Begin
          Select TCFLMONEY
            Into y_sumpay
            From tktcardfqlog
           Where tcfljygs = r_Sh.Shmanaunit
             And tcflmkt = r_Sh.Shmarket
             And tcflsyjid = r_sh.shchr1
             And tcflinvno = r_sh.shchr2
             And tcflzy = 'O' --字母O 消费
             And rownum = 1;
          loldqtype := '%';
        Exception
          When Others Then
            loldqtype := '#';
        End;
    End;
    If loldqtype = '#' Then
      osource.Extend;
      osource(osource.last).qtype := vqtype;
      osource(osource.last).poprule := '0';
      osource(osource.last).qsource := '4';
      osource(osource.last).rate := 1;
    Else
      Select NVL(Sum(TCFLMONEY), 0)
        Into y_sumpay
        From tktcardfqlog
       Where tcfljygs = r_Sh.Shmanaunit
         And tcflmkt = r_Sh.Shmarket
         And tcflsyjid = r_sh.shchr1
         And tcflinvno = r_sh.shchr2
         And tcflzy = 'O' --字母O 消费
         And tcfljetype Like vqtype;
      If y_sumpay = 0 Then
        osource.Extend;
        osource(osource.last).qtype := vqtype;
        osource(osource.last).poprule := '0';
        osource(osource.last).qsource := '4';
        osource(osource.last).rate := 1;
        Return;
      End If;
      For VR In (Select tcflsource,
                        Max(tcflpoprule) tcflpoprule,
                        Sum(tcflmoney) vje
                   From tktcardfqlog
                  Where tcfljygs = r_Sh.Shmanaunit
                    And tcflmkt = r_Sh.Shmarket
                    And tcflsyjid = r_sh.shchr1
                    And tcflinvno = r_sh.shchr2
                    And tcflzy = 'O'
                    And tcfljetype Like loldqtype
                  Group By tcflsource) Loop
        osource.Extend;
        osource(osource.last).qtype := vqtype;
        osource(osource.last).poprule := vr.tcflpoprule;
        osource(osource.last).qsource := vr.tcflsource;
        osource(osource.last).money := vr.vje;
        osource(osource.last).rate := round(vr.vje / y_sumpay, 4);
      End Loop;
    End If;
  End;

  Procedure Upttkt(i_r_Tcft   In Tktcardfqtotal%Rowtype,
                   i_Cardno   In Varchar2,
                   i_Billno   In Number,
                   i_Person   In Varchar2,
                   i_Syjid    In Varchar2,
                   i_Invno    In Varchar2,
                   i_Source   In Varchar2,
                   i_Uptins   In Varchar2,
                   i_Bankcode In Varchar2 Default Null, --银行代码 银行追送时需要
                   i_Sppayno  In Varchar2 Default Null, --追送付款卡号 银行追送时需要
                   i_Precal   In Char Default 'N',
                   i_Debug    In Char Default 'N',
                   i_zy       In Char Default '',
                   o_Retcode  Out Number,
                   o_Retmsg   Out Varchar2,
                   o_Fqlog    Out Row_Tktfqlog) Is
    l_r_Tcft         Tktcardfqtotal%Rowtype;
    l_r_Tcfl         Tktcardfqlog%Rowtype;
    l_Seq            Number;
    l_Ye             Number;
    l_Ins            Number := 0;
    l_Ismz           Char(1); --是否固定面值
    l_Topicstartdate Date;
    l_Topicenddate   Date;
    i                Number;
    l_type           char(1);
  Begin
    o_Fqlog := Row_Tktfqlog();
    o_Fqlog.Extend;
    If i_Debug = 'Y' Then
      Goto Upttktskip;
    End If;
    If i_Uptins Not In ('U', 'I') Then
      Raise_Error(Rtn_Failure, 'i_uptins:' || i_Uptins || '错误！');
      /*o_Retcode := Rtn_Failure;
      o_Retmsg  := 'i_uptins:' || i_Uptins || '错误！';
      Return;*/
    End If;
    l_r_Tcft := i_r_Tcft;
    If l_r_Tcft.Tcftfqje = 0 Then
      Raise_Error(Rtn_Failure, '券金额:' || l_r_Tcft.Tcftfqje || '为零！');
      /*      o_Retcode := Rtn_Failure;
      o_Retmsg  := '券金额:' || l_r_Tcft.Tcftfqje || '为零！';

      Return;*/
    End If;

    If i_Source <> 'b' Then
      Gettktpopinfo(l_r_Tcft.Tcftfstpid,
                    i_Cardno,
                    'ALL',
                    o_Retcode,
                    o_Retmsg,
                    --l_r_tcft.tcftstartdate, l_r_tcft.tcftenddate,
                    l_Topicstartdate,
                    l_Topicenddate,
                    l_r_Tcft.Tcftvipno,
                    l_type,
                    l_r_Tcft.Tcftrulepj,
                    l_r_Tcft.Tcftmkt,
                    l_Ismz,
                    l_r_Tcfl.Tcflchr3);

      If l_r_Tcft.Tcftstartdate Is Null Then
        l_r_Tcft.Tcftstartdate := l_Topicstartdate;
      End If;
      Debug.Debug_New('Gettktpopinfo',
                      o_Retcode || ',' || o_Retmsg || ',' ||
                      l_Topicstartdate || ',' || l_Topicenddate || ',' ||
                      l_r_Tcft.Tcftvipno || l_r_Tcft.Tcfttype || ',' ||
                      l_r_Tcft.Tcftrulepj || ',' || l_r_Tcft.Tcftmkt || ',' ||
                      l_Ismz || ',' || l_r_Tcfl.Tcflchr3 || ',' ||
                      Nvl(i_Source, '#'));
      If l_r_Tcft.Tcftenddate Is Null Then
        l_r_Tcft.Tcftenddate := l_Topicenddate;
      End If;
    Else
      l_r_Tcft.Tcftvipno  := Fgetticketno;
      l_r_Tcft.Tcfttype   := '2';
      l_r_Tcft.Tcftrulepj := '0';
      l_Ismz              := 'Y';
      l_r_Tcfl.Tcflchr3   := 'Y';
      l_r_Tcft.Tcftmkt    := '1';

    End If;
    l_r_Tcft.Tcftstatus  := 'Y';
    l_r_Tcft.Tcftflag    := 'Y';
    l_r_Tcft.Tcftcltpid  := l_r_Tcft.Tcftfstpid;
    l_r_Tcft.Tcftvipexno := l_r_Tcft.Tcftvipno;
    l_r_Tcft.Tcftsource  := i_Source;
    l_r_tcft.tcftismzq   := l_r_tcft.tcftismzq;
    l_r_tcft.tcftqmz     := l_r_tcft.tcftqmz;
    if l_r_tcft.tcftismzq = 'Y' then
      if i_Cardno is not null then
        l_r_tcft.tcftqno := tktfqline.Fgetfqno(l_r_Tcft.Tcftqtype);
      else
        l_r_tcft.tcftqno := l_r_Tcft.Tcftvipno;
      end if;
    end if;
    /*if l_r_tcft.tcftqmz = 'Y' then
      i_Uptins := 'I';
    end if;*/
    If i_Precal <> 'Y' Then
      If i_Uptins = 'U' Then
        Update Tktcardfqtotal
           Set Tcftfqje = Tcftfqje + l_r_Tcft.Tcftfqje
         Where Tcftvipno = l_r_Tcft.Tcftvipno
           And Tcftstartdate = l_r_Tcft.Tcftstartdate
           And Tcftenddate = l_r_Tcft.Tcftenddate
           And Tcftqtype = l_r_Tcft.Tcftqtype --券种A～Z
           And Tcftfstpid = l_r_Tcft.Tcftfstpid
              --AND tcftfqrule = l_r_tcft.tcftfqrule
              --AND tcftrulepj = l_r_tcft.tcftrulepj
           And Tcftstatus = l_r_Tcft.Tcftstatus
           And Tcfttype = l_r_Tcft.Tcfttype --电子券，手工券
           And Tcftmkt = l_r_Tcft.Tcftmkt --券使用范围，--0通用 1本门店 2指定门店
           And Tcftjygs = l_r_Tcft.Tcftjygs
           And Tcftfqmkt = l_r_Tcft.Tcftfqmkt
           And Tcftsource = l_r_Tcft.Tcftsource
           And Tcftflag = l_r_Tcft.Tcftflag
           and nvl(tcftismzq, 'N') = 'N'
           and tcfttype = '1'
           and Rownum = 1
        Returning Tcftseqno, Tcftfqje Into l_Seq, l_Ye;
        If Sql%Rowcount = 0 Or Sql%Rowcount Is Null Then
          l_Ins := 1;
        End If;
      end if;
      If i_Uptins = 'I' Or l_Ins = 1 Then
        Select Seq_Tktcardfqtotal.Nextval Into l_Seq From Dual;
        l_r_Tcft.Tcftseqno := l_Seq;
        l_Ye               := l_r_Tcft.Tcftfqje;
        l_r_tcft.tcftismzq := l_r_tcft.tcftismzq;
        l_r_tcft.tcftqmz   := l_r_tcft.tcftqmz;
        l_r_tcft.tcftqno   := l_r_tcft.tcftqno;
        Insert Into Tktcardfqtotal Values l_r_Tcft;
      End If;
    End If;
    l_r_Tcfl.Tcflsource := i_Source;
    If Trim(i_zy) Is Null Then
      If l_r_Tcfl.Tcflsource In ('1', '3', '9') Then
        l_r_Tcfl.Tcflzy := 'F';
      Elsif l_r_Tcfl.Tcflsource = '2' Then
        l_r_Tcfl.Tcflzy := 'm';
      Elsif l_r_Tcfl.Tcflsource = '4' Then
        l_r_Tcfl.Tcflzy := 'P';
      Elsif l_r_Tcfl.Tcflsource = 'b' Then
        l_r_Tcfl.Tcflzy := 'b';
      End If;
    Else
      l_r_Tcfl.Tcflzy := i_zy;
    End If;
    --插入日志
    i := 1;
    Select Seq_Tktcardfqlog.Nextval,
           l_r_Tcft.Tcftvipno,
           l_r_Tcft.Tcftstartdate,
           l_r_Tcft.Tcftenddate,
           Sysdate,
           l_r_Tcfl.Tcflzy,
           l_r_Tcft.Tcftfstpid,
           l_r_Tcft.Tcftfqrule,
           l_r_Tcft.Tcftfqje,
           l_Ye,
           i_Person,
           i_Syjid,
           i_Invno,
           '',
           i_Billno,
           l_r_Tcft.Tcftqtype,
           l_Seq,
           l_r_Tcft.Tcftfqmkt,
           l_r_Tcft.Tcftjygs,
           l_r_Tcfl.Tcflsource,
           l_r_Tcft.Tcftransno,
           l_r_Tcft.Tcfttype,
           i_Bankcode,
           i_Sppayno,
           l_r_Tcfl.Tcflchr3,
           0,
           l_r_tcft.tcftismzq,
           l_r_tcft.tcftqmz,
           l_r_tcft.tcftqno
      Into o_Fqlog(i).Tcflseqno,
           o_Fqlog(i).Tcflvipno,
           o_Fqlog(i).Tcflstartdate,
           o_Fqlog(i).Tcflenddate,
           o_Fqlog(i).Tcfldate,
           o_Fqlog(i).Tcflzy,
           o_Fqlog(i).Tcflpopid,
           o_Fqlog(i).Tcflpoprule,
           o_Fqlog(i).Tcflmoney,
           o_Fqlog(i).Tcflye,
           o_Fqlog(i).Tcfloper,
           o_Fqlog(i).Tcflsyjid,
           o_Fqlog(i).Tcflinvno,
           o_Fqlog(i).Tcflmemo,
           o_Fqlog(i).Tcflsyjtrace,
           o_Fqlog(i).Tcfljetype,
           o_Fqlog(i).Tcflvipseq,
           o_Fqlog(i).Tcflmkt,
           o_Fqlog(i).Tcfljygs,
           o_Fqlog(i).Tcflsource,
           o_Fqlog(i).Tcfltransno,
           o_Fqlog(i).Tcfltype,
           o_Fqlog(i).Tcflchr1,
           o_Fqlog(i).Tcflchr2,
           o_Fqlog(i).Tcflchr3,

  Procedure Tktseq(p_Htthno  In Varchar2,

    Pragma Autonomous_Transaction;
  Begin
    Savepoint Seq;
    Delete From Tktseqlist
     Where Htthno = p_Htthno
       And Crmmkt = p_Crmmkt;
    Insert Into Tktseqlist
      Select Distinct Tcflvipseq, p_Htthno, p_Crmmkt, Tcflvipno

  /*退货更新券金额*/
 Procedure Salethhctkt(Vbillno  In Number,
                       Vcrmmkt  In Varchar2,
                       Vsyjh    In Varchar2,
                       Vfphm    In Varchar2,
                       Vcardno  In Varchar2,
                       Vperson  In Varchar2,
                       Vqje     In Number,
                       Vqtype   In Varchar2,
                       Vupt     In Varchar2 Default '1', -- 1:upt 0:only calc
                       Vfqsa    In Varchar2 Default 'FQ',
                       Vcallsrc In Varchar2, --调用来源 sp_sendsaleok,
                       --salepopms.salehh, htth_getsalekhje, java_getsalekhje
                       p_Htthno In Number,
                       Vdjlb    In Char,
                       vismzq   in varchar2,
                       Oqye     Out Number,
                       Osign    Out Number,
                       o_Fqlog  Out Row_Tktfqlog) Is
   Cur_Tkt Cursorms;
   Row_Tkt Tktcardfqtotal%Rowtype;
   v_Btye  Number;
   Vcurqje Number;
   --log
   Vseqno Number;
   Vzy    Char(1);
   Vxfseq Number;
   Vsrc   Char(1);
   --l_cardflag    CHAR(2);
   --l_cardownerid VARCHAR2(20);
   l_Sign      Number;
   r_Lasttkt   Tktcardfqtotal%Rowtype;
   l_Bankno    Varchar2(20);
   l_Payno     Varchar2(20);
   Lcardno     Varchar2(20);
   i           Number;
   l_tktseq    Number;
   l_Overnight Date; --通宵营业时间
   l_ye        number;
   l_tmpqje    number;
 Begin
   Debug.Debug_New('Salethhctkt',
                   Vbillno || ',' || Vcrmmkt || ',' || Vsyjh || ',' ||
                   Vfphm || ',' || Vcardno || ',' || Vperson || ',' || Vqje || ',' ||
                   Vqtype || ',' || Vupt || ',' || Vfqsa || ',' || Vcallsrc || ',' ||
                   p_Htthno || ',' || Vdjlb || ',' || vismzq);
   l_Overnight := Fgetbusinessdate(Sysdate, Vcrmmkt);
   Vxfseq      := Vbillno;
   Oqye        := abs(Vqje);
   If Vfqsa = 'SA' Then
     Vsrc := '3';
   Elsif Vfqsa = 'MQ' Then
     --红冲买券2010-11-19增加
     Vsrc := '2';
   Else
     Vsrc := '1';
   End If;
   --
   If Vdjlb = '2' Then
     --红冲销售
     Vzy    := 'O'; --返券 -
     l_Sign := Fgettktzysign(Vzy) * -1;
   Elsif Vdjlb = 'W' Then
     --红冲买券2010-11-19增加
     Vzy    := 'm'; --前台买券
     l_Sign := Fgettktzysign(Vzy) * -1;
   Elsif Vdjlb In ('4', 'G') Then
     --退货
     Vzy := 'K'; --券扣回
     /*if vqje < 0 then
       l_Sign := 1;
     else
       */
     l_Sign := Fgettktzysign(Vzy);
     /*end if;*/
   Else
     --红冲退货
     Vzy    := 'K'; --券扣回
     l_Sign := Fgettktzysign(Vzy) * -1;
   End If;
   Osign := l_Sign;
   --后台退货 在客户端扣除TKTCARDFQTOTAL
   /*If Vcallsrc = 'htth_getsalekhje' And Vupt = '1' Then
     Goto Htth;
   End If;*/
   o_Fqlog := Row_Tktfqlog();
   o_Fqlog.Extend;
   i        := 1;
   l_tktseq := 0;
   /*If Vcardno Is Null Or Vdjlb In ('2', 'W') Then
     l_tktseq := 1;
   End If;*/
   If Vcardno Is Not Null Then
     Open Cur_Tkt For
       Select Tcftseqno,
              Tcftvipno,
              Tcftstartdate,
              Tcftenddate,
              Tcftfqje,
              Tcftfstpid,
              Tcftfqrule,
              Tcftjygs,
              Tcftsource,
              Tcfttype,
              nvl(tcftismzq, 'N') tcftismzq,
              nvl(tcftqmz, 0) tcftqmz,
              nvl(tcftqno, '0') tcftqno
         From Tktcardfqtotal T1 /*, Billmktdetail*/
        Where (Tcftvipno = Vcardno or nvl(tcftqno, '0') = vcardno)
          And Tcftqtype = Vqtype
          And ((Khhis = 'Y' and Tcftenddate >= l_Overnight and tcftfqje > 0) or
              (tcftstartdate <= l_Overnight and Tcftenddate >= l_Overnight and
              Tcftstatus = 'Y') or (vdjlb='2' and nvl(tcftismzq, 'N') = 'Y' and tcftfqje=0))
          And (Tcftmkt = '0' Or (Tcftmkt = '1' And Tcftfqmkt = Vcrmmkt) /*or
                            (Tcftmkt = '2' And Bmdbillid = 'RRR' And Bmdmkt = Vcrmmkt)*/
              )
             /*And Tcftfstpid = Bmdbillno(+)*/
          and ((vismzq = 'N' and tcfttype = '1' and
              nvl(tcftismzq, 'N') = 'N') or
              (vismzq = 'Y' and
              (tcfttype = '2' or
              (tcfttype = '1' and nvl(tcftismzq, 'N') = 'Y'))))
          And ((l_tktseq = 1 And
              Tcftseqno In (Select Seq
                                From Tktseqlist
                               Where Htthno = p_Htthno
                                 And crmmkt = Vcrmmkt)) Or l_tktseq = 0)
        Order By Tcftenddate, Tcftseqno;
   Else
     Open Cur_Tkt For
       Select *
         From (Select Tcftseqno,
                      Tcftvipno,
                      Tcftstartdate,
                      Tcftenddate,
                      Tcftfqje,
                      Tcftfstpid,
                      Tcftfqrule,
                      Tcftjygs,
                      Tcftsource,
                      Tcfttype,
                      nvl(tcftismzq, 'N') tcftismzq,
                      nvl(tcftqmz, 0) tcftqmz,
                      nvl(tcftqno, '0') tcftqno
                 From Tktcardfqtotal T1, Tktseqlist T2
                Where Tcftseqno = T2.Seq
                  And T2.Htthno = p_Htthno
                  And Tcftqtype = Vqtype
                  And (Khhis = 'Y' Or ( /*Tcftstartdate <= Trunc(Sysdate) And */
                       Tcftenddate >= l_Overnight))
                  And (((Khhis = 'Y' Or Tcftstatus = 'Y') And Tcftflag = 'Y' And
                      Nvl(Tcftfqje, 0) > 0) Or Vdjlb = '5')
                  and ((tcfttype = '2' and vismzq = 'Y') or
                      (tcfttype = '1' and nvl(tcftismzq, 'N') = vismzq))
                  And (Tcftmkt = '0' Or
                      (Tcftmkt = '1' And Tcftfqmkt = Vcrmmkt))
               Union All
               Select Tcftseqno,
                      Tcftvipno,
                      Tcftstartdate,
                      Tcftenddate,
                      Tcftfqje,
                      Tcftfstpid,
                      Tcftfqrule,
                      Tcftjygs,
                      Tcftsource,
                      Tcfttype,
                      nvl(tcftismzq, 'N') tcftismzq,
                      nvl(tcftqmz, 0) tcftqmz,
                      nvl(tcftqno, '0') tcftqno
                 From Tktcardfqtotal T1, Billmktdetail, Tktseqlist T2
                Where Tcftseqno = T2.Seq
                  And Tcftqtype = Vqtype
                  And T2.Htthno = p_Htthno
                  And (Khhis = 'Y' Or ( /*Tcftstartdate <= Trunc(Sysdate) And*/
                       Tcftenddate >= Trunc(Sysdate)))
                  And (((Khhis = 'Y' Or Tcftstatus = 'Y') And Tcftflag = 'Y' And
                      Nvl(Tcftfqje, 0) > 0) Or Vdjlb = '5')
                  and ((tcfttype = '2' and vismzq = 'Y') or
                      (tcfttype = '1' and nvl(tcftismzq, 'N') = vismzq))
                  And (Tcftmkt = '2' Or Tcftmkt Is Null)
                  And Tcftfstpid = Bmdbillno
                  And Bmdbillid = 'RRR'
                  And Bmdmkt = Vcrmmkt)
        Order By Tcftenddate, Tcftseqno;
   End If;
   --
   Loop
     Fetch Cur_Tkt
       Into Row_Tkt.Tcftseqno,
            Row_Tkt.Tcftvipno,
            Row_Tkt.Tcftstartdate,
            Row_Tkt.Tcftenddate,
            Row_Tkt.Tcftfqje,
            Row_Tkt.Tcftfstpid,
            Row_Tkt.Tcftfqrule,
            Row_Tkt.Tcftjygs,
            Row_Tkt.Tcftsource,
            Row_Tkt.Tcfttype,
            row_tkt.tcftismzq,
            row_tkt.tcftqmz,

  Procedure Sp_Findtkt(Vmktcode In Varchar2, /* 门店号（逗号分隔）     */
                       Vjygs    In Varchar2, /*经营公司，*/
                       Vtrack1  In Varchar2, /* 空      */
                       Vtrack2  In Varchar2, /* 卡磁道号、券号或'0000'   */
                       Vtrack3  In Varchar2, /* 原终端号,原小票号   */
                       --券类型  '**' 纸券
                       Vmemo   In Varchar2, /*            */
                       Vdjlb   In Varchar2, /*小票 类别       */
                       Vtermno In Varchar2 /* 终端号 退货扣回的时候termno传4，*/,
                       vpasswd in varchar2,/*密码*/
                       --
                       Ret     Out Number, /* 返回值         */
                       Rcode   Out Varchar2, /* 卡号            */
                       Rpasswd Out Varchar2, /* 密码            */
                       Rismz   Out Varchar2, /* 是否面值          */
                       Rissx   Out Varchar2, /* 是否赊帐          */
                       Rislq   Out Varchar2, /* 是否礼券        */
                       Rispw   Out Varchar2, /* 是否输入密码        */
                       Rye     Out Number, /* 返A券余额       */
                       Rhsxe   Out Number, /* 回收限额         */
                       Rsxye   Out Number, /* 赊销余额         */
                       Rfunc   Out Varchar2, /*券种,券名称，余额，汇率，券号，券开始日期，券结束日期，券面值|券种,券名称,余额,汇率，券号，券开始日期，券结束日期，券面值|券种,券名称,余额,汇率，券号，券开始日期，券结束日期，券面值*/
                       Rvalue1 Out Number, /* 面值           */
                       Rvalue2 Out Number, /* 返B券余额        */
                       Rvalue3 Out Number, /* 返利券余额       */
                       Rmsg    Out Varchar2 /*错误信息*/) As
    Vcardno      Varchar2(20);
    Vye          Number;
    Ls_Jygs      Varchar2(20);
    Ls_Mkt       Varchar2(20);
    l_Seq        Number;
    l_Xssj       Char(5);
    l_Exchange   Number;
    l_Dyq        Char(1) := 'N';
    l_Djreason   Varchar2(10); --冻结原因
    l_Cardstatus Char(1);
    l_Cardflag   Varchar(2);
    Khhis        Char(1) := Nvl(Fgetmanapara('Q6'), 'N'); -- 是否允许过期券扣回
    l_Overnight  Date; --通宵营业时间
    /*退货时查找原消费日志的中的券变值倍数 */
    l_Ysyjid Varchar2(10); --退货查询券时 原销售收银机号
    l_Yinvno Varchar2(20); ---退货查询券时 原销售小票
    --l_iscard VARCHAR2(20);
  Begin
    Vcardno := ftracktocard(Vtrack2,vpasswd);
    If Vcardno Is Null Then
      --查找是否手工券
      If Vtrack2 = '0000' Then
        Vcardno := '0000';
        l_Dyq   := 'Y';
      Else
        Begin
          Select 1
            Into l_Seq
            From Tktcardfqtotal
           Where (Tcftvipno = Vtrack2 or tcftqno = Vtrack2)
             And Rownum = 1;
          l_Dyq   := 'Y';
          Vcardno := Vtrack2;
        Exception
          When Others Then
            Ret   := 0;
            Rislq := 'Y';
            Vye   := 0;
            Return;
        End;
      End If;
    Else
      Custinterface.Chkcard(Vcardno, Vdjlb, 'Q');
    End If;
    Ls_Jygs := Vjygs;
    Ls_Mkt  := Vmktcode;
    Ls_Mkt  := Fgetcrmmkt(Ls_Jygs, Ls_Mkt);
    Rfunc   := '';
    --销售时间
    Select To_Char(Sysdate, 'HH24:MI') Into l_Xssj From Dual;
    l_Overnight := Fgetbusinessdate(Sysdate, vmktcode);
    l_Ysyjid    := Fgetpara(',' || Vtrack3 || ',', ',', 1);
    l_Yinvno    := Fgetpara(',' || Vtrack3 || ',', ',', 2);
    Debug.Debug_New('SP_FINDTKT',
                    'Vcardno->' || Vcardno || ',vdjlb->' || vdjlb ||
                    ',Ls_Mkt->' || Ls_Mkt || ',Vtermno->' || Vtermno ||
                    ',l_Overnight->' || to_char(l_Overnight));
    If Vcardno <> '0000' Then
      If vdjlb <> '4' Then
        For Vr In (Select Tcftqtype,
                          Tcfttype,
                          Nvl(Tqname, Tcftqtype) Qname,
                          Round(Sum(Qje), 2) Qje,
                          '0' qno,
                          to_char(sysdate, 'yyyy-mm-dd') qksrq,
                          '2099-12-31' qjsrq,
                          0 qmz
                     From (Select Tcftqtype, Tcfttype, Nvl(Tcftfqje, 0) Qje
                             From Tktcardfqtotal t
                            Where (t.Tcftvipno = Vcardno or
                                  t.tcftqno = Vcardno)
                              And t.Tcftstatus = 'Y'
                              And t.Tcftflag = 'Y'
                              And ((t.Tcftstartdate <= l_Overnight And
                                  t.Tcftenddate >= l_Overnight) Or
                                  (Vtermno = '4' And Khhis = 'Y'))
                              and nvl(t.tcftismzq, 'N') = 'N'
                              And (Tcftmkt = '0' Or
                                  (Tcftmkt = '1' And Tcftfqmkt = Ls_Mkt)) --通用或者本门店
                           Union All
                           Select Tcftqtype, Tcfttype, Nvl(Tcftfqje, 0) Qje
                             From Tktcardfqtotal, Billmktdetail
                            Where (Tcftvipno = Vcardno or tcftqno = Vcardno)
                              And Tcftstatus = 'Y'
                              And Tcftflag = 'Y'
                              And ((Tcftstartdate <= l_Overnight And
                                  Tcftenddate >= l_Overnight) Or
                                  (Vtermno = '4' And Khhis = 'Y'))
                              And (Tcftmkt = '2' Or Tcftmkt Is Null) --指定门店使用
                              And Tcftfstpid = Bmdbillno
                              And Bmdbillid = 'RRR'
                              And Bmdmkt = Ls_Mkt
                           Union All
                           --VIP返利券
                           Select 'a', '1', nvl(Sum(flqtfqje), 0) qje
                             From flqtotal
                            Where flqtvipno = vcardno
                              And flqtstartdate <= trunc(Sysdate)
                              And flqtenddate >= trunc(Sysdate)
                              And flqtstatus = 'Y'
                              And flqtflag = 'Y'
                              And (flqtmkt = '0' Or
                                  (flqtmkt = '1' And flqtfqmkt = ls_mkt))
                              And l_dyq = 'N'
                           Union All
                           --订金券
                           Select TCFTQTYPE,
                                  Tcfttype,
                                  nvl(Sum(tcftfqje), 0) qje
                             From Tktcardfqtotal
                            Where (Tcftvipno = Vcardno or tcftqno = Vcardno)
                              And Tcftstatus = 'Y'
                              And Tcftflag = 'Y'
                              And tcftqtype = 'b'
                            Group By TCFTQTYPE, Tcfttype) Ta,
                          Vw_Qtype
                    Where Tcftqtype = Tqcode(+)
                    Group By Tcftqtype, Tcfttype, Nvl(Tqname, Tcftqtype)
                   union all
                   Select Tcftqtype,
                          Tcfttype,
                          Nvl(Tqname, Tcftqtype) Qname,
                          Nvl(Tcftfqje, 0) Qje,
                          tcftqno,
                          to_char(tcftstartdate, 'yyyy-mm-dd'),
                          to_char(tcftenddate, 'yyyy-mm-dd'),
                          nvl(tcftqmz, 0) qmz
                     From Tktcardfqtotal t, Vw_Qtype
                    Where (t.Tcftvipno = Vcardno or t.tcftqno = Vcardno)
                      And t.Tcftstatus = 'Y'
                      And t.Tcftflag = 'Y'
                      and nvl(t.tcftismzq, 'N') = 'Y'
                      And ((t.Tcftstartdate <= l_Overnight And
                          t.Tcftenddate >= l_Overnight) Or
                          (Vtermno = '4' And Khhis = 'Y'))
                      and nvl(t.tcftismzq, 'N') = 'Y'

  Procedure Sp_WL_Findtkt(Vmktcode In Varchar2, /* 门店号     */
                          Vjygs    In Varchar2, /*经营公司，*/
                          Vtrack1  In Varchar2, /* 磁卡一磁道信息      */
                          Vtrack2  In Varchar2, /* 磁卡二磁道信息     */
                          Vtrack3  In Varchar2, /* 原终端号，原小票号 */
                          --券类型  '**' 纸券
                          Vmemo   In Varchar2, /*            */
                          Vdjlb   In Varchar2, /*小票类别       */
                          Vtermno In Varchar2 /* 退货扣回的时候termno传4，交易类型传销售*/,
                          vpasswd in varchar2,/*密码*/
                          --
                          Ret     Out Number, /* 返回值         */
                          Rcode   Out Varchar2, /* 卡号            */
                          Rpasswd Out Varchar2, /* 密码            */
                          Rismz   Out Varchar2, /* 是否面值          */
                          Rissx   Out Varchar2, /* 是否赊帐          */
                          Rislq   Out Varchar2, /* 是否礼券        */
                          Rispw   Out Varchar2, /* 是否输入密码        */
                          Rye     Out Number, /* 返A券余额       */
                          Rhsxe   Out Number, /* 回收限额         */
                          Rsxye   Out Number, /* 赊销余额         */

                          RMoney  Out Number, /* 面值           */
                          Rmsg    Out Varchar2 /*错误信息*/) As

    Vye          Number;
    Ls_Jygs      Varchar2(20);
    Ls_Mkt       Varchar2(20);
    Ls_ErpMkt    Varchar2(20);
    l_Seq        Number;
    l_Xssj       Char(5);
    l_Exchange   Number;
    l_Dyq        Char(1) := 'N';
    l_Djreason   Varchar2(10); --冻结原因
    l_Cardstatus Char(1);
    l_Cardflag   Varchar(2);
    Khhis        Char(1) := Nvl(Fgetmanapara('Q6'), 'N'); -- 是否允许过期券扣回
    l_Overnight  Date; --通宵营业时间
    l_Flq        Varchar2(100); --返利券余额描述
    /*退货时查找原消费日志的中的券变值倍数 */
    l_Ysyjid Varchar2(10); --退货查询券时 原销售收银机号
    l_Yinvno Varchar2(20); ---退货查询券时 原销售小票
    --l_iscard VARCHAR2(20);
  Begin
    Vcardno := ftracktocard(Vtrack2,vpasswd);
    If Vcardno Is Null Then
      --查找是否手工券
      If Vtrack2 = '0000' Then
        Vcardno := '0000';
        l_Dyq   := 'Y';
      Else
        Begin
          Select 1
            Into l_Seq
            From Tktcardfqtotal
           Where Tcftvipno = Vtrack2
             And Tcfttype = '2'
             And Rownum = 1;
          l_Dyq   := 'Y';
          Vcardno := Vtrack2;
        Exception
          When Others Then
            Ret   := 0;
            Rislq := 'Y';
            Vye   := 0;
            Return;
        End;
      End If;
    Else
      Custinterface.Chkcard(Vcardno, Vdjlb, 'Q');
    End If;
    Ls_Jygs   := Vjygs;
    Ls_ErpMkt := Vmktcode;
    Ls_Mkt    := Fgetcrmmkt(Ls_Jygs, Ls_ErpMkt);
    Rfunc     := '';
    --销售时间
    Select To_Char(Sysdate, 'HH24:MI') Into l_Xssj From Dual;
    l_Overnight := Fgetbusinessdate(Sysdate, vmktcode);
    l_Ysyjid    := Fgetpara(',' || Vtrack3 || ',', ',', 1);
    l_Yinvno    := Fgetpara(',' || Vtrack3 || ',', ',', 2);
    If Vcardno <> '0000' Then
      For Vr In (Select Tcftqtype,
                        Tcfttype,
                        Nvl(Tqname, Tcftqtype) Qname,
                        Round(Sum(Qje), 2) Qje,
                        Tcftenddate Edate
                   From (Select Tcftqtype,
                                Tcfttype,
                                Nvl(Tcftfqje, 0) Qje,
                                Tcftenddate
                           From Tktcardfqtotal t
                          Where t.Tcftvipno = Vcardno
                            And t.Tcftstatus = 'Y'
                            And t.Tcftflag = 'Y'
                            And t.Tcftstartdate <= l_Overnight
                            And t.Tcftenddate >= l_Overnight
                            And (Tcftmkt = '0' Or
                                (Tcftmkt = '1' And Tcftfqmkt = Ls_Mkt)) --通用或者本门店
                         Union All
                         Select Tcftqtype,
                                Tcfttype,
                                Nvl(Tcftfqje, 0) Qje,
                                Tcftenddate
                           From Tktcardfqtotal, Billmktdetail
                          Where Tcftvipno = Vcardno
                            And Tcftstatus = 'Y'
                            And Tcftflag = 'Y'
                            And Tcftstartdate <= l_Overnight
                            And Tcftenddate >= l_Overnight
                            And (Tcftmkt = '2' Or Tcftmkt Is Null) --指定门店使用
                            And Tcftfstpid = Bmdbillno
                            And Bmdbillid = 'RRR'
                            And Bmdmkt = Ls_Mkt
                         Union All
                         --VIP返利券
                         Select 'a',
                                '1',
                                Nvl(Sum(Flqtfqje), 0) Qje,
                                Flqtenddate
                           From Flqtotal
                          Where Flqtvipno = Vcardno
                            And Flqtstartdate <= Trunc(Sysdate)
                            And Flqtenddate >= Trunc(Sysdate)
                            And Flqtstatus = 'Y'
                            And Flqtflag = 'Y'
                            And (Flqtmkt = '0' Or
                                (Flqtmkt = '1' And Flqtfqmkt = Ls_Mkt))
                            And l_Dyq = 'N'
                          Group By Flqtenddate
                         Union All
                         --订金券
                         Select Tcftqtype,
                                Tcfttype,
                                Nvl(Sum(Tcftfqje), 0) Qje,
                                Tcftenddate
                           From Tktcardfqtotal
                          Where Tcftvipno = Vcardno
                            And Tcftstatus = 'Y'
                            And Tcftflag = 'Y'
                            And Tcftqtype = 'b'
                          Group By Tcftqtype, Tcfttype, Tcftenddate) Ta,
                        Vw_Qtype
                  Where Tcftqtype = Tqcode(+)
                  Group By Tcftqtype,
                           Tcfttype,
                           Nvl(Tqname, Tcftqtype),
                           Tcftenddate
                  Order By Tcftqtype) Loop
        l_Exchange := Fgettktbzbs(Ls_Jygs,
                                  Ls_Mkt,
                                  Vr.Tcftqtype,
                                  Vdjlb,
                                  l_Ysyjid,
                                  l_Yinvno);
        vye        := nvl(vye, 0) + Vr.Qje * l_Exchange;
        --券种,券名称,余额,汇率,电子券1/手工券2,有效期|券种,券名称,余额,汇率,电子券1/手工券2,有效期
        Rfunc := Rfunc || Vr.Tcftqtype || ',' || Vr.Qname || ',' ||
                 To_Char(Vr.Qje) || ',' || To_Char(l_Exchange) || ',' ||
                 Vr.Tcfttype || ',' || to_char(Vr.Edate, 'yyyy-mm-dd') || '|';
      End Loop;
      If l_Dyq = 'Y' And Trim(Rfunc) Is Null Then
        vye := 0;
        Begin
          Select Tcftfqje,
                 Tcftqtype || ',' || Nvl(Tqname, Tcftqtype) || ',' ||
                 To_Char(Tcftfqje) || ',1,2,' ||
                 To_Char(Tcftenddate, 'yyyy-mm-dd') || '|'
            Into vye, Rfunc
            From Tktcardfqtotal, Vw_Qtype
           Where Tcftqtype = Tqcode(+)
             And Tcftvipno = Vtrack2
             And Tcfttype = '2'
             And Rownum = 1;
        Exception
          When Others Then
            Rfunc := '';
        End;
      End If;
    Else
      For Vr In (Select * From Tktqtype Where Tqstatus = '1' Order By Tqcode) Loop
        Rfunc := Rfunc || Vr.Tqcode || ',' || Vr.Tqname || ',' || '0' || ',' || '1' || ',' || '2' ||
                 to_char(Sysdate, 'yyyy-mm-dd') || '|';
      End Loop;
    End If;
    --去掉最后面的字符|
    If Length(Rfunc) > 0 Then
      Rfunc := Substr(Rfunc, 1, Length(Rfunc) - 1);
    End If;
    Rcode   := Vcardno;
    Rpasswd := '';
    Rismz   := 'N';
    Rissx   := 'N';
    Rislq   := 'Y';
    Rispw   := 'N';
    Rhsxe   := 0;
    Rsxye   := 0;
    Rye     := Vye;
    RMoney  := 0;
    Ret     := 1;
  Exception
    When No_Data_Found Then
      Ret   := 0;
      Rislq := 'Y';
    When Others Then
      Rislq := 'Y';
      Ret   := 0;
      Rmsg  := Sqlerrm;
      Sp_Opererrors('SP_FINDTKT',
                    '0000',
                    'SP_FINDTKT:' || Vcardno || ',' || Sqlerrm);
  End;

  Procedure salethyk(vseqno    In Number, /* 原log交易流水号       */
                     Vmktcode  In Varchar2, /* 门店号（逗号分隔）  */
                     Vjygs     In Varchar2, /*经营公司，*/
                     Vsyjh     In Varchar2, /* 收银机号(4位)      */
                     Vfphm     In Number, /* 小票号          */
                     Vsyyh     In Varchar2, /* 收银员号          */
                     Vxfseq    In Number, /* 当前小票billno       */
                     Vdjlb     In Char,
                     Ret       Out Number, /* 返回值         */
                     Retmsg    Out Varchar2,
                     o_Fqlog   Out Row_Tktfqlog,
                     Is_Commit In Char Default 'Y') As
    Vzy     char(1) := 'L';
    Vnseqno number;
    i       number;
    row_log Tktcardfqlog%rowtype;
    row_tot tktcardfqtotal%rowtype;
  Begin
    o_Fqlog := Row_Tktfqlog();
    o_Fqlog.Extend;
    i := 1;
    select * into row_log from Tktcardfqlog where tcflseqno = vseqno;
    select *
      into row_tot
      from tktcardfqtotal
     where tcftseqno = row_log.tcflvipseq;
    Select Seq_Tktcardfqlog.Nextval Into Vnseqno From Dual;
    Select Vnseqno,
           row_log.tcflvipseq,
           row_log.tcflvipno,
           row_log.Tcflstartdate,
           row_log.Tcflenddate,
           Vzy,
           row_log.tcflpopid,
           row_log.tcflpoprule,
           Sysdate,
           -1 * row_log.tcflmoney,
           --帐户余额增加
           row_tot.tcftfqje - row_log.tcflmoney,
           Vsyyh,
           to_char(sysdate, 'yyyy-mm-dd'),
           Vsyjh,
           Vfphm,
           Vxfseq,
           row_log.tcfljetype,
           Vmktcode,
           Vjygs,
           row_log.tcflsource,
           row_log.Tcfltype,
           nvl(row_log.Tcflnum3,0),
           row_log.tcflismzq,
           row_log.tcflqmz,
           row_log.tcflqno
      Into o_Fqlog(i).Tcflseqno,
           o_Fqlog(i).Tcflvipseq,
           o_Fqlog(i).Tcflvipno,
           o_Fqlog(i).Tcflstartdate,
           o_Fqlog(i).Tcflenddate,
           o_Fqlog(i).Tcflzy,
           o_Fqlog(i).Tcflpopid,
           o_Fqlog(i).Tcflpoprule,
           o_Fqlog(i).Tcfldate,
           o_Fqlog(i).Tcflmoney,
           o_Fqlog(i).Tcflye,
           o_Fqlog(i).Tcfloper,
           o_Fqlog(i).Tcflmemo,
           o_Fqlog(i).Tcflsyjid,
           o_Fqlog(i).Tcflinvno,
           o_Fqlog(i).Tcflsyjtrace,
           o_Fqlog(i).Tcfljetype,
           o_Fqlog(i).Tcflmkt,
           o_Fqlog(i).Tcfljygs,
           o_Fqlog(i).Tcflsource,
           o_Fqlog(i).Tcfltype,
           o_Fqlog(i).Tcflnum3,
           o_fqlog(i).tcflismzq,
           o_fqlog(i).tcflqmz,
           o_fqlog(i).tcflqno
      From Dual;
    Insert Into Tktcardfqlog Values o_Fqlog (i);
    If Is_Commit = 'Y' Then
      Commit;
    End If;
    Ret    := Rtn_Success;
    retmsg := '预扣成功';
  Exception
    When Others Then
      Errhandle.Errlog();
      Rollback;
      Ret    := Rtn_Failure;
      Retmsg := Sqlerrm;
  End;

              vkjje := vkjje - Vaye;
            Else
              Vjje  := Vkjje;
              vkjje := 0;
            End If;
            --修改余额
            if l_qmz = 0 then
              l_isqmz := 'N';
              Update Tktcardfqtotal
                 Set Tcftfqje   = Tcftfqje - Vjje,
                     Tcftstatus = Decode(Nvl(Tcftfqje, 0) - Vjje,
                                         0,
                                         'N',
                                         'Y')
               Where Tcftseqno = Vvipseqno;
              vye := Nvl(Vrow_Tkt.Tcftfqje, 0) - Vjje;
            else
              l_isqmz := 'Y';
              Update Tktcardfqtotal
                 Set Tcftfqje = 0, Tcftstatus = 'N'
               Where Tcftseqno = Vvipseqno;
              Vye := 0;
            end if;
          
            --如果使用的买券批次，记录分摊任务单
            If Vrow_Tkt.Tcftsource In ('2', '8') Then
              Dbusrpop.Sp_Genmqtask(Ls_Jygs,
                                    Vmktcode,
                                    Vsyjh,
                                    Vfphm,
                                    Vvipseqno,
                                    Vjje);
            End If;
            Select Seq_Tktcardfqlog.Nextval Into Vseqno From Dual;
            Select Vseqno,
                   Vvipseqno,
                   Vcardno,
                   Vrow_Tkt.Tcftstartdate,
                   Vrow_Tkt.Tcftenddate,
                   Vzy,
                   Vrow_Tkt.Tcftfstpid,
                   Vrow_Tkt.Tcftfqrule,
                   Sysdate,
                   -1 * Vjje,
                   --帐户余额减少
                   Vye,
                   Voper,
                   Vczrq,
                   Vsyjh,
                   Vfphm,
                   Vxfseq,
                   Vkjtype,
                   Ls_Mkt,
                   Ls_Jygs,
                   Vrow_Tkt.Tcftsource,
                   Vrow_Tkt.Tcfttype,
                   l_Bzbs,
                   '0',
                   l_isqmz,
                   l_qmz,
                   l_qno
              Into o_Fqlog(i).Tcflseqno,
                   o_Fqlog(i).Tcflvipseq,
                   o_Fqlog(i).Tcflvipno,
                   o_Fqlog(i).Tcflstartdate,
                   o_Fqlog(i).Tcflenddate,
                   o_Fqlog(i).Tcflzy,
                   o_Fqlog(i).Tcflpopid,
                   o_Fqlog(i).Tcflpoprule,
                   o_Fqlog(i).Tcfldate,
                   o_Fqlog(i).Tcflmoney,
                   o_Fqlog(i).Tcflye,
                   o_Fqlog(i).Tcfloper,
                   o_Fqlog(i).Tcflmemo,
                   o_Fqlog(i).Tcflsyjid,
                   o_Fqlog(i).Tcflinvno,
                   o_Fqlog(i).Tcflsyjtrace,
                   o_Fqlog(i).Tcfljetype,
                   o_Fqlog(i).Tcflmkt,
                   o_Fqlog(i).Tcfljygs,
                   o_Fqlog(i).Tcflsource,
                   o_Fqlog(i).Tcfltype,
                   o_Fqlog(i).Tcflnum3,
                   o_fqlog(i).tcflywdh,
                   o_Fqlog(i).tcflismzq,
                   o_fqlog(i).tcflqmz,
                   o_Fqlog(i).tcflqno
              From Dual;
            Insert Into Tktcardfqlog Values o_Fqlog (i);
            o_Fqlog.Extend;
            i := i + 1;
          Else
            Ret := Rtn_Failure;
            close cur_tkt_xs;
            Raise_Error(Ret,
                        '交易已经冲正,卡号为' || Vcardno || '返券类型:' ||
                        Nvl(Vkjtype, '') || '交易金额为' || To_Char(Vje));
          End If;
        End Loop;
        Close Cur_Tkt_xs;
        If Abs(Vkjje) > 0 and vdjlb = '1' Then
          Ret := Rtn_Failure;
          Raise_Error(Ret,
                      'VIP卡返券金额余额不足，交易失败！卡号为' || Vcardno || '返券类型:' ||
                      Nvl(Vkjtype, '') || '交易金额为' || To_Char(Vje));
        End If;
        /*else
        --退货券扣回
        null;
        end if;*/
      end if;
    else
      --销售冲正、退货冲正,检查冲正的数据是否一致
      If Vxf > 0 Then
        Select Nvl(Sum(Tcflmoney), 0)
          Into Vnum
          From Tktcardfqlog
         Where Tcflsyjid = Vsyjh
           And Tcflsyjtrace = Vxfseq
           And Tcflmemo = Vczrq
           And Tcflzy In ('O', 'P', 'n', 'L')
           And Tcflvipno = Vcardno;
        If Abs(Vnum) <> Abs(Vje) Then
          Ret := Rtn_Failure;
          Raise_Error(Ret,
                      '冲正金额与原来交易不一致，冲正失败！交易序号' || Vxfseq || ',卡号为' ||
                      Vcardno || '返券类型:' || Nvl(Vkjtype, '') || '交易金额为' ||
                      To_Char(Vje));
        End If;
      End If;
      --如果找到原交易且无冲正,则修改卡余额
      If Vxf > 0 And Vcz = 0 Then
        Open Cur_Tktlog;
        Loop
          Fetch Cur_Tktlog
            Into Vrow_Tktlog;
          Exit When Cur_Tktlog%Notfound Or Cur_Tktlog%Notfound Is Null;
          Vczrq := Vrow_Tktlog.Tcflmemo;
          Select Vrow_Tktlog.Tcflmoney * -1 Into Vkjje From Dual;
          Vkjtype   := Vrow_Tktlog.Tcfljetype;
          Vvipseqno := Vrow_Tktlog.Tcflvipseq;
          --修改券金额
          Select Tcftfqje
            Into Vye
            From Tktcardfqtotal
           Where Tcftseqno = Vvipseqno;
          Update Tktcardfqtotal
             Set Tcftfqje   = Tcftfqje + Vkjje,
                 Tcftstatus = Decode(Nvl(Tcftfqje, 0) + Vkjje, 0, 'N', 'Y')
           Where Tcftseqno = Vvipseqno;
          If Vrow_Tktlog.Tcflsource In ('2', '8') Then
            --Dbusrpop.Sp_Genmqtask(Nvl(Fgetpara(',' || Vmktcode || ',', ',', 1), ''), Nvl(Fgetpara(',' || Vmktcode || ',', ',', 2), ''), Vsyjh, Vfphm, Vrow_Tktlog.Tcflvipseq, Vkjje);
            Dbusrpop.Sp_Genmqtask(Nvl(Fgetpara(',' || Vmktcode || ',',
                                               ',',
                                               1),
                                      ''),
                                  Nvl(Fgetpara(',' || Vmktcode || ',',
                                               ',',
                                               2),
                                      ''),
                                  Vsyjh,
                                  Vfphm,
                                  Vrow_Tktlog.Tcflvipseq,
                                  -Vkjje);
          End If;
          --获得序列
          Select Seq_Tktcardfqlog.Nextval Into Vseqno From Dual;
          --摘要
          If Vrow_Tktlog.Tcflzy = 'O' Then
            Vzy := 'U'; --消费冲正
          Elsif Vrow_Tktlog.Tcflzy = 'P' Then
            Vzy := 'V'; --退货冲正
          Elsif Vrow_Tktlog.Tcflzy = 'n' Then
            Vzy := 'z'; --退买券冲正
          Elsif Vrow_Tktlog.Tcflzy = 'L' Then
            vZy := 'l'; --补现
          End If;
          Select Vseqno,
                 Vvipseqno,
                 Vcardno,
                 Vrow_Tktlog.Tcflstartdate,
                 Vrow_Tktlog.Tcflenddate,
                 Vzy,
                 Vrow_Tktlog.Tcflpopid,
                 Vrow_Tktlog.Tcflpoprule,
                 Sysdate,
                 Vkjje,
                 Vye + Vkjje,
                 Voper,
                 Vczrq,
                 Vsyjh,
                 Vfphm,
                 Vxfseq,
                 Vkjtype,
                 Ls_Mkt,
                 Ls_Jygs,
                 Vrow_Tktlog.Tcflsource,
                 Vrow_Tktlog.Tcfltype,
                 l_Bzbs
            Into o_Fqlog(i).Tcflseqno,
                 o_Fqlog(i).Tcflvipseq,
                 o_Fqlog(i).Tcflvipno,
                 o_Fqlog(i).Tcflstartdate,
                 o_Fqlog(i).Tcflenddate,
                 o_Fqlog(i).Tcflzy,
                 o_Fqlog(i).Tcflpopid,
                 o_Fqlog(i).Tcflpoprule,
                 o_Fqlog(i).Tcfldate,
                 o_Fqlog(i).Tcflmoney,
                 o_Fqlog(i).Tcflye,
                 o_Fqlog(i).Tcfloper,
                 o_Fqlog(i).Tcflmemo,
                 o_Fqlog(i).Tcflsyjid,
                 o_Fqlog(i).Tcflinvno,
                 o_Fqlog(i).Tcflsyjtrace,
                 o_Fqlog(i).Tcfljetype,
                 o_Fqlog(i).Tcflmkt,
                 o_Fqlog(i).Tcfljygs,
                 o_Fqlog(i).Tcflsource,
                 o_Fqlog(i).Tcfltype,
                 o_Fqlog(i).Tcflnum3
            From Dual;
          Insert Into Tktcardfqlog Values o_Fqlog (i);
          o_Fqlog.Extend;
          i   := i + 1;
          Ret := Rtn_Success;
        End Loop;
      Else
        Ret := Rtn_Success;
      end if;
    end if;
    If Is_Commit = 'Y' Then
      Commit;
    End If;
  Exception
    When Others Then
      Errhandle.Errlog();
      Rollback;
      Ret    := Rtn_Failure;
      Retmsg := Sqlerrm;
      Sp_Ora_Error(Vsyjh,
                   'SP_SENDTKT:' || To_Char(Vfphm, '0000000') || ',' ||
                   Vcardno || ',' || Sqlerrm);
  End;

  Procedure Sp_NewTkt(vJygs   In varchar2,
                      vMkt    In varchar2,
                      vSyjid  In Varchar2,
                      vInvno  In Varchar2,
                      vOper   In Varchar2,
                      vBillno In Number,
                      vQType  In Char,
                      vje     In Number) As
    l_Days     Number;
    l_r_Tcft   Tktcardfqtotal%Rowtype;
    l_tktfqlog Row_Tktfqlog;
    l_Retcode  Number;
    l_Retmsg   Varchar(2000);
    r_source   b_source;
    ltemp      Number;
    lsum       Number;
    l_Transno  Number;
  Begin
    l_r_Tcft := Null;
    Begin
      Select Tpiid
        Into l_r_Tcft.Tcftfstpid
        From Tktpopinfo, Billmktdetail
       Where Tpistartdate <= Trunc(Sysdate)
         And Tpienddate >= Trunc(Sysdate)
         And Tpiid = Bmdbillno
         And Bmdbillid = 'RRS'
         And Bmdmkt = vMkt;
    Exception
      When Others Then
        Select Nvl(Max(Tpiid), '0')
          Into l_r_Tcft.Tcftfstpid
          From Tktpopinfo, Billmktdetail
         Where Tpiid = Bmdbillno
           And Bmdbillid = 'RRS'
           And Bmdmkt = vMkt;
        --b_outret := rtn_noaction;
      --RETURN;
    End;
    l_Days              := To_Number(Fgetmanapara('Q1'));
    l_r_Tcft.Tcftfqrule := '0';
    l_r_Tcft.Tcftfqje   := Vje;
    l_r_Tcft.Tcftqtype  := vQType;
    l_r_Tcft.Tcftjygs   := vJygs;
    l_r_Tcft.Tcftfqmkt  := vMkt;
    Select Seq_Popms.Nextval Into l_Transno From Dual;
    l_r_Tcft.Tcftransno := l_Transno;
    If l_Days >= 0 Then
      l_r_Tcft.Tcftstartdate := Trunc(Sysdate);
      l_r_Tcft.Tcftenddate   := Trunc(Sysdate) + l_Days;
    Else
      l_r_Tcft.Tcftstartdate := Fgetqexpiredate(l_r_Tcft.Tcftfstpid,
                                                l_r_Tcft.Tcftqtype,
                                                '1');
      l_r_Tcft.Tcftenddate   := Fgetqexpiredate(l_r_Tcft.Tcftfstpid,
                                                l_r_Tcft.Tcftqtype,
                                                '2');
    End If;
    ----20101220更新
    gettktsource(vJygs, vMkt, vSyjid, vInvno, vQType, r_source);
    ltemp := 0;
    lsum  := 0;
    For j In 1 .. r_source.count Loop
      ltemp := round(Vje * r_source(j).rate, 2);
      lsum  := lsum + ltemp;
      If j = r_source.count Then
        ltemp := ltemp + (vje - lsum);
      End If;
      l_r_Tcft.Tcftfqje   := ltemp;
      l_r_Tcft.Tcftfqrule := r_source(j).poprule;
      Begin
        --Salepopms.Init;

        Upttkt(i_r_Tcft   => l_r_Tcft,
               i_Cardno   => '',
               i_Billno   => vBillno,
               i_Person   => vOper,
               i_Syjid    => vSyjid,
               i_Invno    => vInvno,
               i_Source   => r_source(j).qsource,
               i_Uptins   => 'I',
               i_BankCode => '',
               i_Sppayno  => '',
               i_Precal   => 'N',
               i_Debug    => 'N',
               i_zy       => 'P',
               o_Retcode  => l_Retcode,
               o_Retmsg   => l_Retmsg,
               o_Fqlog    => l_tktfqlog);
        If l_Retcode <> Rtn_Success Then
          Rollback;
          Raise_Error(l_Retcode, l_Retmsg);
        End If;
      Exception
        When Others Then
          Rollback;
          Raise_Error(Rtn_Failure, Sqlerrm);
      End;
    End Loop;
  End;

  Procedure Sp_WL_Sendtkt(Vtrace    In Number, /* 交易流水号(查询时为空)            */
                          Vtype     In Varchar2, /* 交易类型,'01'-消费,'02'-消费冲正,'03'-退货,'04'-退货冲正,'05'-查询 */
                          Vmktcode  In Varchar2, /*经营公司,门店*/
                          Vtermno   In Varchar2, /* 终端号   */
                          Vsyyh     In Varchar2, /* 收银员                     */
                          Vinvno    In Varchar2, /* 小票号                     */
                          Vamount   In Number, /* 金额 (查询时为空)           */
                          Vtrack1   In Varchar2, /* 磁道一                         */
                          Vtrack2   In Varchar2, /* 磁道二  消费时填电子券卡号或纸券券号，查询时填磁道号或纸券号，纸券退货时填'0000'   */
                          Vtrack3   In Varchar2, /* 原终端号，原小票号                    */
                          Vpasswd   In Varchar2, /* 密码 (未使用)                         */
                          Vloginseq In Varchar2, /* 签到时返回的签到序号*/
                          Vmemo     In Out Varchar2, /* 券种,,变值倍数 */
                          Ret       Out Number, /* 返回值:  0-执行失败，1-执行成功     */
                          RetCode   Out Varchar2, /* 返回代码 00-交易成功，21-流水号重复 */
                          Retmsg    Out Varchar2, /* 返回提示信息                         */
                          Rcardno   Out Varchar2, /* 卡号或纸券券号                        */
                          RAmount   Out Number, /* 流水号重复时返回新的流水号                */
                          curdate   Out Varchar2, /*(未使用)*/
                          curtime   Out Varchar2, /*(未使用)*/
                          Rcardname Out Varchar2, /* 持卡人 (未使用)                      */
                          Rispw     Out Varchar2, /* 是否输入密码 (未使用)                */
                          Rmoney    Out Number, /* 面值 返券卡没有面值                  */
                          Rmemo     Out Varchar2, /* 券信息，格式：券种,券名称,余额,汇率,券类型,有效期（YYYY-MM-DD）|券种,券名称,余额,汇率,券类型,有效期（YYYY-MM-DD）|... */
                          Rmemo1    Out Varchar2, /* 券信息（续），格式：券种,券名称,余额,汇率,券类型,有效期（YYYY-MM-DD）|券种,券名称,余额,汇率,券类型,有效期（YYYY-MM-DD）|...  */
                          Rmemo2    Out Varchar2 /* 保留未使用 */) As
    --返回值大于10 表示前台需要重置本地流水号
    Rismz       Char(1);
    Rissx       Char(1);
    Rislq       Char(1);
    Rhsxe       Number;
    Rsxye       Number;
    Vseqno      Number;
    Voper       Varchar2(15);
    Vaye        Number;
    Vye         Number;
    Vjje        Number;
    Vzy         Char(1);
    Vxf         Number;
    Vcz         Number;
    Vkjje       Number;
    Vkjtype     Char(1);
    Vnum        Number;
    Vczrq       Varchar2(10);
    Vcardno     Varchar2(20);
    Vvipseqno   Number;
    Ls_Jygs     Varchar2(20);
    l_Days      Number := 30;
    Ls_Mkt      Varchar2(20);
    Ls_Erpmkt   Varchar2(20);
    Vrow_Tkt    Tktcardfqtotal%Rowtype;
    r_Shth      Sellheadth%Rowtype;
    Lzxbl       Number;
    Khhis       Char(1) := Nvl(Fgetmanapara('Q6'), 'N');
    i           Number;
    l_Overnight Date; --通宵营业时间
    l_Fqlog     Row_Tktfqlog;
    Vctype      Char(1);
    Vdjlb       Char(1);
    Vqtype      Char(1);
    Cursor Cur_Tkt Is
      Select *
        From Tktcardfqtotal t
       Where Tcftvipno = Vcardno
         And ((Tcftstartdate <= l_Overnight And Tcftenddate >= l_Overnight) Or
             (Vdjlb In ('4', 'Y') And Khhis = 'Y'))
         And Tcftstatus = 'Y'
         And Tcftflag = 'Y'
         And Tcftqtype = Vkjtype
         And Nvl(Tcftfqje, 0) <> 0
         And ((Tcftmkt = '0' Or (Tcftmkt = '1' And Tcftfqmkt = Ls_Mkt)) Or
             ((Tcftmkt = '2' Or Tcftmkt Is Null) And Exists
              (Select 1
                  From Billmktdetail
                 Where Bmdbillno = t.Tcftfstpid
                   And Bmdbillid = 'RRR'
                   And Bmdmkt = Ls_Mkt)))
       Order By Tcftenddate;
    Vrow_Tktlog Tktcardfqlog%Rowtype;
    Cursor Cur_Tktlog Is
      Select *
        From Tktcardfqlog
       Where Tcflvipno = Vcardno
         And Tcflsyjid = Vtermno
         And Tcflsyjtrace = Vtrace
         And Tcflmemo = Vczrq
         And tcflmkt = Ls_Mkt
         And tcfljygs = ls_jygs
         And (Tcflzy In ('O', 'P', 'L', 'n'))
      --(Tcflzy = 'O' Or Tcflzy = 'P')
       Order By Tcflseqno;
    l_Exists Number;
    r_Tkt    Tktcardfqtotal%Rowtype;
    l_Qstart Date;
    l_Qend   Date;
    l_Source Char(1);
    l_Bzbs   Number; --变值倍数
    y_sumpay Number;
    ltemp    Number;
    lsum     Number;
    r_source b_source;
  Begin

    If Vtype = '01' Then
      Vctype := 'A'; --消费
      Vdjlb  := '1';
    Elsif Vtype = '02' Then
      Vctype := 'B'; --消费冲正
      Vdjlb  := '2';
    Elsif Vtype = '03' Then
      Vctype := 'A'; --退货
      Vdjlb  := '4';
    Elsif Vtype = '04' Then
      Vctype := 'B'; --退货冲正
      Vdjlb  := '5';
    Elsif Vtype = '05' Then
      If Trim(vtrack3) = '' Then
        Vdjlb := '1';
      Else
        Vdjlb := '4';
      End If;
    Else
      ret     := 1;
      Retcode := '02';
      Retmsg  := '交易类型错误';
      Return;
    End If;
    Ls_Jygs   := Nvl(Fgetpara(',' || Vmktcode || ',', ',', 1), '#');
    Ls_Erpmkt := Nvl(Fgetpara(',' || Vmktcode || ',', ',', 2), '#');
    Ls_Mkt    := Fgetcrmmkt(Ls_Jygs, Ls_Erpmkt);
    If Vtype = '05' Then
      Sp_WL_Findtkt(Ls_Erpmkt,
                    Ls_Jygs,
                    Vtrack1,
                    Vtrack2,
                    Vtrack3,
                    Vmemo,
                    Vdjlb,
                    Vtermno,
                    Vpasswd,
                    Ret,
                    Rcardno,
                    Rmemo1,
                    Rismz,
                    Rissx,
                    Rislq,
                    Rispw,
                    RAmount,
                    Rhsxe,
                    Rsxye,
                    Rmemo,
                    Rmoney,
                    Retmsg);
      If Ret = 1 Then
        Retcode := '00';
        Retmsg  := 'XX成功';
        If length(Rmemo) > 240 Then
          Rmemo1 := substr(substr(Rmemo, 241), 1, 240);
          Rmemo  := substr(Rmemo, 1, 240);
        End If;
        Debug.Debug_New('SP_WL_Sendfjk_End',
                        Vtype || '|' || Vtrace || '|' || Vtermno || '|' ||
                        Vinvno || '|' || Vsyyh || '|' || Vdjlb || '|' ||
                        Vamount || '|' || Vtrack2 || '|' || Vtrack3 || '|' || '|' ||
                        Vmemo || '|' || Retmsg);
      Elsif Ret <> 1 Then
        ret     := 1;
        Retcode := '04';
        Retmsg  := Nvl(Retmsg, '找不到该返券卡信息');
        Debug.Debug_New('SP_WL_Sendfjk_Err',
                        Vtype || '|' || Vtrace || '|' || Vtermno || '|' ||
                        Vinvno || '|' || Vsyyh || '|' || Vdjlb || '|' ||
                        Vamount || '|' || Vtrack2 || '|' || Vtrack3 || '|' || '|' ||
                        Vmemo || '|' || Retmsg);
      End If;
      Return;
    End If;

    Rcardno := Vtrack2;
    Vqtype  := Nvl(Fgetpara(',' || Vmemo || ',', ',', 1), '#');
    If Vqtype = '#' Then
      If Vctype = 'A' Then
        ret     := 1;
        Retcode := '05';
        Retmsg  := Nvl(Retmsg, '交易失败:未指定券种！');
      Else
        ret     := 1;
        Retcode := '06';
        Retmsg  := '冲正失败:未指定券种！';
      End If;
    End If;

    l_Days  := To_Number(Fgetmanapara('Q1'));
    l_Fqlog := Row_Tktfqlog();
    l_Fqlog.Extend;
    i := 1;
    --
    l_Bzbs := To_Number(Nvl(Fgetpara(',' || Vmemo || ',', ',', 3), '1'));
    If l_Bzbs = 0 Then
      l_Bzbs := 1;
    End If;
    Vkjje := Vamount;
    --记帐时直接传卡号
    If Vtrack1 = 'CARDNO' Then
      Vcardno := Vtrack2;
    Else
      Vcardno := ftracktocard(vtrack2,Vpasswd);
    End If;

    If Vmemo Is Null And Trim(Vkjtype) Is Null Then
      Vkjtype := '#';
    Else
      Vkjtype := Nvl(Fgetpara(',' || Vmemo || ',', ',', 1), ''); --substr(vfunc, 1, 1);
    End If;
    /*如果是返利券，执行返利券过程，并返回*/
    If Vkjtype = Fgetflqtype Then
      Return;
    End If;

    Vczrq       := To_Char(Trunc(Sysdate), 'YYYY-MM-DD');
    l_Overnight := Fgetbusinessdate(Sysdate, vmktcode);
    --获得相应的原消费和冲正记录

    Select Nvl(Sum(Decode(Tcflzy, 'O', 1, 'P', 1, 'n', 1, 'L', 1, 0)), 0),
           Nvl(Sum(Decode(Tcflzy, 'U', 1, 'V', 1, 'z', 1, 'l', 1, 0)), 0)
      Into Vxf, Vcz
      From Tktcardfqlog
     Where Tcflmkt = Ls_Mkt
       And Tcflsyjid = Vtermno
       And Tcflsyjtrace = Vtrace
       And Tcflmemo = Vczrq;
    If Vctype = 'A' Then
      --消费，退货
      If Vxf > 0 Then
        --检查流水号是否重复
        Ret     := 1;
        RetCode := '21';
        RAmount := Vtrace + 1000; --当天收银机重装后，序列号可能重复，返回值加1000重置收银机序列
        Retmsg  := '返券交易序号重复，交易失败！';
        Return;
      End If;
      --
      If Vkjtype <> 'b' Then
        Begin
          Select 1 Into l_Exists From Tktqtype Where Tqcode = Vkjtype;
        Exception
          When No_Data_Found Then
            Ret     := 1;
            RetCode := '31';
            RAmount := 0;
            Retmsg  := '券种[' || Nvl(Vkjtype, '') || ']未定义!';
            Return;
        End;
      End If;

      --如果是退货，增加一个新的券批次
      If Vkjje < 0 Then
        --退货或者是 红冲销售 ，传入金额 小于0
        ----20101220更新
        Vaye := Abs(Vkjje);

          Select Nvl(Max(Tcftseqno), 0)

             And Tcftstartdate <= l_Overnight
             And Tcftenddate >= l_Overnight

             And Tcftqtype = Vkjtype

        If Vvipseqno = 0 Then
          If l_Days >= 0 Then

            l_Qend   := Trunc(Sysdate) + l_Days; --退货券有效期 开始日期
          Else
            --IF l_days < 0 THEN 小于0则 使用原有效期
            Begin
              Select Tcftstartdate, Tcftenddate
                Into l_Qstart, l_Qend
                From Tktcardfqtotal
               Where Tcftvipno = Vcardno
                 And Tcftqtype = Vkjtype
                 And (Tcftmkt = '0' Or
                     (Tcftmkt = '1' And Tcftfqmkt = Ls_Mkt))
                 And Rownum = 1;
            Exception
              When No_Data_Found Then
                l_Qstart := To_Date('1900-01-01', 'yyyy-mm-dd');
                l_Qend   := To_Date('1900-01-01', 'yyyy-mm-dd');
            End;
          End If;

          gettktsource(ls_Jygs,
                       Ls_Erpmkt,
                       Vtermno,
                       Vinvno,
                       Vkjtype,
                       r_source);
          ltemp := 0;
          lsum  := 0;
          For j In 1 .. r_source.count Loop
            ltemp := round(Vaye * r_source(j).rate, 2);
            lsum  := lsum + ltemp;
            If j = r_source.count Then
              ltemp := ltemp + (Vaye - lsum);
            End If;
            If Vcardno = '0000' Then

            Select Seq_Tktcardfqtotal.Nextval

            r_Tkt.Tcftenddate   := l_Qend;
            r_Tkt.Tcftqtype     := Vkjtype;
            r_Tkt.Tcftfqje      := ltemp;
            r_Tkt.Tcftstatus    := 'Y';
            r_Tkt.Tcftvipexno   := r_Tkt.Tcftvipno;
            r_Tkt.Tcftmemo      := Null;
            r_Tkt.Tcftzpno      := Null;

            r_Tkt.Tcftransno    := 0;
            r_Tkt.Tcftqmz       := 0;
            r_Tkt.Tcftmkt       := '0';
            r_Tkt.Tcftjygs      := Ls_Jygs;
            r_Tkt.Tcftfqmkt     := Ls_Mkt;
            r_Tkt.Tcftfqrule    := r_source(j).poprule;
            r_Tkt.Tcftrulepj    := '0';
            r_Tkt.Tcftsource    := r_source(j).qsource;
            Insert Into Tktcardfqtotal Values r_Tkt;
            --券种,券名称,余额,汇率,电子券1/手工券2,有效期|券种,券名称,余额,汇率,电子券1/手工券2,有效期
            Select Rmemo || Vkjtype || ',' || Nvl(Tqname, Vkjtype) || ',' ||

              Into Rmemo
              From Vw_Qtype
             Where Tqcode = Vkjtype;
            If Vctype = 'A' Then
              Vzy := 'P'; --退货记券退货
            Else
              Vzy := 'O'; --红冲销售记负的券消费
            End If;
            Select Seq_Tktcardfqlog.Nextval Into Vseqno From Dual;
            Select Vseqno,
                   r_Tkt.Tcftseqno,
                   r_Tkt.Tcftvipno,
                   r_Tkt.Tcftstartdate,
                   r_Tkt.Tcftenddate,
                   Vzy,
                   r_Tkt.Tcftfstpid,
                   r_Tkt.Tcftfqrule,
                   Sysdate,
                   Vaye,
                   --帐户余额增加
                   r_Tkt.Tcftfqje,
                   vsyyh,
                   Vczrq,
                   Vtermno,
                   Vinvno,
                   Vtrace,
                   r_Tkt.Tcftqtype,
                   Ls_Mkt,
                   Ls_Jygs,
                   r_source(j).qsource,
                   '1',
                   l_Bzbs
              Into l_Fqlog(i).Tcflseqno,
                   l_Fqlog(i).Tcflvipseq,
                   l_Fqlog(i).Tcflvipno,
                   l_Fqlog(i).Tcflstartdate,
                   l_Fqlog(i).Tcflenddate,
                   l_Fqlog(i).Tcflzy,
                   l_Fqlog(i).Tcflpopid,
                   l_Fqlog(i).Tcflpoprule,
                   l_Fqlog(i).Tcfldate,
                   l_Fqlog(i).Tcflmoney,
                   l_Fqlog(i).Tcflye,
                   l_Fqlog(i).Tcfloper,
                   l_Fqlog(i).Tcflmemo,
                   l_Fqlog(i).Tcflsyjid,
                   l_Fqlog(i).Tcflinvno,

          Rcardno := r_Tkt.Tcftvipno;
          If Length(Rmemo) > 0 Then

        End If;
        If Vvipseqno > 0 Then
          Select *
            Into r_Tkt
            From Tktcardfqtotal
           Where Tcftseqno = Vvipseqno;
          If Vctype = 'A' Then
            Vzy := 'P'; --退货记券退货
          Else
            Vzy := 'O'; --红冲销售记负的券消费

                 Vvipseqno,

            Update Tktcardfqtotal

                   Tcftstatus = Decode(Nvl(Tcftfqje, 0) - Vjje,
                                       0,
                                       'N',
                                       decode(tcfttype, '2', 'N', 'Y'))
             Where Tcftseqno = Vvipseqno;
            /*--如果使用的买券批次，记录分摊任务单
            If Vrow_Tkt.Tcftsource In ('2', '8') Then
              Dbusrpop.Sp_Genmqtask(Ls_Jygs, Vmktcode, Vtermno, Vinvno, Vvipseqno, Vjje);

            Vye := Vaye - Vjje;
            --
            If Abs(Vjje) > 0 Then
              --获得序列
              Select Seq_Tktcardfqlog.Nextval Into Vseqno From Dual;
              --摘要
              If Vctype = 'A' Then
                -- 2010-11-26
                Vzy := 'O';

              Select Vseqno,
                     Vvipseqno,
                     Vcardno,
                     Vrow_Tkt.Tcftstartdate,
                     Vrow_Tkt.Tcftenddate,
                     Vzy,
                     Vrow_Tkt.Tcftfstpid,
                     Vrow_Tkt.Tcftfqrule,
                     Sysdate,
                     -1 * Vjje,
                     --帐户余额减少

                     Vczrq,

                     Vinvno,
                     Vtrace,

                     Ls_Mkt,

                     Vrow_Tkt.Tcftsource,
                     Vrow_Tkt.Tcfttype,
                     l_Bzbs

      If Vxf > 0 Then

           And Tcflmemo = Vczrq
           And

           And Tcflvipno = Vcardno;

                     To_Char(Vamount);

      --如果找到原交易且无冲正,则修改卡余额
      If Vxf > 0 And Vcz = 0 Then
        Open Cur_Tktlog;
        Loop
          Fetch Cur_Tktlog
            Into Vrow_Tktlog;
          Exit When Cur_Tktlog%Notfound Or Cur_Tktlog%Notfound Is Null;
          Vczrq := Vrow_Tktlog.Tcflmemo;
          Select Vrow_Tktlog.Tcflmoney * -1 Into Vkjje From Dual;
          Vkjtype   := Vrow_Tktlog.Tcfljetype;
          Vvipseqno := Vrow_Tktlog.Tcflvipseq;

          Select Tcftfqje

            From Tktcardfqtotal
           Where Tcftseqno = Vvipseqno;

             Set Tcftfqje   = Tcftfqje + Vkjje,
                 Tcftstatus = Decode(Nvl(Tcftfqje, 0) + Vkjje, 0, 'N', 'Y')
           Where Tcftseqno = Vvipseqno;
          /*If Vrow_Tktlog.Tcflsource In ('2', '8') Then
            --Dbusrpop.Sp_Genmqtask(Nvl(Fgetpara(',' || Vmktcode || ',', ',', 1), ''), Nvl(Fgetpara(',' || Vmktcode || ',', ',', 2), ''), Vsyjh, Vfphm, Vrow_Tktlog.Tcflvipseq, Vkjje);
            Dbusrpop.Sp_Genmqtask(Nvl(Fgetpara(',' || Vmktcode || ',', ',', 1), ''), Nvl(Fgetpara(',' || Vmktcode || ',', ',', 2), ''), Vtermno, Vinvno, Vrow_Tktlog.Tcflvipseq, -Vkjje);
          End If;*/
          --获得序列
          Select Seq_Tktcardfqlog.Nextval Into Vseqno From Dual;

          If Vrow_Tktlog.Tcflzy = 'O' Then
            Vzy := 'U'; --消费冲正
          Elsif Vrow_Tktlog.Tcflzy = 'P' Then
            Vzy := 'V'; --退货冲正
          Elsif Vrow_Tktlog.Tcflzy = 'n' Then

                 Vkjje,
                 Vye + Vkjje,
                 vsyyh,
                 Vczrq,
                 Vtermno,
                 Vinvno,
                 Vtrace,
                 Vkjtype,
                 Ls_Mkt,

          i := i + 1;
        End Loop;
      Else
        If vxf > 0 And vcz > 0 Then
          Ret     := 1;
          RetCode := '35';

          Return;
        Elsif vxf = 0 Then
          Ret     := 1;
          RetCode := '36';
          Retmsg  := '原交易不存在，冲正失败！';
          Return;
        End If;
      End If;
    End If;
    --Commit;
    Select nvl(Sum(tcftfqje), 0)
      Into RAmount
      From Tktcardfqtotal t
     Where Tcftvipno = Vcardno
       And ((Tcftstartdate <= l_Overnight And Tcftenddate >= l_Overnight) Or

       And Tcftstatus = 'Y'
       And Tcftflag = 'Y'

       And ((Tcftmkt = '0' Or (Tcftmkt = '1' And Tcftfqmkt = Ls_Mkt)) Or

            (Select 1
                From Billmktdetail
               Where Bmdbillno = t.Tcftfstpid
                 And Bmdbillid = 'RRR'
                 And Bmdmkt = Ls_Mkt)));
    ret     := 1;
    Retcode := '00';
    Retmsg  := '成功';

    /*Cardbase.Debug('SP_WL_Sendfjk_End',

/*  买券和退券不支持红冲
  \*买券和退券红冲根据原交易，全部冲回*\
    Procedure Sp_Mqhc
    (
      Vjygs     In Varchar2, --经营公司
      Vmkt      In Varchar2, --门店
      Voldsyjid In Varchar2, --旧收银机号
      Voldinvno In Varchar2, --旧小票号
      Voldzy    In Char, --旧的摘要
      Vnewsyjid In Varchar2, --新的收银机号
      Vnewinvno In Varchar2, --新的小票号
      Vnewzy    In Char, --新的摘要
      Vsource   In Varchar2, --券来源
      --Vsign      In Number, --符号，(券交易LOG金额*此符号)
      Voper      In Varchar2, --操作人
      Voptype    In Char Default 'A', --操作类型()
      o_Rowfqlog Out Row_Tktfqlog,
      o_Net      Out Number,
      o_Netmsg   Out Varchar2
    ) Is
      l_Can  Char(1);
      l_Sign Number;
    Begin
      l_Sign := Fgettktzysign(Voldzy);
      \*如果原来是加券，需要判断券金额是否够*\
      If l_Sign > 0 Then
        Begin
          Select 'Y'
          Into   l_Can
          From   (Select Tcflvipno, Tcfljetype, Sum(Tcflmoney) Tcflmoney
                   From   Tktcardfqlog
                   Where  Tcflzy = Voldzy And
                          Tcflinvno = Voldinvno And
                          Tcflsyjid = Voldsyjid And
                          Tcfljygs = Vjygs And
                          Tcflmkt = Vmkt And
                          Tcflsource = Vsource
                   Group  By Tcflvipno, Tcfljetype) Tktcardfqlog,
                 (Select Tcftvipno, Tcftqtype, Sum(Tcftfqje) Tcftfqje
                   From   Tktcardfqtotal
                   Where  Tcftstatus = 'Y' And
                          Tcftsource = Vsource
