CREATE OR REPLACE PACKAGE BODY JAVA_GETpopinfo
as
  PROCEDURE GETpopinfo
 (vmkt      in varchar2, 
  vmanaunit in varchar2,
   vbillno   in varchar2,
   rc        OUT RETCURSOR) as
begin
    OPEN rc FOR
        select ppiseq,
               ppibillno,
               ppimode,
               ppibarcode,
               ppimfid,
               ppicatid,
               ppippcode,
               ppispec,
               to_char(ppistartdate,'YYYY/MM/DD'),
               to_char(ppienddate,'YYYY/MM/DD'),
               ppistarttime,
               ppiendtime,
               ppizkfd,
               ppispace,
               ppinewsj,
               ppinewhyj,
               ppinewrate,
               ppinewhyrate,
               ppinewpfj,
               ppinewpfrate,
               ppihyzkfd,
               ppipfzkfd,
               pphdjlb,
               nvl(ppimaxnum,0),
               ppipresentcode,
               ppititle,
               nvl(ppimaxnum,0),
               ppipresentcode,
               ppipresentunit,
               nvl(ppipresentjs,0),
               nvl(ppipresentsl,0),
               nvl(ppipresentxl,0),
               nvl(ppipresentjg,0),
               pphjc,
               pphstr1,
               pphstr2,
               pphstr3,
               pphstr4,
               pphstr5,
               pphstr6,
               pphiszsz,
               pphisptgz,
               pphistjn
        from dbusrpop.pospopinfo_r
        where ppibillno= vbillno and pphmanaunit = vmanaunit and pphmarket = vmkt;
end GETpopinfo;


END JAVA_GETpopinfo;
