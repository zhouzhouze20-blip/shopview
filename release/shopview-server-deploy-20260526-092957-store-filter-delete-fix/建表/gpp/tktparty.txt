  Function Fgetfqno(Qtype In Varchar2) Return Varchar2 Is
    Ll_Qtype Number;
    Ll_Seqno Number;
    Ll_Rand  Number;
    Ls_Qno   Varchar2(20);
  Begin
    Select Ascii(Qtype),
           Seq_Fqtotal.Nextval,
           Trunc(Dbms_Random.Value(0, 10000))
      Into Ll_Qtype, Ll_Seqno, Ll_Rand
      From Dual;
    Ls_Qno := Trim(To_Char(Ll_Qtype)) ||
              Trim(To_Char(Ll_Seqno, '00000000')) ||
              Trim(To_Char(Ll_Rand, '0000'));
    Return Ls_Qno;
  End;

  Procedure tktsale(p_type   in varchar2, --类型01、销售02、冲正销售03、退货04、冲正退货
                    p_qno    in varchar2, --券号
                    p_cardno in varchar2 default '0', --会员卡号
                    p_seq    in varchar2 default '0', --交易流水号
                    p_qje    in number default 0, --券金额
                    p_qtype  in varchar2 default '0', --券种
                    p_ismzq  in varchar2 default 'N', --是否面值券
                    p_qmz    in number default 0, --券面值
                    p_source in varchar2, --来源第三方标识
                    p_oper   in varchar2, --操作人
                    p_mkt    in varchar2 default '0', --来源门店
                    p_yseq   in varchar2 default '0', --原交易流水号，用于冲正
                    ret      out number, --返回值1失败0成功
                    retmsg   out varchar2, --返回信息
                    retqno   out varchar2 --退货时返回新的面值券券号
                    ) is
    Row_Total Tktcardfqtotal%Rowtype;
    Row_Log   Tktcardfqlog%Rowtype;
    Cursor Cur_Tkt(v_qno    in varchar2,
                   v_cardno in varchar2,
                   v_qtype  in varchar2,
                   v_mkt    in varchar2,
                   v_ismzq  in varchar2) Is
      select *
        from tktcardfqtotal
       where (tcftqno = v_qno or
             (tcftvipno = v_cardno and tcftqtype = v_qtype))
         and tcftstartdate <= trunc(sysdate)
         and tcftenddate >= trunc(sysdate)
         and tcftstatus = 'Y'
         and tcftfqje > 0
         and (tcftmkt = '0' or (tcftmkt <> '0' and tcftfqmkt = v_mkt))
         and ((v_ismzq = 'N' and tcfttype = '1' and
             nvl(tcftismzq, 'N') = v_ismzq) or
             (v_ismzq = 'Y' and nvl(tcftismzq, 'N') = v_ismzq))
       order by tcftenddate, tcftseqno;
    Cursor cur_cz(v_qno   in varchar2,
                  v_syjid in varchar2,
                  v_invno in varchar2) Is
      select *
        from tktcardfqlog
       where ((nvl(v_qno, '0') <> '0' and tcflqno = v_qno) or
             (nvl(v_qno, '0') = '0' and tcflsyjid = v_syjid and
             tcflinvno = v_invno))
         and tcfldate >= trunc(sysdate)
         and tcfldate < trunc(sysdate) + 1
       order by tcflseqno desc;
    row_tkt  cur_tkt%rowtype;
    row_cz   cur_cz%rowtype;
    l_qje    number := 0;
    l_zy     char(1); --摘要
    l_money  number := 0; --发生券金额
    l_ye     number := 0; --余额
    l_exists number := 0;
    l_sign   number := 1;
  Begin
    ret    := 0;
    retmsg := '成功';
    if p_type = '01' or p_type = '02' or p_type = '03' or p_type = '04' then
      if (p_type = '01' or p_type = '03') and nvl(p_qno, '0') = '0' then
        if nvl(p_cardno, '0') = '0' or nvl(p_qtype, '0') = '0' then
          ret    := 1;
          retmsg := '消费退货券号为0时，会员卡号或券种不允许为空或0';
          return;
        end if;
      end if;
      if (p_type = '02' or p_type = '04') and nvl(p_qno, '0') = '0' and
         (nvl(p_seq, '0') = '0' or nvl(p_yseq, '0') = '0') then
        ret    := 1;
        retmsg := '冲正消费或冲正退货券号为0时，交易流水号或原交易流水号不允许为空或0';
        return;
      end if;
    
      l_qje := abs(p_qje);
      if p_type = '01' or p_type = '03' then
        if p_type = '01' then
          l_zy := 'O'; --销售
          --判断传入帐户中的余额是否足够扣减
          select count(*)
            into l_exists
            from tktcardfqtotal
           where (tcftqno = p_qno or
                 (tcftvipno = p_cardno and tcftqtype = p_qtype))
             and tcftstartdate <= trunc(sysdate)
             and tcftenddate >= trunc(sysdate)
             and tcftstatus = 'Y'
             and tcftfqje > 0
             and (tcftmkt = '0' or (tcftmkt <> '0' and tcftfqmkt = p_mkt))
             and ((p_ismzq = 'N' and tcfttype = '1' and
                 nvl(tcftismzq, 'N') = p_ismzq) or
                 (p_ismzq = 'Y' and nvl(tcftismzq, 'N') = p_ismzq))
           having sum(tcftfqje) >= l_qje;
          if l_exists > 0 then
            Open Cur_Tkt(p_qno, p_cardno, p_qtype, p_mkt, p_ismzq);
            Loop
              Fetch Cur_Tkt
                Into row_tkt;
              Exit When Cur_Tkt%Notfound Or Cur_Tkt%Notfound Is Null or l_qje <= 0;
              --减券金额
              if nvl(p_qno, '0') <> '0' then
                --根据券号消减面值券
                update tktcardfqtotal
                   set tcftfqje = 0, tcftstatus = 'N'
                 where tcftseqno = row_tkt.tcftseqno;
                l_money := -1 * l_qje;
                l_ye    := 0;
                l_qje   := 0;
              else
                --根据金额消减余额券
                if l_qje - row_tkt.tcftfqje <= 0 then
                  update tktcardfqtotal
                     set tcftfqje   = tcftfqje - l_qje,
                         tcftstatus = decode(tcftfqje - l_qje, 0, 'N', 'Y')
                   where tcftseqno = row_tkt.tcftseqno;
                  l_money := -1 * l_qje;
                  l_ye    := row_tkt.tcftfqje - l_qje;
                  l_qje   := 0;
                else
                  update tktcardfqtotal
                     set tcftfqje = 0, tcftstatus = 'N'
                   where tcftseqno = row_tkt.tcftseqno;
                  l_money := -1 * row_tkt.tcftfqje;
                  l_ye    := 0;
                  l_qje   := l_qje - row_tkt.tcftfqje;
                end if;
              end if;
              --记录日志
              Select Seq_Tktcardfqlog.Nextval
                Into Row_Log.Tcflseqno
                From Dual;
              Row_Log.Tcflvipno     := row_tkt.tcftvipno;
              row_log.Tcflvipseq    := row_tkt.tcftseqno;
              row_log.tcflstartdate := row_tkt.tcftstartdate;
              row_log.tcflenddate   := row_tkt.tcftenddate;
              row_log.tcflzy        := l_zy;
              row_log.tcfldate      := sysdate;
              row_log.tcflpopid     := row_tkt.tcftfstpid;
              row_log.tcflpoprule   := row_tkt.tcftfqrule;
              row_log.tcflmoney     := l_money;
              row_log.tcflye        := l_ye;
              row_log.tcfloper      := p_oper;
              row_log.tcflmemo      := '由第三方发起的消费';
              row_log.tcfljetype    := row_tkt.tcftqtype;
              row_log.tcflmkt       := p_mkt;
              row_log.tcfljygs      := '002';
              row_log.tcflsource    := '6';
              row_log.tcfltype      := row_tkt.tcfttype;
              row_log.tcflismzq     := row_tkt.tcftismzq;
              row_log.tcflqmz       := row_tkt.tcftqmz;
              row_log.tcflqno       := row_tkt.tcftqno;
              row_log.tcflsyjid     := p_source;
              row_log.tcflinvno     := p_seq;
              Inserttotlog(row_log);
            end loop;
            close cur_tkt;
          else
            ret    := 1;
            retmsg := '会员卡号或券号下的余额不足';
            return;
          end if;
        else
          l_zy     := 'P'; --退货
          l_exists := 0;
          if nvl(p_ismzq, 'N') = 'N' and nvl(p_qno, '0') = '0' then
            --是否面值券为N并且券号为0时则是余额券，判断这个会员卡上是否有这个券种，并且是有效的
            select count(*)
              into l_exists
              from tktcardfqtotal
             where ((tcftvipno = p_cardno and tcftqtype = p_qtype) or
                   nvl(tcftqno, '0') = '0')
               and tcftenddate >= trunc(sysdate)
               and nvl(tcftismzq, 'N') = 'N'
               and tcfttype = '1';
            if l_exists > 0 then
              select *
                into Row_Total
                from tktcardfqtotal
               where tcftseqno =
                     (select max(tcftseqno)
                        from tktcardfqtotal
                       where ((tcftvipno = p_cardno and tcftqtype = p_qtype) or
                             nvl(tcftqno, '0') = '0')
                         and tcftenddate >= trunc(sysdate)
                         and nvl(tcftismzq, 'N') = 'N'
                         and tcfttype = '1');
              l_money := l_qje;
              l_ye    := Row_Total.tcftfqje + l_qje;
              update tktcardfqtotal
                 set Tcftfqje   = l_ye,
                     tcftstatus = decode(l_ye, 0, 'N', 'Y')
               where tcftseqno = Row_Total.Tcftseqno;
            else
              l_money := l_qje;
              l_ye    := l_qje;
            end if;
          else
            select count(*)
              into l_exists
              from tktcardfqtotal
             where tcftqno = p_qno;
            if l_exists > 0 then
              select *
                into Row_Total
                from tktcardfqtotal
               where tcftqno = p_qno;
              Select Seq_Tktcardfqtotal.Nextval
                Into Row_Total.Tcftseqno
                From Dual;
              row_total.tcftstartdate := trunc(sysdate);
              row_total.tcftenddate   := trunc(sysdate) +
                                         to_number(fgetmanapara('Q1'));
              row_total.tcftfqje      := row_total.tcftqmz;
              row_total.tcftstatus    := 'Y';
              Row_Total.tcftqno := Fgetfqno(p_qtype);
              retqno:=Row_Total.tcftqno;
              Inserttotal(Row_Total);
            else
              l_money := l_qje;
              l_ye    := l_qje;
            end if;
          end if;
          if l_exists = 0 then
            Select Seq_Tktcardfqtotal.Nextval
              Into Row_Total.Tcftseqno
              From Dual;
            if nvl(p_ismzq, 'N') = 'N' and nvl(p_qno, '0') = '0' and
               nvl(p_cardno, '0') <> '0' then
              --余额券              
              Row_Total.Tcftvipno   := p_cardno;
              Row_Total.Tcftvipexno := p_cardno;
              Row_Total.Tcfttype    := '1';
              Row_Total.Tcftqno     := '0';
            else
              --面值券
              Row_Total.Tcftvipno   := nvl(p_cardno, '0');
              Row_Total.Tcftvipexno := nvl(p_cardno, '0');
              Row_Total.Tcfttype    := '1';
              Row_Total.Tcftqno     := Fgetfqno(p_qtype);
              retqno:=Row_Total.tcftqno;
            end if;
            Row_Total.Tcftqtype  := p_qtype;
            Row_Total.Tcftfqje   := l_ye;
            Row_Total.Tcftstatus := 'Y';
            Row_Total.Tcftmemo   := '由第三方发起的退货:' || p_source;
            Row_Total.Tcftfstpid := '0';
            Row_Total.Tcftflag   := 'Y';
            Row_Total.Tcftransno := 0;
            Row_Total.Tcftqmz    := p_qmz;
            Row_Total.Tcftjygs   := '002';
            if p_mkt is null or p_mkt = '0' then
              Row_Total.Tcftmkt   := '0';
              Row_Total.Tcftfqmkt := '0';
            else
              Row_Total.Tcftmkt   := '1';
              Row_Total.Tcftfqmkt := p_mkt;
            end if;
            Row_Total.Tcftsource := '6';
            Row_Total.Tcftfqrule := '0';
            Row_Total.Tcftrulepj := '0';
            Row_Total.Tcftismzq  := p_ismzq;
            Inserttotal(Row_Total);
          end if;
          --记录日志
          Select Seq_Tktcardfqlog.Nextval Into Row_Log.Tcflseqno From Dual;
          Row_Log.Tcflvipno     := Row_Total.tcftvipno;
          row_log.Tcflvipseq    := Row_Total.tcftseqno;
          row_log.tcflstartdate := Row_Total.tcftstartdate;
          row_log.tcflenddate   := Row_Total.tcftenddate;
          row_log.tcflzy        := l_zy;
          row_log.tcfldate      := sysdate;
          row_log.tcflpopid     := Row_Total.tcftfstpid;
          row_log.tcflpoprule   := Row_Total.tcftfqrule;
          row_log.tcflmoney     := l_money;
          row_log.tcflye        := l_ye;
          row_log.tcfloper      := p_oper;
          row_log.tcflmemo      := '由第三方发起的退货';
          row_log.tcfljetype    := row_tkt.tcftqtype;
          row_log.tcflmkt       := p_mkt;
          row_log.tcfljygs      := '002';
          row_log.tcflsource    := '6';
          row_log.tcfltype      := row_tkt.tcfttype;
          row_log.tcflismzq     := row_tkt.tcftismzq;
          row_log.tcflqmz       := row_tkt.tcftqmz;
          row_log.tcflqno       := row_tkt.tcftqno;
          row_log.tcflsyjid     := p_source;
          row_log.tcflinvno     := p_seq;
          Inserttotlog(row_log);
        end if;
      else
        --冲正销售/冲正退货,冲正只允许当天
        if p_type = '02' then
          l_zy   := 'U';
          l_sign := -1;
        else
          l_zy   := 'V';
          l_sign := 1;
        end if;
        --根据传入的原交易流水号找到原交易，并原样冲回      
        Open Cur_cz(p_qno, p_yseq, p_source);
        Loop
          Fetch Cur_cz
            Into row_cz;
          Exit When Cur_cz%Notfound Or Cur_cz%Notfound Is Null;
          select *
            into Row_Total
            from tktcardfqtotal
           where tcftseqno = row_cz.tcflvipseq;
          update tktcardfqtotal
             set tcftfqje   = tcftfqje + l_sign * row_cz.tcflmoney,
                 tcftstatus = decode(tcftfqje + l_sign * row_cz.tcflmoney,
                                     0,
                                     'N',
                                     'Y')
           where tcftseqno = row_cz.tcflvipseq;
          l_money := l_sign * row_cz.tcflmoney;
          l_ye    := Row_Total.tcftfqje + l_sign * row_cz.tcflmoney;
          --记录日志
          Select Seq_Tktcardfqlog.Nextval Into Row_Log.Tcflseqno From Dual;
          Row_Log.Tcflvipno     := Row_Total.tcftvipno;
          row_log.Tcflvipseq    := Row_Total.tcftseqno;
          row_log.tcflstartdate := Row_Total.tcftstartdate;
          row_log.tcflenddate   := Row_Total.tcftenddate;
          row_log.tcflzy        := l_zy;
          row_log.tcfldate      := sysdate;
          row_log.tcflpopid     := Row_Total.tcftfstpid;
          row_log.tcflpoprule   := Row_Total.tcftfqrule;
          row_log.tcflmoney     := l_money;
          row_log.tcflye        := l_ye;
          row_log.tcfloper      := p_oper;
          row_log.tcflmemo      := p_yseq;
          row_log.tcfljetype    := Row_Total.tcftqtype;
          row_log.tcflmkt       := p_mkt;
          row_log.tcfljygs      := '002';
          row_log.tcflsource    := '6';
          row_log.tcfltype      := row_tkt.tcfttype;
          row_log.tcflismzq     := row_tkt.tcftismzq;
          row_log.tcflqmz       := row_tkt.tcftqmz;
          row_log.tcflqno       := row_tkt.tcftqno;
          row_log.tcflsyjid     := p_source;
          row_log.tcflinvno     := p_seq;
          Inserttotlog(row_log);
        end loop;
        close cur_cz;
      end if;
    else
      ret    := 1;
      retmsg := '交易类型不存在';
      return;
    end if;
  Exception
    When Others Then
      ret    := 1;
      retmsg := '未知错误:' || Sqlerrm;
      return;
  End tktsale;

  Procedure tktsearch(p_cardno in varchar2, --会员卡号/券号
                      p_mkt    in varchar2 default '0', --门店
                      rc       Out RETCURSOR) is
  Begin
    open rc for
      select qtype, nvl(tqname, qtype) qname, qje, qksrq, qjsrq, qmz, qno
        from (select tcftqtype qtype,
                     sum(tcftfqje) qje,
                     '' qksrq,
                     '' qjsrq,
                     0 qmz,
                     '0' qno
                from tktcardfqtotal
               where (tcftvipno = p_cardno or tcftqno = p_cardno)
                 and tcftfqje > 0
                 and tcfttype = '1'
                 and nvl(tcftismzq, 'N') = 'N'
               group by tcftqtype
              union all
              select tcftqtype qtype,
                     tcftfqje qje,
                     to_char(tcftstartdate, 'yyyy-mm-dd') qksrq,
                     to_char(tcftenddate, 'yyyy-mm-dd') qjsrq,
                     tcftqmz,
                     tcftqno qno
                from tktcardfqtotal
               where (tcftvipno = p_cardno or tcftqno = p_cardno)
                 and tcftfqje > 0
                 and ((tcfttype = '1' and nvl(tcftismzq, 'N') = 'Y') or
                     tcfttype = '2')),
             vw_qtype
       where qtype = tqcode(+);
  Exception
    When Others Then
      open rc for
        select '0' qtype,
               '' qname,
               0 qje,
               '' qksrq,
               '' qjsrq,
               0 qmz,
               '0' qno
          from dual;
  End tktsearch;

  Procedure tktactivation(p_qno    in varchar2, --券号
                          p_cardno in varchar2 default '0', --会员卡号
                          p_oper   in varchar2, --操作人
                          p_source in varchar2, --来源
                          ret      out number, --返回值1失败0成功
                          retmsg   out varchar2, --返回信息
                          retqno   out number --返回新券号
                          ) is
    Row_List  Fqlinelist%Rowtype;
    Row_Total Tktcardfqtotal%Rowtype;
    Row_Log   Tktcardfqlog%Rowtype;
    l_num     number;
  Begin
    retqno := '0';
    --判断券号是否存在或是否已经激活了
    select count(*)
      into l_num
      from Fqlinelist
     Where Fqlqno = p_qno
       and fqlyqno is null;
    if l_num > 0 then
      Select * Into Row_List From Fqlinelist Where Fqlqno = p_qno;
      if Row_List.Fqlxqmode = '1' then
        --判断有效期是否已经过期
        if row_list.fqljsrq < trunc(sysdate) then
          ret    := 1;
          retmsg := '券号【' || p_qno || '】有效期已过，无法激活';
          return;
        end if;
        Row_Total.Tcftstartdate := Row_List.Fqlksrq;
        Row_Total.Tcftenddate   := Row_List.Fqljsrq;
      else
        Row_Total.Tcftstartdate := Trunc(Sysdate);
        Select Decode(Row_List.Fqlxqmode,
                      '2',
                      Trunc(Sysdate) + (Row_List.Fqldays - 1),
                      To_Date('2099-12-31', 'yyyy-mm-dd'))
          Into Row_Total.Tcftenddate
          From Dual;
      end if;
      Select Seq_Tktcardfqtotal.Nextval, Seq_Tktcardfqlog.Nextval
        Into Row_Total.Tcftseqno, Row_Log.Tcflseqno
        From Dual;
      if p_cardno is null or p_cardno = '0' then
        Row_Total.Tcftvipno   := '0';
        Row_Total.Tcftvipexno := '0';
        Row_Total.Tcfttype    := '2';
      else
        Row_Total.Tcftvipno   := p_cardno;
        Row_Total.Tcftvipexno := p_cardno;
        Row_Total.Tcfttype    := '1';
      end if;
      Row_Total.Tcftqtype  := row_list.fqlqtype;
      Row_Total.Tcftfqje   := row_list.fqlqje;
      Row_Total.Tcftstatus := 'Y';
      Row_Total.Tcftmemo   := '线上券激活:' || p_source;
      Row_Total.Tcftfstpid := '0';
      Row_Total.Tcftflag   := 'Y';
      Row_Total.Tcftransno := 0;
      Row_Total.Tcftqmz    := row_list.fqlqmz;
      Row_Total.Tcftjygs   := '002';
      if row_list.fqlmkt is null or row_list.fqlmkt = '0' then
        Row_Total.Tcftmkt   := '0';
        Row_Total.Tcftfqmkt := '0';
      else
        Row_Total.Tcftmkt   := '1';
        Row_Total.Tcftfqmkt := row_list.fqlmkt;
      end if;
      Row_Total.Tcftsource := '6';
      Row_Total.Tcftfqrule := '0';
      Row_Total.Tcftrulepj := '0';
      Row_Total.Tcftismzq  := row_list.fqlismzq;
      Row_Total.Tcftqno    := Fgetfqno(row_list.fqlqtype);
      Inserttotal(Row_Total);
    
      --记录日志
      Row_Log.Tcflvipno     := Row_Total.Tcftvipno;
      Row_Log.Tcflstartdate := Row_Total.Tcftstartdate;
      Row_Log.Tcflenddate   := Row_Total.Tcftenddate;
      Row_Log.Tcfldate      := Sysdate;
      Row_Log.Tcflzy        := 'X';
      Row_Log.Tcflpopid     := '0';
      Row_Log.Tcflpoprule   := '0';
      Row_Log.Tcflmoney     := Row_Total.Tcftfqje;
      Row_Log.Tcflye        := Row_Total.Tcftfqje;
      Row_Log.Tcfloper      := p_oper;
      Row_Log.Tcfljetype    := Row_Total.Tcftqtype;
      Row_Log.Tcflvipseq    := Row_Total.Tcftseqno;
      Row_Log.Tcflmkt       := Row_Total.Tcftfqmkt;
      Row_Log.Tcfljygs      := Row_Total.Tcftjygs;
      Row_Log.Tcflsource    := Row_Total.Tcftsource;
      Row_Log.Tcfltype      := Row_Total.tcfttype;
      Row_Log.Tcflywdh      := p_source;
      Row_Log.Tcflismzq     := Row_Total.Tcftismzq;
      Row_Log.Tcflqmz       := Row_Total.Tcftqmz;
      Row_Log.Tcflqno       := Row_Total.Tcftqno;
      Inserttotlog(Row_Log);
    
      Update Fqlinelist
         Set Fqlyqno = Row_Total.Tcftqno
       Where Fqlqno = p_qno
         And Fqlyqno Is Null;
      ret    := 0;
      retmsg := '券号【' || p_qno || '】已经生成新的券与【' || Row_Total.Tcftqno || '】';
      retqno := Row_Total.Tcftqno;
      return;
    else
      ret    := 1;
      retmsg := '券号【' || p_qno || '】不存在或已被激活';
      return;
    end if;
  Exception
    When Others Then
      ret    := 1;
      retmsg := '券号【' || p_qno || '】激活失败';
      return;
  End tktactivation;
