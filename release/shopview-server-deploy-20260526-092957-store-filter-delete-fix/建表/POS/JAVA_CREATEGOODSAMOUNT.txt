CREATE OR REPLACE PACKAGE BODY JAVA_CREATEGOODSAMOUNT AS
  -- 返回结果集为
  -- 商品编码,柜组,规格单位,批量售价,批量会员价,批量数量,开始日期,结束日期,是否折上折
  PROCEDURE CREATEGOODSAMOUNT(rc OUT RETCURSOR) AS
  BEGIN
    OPEN rc FOR
    --select pgagdid,pgamfid,pgauid,pgasj,pgahyj,pgamount,to_char(ggastadate,'YYYY/MM/DD'),to_char(ggaenddate,'YYYY/MM/DD'),pgiszsz from posgoodsamount where (ggaenddate >= trunc(sysdate) and pgamount = 0) or pgamount = 1;
      select tc.pgagdid,
             tc.pgamfid,
             tc.pgauid,
             tc.pgasj,
             tc.pgahyj,
             tc.pgamount,
             to_char(tc.ggastadate, 'YYYY/MM/DD'),
             to_char(tc.ggaenddate, 'YYYY/MM/DD'),
             tc.pgiszsz
        from (select max(tb.rowid) roworder,
                     tb.pgagdid,
                     tb.pgamfid,
                     tb.pgauid,
                     tb.pgasj
                from (select pgagdid,
                             pgamfid,
                             pgauid,
                             pgamount,
                             max(pgasj) pgasj

                        from posgoodsamount
                       where (ggaenddate >= trunc(sysdate) and pgamount = 0)
                          or pgamount = 1
                       group by pgagdid, pgamfid, pgauid, pgamount) ta,
                     posgoodsamount tb
               where ta.pgagdid = tb.pgagdid
                 and ta.pgamfid = tb.pgamfid
                 and ta.pgauid = tb.pgauid
                 and ta.pgamount = tb.pgamount
                 and ta.pgasj = tb.pgasj
                 and ((tb.ggaenddate >= trunc(sysdate) and tb.pgamount = 0)
                          or tb.pgamount = 1)
               group by tb.pgagdid,
                        tb.pgamfid,
                        tb.pgauid,
                        tb.pgasj,
                        tb.rowid) td,
             posgoodsamount tc
       where tc.rowid = td.roworder
         and ((tc.ggaenddate >= trunc(sysdate) and tc.pgamount = 0) or
             tc.pgamount = 1);
  END CREATEGOODSAMOUNT;

END JAVA_CREATEGOODSAMOUNT;
