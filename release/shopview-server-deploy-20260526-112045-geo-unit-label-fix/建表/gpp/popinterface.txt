create or replace package body popinterface is

  PROCEDURE sp_findzksarule
  (
    i_date   in date, /*日期*/
    jygs     IN VARCHAR2, /* 经营公司*/
    mkt      in varchar2, /*门店号*/
    code     IN VARCHAR2, /* 商品条码          */
    gz       IN VARCHAR2, /* 商品柜组          */
    catid    IN VARCHAR2, /* 商品品类          */
    pp       IN VARCHAR2, /* 商品品牌          */
    spsx     in varchar2, /* 商品属性码          */
    track1   IN VARCHAR2, /* 磁道1            */
    bankno   in varchar2,
/*    track2   IN VARCHAR2, \* 磁道2            *\
    track3   IN VARCHAR2, \* 磁道3            *\*/
    retcode  OUT NUMBER, /* 返回值          */
    retmsg   OUT VARCHAR2, /* 返回提示          */
    retzkl   out number, /*折扣率*/
    retseq   out varchar2 /*规则序号*/
  ) as
    l_gbid varchar2(30);
    l_seq  number;
  begin
    l_gbid := code;
    SELECT nvl(MAX(sriseqno), 0)
    INTO   l_seq
    FROM   zksaruleinfo
    WHERE  sriksrq <= i_date AND
           srijsrq >= i_date AND
           /*((sriyhkrule ='1' and  (substr(track1,sritrstart,sritrlenth) LIKE REPLACE(srino1, '*', '%') AND
           instr(srino1, '*') > 0) OR
           (srino1 <= substr(track1,sritrstart,sritrlenth) AND srino2 >= substr(track1,sritrstart,sritrlenth) AND
           instr(srino1, '*') = 0)) 
           or (sriyhkrule ='2' and  (substr(track2,sritrstart,sritrlenth) LIKE REPLACE(srino1, '*', '%') AND
           instr(srino1, '*') > 0) OR
           (srino1 <= substr(track2,sritrstart,sritrlenth) AND srino2 >= substr(track2,sritrstart,sritrlenth) AND
           instr(srino1, '*') = 0)) 
           or (sriyhkrule ='3' and  (substr(track3,sritrstart,sritrlenth) LIKE REPLACE(srino1, '*', '%') AND
           instr(srino1, '*') > 0) OR
           (srino1 <= substr(track3,sritrstart,sritrlenth) AND srino2 >= substr(track3,sritrstart,sritrlenth) AND
           instr(srino1, '*') = 0)) 
           )*/ 
           ( (track1 LIKE REPLACE(srino1, '*', '%') AND
           instr(srino1, '*') > 0) OR
           (srino1 <= track1 AND srino2 >= track1 AND
           instr(srino1, '*') = 0)) 
           and SRIBANKCODE = bankno and
           srijygs = jygs --经营公司
           AND
           srimkt = mkt AND
           ((srigdid = l_gbid AND (srimfid = gz OR srimfid = 'ALL') AND
           srimode = '1') OR ((srimfid = gz OR srimfid = 'ALL') AND
           (srippcode = pp OR srippcode = 'ALL') AND
           (sricatid = catid OR sricatid = 'ALL') AND
           (srispsx = spsx OR srispsx = '0' OR
           srispsx IS NULL) AND srimode = '2'));
    IF l_seq = 0 THEN
      retcode := 0;
      retmsg  := '没有找到规则';
      retzkl  := 0;
      retseq  := 0;
    else
      retseq  := l_seq;
      retcode := 0;
      retmsg  := '成功';
      select SRZKBL
      into   retzkl
      from   zksaruleinfo
      where  sriseqno = retseq;
    END IF;
  exception
    when others then
      retcode := 1;
      retmsg  := SQLERRM;
      retzkl  := 0;
      retseq  := 0;
  end;

begin
  null;
end popinterface;
