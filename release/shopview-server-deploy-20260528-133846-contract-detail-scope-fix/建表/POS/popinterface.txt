CREATE OR REPLACE PACKAGE BODY popinterface IS
   --退货扣回券折现比率
  FUNCTION fgetthkhzxl(vqtype IN VARCHAR2) RETURN NUMBER IS
    l_rate NUMBER;
  BEGIN
    SELECT nvl(tqzxbl, 1) INTO l_rate FROM tktqtype WHERE tqcode = vqtype;
    RETURN l_rate;
  EXCEPTION
    WHEN OTHERS THEN
      RETURN 1;
  END;
  

END;
