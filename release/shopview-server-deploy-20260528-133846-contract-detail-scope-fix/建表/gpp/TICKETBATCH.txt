  End;

  --初始化接口集
  --raise_application_error
  Procedure init(ex in out ex_type) is
    row_fq tktcardfqtotal%rowtype;
  Begin
    ib_isfouceseqno := FALSE;
    If isnull(ex.efbvipno) or isnull(ex.EFBOPERTYPE) or isnull(ex.EFBCARDZY)
       or isnull(ex.efbmkt) Then
       raise_application_error(errcode,ex.efbvipno||'接口参数必须完整！');
    End If;
    If ex.efbopertype not in (OT_ADD,OT_ADJ,OT_DEC) Then
       raise_application_error(errcode,ex.efbvipno||'不支持的处理类型！');
    End If;
    If isnull(ex.efbseqno) Then ex.efbseqno := 0; End If; --券批次
    --指定券批次
    If ex.efbseqno > 0 Then --指定券批次号 ,以券批次信息为准
       ib_isfouceseqno := TRUE;
       select * into row_fq from tktcardfqtotal  where tcftseqno = ex.efbseqno;
       If ex.efbvipno <> row_fq.tcftvipno Then
          raise_application_error(errcode,ex.efbvipno||'卡号与批次卡号不符！');
          
       End If;
       ex.efbstartdate := row_fq.tcftstartdate;
       If ex.efbopertype = OT_ADJ Then
          ex.efbfaqje     := nvl(row_fq.tcftfqje,0);
          --efbmemo 外部传入
       Else
           --efbfqje  外部传入
           ex.efbenddate := row_fq.tcftenddate;
           ex.efbmemo := row_fq.tcftmemo;
       End If; 
       ex.efbzpno := row_fq.tcftzpno;
       ex.efbtpid := row_fq.tcftfstpid;
       ex.efbtransno  := row_fq.tcftransno;
       ex.efbtype     := row_fq.tcfttype;
       ex.efbmz       := row_fq.tcfttype;
       ex.efbareatype := row_fq.tcftmkt;
       ex.efbjygs     := row_fq.tcftjygs;
       ex.efbfqmkt    := row_fq.tcftfqmkt;
       ex.efbfqrule   := row_fq.tcftfqrule;    
       ex.efbrulepj   := row_fq.tcftrulepj;
       ex.efbsource   := row_fq.tcftsource;
       --ex.efbcardzy 外部传入
       --efbmkt  外部传入
       --efboper 外部传入
       --efbsyjid  外部传入
       --efbinvno 外部传入
       --efbsyjtrace 外部传入
       --efblqchg 外部传入
    Else  --不指定券批次号，以接口信息为准
      null;
    End If;
    If isnull(ex.efbfaqje) Then ex.efbfaqje := 0; End If;
    If isnull(ex.efbfbqje) Then ex.efbfaqje := 0; End If;
    If isnull(ex.efbflqje) Then ex.efbfaqje := 0; End If;
    If ex.efbfaqje = 0 and ex.efbfbqje = 0 and ex.efbflqje = 0 and ex.efbopertype <> OT_ADJ Then
       raise_application_error(errcode,ex.efbvipno||'券金额为0！');
    End If;
      
  End;

      p_log.tcflmkt := p_ex.efbmkt; --发生门店号

      If p_ex.efbfaqje <> 0 Then

  procedure NewBatch(p_ex in out ex_type) is

  Procedure submit(p_row in out ex_interface) is
  ex  ex_type;
  Begin
     ex := p_row;
     trace('submit',ex);
     init(ex);       
     If ib_isfouceseqno = True  Then --指定批次增加
        If ex.efbopertype = OT_DEC Then
           DecreaseBatch(ex);
        ElsIf ex.efbopertype = OT_ADD THEN
           IncreaseBatch(ex);
        ElSIF ex.efbopertype = OT_ADJ THEN
           AdjBatch(ex);
        END IF;
        RETURN;
     End If;
     
     p_row := ex;
  Exception When Others Then
     Rollback;
     raise;     
  End;
