CREATE OR REPLACE PACKAGE BODY JAVA_CREATEGOODSPOP
AS
  -- 返回结果集为
  -- 序号,优惠单号,优惠类型,促销规则，促销模式,商品编码,柜组,多单位代码,商品品类,商品品牌,销售数量,电子秤商品的优惠的时间间隔(秒),开始日期,结束日期,开始时间,结束时间,优惠零售价,优惠会员价,优惠批发价,优惠零售价折扣率,优惠会员价折扣率,优惠批发价折扣率,优惠零售价折扣分担,优惠会员价折扣分担,优惠批发价折扣分担,备用，备用，备用，备用，备用
  PROCEDURE CREATEGOODSPOP
    (
      rc OUT RETCURSOR
    )
  AS
  BEGIN
    OPEN rc FOR
      select ppiseq,ppibillno,ppimode,ppipresentcode,ppipresentunit,ppibarcode,ppimfid,ppispec,ppicatid,ppippcode,PPIMAXNUM,ppispace,to_char(ppistartdate,'YYYY/MM/DD'),to_char(ppienddate,'YYYY/MM/DD'),ppistarttime,ppiendtime,ppinewsj,ppinewhyj,ppinewpfj,ppinewrate,ppinewhyrate,ppinewpfrate,ppizkfd,ppihyzkfd,ppipfzkfd,ppititle,ppichr1,PPISUPID ,0,0 from pospopinfo where ppienddate >= trunc(sysdate) or PPIPRESENTCODE = 'QFW';
  END CREATEGOODSPOP;

END JAVA_CREATEGOODSPOP;
