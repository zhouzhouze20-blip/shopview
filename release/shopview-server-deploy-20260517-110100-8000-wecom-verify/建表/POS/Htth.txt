  Function f_Getstringpara(Vstr     In Char,
                           Vflag    In Char,
                           Vn       In Number,
                           Vdefault In Char) Return Char As
    l_1    Number; --1
    l_2    Number; --8
    l_Loop Number;
    Lret   Varchar2(40);
    l_Str  Varchar2(255);
  Begin
    l_1    := 0;
    l_Loop := 1;
    l_Str  := Vflag || Vstr || Vflag;
    Loop
      l_1 := Instr(l_Str, Vflag, l_1 + 1);
      If l_1 = 0 Then
        Exit;
      End If;
      If l_Loop = Vn Then
        Exit;
      End If;
      l_Loop := l_Loop + 1;
    End Loop;
    l_2 := Instr(l_Str, Vflag, l_1 + 1);
    If l_1 = 0 Or l_2 = 0 Then
      Lret := '';
    Else
      Lret := Substr(l_Str, l_1 + 1, l_2 - l_1 - 1);
    End If;
    Return Nvl(Lret, Vdefault);
  End f_Getstringpara;

  Function f_Gethgstring(Vje In Number, Vstr In Char) Return Char As
    Ljf       Number;
    Lje       Number;
    Lkjf      Number;
    Vhgbillno Varchar2(20);
    Vhggdid   Varchar2(20);
    Vhgsl     Number;
    --lret      VARCHAR2(40);
    Lstr    Varchar2(255);
    Ljftype Varchar2(20);
  Begin
    Ljftype   := f_Getstringpara(Vstr, ',', 1, '');
    Ljf       := To_Number(f_Getstringpara(Vstr, ',', 2, ''));
    Lje       := To_Number(f_Getstringpara(Vstr, ',', 3, ''));
    Vhgbillno := f_Getstringpara(Vstr, ',', 4, '');
    Lkjf      := Round(Vje * (Ljf / Lje), 0);
    Lstr      := Ljftype || ',' || To_Char(Lkjf) || ',' || To_Char(Ljf) || ',' ||
                 To_Char(Lje);
    --积分换购
    If Vhgbillno Is Not Null Then
      Vhggdid := f_Getstringpara(Vstr, ',', 5, '');
      Vhgsl   := Round(Lkjf / Ljf, 2);
      Lstr    := Lstr || ',' || Vhgbillno || ',' || Vhggdid || ',' ||
                 To_Char(Vhgsl);
    End If;
    Return Lstr;
  End f_Gethgstring;

  Function f_Isdzq(p_Payerid In Varchar2) Return Char Is
  Begin
    If Substr(p_Payerid, 1, 1) Between 'A' And 'Z' Then
      Return 'Y';
    Elsif Substr(p_Payerid, 1, 1) = 'b' Then
      Return 'Y';
    Else
      Return 'N';
    End If;
  End f_Isdzq;

  Function Fgetoriginalinv_Hh(Vsyjid In Varchar2,
                              Vinvno In Varchar2,
                              Vtype  In Char) Return Varchar2 Is
    /*退货单号对应的原销售收银机和小票号*/
    Lsyjid  Char(4);
    Linvno  Number;
    Lhhflag Char(1);
    Lbillno Number;
    Ldjlb   Char(1);
    Lysyjid Char(4);
    Lyinvno Number;
    Lishh   Boolean;
  Begin
    Lsyjid := Vsyjid;
    Linvno := To_Number(Vinvno);
    Loop
      Select Shishh, Shbillno, Shdjlb
        Into Lhhflag, Lbillno, Ldjlb
        From Sellhead
       Where Shsyjid = Lsyjid
         And Shinvno = Linvno;
      If Lhhflag = 'Y' Then
        If Lsyjid = Vsyjid And Linvno = To_Number(Vinvno) Then
          Lishh := True;
        End If;
        Select Sdysyjid, Sdyinvno
          Into Lysyjid, Lyinvno
          From Selldetail
         Where Sdbillno = Lbillno
           And Rownum = 1;
      Else
        Lysyjid := Null;
        Lyinvno := Null;
      End If;
      If Lysyjid Is Null And (Lyinvno = 0 Or Lyinvno Is Null) Then
        Exit;
      Else
        Lsyjid := Lysyjid;
        Linvno := Lyinvno;
      End If;
    End Loop;
    If Vtype = '1' Then
      Return Lsyjid;
    Else
      Return To_Char(Linvno);
    End If;
  Exception
    When Others Then
      If Vtype = '1' Then
        Return Vsyjid;
      Else
        Return Vinvno;
      End If;
  End;

  Procedure Sp_Htthougtpay(Vsyjid In Varchar2,
                           Vzje   In Number,
                           vmkt   IN VARCHAR2,
                           Rysje  Out Number,
                           Rsysy  Out Number) Is
    Vsswrfs Char(1);
  Begin
    --收银截断方式
    Begin
      Select Syjrmode Into Vsswrfs From Syjmain Where SYJMKT=vmkt AND Syjid = Vsyjid;
    Exception
      When Others Then
        Vsswrfs := '1';
    End;
    If Vsswrfs = '0' Then
      --精确到分
      Rysje := Vzje;
      Rsysy := 0;
    Elsif Vsswrfs = '1' Then
      --四舍五入到角
      Rysje := Round(Vzje, 1);
      Rsysy := Round(Rysje - Vzje, 2);
    Elsif Vsswrfs = '2' Then
      ---截断到角
      Rysje := Trunc(Vzje, 1);
      Rsysy := Round(Rysje - Vzje, 2);
    Elsif Vsswrfs = '3' Then
      --四舍五入到元
      Rysje := Round(Vzje);
      Rsysy := Round(Rysje - Vzje, 2);
    Elsif Vsswrfs = '4' Then
      --截断到元
      Rysje := Trunc(Vzje);
      Rsysy := Round(Rysje - Vzje, 2);

    Elsif Vsswrfs = '5' Then
      --进位到角
      If Vzje > Trunc(Vzje, 1) Then
        Rysje := Trunc(Vzje, 1) + 0.1;
      Else
        Rysje := Vzje;
      End If;
      Rsysy := Round(Rysje - Vzje, 2);
    Elsif Vsswrfs = '6' Then
      --进位到元
      If Vzje > Trunc(Vzje) Then
        Rysje := Trunc(Vzje) + 1;
      Else
        Rysje := Vzje;
      End If;
      Rsysy := Round(Rysje - Vzje, 2);
    Else
      --截断到角
      Rysje := Trunc(Vzje, 1);
      Rsysy := Round(Rysje - Vzje, 2);
    End If;
  End;

                               Length(Vr.Sdjfrule));
            If Instr(l_Jfrule, ';', 1) > 0 Then
              l_Jfrule := Substr(l_Jfrule, 1, Instr(l_Jfrule, ';', 1) - 1);
            End If;
            l_Jfjs := To_Number(f_Getstringpara(l_Jfrule, '|', 4, Null));
            l_Jejs := To_Number(f_Getstringpara(l_Jfrule, '|', 5, Null));
            Select Round(l_Money * Decode(l_Jejs, 0, 0, (l_Jfjs / l_Jejs)),
                         2) d
              Into l_Jfmoney
              From Dual;

          Elsif Vr.Sppmcode = '0509' Then
            l_Payid := f_Gethgstring(l_Money, Vr.Sppayerid);
            --l_payid :=  vr.sppayerid;
          Else
            If Vr.Sppmcode = '04' Then
              Vr.Sppmcode := '0400';
            End If;
            l_Payid  := Vr.Sppayerid;
            l_Jftype := '@@';
          End If;
          If Vr.Ppmtype = '4' Or Vr.Spno = 'AUTO' Then
            l_Payno := '';
          Else
            l_Payno := Vr.Spno;
          End If;
          If Vr.Sppmcode = '0508' Then
            Update Htthsellpayment
               Set Spmoney   = Spmoney + l_Money, --round(l_money / vr.sperate, 1),
                   Spjfmoney = Spjfmoney + l_Jfmoney
             Where Spbillno = p_Newbillno
               And Sppmcode = Vr.Sppmcode
               And Sprowno > 0

                  /*Nvl(SpPayerId, '@@') = Nvl(l_payid, '@@')*/

             And Sprowno > 0
             And Sppmcode In
                 (Select Ppmcode From Pospaymode Where Ppmtype <> '5')
             And Rownum = 1;
        End If;
      End If;
      l_Qyy := Nvl(l_Qyy, 0) + l_Sqyy;
      --Out_Sysy := Out_Sysy + l_Sqyy;
    End Loop;
    If l_Qyy >= Out_Sysy Then
      Out_Sysy := l_Qyy;
    End If;

    Update Htthsellpayment

     Where Spbillno = p_Newbillno
       And Sppmcode = '0508';

    If (Out_Sysy <> 0 And Out_Sysy <> l_Qyy) Or

  Procedure Checkkh(p_Billno    In Number,
                    p_Upt       In Varchar2 Default '0',
                    p_Debug     In Varchar2 Default '0', --0:正常处理 1:断点1, 2断点2
                    o_Ret       Out Number,
                    o_Retmsg    Out Varchar2,
                    o_Jfkh      Out Number,
                    o_Jfkhdesc  Out Varchar2,
                    o_Jfmemo    Out Varchar2,
                    o_Qkh       Out Number,
                    o_Qkhdesc   Out Varchar2,
                    o_Fqmemo    Out Varchar2,
                    o_Descmemo  Out Varchar2,
                    o_Crmbillno Out Number,
                    o_Mjjfhk    Out Number) As
    Row_h Htthsellhead%Rowtype;
    -- 生成CRM小票信息,计算扣回
    l_Crm_Billno Number;
    l_Pos_Billno Number; --返券需要在POS计算扣回
    l_Tmp        Number;
    l_Qtmemo     Varchar2(100);
  Begin
    Begin
      Select * Into Row_h From Htthsellhead Where Shbillno = p_Billno;
    Exception
      When Others Then
        Raise_Application_Error(Errcode, p_Billno || '->无此退货小票！');
    End;
    Currentmkt := Row_h.Shmarket;
    /*If Row_h.Shsyjid = '0000' Then
      o_Ret       := 0;
      o_Retmsg    := '';
      o_Jfkh      := 0;
      o_Jfkhdesc  := '';
      o_Jfmemo    := '';
      o_Qkh       := 0;
      o_Qkhdesc   := '';
      o_Fqmemo    := '';
      o_Descmemo  := '';
      o_Crmbillno := '';
      Return;
    End If;*/
    --1生成CRM小票,2调用扣回过程
    Gencrmbill(Row_h, l_Crm_Billno, o_Ret, o_Retmsg);
    Genpossellbill(Row_h, l_Pos_Billno, o_Ret, o_Retmsg);
    If o_Ret <> 0 Then
      Raise_Application_Error(Errcode, p_Billno || '->' || o_Retmsg);
    End If;
    Update Htthsellhead
       Set Shcrmbillno = l_Crm_Billno, Shposbillno = l_Pos_Billno
     Where Shbillno = p_Billno;
    --
    If p_Debug = '1' Then
      If p_Upt = '0' Then
        Rollback;
      Else
        Delete From Sellhead_Crm Where Shbillno = l_Crm_Billno;
        Delete From Dbusrposnew.Salehead Where Billno = l_Pos_Billno;
        Delete From Sellmail Where Smkey = l_Pos_Billno;
        Delete From Sellhead Where shbillno=l_Pos_Billno;
        Commit;
      End If;
      Return;
    End If;
    Htth_Getsalekhje(l_Crm_Billno,
                     l_Pos_Billno,
                     p_Upt,
                     --1:upt 0:only calc
                     p_Billno,
                     --后台退货billno
                     --
                     o_Ret,
                     o_Retmsg,
                     o_Jfkh,
                     o_Jfkhdesc,
                     o_Jfmemo,
                     o_Qkh,
                     o_Qkhdesc,
                     o_Fqmemo,
                     l_Tmp,
                     o_Descmemo,
                     o_Mjjfhk);
    o_Crmbillno := l_Crm_Billno;
    If o_Ret <> 0 Then
      Rollback;
      Raise_Application_Error(-20009, o_Retmsg);
    End If;
    If p_Debug = '2' Then
      If p_Upt = '0' Then
        Rollback;
      Else
        Commit;
      End If;
      Return;
    End If;
    If p_Upt = '0' Then
      Rollback;
    Else
      --审核记录
      Insert Into Sellhead_Htth_Crm
        Select * From Sellhead_Crm Where Shbillno = l_Crm_Billno;
      Insert Into Selldetail_Htth_Crm
        Select * From Selldetail_Crm Where Sdbillno = l_Crm_Billno;
      Insert Into Sellpayment_Htth_Crm
        Select spbillno,sprowno,sppmcode,spmoney,sperate,spno,sppayerid,sppayyy,spispop From Sellpayment_Crm Where Spbillno = l_Crm_Billno;
      Insert Into Salehead_Htth
        Select * From Dbusrposnew.Salehead Where Billno = l_Pos_Billno;
      Update Salehead_Htth
         Set Crmbillno = p_Billno
       Where Billno = l_Pos_Billno;
      /*      Insert Into Selldetail_Htth
        Select * From Selldetail Where Sdbillno = l_Pos_Billno;
      Insert Into Sellpayment_Htth
        Select * From Sellpayment Where Spbillno = l_Pos_Billno;*/
    End If;
    --生成用于计算扣回的CRM小票要删除
    -- 退货完成 或者 取消时再删除
    --by zhouwd 只要是后台退货的小票,计算完后，都删除
    --If p_upt = '0' Then
    Delete From Sellhead_Crm Where Shbillno = l_Crm_Billno;
    --delete from crm_sellmail where shbillno = l_crm_billno;
    Delete From Dbusrposnew.Salehead Where Billno = l_Pos_Billno;
    Delete From Sellmail Where Smkey = l_Pos_Billno;
    Delete From Sellhead Where shbillno=l_Pos_Billno;
    --End If;
    --Commit;
  Exception
    When Others Then
      Raise;
  End;

    Select Max(Shbillno)

     Where Shnum2 = r_Ch.Htthno;
    Select Sum(Spmoney * Sperate * Nvl(Spnum1, 1))
      Into l_Totje
      From Htthsellpayment
     Where Spbillno = Row_h.Shbillno
       And Sprowno < 0
       And (Exists (Select * From Pospaymode Where Sppmcode = Ppmcode) Or

  -- 取消当日 已审核/未处理 的单据
  Procedure Daycancel(p_Date In Date) Is
  Begin
    For Vr In (Select Shbillno
                 From Htthsellhead
                Where Shstatus = 'Y'
                  And Shisth = 'N'
                  And Shdate >= p_Date
                  And Shdate < p_Date + 1) Loop
      Begin
        Cancel(Vr.Shbillno, 'system');
      Exception
        When Others Then
          Sp_Opererrors('后台退货取消', Sqlcode, Sqlerrm);
      End;
    End Loop;
    Commit;
  Exception
    When Others Then
      Null;
  End;
