CREATE OR REPLACE Package Body Sellimp Is

  Errcode Numeric(8) := -20099;

  Procedure Raiseerr(Msg In Varchar2) As
  Begin
    Rollback;
    Raise_Application_Error(Errcode, Msg);
  End;

  Function Fgetpaycode(V_Code In Varchar2) Return Varchar2 As
    L_Paycode Varchar2(10);
    L_Exists  Char(1);
  Begin
    Begin
      Select Npmcode
        Into L_Paycode
        From Paymodeconver_Bd
       Where Opmcode = V_Code;
    Exception
      When Others Then
        L_Paycode := V_Code;
    End;
    Begin
      Select 'Y' Into L_Exists From Paymode Where Pmcode = L_Paycode;
    Exception
      When Others Then
        Raiseerr('没有定义付款方式代码[' || V_Code || ']');
    End;
    Return L_Paycode;
  End;

  --获取付款方式是否支持溢余 zhangheng 2013-12-12
  Function Fgetpayyy(V_Paycode In Varchar2) Return Varchar2 As
    L_Yon Char(1);
  Begin
    Select Pmisyy Into L_Yon From Paymode Where Pmcode = V_Paycode;
    Return L_Yon;
  End;

  Function Fgetpaycodename(V_Code In Varchar2) Return Varchar2 As
    L_Paycode Varchar2(10);
    L_Name    Varchar2(30);
  Begin
    L_Paycode := Fgetpaycode(V_Code);
    If L_Paycode Is Not Null Then
      Select Pmname Into L_Name From Paymode Where Pmcode = L_Paycode;
    End If;
    Return L_Name;
  End;

  /*小票补登-一张小票*/
  Procedure Approve4one(V_Billno Varchar2,
                        V_Oper   Varchar2,
                        V_Billid Varchar2 Default 'XP1',
                        V_Djlb   Varchar2 Default '1') Is
    Rh         Pos_Sellhthead%Rowtype;
    L_Ysamount Number := 0;
    L_Ssamount Number := 0;
    L_Djlb     Varchar2(1) := '0';
    L_Djlbtmp  Varchar2(1) := '0';
    L_Billno   Number := 0;
    L_Rowno    Number := 0;
    L_Mkt      Varchar2(20);
    L_Yon      Char(1);
    L_Yy       Number;
  Begin
    Begin
      Select * Into Rh From Pos_Sellhthead Where Seqno = V_Billno;
    Exception
      When Others Then
        Raiseerr('单据[' || V_Billno || ']已不存在,请检查!');
    End;
    If Rh.Flag = 'Y' Then
      Raiseerr('单据[' || V_Billno || ']已审核,请检查!');
    End If;
    If Rh.Billtype <> 'XP1' Then
      Raiseerr('单据[' || V_Billno || ']单据类型错误!');
    End If;
    /*初始化单号和门店*/
    L_Billno := Improtsaleinterface.Getsalebillno;
    If L_Billno <= 0 Then
      Raiseerr('获取小票序号错误,请重新尝试!');
    End If;
    L_Mkt   := Rh.Str3;
    L_Rowno := 0;
    /*插入明细*/
    For Vr In (Select * From Pos_Sellhtdet Where Seqno = V_Billno) Loop
      L_Rowno := L_Rowno + 1;
      Improtsaleinterface.Insertsalegoods(L_Billno,
                                          L_Rowno,
                                          L_Mkt,
                                          Rh.Spbm,
                                          Vr.Gdid,
                                          Abs(Vr.Sl),
                                          Abs(Round(Vr.Je / Vr.Sl, 4)),
                                          Abs(Vr.Je));
      L_Ysamount := L_Ysamount + Vr.Je;
      --If l_Ysamount < 0 Then
      If Vr.Sl < 0 Then
        L_Djlbtmp := Improtsaleinterface.Trans_Th;
      Else
        L_Djlbtmp := Improtsaleinterface.Trans_Xs;
      End If;
      If L_Djlb = 0 Then
        L_Djlb := L_Djlbtmp;
      Else
        If L_Djlb <> L_Djlbtmp Then
          Raiseerr('明细数据不能既有负数也有正数,请检查!');
        End If;
      End If;
    End Loop;
    /*插入付款方式*/
    L_Rowno := 0;
    For Vr In (Select * From Pos_Sellhtpay Where Seqno = V_Billno) Loop
      L_Rowno := L_Rowno + 1;
      Improtsaleinterface.Insertsalepay(L_Billno,
                                        L_Rowno,
                                        L_Mkt,
                                        Vr.Payid,
                                        Abs(Vr.Paymoney));
      /*If (Vr.Paymoney < 0 And l_Djlb = Improtsaleinterface.Trans_Xs) Or
         (Vr.Paymoney > 0 And l_Djlb = Improtsaleinterface.Trans_Th) Then
        Raiseerr('付款方式金额与明细金额不相符!');
      End If;*/
      L_Ssamount := L_Ssamount + Vr.Paymoney;
    End Loop;
    If (L_Ysamount > L_Ssamount) Then
      Raiseerr('付款方式金额与明细金额不相符!');
    End If;
    /*修改溢余sellpayment*/ --zhangheng 2013-12-13
    L_Yy    := L_Ssamount - L_Ysamount;
    L_Rowno := 0;
    For Sr In (Select * From Pos_Sellhtpay Where Seqno = V_Billno) Loop
      L_Rowno := L_Rowno + 1;
      L_Yon   := Fgetpayyy(Fgetpaycode(Sr.Payid));
      If L_Yon = 'Y' Then
        If L_Yy > 0 Then
          Update Sellpayment
             Set Sppayerid = Sr.Payid,
                 Sppayyy   = Least(L_Yy, Abs(Sr.Paymoney))
           Where Spbillno = L_Billno
             And Sprowno = L_Rowno;
          L_Yy := L_Yy - Least(L_Yy, Abs(Sr.Paymoney));
        End If;
      End If;
    End Loop;
    Improtsaleinterface.Insertsalehead(L_Billno,
                                       L_Mkt,
                                       L_Djlb,
                                       L_Ysamount,
                                       L_Ssamount,
                                       Rh.Custno,
                                       Rh.Syy,
                                       Rh.Xsrq,
                                       Rh.Syj);
    Improtsaleinterface.Insertsaleok(L_Billno);
    Update Pos_Sellhthead
       Set Flag = 'Y', Shr = V_Oper, Shrq = Sysdate, Num3 = L_Billno
     Where Seqno = V_Billno;
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End;

  Procedure Approve(V_Billno Varchar2,
                    V_Oper   Varchar2,
                    V_Billid Varchar2 Default 'XP1',
                    V_Djlb   Varchar2 Default '1') As
    V_Sdtbillno Dbusrset.Supsettledettot.Sdtbillno@Sys%Type;
    V_Para1     Manapara.MPVALUE%Type;
    V_Para2     Privatepara.Pparavalue%Type;
  Begin
    ----[[[[增加参数控制[SA]，销售补登审核时，如果发生日期所在期间有生成包含【04抽成】费用的账单，则不允许审核。
    --Manapara参数表的SETTLE_DATETYPE参数设置为【2-发生日期】时有效
    --2015.08.24 =  =P~
    Select MPVALUE Into V_Para1 From dbusrmall.Manapara@sys Where MPID='SETTLE_DATETYPE';
    V_Para2 := Dbusrsys.Fgetprivatepara@Sys('SA');
    If V_Para1 = '2' And V_Para2 = 'Y' Then
      Begin
        Select Supsettledettot.Sdtbillno
          Into V_Sdtbillno
          From Pos_Sellhthead               Poshhd,
               Pos_Sellhtdet                Posshdet,
               Dbusrset.Supsettledettot@Sys
         Where Poshhd.Seqno = Posshdet.Seqno
           And Sdtitemcode = '04'
           And Sdtmfid =
               Decode(Posshdet.Mfid, '#', Poshhd.Spbm, Posshdet.Mfid)
           And Sdtsupid =
               Decode(Posshdet.Supid, '#', Poshhd.Str1, Posshdet.Supid)
           And Trunc(Sdtstartdate) <= Trunc(Xsrq)
           And Trunc(Sdtenddate) >= Trunc(Xsrq)
           And Poshhd.Seqno = V_Billno
           And Rownum = 1;
        Raise_Application_Error(-20001,
                                '销售日期所在期间已生成包含【04抽成】费用的账单:' || V_Sdtbillno ||
                                '，不允许审核');
      Exception
        When No_Data_Found Then
          Null;
      End;
    End If;
    ----]]]]--------------------------------------------------
    If V_Billid = 'XP1' Then
      Approve4one(V_Billno, V_Oper, V_Billid, V_Djlb);
    Elsif V_Billid = 'XP2' Then
      Approve4pl(V_Billno, V_Oper, V_Billid, V_Djlb);
    End If;
  End;

End;
