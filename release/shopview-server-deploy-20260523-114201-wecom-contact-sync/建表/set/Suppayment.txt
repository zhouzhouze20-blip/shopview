  /*生成付款批次临时表*/
  Procedure Genpaybatchtemp(P_Supid   In Varchar2,
                            P_Wmid    In Varchar2,
                            P_Mfid    In Varchar2,
                            O_Batchno Out Number) Is
    L_Batchno Number;
    L_Batchnoid Number;
    --L_Billno  Varchar2(15);
    L_Billno Suppayhead.Sphbillno%Type; --Suppayhead.Sphbillno已调整为20码,变量同步调整 =  =P~ 2016.03.30
     L_SPHSUPID Suppayhead.SPHSUPID%Type;
     L_SPHWMID Suppayhead.SPHWMID%Type;
     L_SPHMFID Suppayhead.SPHMFID%Type;
  Begin
    insert into a values(P_Supid||';'||P_Wmid||';'||P_Mfid,sysdate);
    commit;
    Begin
      Select Sphbillno,SPHSUPID,SPHWMID,SPHMFID
        Into L_Billno,L_SPHSUPID,L_SPHWMID,L_SPHMFID
        From Suppayhead
       Where Sphsupid Like Nvl(P_Supid, '%')--= P_Supid
       --  And Sphwmid Like Nvl(P_Wmid, '%')
         And Sphmfid Like Nvl(P_Mfid, '%')--= P_Mfid
         And Sphflag <> 'Y' and SPHMKT=Currentmkt
         And Rownum = 1;
      If L_Billno Is Not Null Then
        Raise_Application_Error(Errcode,
                                '->供应商[' || L_SPHSUPID || ']经营方式[' || L_SPHWMID || ']
                      柜组[' || L_SPHMFID || ']有未审核的付款单[' ||
                                L_Billno || ']存在,
                      请先审核或删除此付款单后再生成!');
      End If;
    Exception
      When No_Data_Found Then
        Null;
    End;
    Select Seq_Paybatchid.Nextval Into L_Batchnoid From Dual;
    for vr in (select SPDSUPID,SPDWMID,Fgetmframepcode(Spdmfid, 3) SPHKQ from Settlepaydet,supplierbase 
       Where SPDSUPID Like Nvl(P_Supid, '%')--= P_Supid
       --  And Sphwmid Like Nvl(P_Wmid, '%')
       and SPDMKT=Currentmkt and SBID=SPDSUPID and SBSTATUS='Y'
         And Fgetmframepcode(Spdmfid, 3) Like Nvl(P_Mfid, '%')--= P_Mfid
          group by SPDSUPID,SPDWMID,Fgetmframepcode(Spdmfid, 3)) loop
    Select Seq_Paybatch.Nextval Into L_Batchno From Dual;
    /*插入销售，提成*/
    Insert Into Paybatchtemp
      (Pbseq,
       Pbsupid,
       Pbwmid,
       Pbmkt,
       Pbmfid,
       Pbmcontno,
       Pbcontno,
       Pbjsno,
       Pbbillno,
       Pbdjlb,
       Pbbilltype,
       Pbjssdate,
       Pbjsedate,
       Pbxssr,
       Pbtc,
       Pbyf,
       Pbpay,
       Pbsf,
       Pbdate,
       Pbisgen,
       Pbpaybillno,
       Pbpaystatus,
       Pbfeetype,
       Pbjh,
       Pbzkfd,
       Pbyfk,
       Pbsy,
       Pbzj,
       Pbfy1,
       Pbfy2,
       Pbkp,
       Pbn1,
       Pbn2,
       Pbn3,
       Pbn4,
       Pbbckp,
       Pbykp)
      Select L_Batchno,
             Spdsupid,
             Spdwmid,
             Spdmkt,
             Spdmfid,
             Spdmcontno,
             Spdcontno,
             Spdbillno,
             Spdbillno,
             Spddjlb,
             Decode(Spddjlb, 'P', 'P', 'J') As Billtype,
             Spdjssdate + 1 As Sshlastdate,
             Spdjsedate,
             Spdxssr,
             Spdtc,
             Spdyf,
             Spdsf,
             Spdyf - Spdsf + Spdyfk,
             Sysdate,
             'N',
             Null,
             Null,
             'Y',
             Spdjh,
             Spdzkfd,
             Spdyfk,
             Spdsy,
             Spdzj,
             Spdfy1,
             Spdfy2,
             Spdkp,
             Spdn1,
             Spdn2,
             Spdn3,
             L_Batchnoid,--Spdn4,
             Spdkp - Spdykp,
             Spdykp --已经开票
        From Settlepaydet, View_Mframe_All
       Where (Spdyf <> Spdsf and round(Spdkp -spdfy2 - Spdykp,2) <> 0)
         --And round(Spdkp -spdfy2 - Spdykp,2) <> 0
         And Spdsupid =vr.SPDSUPID--= P_Supid
         And Spdwmid =vr.SPDWMID
         and SPDMKT=Currentmkt
         And Fgetmframepcode(Spdmfid, 3) = vr.SPHKQ
         And Spdmfid = Class3;
    
    end loop;
    O_Batchno := L_Batchnoid;--L_Batchno;
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End;
Procedure Genpaymentbatch(P_Seqid In Number,P_Person    In Char,P_Paybillno Out Varchar2) As
  Out_Billno  Varchar2(20);
begin
      --update  Paybatchtemp set Pbselected= 'Y'
      -- Where Pbisgen = 'N' and PBN4=P_Seqid;
      for vr in (select distinct PBSEQ from Paybatchtemp  Where Pbisgen = 'N' and Pbselected= 'Y' and PBN4=P_Seqid) loop
         Genpayment(vr.pbseq,P_Person,'N',Out_Billno);
      End Loop;
    P_Paybillno:=Out_Billno;
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;      
end;                      
  ------------------------------------------------------------------------------
  -- 生成供应商付款单
  --   1、检查发票金额是否足够
  --   2、循环锁定并检查结算单
  --   3、循环锁定并检查发票，减少余额，记录对应表，修改结算单标志
  --   4、生成付款单
  ------------------------------------------------------------------------------
  Procedure Genpayment(P_Seq       In Number,
                       P_Person    In Char,
                       P_Checkfp   In Varchar2 Default 'Y',
                       P_Paybillno Out Varchar2) As
    V_Cnt  Number;
    V_Cnt2 Number;
     l_fkdate date;
    Cursor C_Transbills(V_Seq Number) Is
      Select *
        From Paybatch
       Where Pbseq = V_Seq
       Order By Pbbckp --添加排序,红字发票和篮字发票混勾时从pbbckp小的开始勾兑 2016.04.05 =  =P~
         For Update;

    Cursor C_Invoice(V_Mkt   In Varchar2,
                     V_Supid In Varchar2,
                     V_Mfid  In Varchar2) Is
      Select *
        From Supinvoicehd
       Where (Nvl(Sihmarket, '0') = V_Mkt Or V_Mkt = '0' Or
             Nvl(Sihmarket, '0') = '0')
         And Sihsupid = V_Supid
         and sihhsye <> 0
         And Sihflag = 'Y'
         And (Nvl(Sihmfid, '0') = V_Mfid Or V_Mfid = '0' Or
             Nvl(Sihmfid, '0') = '0')
            /*And (Sihhsye - Nvl(Sihnum1, 0)) > 0*/ --允许付款单与红字发票勾兑2016.03.31 =  =P~
         And Person5 = 'Y'
       Order By (Case
                  When Sihhsye > 0 Then
                   1
                  Else
                   0
                End),
                Abs(Sihhsye) --添加排序,从发票余额小的开始勾兑 2016.04.05 =  =P~
         For Update;
    Row_Paymenttemp   C_Transbills%Rowtype;
    Row_Invoice       C_Invoice%Rowtype;
    Row_Supsettlehead Supsettlehead%Rowtype;
    Row_Suppayhead    Suppayhead%Rowtype;
    Row_Usage         Supinvoiceusage%Rowtype;
    Lisms             Varchar2(20);
    Row_Sb            Supplierbase%Rowtype;
    Lcount            Number;
    R_Cm              Contmain%Rowtype;
    L_Totyfk          Number;
    L_Totfpye         Number;
    L_Allfpje         Number;
    L_Hsye            Number;
    L_Bhsye           Number;
    L_Totpbsf         Number; --本次付款合计
    --L_Billno          Varchar2(15);
    L_Billno Suppayhead.Sphbillno%Type; --Suppayhead.Sphbillno已调整为20码,变量同步调整 =  =P~ 2016.03.30
    L_Kp     Varchar2(1);
  Begin
    Row_Suppayhead := Null;

    Insert Into Paybatch
      (Pbseq, --1
       Pbsupid, --2
       Pbwmid, --3
       Pbmkt, --4
       Pbmfid, --5
       Pbmcontno, --6
       Pbcontno, --7
       Pbjsno, --8
       Pbbillno, --9
       Pbdjlb, --10
       Pbfeetype, --11
       Pbbilltype, --12
       Pbjssdate, --13
       Pbjsedate, --14
       Pbxssr, --15
       Pbtc, --16
       Pbyf, --17
       Pbpay, --18
       Pbsf, --19
       Pbdate, --20
       Pbisgen, --21
       Pbpaybillno, --22
       Pbpaystatus, --23
       Pbselected, --24
       Pbkp, --25
       Pbbckp, --26
       Pbykp, --27
       Pbjh, --28
       Pbzkfd, --29
       Pbyfk, --30
       Pbsy, --31
       Pbzj, --32
       Pbfy1, --33
       Pbfy2, --34
       Pbn1, --35
       Pbn2, --36
       Pbn3, --37
       Pbn4) --38
      Select Pbseq, --1
             Pbsupid, --2
             Pbwmid, --3
             Pbmkt, --4
             Pbmfid, --5
             Pbmcontno, --6
             Pbcontno, --7
             Pbjsno, --8
             Pbbillno, --9
             Pbdjlb, --10
             Pbfeetype, --11
             Pbbilltype, --12
             Pbjssdate, --13
             Pbjsedate, --14
             Pbxssr, --15
             Pbtc, --16
             Pbyf, --17
             Pbpay, --18
             Pbsf, --19
             Pbdate, --20
             Pbisgen, --21
             Pbpaybillno, --22
             Pbpaystatus, --23
             Pbselected, --24
             Pbkp, --25
             Least(pbsf,pbbckp),
             /*(Case
               --When Pbkp >= 0 And Pbsf >= 0 Then
               When Pbkp >= 0 And Pbsf >= 0 And PBSF+PBPAY>=PBYF Then --付款完成时,开票金额按剩余应开票金额开票 =  =P~ 2017.03.06
                --Least(Pbkp - Pbykp, Pbsf)
                Pbkp - Pbykp
               When Pbkp >= 0 And Pbsf >= 0 And PBSF+PBPAY < PBYF Then --未付完款时,开票金额按本次付款金额开票 =  =P~ 2017.03.06
                Pbsf
               When Pbkp >= 0 And Pbsf < 0 Then
                Pbkp - Pbykp
               --When Pbkp < 0 And Pbsf < 0 Then
               When Pbkp < 0 And Pbsf < 0 And PBSF+PBPAY > PBYF Then --未付完款时,开票金额按本次付款金额开票 =  =P~ 2017.03.06
                --Greatest(Pbkp - Pbykp, Pbsf)
                Pbsf
               When Pbkp < 0 And Pbsf < 0 And PBSF+PBPAY <=PBYF Then --付款完成时,开票金额按剩余应开票金额开票 =  =P~ 2017.03.06
                Pbkp - Pbykp
               When Pbkp < 0 And Pbsf >= 0 Then
                Pbkp - Pbykp
             End), */--26  --本次开票金额调整,剩余应开票金额和本次付款取abs低的 2016.04.15 =  =P~
             Pbykp, --27
             Pbjh, --28
             Pbzkfd, --29
             Pbyfk, --30
             Pbsy, --31
             Pbzj, --32
             Pbfy1, --33
             Pbfy2, --34
             Pbn1, --35
             Pbn2, --36
             Pbn3, --37
             Pbn4 --38
        From Paybatchtemp
       Where Pbisgen = 'N'
         And Pbselected = 'Y'
         And Pbseq = P_Seq;
    Delete From Paybatchtemp Where Pbseq = P_Seq;

    --处理的供应商，门店
    Begin
      Select Pbsupid, Pbmkt, Fgetmframepcode(Pbmfid, 3)
        Into Row_Suppayhead.Sphsupid,
             Row_Suppayhead.Sphmkt,
             Row_Suppayhead.Sphmfid
        From Paybatch
       Where Pbseq = P_Seq
         And Rownum = 1;
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                '->1.生成供应商付款单序列' || P_Seq || '无有效数据!');
    End;

    Begin
      ----[[[[Row_Suppayhead.Sphmfid仅为第一个柜台的资料,无法检查所有的柜台,改为循环检查=  =P~ 2015.08.18
      /*Select Sphbillno
        Into L_Billno
        From Suppayhead
       Where Sphsupid = Row_Suppayhead.Sphsupid
         And Sphmfid = Row_Suppayhead.Sphmfid
         And Sphflag <> Flagapproved
         And Rownum = 1;
      If L_Billno Is Not Null Then
        Raise_Application_Error(Errcode,
                                '->供应商[' || Row_Suppayhead.Sphsupid ||
                                ']柜组[' || Row_Suppayhead.Sphmfid ||
                                ']有未审核的付款单[' || L_Billno ||
                                ']存在,
                      请先审核或删除此付款单后再生成!');
      End If;*/
      For Vr In (Select Distinct Pbsupid,
                                 Pbmkt,
                                 Pbmfid,
                                 Fgetmframepcode(Pbmfid, 3) Sphmfid
                   From Paybatch
                  Where Pbseq = P_Seq) Loop
        Begin
          Select Sphbillno
            Into L_Billno
            From Suppayhead
           Where Sphsupid = Vr.Pbsupid
             And Sphmfid = Vr.Sphmfid
             And Sphflag <> Flagapproved
             And Rownum = 1;
          If L_Billno Is Not Null Then
            Raise_Application_Error(Errcode,
                                    '->供应商[' || Vr.Pbsupid || ']柜组[' ||
                                    Vr.Sphmfid || ']有未审核的付款单[' || L_Billno ||
                                    ']存在,
                      请先审核或删除此付款单后再生成!');
          End If;
        Exception
          When No_Data_Found Then
            Null;
        End;
      End Loop;
      ----]]]]------------------------------------------------------------------------------------
    Exception
      When No_Data_Found Then
        Null;
    End;
    --供应商银行帐号信息
    Begin
      Select *
        Into Row_Sb
        From Supplierbase
       Where Sbid = Row_Suppayhead.Sphsupid
         And Sbstatus = 'Y'
         And Sbflag In ('0', '1', '2');
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                '->2.供应商' || Row_Suppayhead.Sphsupid ||
                                '不存在或已停止付款!');
    End;
    -- 扣发票的金额不超过实付金额
    -- Update Paybatch Set pbbckp = least((pbkp - pbykp), pbsf) Where Pbseq = p_Seq;
    -- 2010.11.27，开票金额可能超过实际付款

    --检查本次开票金额总和是否超过发票余额*/
    Begin
      --本次开票金额Pbbckp在写入Paybatch时进行计算 2016.04.15 =  =P~
      Select Sum(Decode(Cmweek2, 'Y', 1, 0) * Pbbckp)
       Into L_Totyfk
       From Paybatch, Contmain
      Where Pbseq = P_Seq
        And Pbmcontno = Cmcontno
        And Nvl(Pbselected, 'N') = 'Y';
      Select Nvl(Sum(Sihhsye), 0)
        Into L_Totfpye
        From Supinvoicehd
       Where Sihmarket = Currentmkt
         And Sihsupid = Row_Suppayhead.Sphsupid
         And (Nvl(Sihmfid, '0') = Row_Suppayhead.Sphmfid Or
             Nvl(Sihmfid, '0') = '0')
         And Sihflag = 'Y'
         And Person5 = 'Y';

      --[[[[添加允许勾兑红字发票后,发票总金额符号需要与结算单本次开票金额符号保持一致 =  =P~ 2016.04.05
      If L_Totyfk * L_Totfpye < 0 And B_Needinvoice = 'Y' Then
        Raise_Application_Error(Errcode,
                                '->2.供应商' || Row_Suppayhead.Sphsupid ||
                                ',柜组' || Row_Suppayhead.Sphmfid ||
                                ',本次开票金额[' || To_Char(L_Totyfk) ||
                                ']与发票余额[' || To_Char(L_Totfpye) || ']发票类型' ||
                                '不一致，不能付款!');
      End If;
      If Abs(L_Totyfk) > Abs(L_Totfpye) And B_Needinvoice = 'Y' Then
        --添加abs用以比较负金额 =  =P~ 2016.04.05
        Raise_Application_Error(Errcode,
                                '->2.供应商' || Row_Suppayhead.Sphsupid ||
                                ',柜组' || Row_Suppayhead.Sphmfid ||
                                ',发票余额不足，不能付款!' || ',本次开票金额[' ||
                                To_Char(L_Totyfk) || ']发票余额[' ||
                                To_Char(L_Totfpye) || ']');
      End If;
      --]]]]-----------------------------------------------------------------------------
    Exception
      When Others Then
        Raise;
    End;

    Row_Suppayhead.Sphmoney := 0;
    --按处理序列生成付款单
    Open C_Transbills(P_Seq);
    Loop
      Fetch C_Transbills
        Into Row_Paymenttemp;
      Exit When C_Transbills%Notfound Or C_Transbills%Notfound Is Null;
      Begin
        Select *
          Into Row_Supsettlehead
          From Supsettlehead
         Where Sshbillno = Row_Paymenttemp.Pbjsno
           And Sshflag = 'Y'
           For Update;
      Exception
        When Others Then
          Raise_Application_Error(Errcode,
                                  '->3.供应商结算单' || Row_Paymenttemp.Pbjsno ||
                                  '不存在或不为审核状态!');
      End;
      Begin
        Select *
          Into R_Cm
          From Contmain
         Where Cmcontno = Row_Paymenttemp.Pbcontno;
        If Row_Supsettlehead.Sshthisdate + R_Cm.Cmzqts3 > Trunc(Sysdate) Then
          Raise_Application_Error(Errcode,
                                  '->5.供应商结算单' || Row_Paymenttemp.Pbjsno ||
                                  '中未达到付款帐期！!');
        End If;
      Exception
        When No_Data_Found Then
          /*Raise_Application_Error(Errcode,
          '->3.供应商结算单' || Row_Paymenttemp.Pbjsno ||
           '合同不存在!');*/
          Null;
        When Others Then
          Raise;
      End;
      -- End If;
      If Row_Paymenttemp.Pbsupid <> Row_Suppayhead.Sphsupid Then
        Raise_Application_Error(Errcode,
                                '->4.供应商结算单' || Row_Paymenttemp.Pbjsno ||
                                '中供应商与处理供应商不一致!');
      End If;
      If Row_Paymenttemp.Pbmkt <> Row_Suppayhead.Sphmkt Then
        Raise_Application_Error(Errcode,
                                '->5.供应商结算单' || Row_Paymenttemp.Pbjsno ||
                                '中门店与处理门店不一致!');
      End If;
      Row_Suppayhead.Sphmoney := Row_Suppayhead.Sphmoney +
                                 Nvl(Row_Paymenttemp.Pbsf, 0);
      /*Row_Suppayhead.Sphvc1   := To_Char(Row_Supsettlehead.Sshthisdate,
      'YYYYMMDD');*/
      Select Cmweek2
        Into L_Kp
        From Contmain
       Where Cmcontno = Row_Paymenttemp.Pbmcontno;
      If Nvl(Row_Paymenttemp.Pbfeetype, 'Y') = 'Y' Then
        --更新开票金额
        --l_Totyfk := ;
        --[[[[本次开票金额调整Pbbckp在写入Paybatch时进行计算 2016.04.15 =  =P~
        Select Nvl(Row_Paymenttemp.Pbbckp * Decode(L_Kp, 'Y', 1, 0), 0)
        Into L_Totyfk
        From Dual;
        --]]]]-----------------------------------
        Open C_Invoice(Currentmkt,
                       Row_Paymenttemp.Pbsupid,
                       Fgetmframepcode(Row_Paymenttemp.Pbmfid, 3));
        Loop
          Fetch C_Invoice
            Into Row_Invoice;

          Exit When C_Invoice%Notfound Or C_Invoice%Notfound Is Null;

          --[[[[查询发票游标记录行数,需要与前面C_Invoice定义的SQL保持一致
          --参数取open游标时使用的参数 =  =P~ 2016.04.05
          Select Count(*)
            Into V_Cnt
            From Supinvoicehd
           Where (Nvl(Sihmarket, '0') = Currentmkt Or Currentmkt = '0' Or
                 Nvl(Sihmarket, '0') = '0')
             And Sihsupid = Row_Paymenttemp.Pbsupid
             And Sihflag = 'Y'
             And (Nvl(Sihmfid, '0') =
                 Fgetmframepcode(Row_Paymenttemp.Pbmfid, 3) Or
                 Fgetmframepcode(Row_Paymenttemp.Pbmfid, 3) = '0' Or
                 Nvl(Sihmfid, '0') = '0')
             And Person5 = 'Y';
          --]]]]---------------------------------------------------------------------
          --[[[[红字发票,余额为负,需要取大值 =  =P~ 2016.03.31
          /*L_Totfpye := Least(L_Totyfk,
          Row_Invoice.Sihhsye -
          Nvl(Row_Invoice.Sihnum1, 0));*/
          V_Cnt2 := C_Invoice%Rowcount;
          If V_Cnt2 = V_Cnt Then
            --最后一张发票,应开票金额直接记入
            L_Totfpye := L_Totyfk;
          Else
            --根据发票勾兑顺序,不是最后一张发票时,红字发票不参与正数金额勾兑,正常发票可能参与负数金额勾兑
            If Row_Invoice.Sihhsye - Nvl(Row_Invoice.Sihnum1, 0) > 0 Then
              L_Totfpye := Least(L_Totyfk,
                                 Row_Invoice.Sihhsye -
                                 Nvl(Row_Invoice.Sihnum1, 0));
            Elsif Row_Invoice.Sihhsye - Nvl(Row_Invoice.Sihnum1, 0) < 0 Then
              If L_Totyfk < 0 Then
                L_Totfpye := Greatest(L_Totyfk,
                                      Row_Invoice.Sihhsye -
                                      Nvl(Row_Invoice.Sihnum1, 0));
              End If;
            End If;
          End If;
          --]]]]------------------------
          If L_Totfpye <> 0 Then
            Row_Usage            := Null;
            Row_Usage.Siubillno  := Row_Invoice.Sihbillno;
            Row_Usage.Siupayno   := '#';
            Row_Usage.Siusetno   := Row_Paymenttemp.Pbjsno;
            Row_Usage.Siuhsje    := L_Totfpye;
            Row_Usage.Siubhsje   := Round(L_Totfpye /
                                          (1 + Row_Invoice.Sihtax),
                                          4);
            Row_Usage.Siuseq     := P_Seq;
            Row_Usage.Siugendate := Sysdate;
            Row_Usage.Siuoper    := P_Person;
            Row_Usage.Siuchr1    := Row_Paymenttemp.Pbdjlb;
            Row_Usage.Siuhsye    := L_Hsye;
            Row_Usage.Siubhsye   := L_Bhsye;
            Row_Usage.Simfid     := Row_Paymenttemp.Pbmfid;
            Row_Usage.Siwmid     := Row_Paymenttemp.Pbwmid;
            Row_Usage.Sibilltype := Row_Paymenttemp.Pbbilltype;
            Row_Usage.Sicontno   := Row_Paymenttemp.Pbcontno;
            Row_Usage.Sibillno   := Row_Paymenttemp.Pbbillno;
            Row_Usage.Sistatus   := 'N';
            L_Totyfk             := L_Totyfk - L_Totfpye;
            Update Supinvoicehd
               Set Sihnum1 = Nvl(Sihnum1, 0) + L_Totfpye,person1=nvl(person1,'A')
             Where Sihbillno = Row_Invoice.Sihbillno;
            Insert Into Supinvoiceusage Values Row_Usage;
          End If;
        End Loop;
        Close C_Invoice;
        If L_Totyfk <> 0 And B_Needinvoice = 'Y' Then
          Raise_Application_Error(Errcode, '->6.发票不够扣减，请检查!');
        End If;
      End If;
    End Loop;
    Close C_Transbills;
    If Abs(Row_Suppayhead.Sphmoney) < 0.001 Then
      Raise_Application_Error(Errcode,Row_Suppayhead.Sphsupid||'->9.供应商本期无付款金额!');
    End If;
    Sp_Billseq('PAY', Row_Suppayhead.Sphbillno, 'N');
    Sp_Billnum('PAY',
               Row_Suppayhead.Sphbillno,
               Row_Suppayhead.Sphdjbh,
               'N');
    Row_Suppayhead.Sphflag       := Flagautoed;
    Row_Suppayhead.Sphmoneyupper := Fmoney2chn(Row_Suppayhead.Sphmoney);
    --审核时重算当天汇率、本币金额
    Begin
      Select Sbvc2
        Into Lisms
        From Supplierbase
       Where Sbid = Row_Suppayhead.Sphsupid;
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                '->10.对应供应商【' || Row_Suppayhead.Sphsupid ||
                                '】的主档信息不存在，请检查!');
    End;
    --IF lisms <> 'Y' THEN
    Row_Suppayhead.Sphbz         := 'RMB'; --currency.Fgetcurrencynative;
    Row_Suppayhead.Sphhl         := 1;
    Row_Suppayhead.Sphbbje       := Round(Row_Suppayhead.Sphhl *
                                          Row_Suppayhead.Sphmoney,
                                          4);
    Row_Suppayhead.Sphbank       := Row_Sb.Sbbank;
    Row_Suppayhead.Sphaccntno    := Row_Sb.Sbaccntno;
    Row_Suppayhead.Sphtaxno      := Row_Sb.Sbtaxno;
    Row_Suppayhead.Sphmktbank    := Fgetprivateinfo('03');
    Row_Suppayhead.Sphmktaccntno := Fgetprivateinfo('04');
    Row_Suppayhead.Sphmkttaxno   := Fgetprivateinfo('06');
    Row_Suppayhead.Sphpaydate    := Currentdate;
    Row_Suppayhead.Inputor       := P_Person;
    Row_Suppayhead.Inputdate     := Currentdate;
    Row_Suppayhead.Sphn2         := P_Seq;
     Row_Suppayhead.Sphvc1        := '9906';
    Inserthead(Row_Suppayhead);
    /* --处理的结算单、发票处于等待付款
    Select Count(*)
    Into   Lcount
    From   Genpaymenttemp
    Where  Gptseq = p_Seq And
           Gptflag = 'N' And
           Gpttype = '1';
    If Lcount > 0 Then
      Update Supsettlehead
      Set    Sshpayflag = 'S', Sshpayno = Row_Suppayhead.Sphbillno
      Where  Sshflag = 'Y' And
             Sshbillno In (Select Gptbillno
                           From   Genpaymenttemp
                           Where  Gptseq = p_Seq And
                                  Gptflag = 'N' And
                                  Gpttype = '1');
      If Sql%Rowcount = 0 Or Sql%Rowcount Is Null Then
        Raise_Application_Error(Errcode, '->11.修改结算单标志失败!');
      End If;
    End If;*/
    --生成付款单后处理中间表
    Update Paybatch
       Set Pbpaystatus = 'N', Pbpaybillno = Row_Suppayhead.Sphbillno
     Where Pbseq = P_Seq;
    If Sql%Rowcount = 0 Or Sql%Rowcount Is Null Then
      Raise_Application_Error(Errcode, '->13.生成付款单环节失败，请重试!');
    End If;
    --更新付款单号
    Update Supinvoiceusage
       Set Siupayno = Row_Suppayhead.Sphbillno
     Where Siuseq = P_Seq;
    --Commit;
    P_Paybillno := Row_Suppayhead.Sphbillno;
    --更新付款日期
    begin
      select SSHPLANPAYDATE
        into l_fkdate
        from supsettlehead,
             (select max(pbjsno) jsno
                from Paybatch
               where Pbpaybillno = Row_Suppayhead.Sphbillno) a
       where SSHBILLNO = a.jsno;
    Exception
      When Others Then
        l_fkdate := Row_Suppayhead.Sphpaydate;
    end;
    update Suppayhead
       set Sphpaydate = l_fkdate
     where Sphbillno = Row_Suppayhead.Sphbillno;
  Exception
    When Others Then
      --Rollback;
      Raise;
  End Genpayment;

  --审核(付款)付款单
 Procedure Approve(P_Billno In Char,
                    P_Person In Char,
                    P_Billid In Char,
                    P_Djlb   In Char) Is
    Row_Hd     Suppayhead%Rowtype;
    Lisms      Varchar2(20);
    Lbz        Varchar2(10);
    Lhl        Number;
    Lbbje      Number;
    Lcount     Number;
    L_Money    Number;
    L_Retmoney Number;
    Row_Ssc    Supsetcharge%Rowtype;
    L_Hsye     Number;
    L_Bhsye    Number;
    row_p      nc_temp_cashpay%Rowtype;
    Lseq       Number;
    lsupsub    varchar2(20);
    lmfsub     varchar2(20);
    lzt        varchar2(20);
    LDATE      DATE;
    lyfk       number;
  Begin
    Begin
      Select *
        Into Row_Hd
        From Suppayhead
       Where Sphbillno = P_Billno
         For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, P_Billno || '->01.单据不存在!');
    End;
    Begin
    select sum(PBYFK) into lyfk from paybatch where PBPAYBILLNO=P_Billno;
     Exception
      When No_Data_Found Then
        lyfk:=0;
    End;
    If (P_Billid <> 'PAY' and P_Billid <> 'PBY') Or P_Djlb <> '1' Then
      Raise_Application_Error(Errcode, P_Billno || '->02.单据类别不一致!');
    End If;
    If Currentmkt <> Row_Hd.Sphmkt /*Or Currentmkttype <> Stock.Mf_Mkt*/
     Then
      Raise_Application_Error(Errcode,
                              P_Billno || '->03.付款单只能在门店审核!');
    End If;
    If Row_Hd.Sphflag Not In (Flagediting, Flagautoed) Then
      Raise_Application_Error(Errcode,
                              P_Billno || '->04.单据已经付款或单据状态不一致!');
    End If;
    --审核时重算当天汇率、本币金额
    Begin
      Select Sbvc2
        Into Lisms
        From Supplierbase
       Where Sbid = Row_Hd.Sphsupid;
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                '->05.付款单中对应供应商【' || Row_Hd.Sphsupid ||
                                '】不存在，请检查!');
    End;
    Lbz   := 'RMB'; --currency.Fgetcurrencynative;
    Lhl   := 1;
    Lbbje := Round(Lhl * Row_Hd.Sphmoney, 4);
    --处理结算单、发票处于已付款状态
    For Vr In (Select * From Paybatch Where Pbpaybillno = P_Billno) Loop
      If Vr.Pbbilltype = 'J' Then
        Update Settlepaydet
           Set Spdsf       = Spdsf + Vr.Pbsf,
               Spdykp      = Spdykp + Vr.Pbbckp,
               Spdlastdate = Sysdate
         Where Spdsupid = Vr.Pbsupid
           And Spdwmid = Vr.Pbwmid
           And Spdmkt = Vr.Pbmkt
           And Spdmfid = Vr.Pbmfid
           And Spdmcontno = Vr.Pbmcontno
           And Spdcontno = Vr.Pbcontno
           And Spdbillno = Vr.Pbbillno
           And Spddjlb = Vr.Pbdjlb;
        /*更新当前付款的结算单中整体的付款情况*/
        Update Supsettlehead
           Set Paydate = Currentdate, Sshpayflag = 'Y'
         Where Sshflag = 'Y'
           And Sshbillno = Vr.Pbbillno
           And Not Exists (Select 'X'
                  From Settlepaydet
                 Where /*Spdsupid = Vr.Pbsupid And
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  Spdwmid = Vr.Pbwmid And
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  Spdmkt = Vr.Pbmkt And
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  Spdmfid = Vr.Pbmfid And
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  Spdmcontno = Vr.Pbmcontno And
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  Spdcontno = Vr.Pbcontno And
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  Spdbillno = Vr.Pbbillno And
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  Spddjlb = Vr.Pbdjlb And*/
                --2014.3.25 ，需要全部都付完款才能设置付款标志
                 Spdyf+SPDYFK <> Spdsf
              And Spdbillno = Sshbillno);
        For Row_Ssc In (Select *
                          From Supsetcharge
                         Where Sscjsno = /*Vr.Pbbillno*/
                               Vr.Pbjsno --2015.08.11 =  =P~
                        ) Loop
          If Row_Ssc.Sscrettype = '2' Then
            Update Supsetcharge
               Set Sscretmoney   = Nvl(Sscretmoney, 0) - Row_Ssc.Sscmoney,
                   Sscrecretdate = Trunc(Sysdate),
                   Sscretstatus  = Row_Ssc.Sscretstatus
             Where Sscbillno = Row_Ssc.Sscyretbillno
            Returning Sscretmoney, Sscmoney Into L_Money, L_Retmoney;
            If L_Money <= L_Retmoney Then
              Update Supsetcharge
                 Set Sscretstatus = 'Y'
               Where Sscbillno = Row_Ssc.Sscyretbillno;
            End If;
          End If;
          --[[[[ Vr.Pbbillno记录的是付款单号,Sscbillno是费用单号,导致查不到资料=  =P~ 2015.08.11
          --Update Supsetcharge Set Sscflag = 'G' Where Sscbillno = Vr.Pbbillno;
          Update Supsetcharge
             Set Sscflag = 'G', Sscpaybillno = Vr.Pbpaybillno --记录付款单号,便于反审操作,与直接在结算单上付款的费用做区分
           Where Sscbillno = Row_Ssc.Sscbillno
             And Sscmfid = Vr.Pbmfid
             And Sscjsno = Row_Ssc.Sscjsno
             And Sscflag <> 'G';
          --]]]]-------------------------------------
        End Loop;
      Else
        Raise_Application_Error(Errcode,
                                P_Billno || '->06.付款明细中错误的单据类型!');
      End If;
    End Loop;
    --更新发票用途对应表
    For Vr In (Select Supinvoiceusage.*, Rowid Row_Id
                 From Supinvoiceusage
                Where Siupayno = P_Billno) Loop
      Update Supinvoicehd
         Set Sihnum1  = Sihnum1 - Vr.Siuhsje, --减发票待处理金额
             Sihhsye  = Sihhsye - Vr.Siuhsje,
             Sihbhsye = Sihbhsye - Vr.Siubhsje
       Where Sihbillno = Vr.Siubillno
      Returning Sihhsye, Sihbhsye Into L_Hsye, L_Bhsye;
      Update Supinvoiceusage
         Set Siupaydate = Sysdate,
             Sistatus   = 'Y',
             Siuhsye    = L_Hsye,
             Siubhsye   = L_Bhsye
       Where Rowid = Vr.Row_Id;
    End Loop;
    Update Paybatch Set Pbpaystatus = 'Y' Where Pbpaybillno = P_Billno;
    Update Suppayhead
       Set Sphhl     = Lhl,
           Sphbz     = Lbz,
           Sphbbje   = Lbbje,
           Sphflag   = Flagapproved,
           Auditor   = P_Person,
           Auditdate = Trunc(Sysdate)
     Where Sphbillno = P_Billno;
    If Sql%Rowcount = 0 Or Sql%Rowcount Is Null Then
      Raise_Application_Error(Errcode, '->08.修改供应商付款单标志不成功!');
    End If;
  
    --ins_midcashpay(P_Billno);
    select seq_nc_temp_cashpay.nextval into lseq from dual;
    row_p.id := lseq;
    begin
      select mfsubject
        into lzt
        from manaframe
       where mfcode = Row_Hd.Sphmkt
         and rownum = 1;
    exception
      when others then
        Raise_Application_Error(Errcode, '->08.门店找不到对应核算信息!');
    end;
    if Row_hd.Sphvc1 = '9906' then
      row_p.corp       := lzt;
      row_p.tradetype  := '0';
      row_p.settlecode := '01';
      row_p.billdate   := to_char(sysdate, 'yyyy-mm-dd');
      row_p.headid     := lseq;
      row_p.bodyid     := 1;
      select nvl(sbsubject, sbid)
        into lsupsub
        from supplierbase
       where sbid = Row_Hd.Sphsupid;
      row_p.cvcode := lsupsub;
      /*select count(distinct pbmfid) into lcount from paybatch where pbpaybillno = p_billno;
      if lcount > 1 then
        Raise_Application_Error(Errcode, '->08.付款单不允许包含多个柜组数据!');
      end if;*/
      select nvl(mfsubject, mfcode)
        into lmfsub
        from manaframe
       where mfcode = Row_Hd.Sphmfid;
      row_p.deptcode       := lmfsub;
      row_p.paybkaccount   := Row_Hd.Sphmktaccntno;
      row_p.paybkname      := Row_Hd.Sphmktbank;
      row_p.recebkaccount  := Row_Hd.Sphaccntno;
      row_p.recebkname     := Row_Hd.Sphbank;
      row_p.crc            := '人民币';
      row_p.rate           := 1;
      row_p.digest         := fgetpayzy(Row_Hd.Sphbillno);
      row_p.payoriginalmny := Row_Hd.Sphmoney-lyfk;
      row_p.paylocalmny    := Row_Hd.Sphmoney-lyfk;
      row_p.billflag       := '0';
      row_p.sourcebillcode := Row_Hd.Sphbillno;
      row_p.iflag          := 'N';
      row_p.note           := '';
      row_p.ts             := to_char(sysdate, 'yyyy-mm-dd hh24:mi:ss');
      row_p.billcode       := null;
      LDATE                := fgetplanpaydate(Row_Hd.Sphbillno);
      row_p.prepaydate     := to_char(LDATE,'yyyy-mm-dd');
      row_p.cashflowcode   := null;
      IF Row_Hd.Sphaccntno IS NULL OR Row_Hd.Sphbank IS NULL THEN
        Raise_Application_Error(Errcode, '->09.供应商开户银行为空或银行账户为空!');
      END IF;
      insert into nc_temp_cashpay values row_p;
    end if;
  
    --审核已付已收帐
    --settleaccpayday.approve(p_billno,settleaccpayday.sourcefkd);
    Supcamana.Paygen(P_Billno, 'PAY');
    Commit;
  Exception
    When Others Then
      Raise;
  End Approve;

      select ztdm into lzt from cwzzzt where ztdymarket = Row_Hd.Sphmkt and rownum = 1;
    exception when others then
      Raise_Application_Error(Errcode, '->08.门店找不到对应帐套信息!');
    end;  
    row_p.corp                 := lzt;
    row_p.tradetype            := '0';
    row_p.settlecode           := '01';
    row_p.billdate             := to_char(sysdate, 'yyyy-mm-dd');
    row_p.headid               := lseq;
    row_p.bodyid               := 1;
    select nvl(sbsubject,sbid) into lsupsub from supplierbase where sbid = Row_Hd.Sphsupid;
    row_p.cvcode               := lsupsub;
    /*select count(distinct pbmfid) into lcount from paybatch where pbpaybillno = p_billno;
    if lcount > 1 then
      Raise_Application_Error(Errcode, '->08.付款单不允许包含多个柜组数据!');
    end if;*/
    select nvl(mfsubject,mfcode) into lmfsub from manaframe where mfcode = Row_Hd.Sphmfid;

    row_p.paybkaccount         := Row_Hd.Sphmktaccntno;

    DELETE FROM suppayhead WHERE sphbillno = p_billno;
    For Vr In (Select Supinvoiceusage.*, Rowid Row_Id
                 From Supinvoiceusage
                Where Siupayno = P_Billno) Loop
      Update Supinvoicehd
         Set Sihnum1 = Sihnum1 - Vr.Siuhsje --减发票待处理金额
       Where Sihbillno = Vr.Siubillno;
    End Loop;
    Delete From Supinvoiceusage Where Siupayno = P_Billno;

    Commit;
  Exception
    When Others Then
      Raise;
  End Cancel;

  Procedure ins_midcashpay(P_Billno In Char) Is
    Row_Hd Suppayhead%Rowtype;   
    Lcount    Number;
    row_p     nc_temp_cashpay%Rowtype;
    Lseq      Number;
    lsupsub   varchar2(20);
    lmfsub    varchar2(20);
    lzt       varchar2(20);
  Begin
    Begin
      Select *
        Into Row_Hd
        From Suppayhead
       Where Sphbillno = P_Billno
         For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, P_Billno || '->01 单据不存在!');
    End;

    If Row_Hd.Sphflag Not In (Flagapproved) Then
      Raise_Application_Error(Errcode,
                              P_Billno || '->04.单据不是已付款状态!');
    End If;
    
    select seq_nc_temp_cashpay.nextval into lseq from dual;
    row_p.id                   := lseq;
    begin
      select ztdm into lzt from cwzzzt where ztdymarket = Row_Hd.Sphmkt and rownum = 1;
    exception when others then
      Raise_Application_Error(Errcode, '->08.门店找不到对应帐套信息!');
    end;  
    row_p.corp                 := lzt;
    row_p.tradetype            := '0';
    row_p.settlecode           := '01';
    row_p.billdate             := to_char(Row_Hd.Auditdate, 'yyyy-mm-dd');
    row_p.headid               := lseq;
    row_p.bodyid               := 1;
    select nvl(sbsubject,sbid) into lsupsub from supplierbase where sbid = Row_Hd.Sphsupid;
    row_p.cvcode               := lsupsub;
    /*select count(distinct pbmfid) into lcount from paybatch where pbpaybillno = p_billno;
    if lcount > 1 then
      Raise_Application_Error(Errcode, '->08.付款单不允许包含多个柜组数据!');
    end if;*/
    select nvl(mfsubject,mfcode) into lmfsub from manaframe where mfcode = Row_Hd.Sphmfid;
    row_p.deptcode             := lmfsub;
    row_p.paybkaccount         := Row_Hd.Sphmktaccntno;
    row_p.paybkname            := Row_Hd.Sphmktbank;
    row_p.recebkaccount        := Row_Hd.Sphaccntno;
    row_p.recebkname           := Row_Hd.Sphbank;
    row_p.crc                  := '人民币';
    row_p.rate                 := 1;
    row_p.digest               := '付款';
    row_p.payoriginalmny       := Row_Hd.Sphmoney;
    row_p.paylocalmny          := Row_Hd.Sphmoney;
    row_p.billflag             := '0';
    row_p.sourcebillcode       := Row_Hd.Sphbillno;
    row_p.iflag                := 'N';
    row_p.note                 := '';
    row_p.ts                   := to_char(Row_Hd.Auditdate, 'yyyy-mm-dd hh24:mi:ss');
    row_p.billcode             := null;
    row_p.prepaydate           := null;
    row_p.cashflowcode         := null;
    insert into nc_temp_cashpay values row_p;     
       
  End;  
  Procedure Noapprove(P_Billno In Char,
                      P_Person In Char,
                      P_Billid In Char,
                      P_Djlb   In Char) Is
    Row_Hd  Suppayhead%Rowtype;
    Lcount  Number;
    row_p   nc_temp_cashpay%Rowtype;
    Lseq    Number;
    lsupsub varchar2(20);
    lmfsub  varchar2(20);
    lzt     varchar2(20);
    LDATE   DATE;
    lyfk       number;
    Cursor C_Usage Is
      Select * From Supinvoiceusage Where Siupayno = P_Billno;
    R_Usage C_Usage%Rowtype;
  Begin
    Begin
      Select *
        Into Row_Hd
        From Suppayhead
       Where Sphbillno = P_Billno
         For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, P_Billno || '->01 单据不存在!');
    End;
     Begin
    select sum(PBYFK) into lyfk from paybatch where PBPAYBILLNO=P_Billno;
     Exception
      When No_Data_Found Then
        lyfk:=0;
    End;
    If (P_Billid <> 'PAY' and P_Billid <> 'PBY') Or P_Djlb <> '1' Then
      Raise_Application_Error(Errcode, P_Billno || '->02.单据类别不一致!');
    End If;
    If Currentmkt <> Row_Hd.Sphmkt Then
      Raise_Application_Error(Errcode,
                              P_Billno || '->03.付款单只能在门店删除!');
    End If;
    If Row_Hd.Sphflag Not In (Flagapproved) Then
      Raise_Application_Error(Errcode,
                              P_Billno || '->04.单据已经审核或单据状态不一致!');
    End If;
    For Vr In (Select * From Paybatch Where Pbpaybillno = P_Billno) Loop
      If Vr.Pbbilltype = 'J' Then
        Update Settlepaydet
           Set Spdsf       = Spdsf - Vr.Pbsf,
               Spdykp      = Spdykp - Vr.Pbbckp,
               Spdlastdate = Sysdate
         Where Spdsupid = Vr.Pbsupid
           And Spdwmid = Vr.Pbwmid
           And Spdmkt = Vr.Pbmkt
           And Spdmfid = Vr.Pbmfid
           And Spdmcontno = Vr.Pbmcontno
           And Spdcontno = Vr.Pbcontno
           And Spdbillno = Vr.Pbbillno
           And Spddjlb = Vr.Pbdjlb;
        /*更新当前付款的结算单中整体的付款情况*/
        Update Supsettlehead
           Set Paydate = Null, Sshpayflag = 'N'
         Where Sshflag = 'Y'
           And Sshbillno = Vr.Pbbillno
           And Sshpayflag = 'Y';
        For Row_Ssc In (Select *
                          From Supsetcharge
                         Where Sscjsno = Vr.Pbbillno
                           And Sscpaybillno = Vr.Pbpaybillno --2015.08.11 =  =P~
                        ) Loop
        
          If Row_Ssc.Sscrettype = '2' Then
            Update Supsetcharge
               Set Sscretmoney   = Nvl(Sscretmoney, 0) + Row_Ssc.Sscmoney,
                   Sscrecretdate = Null,
                   Sscretstatus  = 'N'
             Where Sscbillno = Row_Ssc.Sscyretbillno;
          End If;
          --[[[[ Vr.Pbbillno记录的是付款单号,Sscbillno是费用单号,导致查不到资料=  =P~ 2015.08.11
          --Update Supsetcharge Set Sscflag = 'Y' Where Sscbillno = Vr.Pbbillno;
          Update Supsetcharge
             Set Sscflag = 'Y', Sscpaybillno = '' --反审时付款单号置空
           Where Sscbillno = Row_Ssc.Sscbillno
             And Sscmfid = Vr.Pbmfid
             And Sscjsno = Row_Ssc.Sscjsno
             And Sscpaybillno = Vr.Pbpaybillno; --审核时会在费用单上记录付款单号,反审时据此改写费用单的状态,与直接在结算单上收费做区分
        --]]]]-----------------------------------------------------
        End Loop;
      Else
        Raise_Application_Error(Errcode,
                                P_Billno || '->06.付款明细中错误的单据类型!');
      End If;
    End Loop;
    --更新发票用途对应表
    For Vr In (Select Supinvoiceusage.*, Rowid Row_Id
                 From Supinvoiceusage
                Where Siupayno = P_Billno) Loop
      Update Supinvoicehd
         Set Sihnum1  = Sihnum1 + Vr.Siuhsje, --减发票待处理金额
             Sihhsye  = Sihhsye + Vr.Siuhsje,
             Sihbhsye = Sihbhsye + Vr.Siubhsje
       Where Sihbillno = Vr.Siubillno;
      Update Supinvoiceusage
         Set Siupaydate = Null, Sistatus = 'N', Siuhsye = 0, Siubhsye = 0
       Where Rowid = Vr.Row_Id;
    End Loop;
    Update Paybatch Set Pbpaystatus = 'N' Where Pbpaybillno = P_Billno;
    Update Suppayhead
       Set Sphhl     = Null,
           Sphbz     = Null,
           Sphbbje   = Null,
           Sphflag   = Flagautoed,
           Auditor   = Null,
           Auditdate = Null
     Where Sphbillno = P_Billno;
    If Sql%Rowcount = 0 Or Sql%Rowcount Is Null Then
      Raise_Application_Error(Errcode, '->08.修改供应商付款单标志不成功!');
    End If;
    --应收应付
    Supcamana.Paygen(P_Billno, 'CANCEL');
  
    --ins_midcashpay(P_Billno);
    select seq_nc_temp_cashpay.nextval into lseq from dual;
    row_p.id := lseq;
    begin
      select mfsubject
        into lzt
        from manaframe
       where mfcode = Row_Hd.Sphmkt
         and rownum = 1;
    exception
      when others then
        Raise_Application_Error(Errcode, '->08.门店找不到对应核算信息!');
    end;
    if Row_hd.Sphvc1 = '9906' then
      row_p.corp       := lzt;
      row_p.tradetype  := '0';
      row_p.settlecode := '01';
      row_p.billdate   := to_char(sysdate, 'yyyy-mm-dd');
      row_p.headid     := lseq;
      row_p.bodyid     := 1;
      select nvl(sbsubject, sbid)
        into lsupsub
        from supplierbase
       where sbid = Row_Hd.Sphsupid;
      row_p.cvcode := lsupsub;
      /*select count(distinct pbmfid) into lcount from paybatch where pbpaybillno = p_billno;
      if lcount > 1 then
        Raise_Application_Error(Errcode, '->08.付款单不允许包含多个柜组数据!');
      end if;*/
      select nvl(mfsubject, mfcode)
        into lmfsub
        from manaframe
       where mfcode = Row_Hd.Sphmfid;
      row_p.deptcode       := lmfsub;
      row_p.paybkaccount   := Row_Hd.Sphmktaccntno;
      row_p.paybkname      := Row_Hd.Sphmktbank;
      row_p.recebkaccount  := Row_Hd.Sphaccntno;
      row_p.recebkname     := Row_Hd.Sphbank;
      row_p.crc            := '人民币';
      row_p.rate           := 1;
      row_p.digest         := fgetpayzy(Row_Hd.Sphbillno);
      row_p.payoriginalmny := (Row_Hd.Sphmoney-lyfk) * (-1);
      row_p.paylocalmny    := (Row_Hd.Sphmoney-lyfk) * (-1);
      row_p.billflag       := '0';
      row_p.sourcebillcode := Row_Hd.Sphbillno;
      row_p.iflag          := 'N';
      row_p.note           := '';
      row_p.ts             := to_char(sysdate, 'yyyy-mm-dd hh24:mi:ss');
      row_p.billcode       := null;
      LDATE                := fgetplanpaydate(Row_Hd.Sphbillno);
      row_p.prepaydate     := to_char(LDATE,'yyyy-mm-dd');
      row_p.cashflowcode   := null;
      insert into nc_temp_cashpay values row_p;
    end if;
    Commit;
  Exception
    When Others Then
      Raise;
  End Noapprove;

Begin
  Currentmkt     := Fgetsysplace;
  Currentdate    := Fgeterpdate(Currentmkt);
  Currentmkttype := Fgetsystype(Currentmkt);
End Suppayment;
