CREATE OR REPLACE Package Body posdayinit Is
  b_sysdate  Date;
  b_sysplace Varchar2(20);
  b_sysid    Varchar2(20);

  -- Initialization
  Procedure init Is
  Begin
    b_sysdate  := trunc(Sysdate);
    b_sysplace := fgetsysplace();
    b_sysid    := fgetsysid();
  End init;

  --执行任务
  Procedure exec_initdata Is
  Begin
    --POS管理登录日志
    --Delete From operlogin Where oltime < Sysdate - 0.01;

    --初始化POS机状态
    Update syjmain
       Set syjcurstatus = '5', --关机
           syjcurtime   = Sysdate,
           syjreceiver  = Null,
           syjxsbs      = 0,
           syjxssumje   = 0,
           syjcashje    = 0;
    Commit;

    --调整商品状态标志
    Begin
      sp_adjustallgoodsflag;
    Exception
      When Others Then
        Rollback;
    End;

    --清除POS工作日志和异常日志
    Delete From posoperlog Where poldate < b_sysdate - 30;
    Commit;
    Delete From posopererrors Where poedate < b_sysdate - 7;
    Commit;

    --清除历史通知
    Delete From posinfo Where piinfodate < b_sysdate - pibltime;
    Commit;

    --清除过期的优惠信息
    Delete From pospopinfo Where ppienddate < b_sysdate;
    Commit;

    --清除过期的批量信息
    Delete From posgoodsamount
    Where ggaenddate < b_sysdate and pgamount<>'1';--保留柜组VIP折扣设置信息
    Commit;

    Delete From opererrors Where oedate < b_sysdate - 30;
    Commit;

    --清除过期的更新日志
    --DELETE FROM SYJCMDMAIL WHERE SCMDATE < b_sysdate - 1;
    Execute Immediate 'truncate table DBUSRPOSNEW.SYJCMDMAIL';
    --清除日志
    Delete From debug_log Where logdate  < b_sysdate - 93;
    Commit;
  End exec_initdata;

  Procedure submit Is
  Begin
    init;
    exec_initdata;
    Commit;
    --sp_recompute_jf('%', trunc(Sysdate) - 3, trunc(Sysdate), 0);
    --Commit;
    --生成长短款
    /*Begin
      sp_cdkrep_day(b_sysdate - 1);
      sp_cdkpm_day(b_sysdate - 1);
      Commit;
    Exception
      When Others Then
        Rollback;
    End;*/
    Begin
       Cardcrm_Pos.SubmitPay;
       Commit;
    Exception When Others Then
       Rollback;
    End;
    
    Begin
       sp_syjmain_mall(b_SysDate);
       Commit;
    Exception When Others Then
       Rollback;
    End;
    
    -- sp_savespacelog(trunc(sysdate));
  Exception
    When Others Then
      Rollback;
      sp_opererrors('日初始化', Sqlcode, b_sysplace || '--' || Sqlerrm);
      Raise;
  End submit;

  --执行过程
  Procedure run Is
  Begin
    submit;
  Exception
    When Others Then
      dbms_output.put_line(substr(b_sysplace || '--' || Sqlerrm, 1, 255));
  End run;

  Procedure job_submit(i_time In Number) As
    jobid  Binary_Integer;
    l_time Number;
  Begin
    l_time := 1 + round(i_time / 24, 3);
    dbms_job.submit(job       => jobid,
                    what      => 'POSDAYINIT.RUN;',
                    next_date => trunc(Sysdate) + l_time,
                    Interval  => 'TRUNC(SYSDATE) + ' || to_char(l_time),
                    no_parse  => Null);
    Commit;
    dbms_output.put_line('JobID = ' || jobid);
  End job_submit;

  Procedure job_remove(jobid Binary_Integer) As
  Begin
    dbms_job.remove(job => jobid);
    Commit;
  End job_remove;
End posdayinit;
