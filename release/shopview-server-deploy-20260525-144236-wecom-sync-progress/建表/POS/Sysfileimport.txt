CREATE OR REPLACE Package Body Sysfileimport Is

  /*得到序号*/
  Function Fgetseqno(v_Id In Varchar2) Return Number As
    l_Seq Number;
  Begin
    Select Seq_Importfile.Nextval Into l_Seq From Dual;
    Return l_Seq;
  Exception
    When Others Then
      Return 0;
  End;

  /*开始执行*/
  Procedure Sp_Startimport
  (
    v_Seqno    In Number,
    v_Id       In Varchar2,
    v_Filename In Varchar2
  ) As
  Begin
    Insert Into Sys_File_Import_Master
      (Imseq, Imid, Imdate, Imfilename, Imstatus, Imtimes)
    Values
      (v_Seqno, v_Id, Sysdate, v_Filename, 'N', 0);
    Commit;
  End;

  /*插入日志*/
  Procedure Sp_Importlog
  (
    v_Seqno    In Number,
    v_Id       In Varchar2,
    v_Type     In Varchar2, --1 -成功 2-失败
    v_Key      In Varchar2,
    v_Msg      In Varchar2,
    v_Filename In Varchar2
  ) As
    Pragma Autonomous_Transaction;
  Begin
    Savepoint Debug;
    Insert Into Interfacelog
      (Ilseq, Ildate, Iltypeid, Ilmode, Ilkey, Ilmfid, Ilfilename)
      Select v_Seqno, Sysdate, v_Type, v_Id, v_Msg, v_Key, v_Filename
      From   Dual;
    Commit;
  Exception
    When Others Then
      Rollback To Debug;
      Null;
  End;

  /*结束导入
   动态执行过程，
  */
  Procedure Sp_Endimport
  (
    v_Seqno In Number,
    v_Id    In Varchar2
  ) As
    l_Proc Varchar(40);
  Begin
    Select Nvl(Sicallproc, '#')
    Into   l_Proc
    From   Sys_File_Import_Dbconfig
    Where  Siid = v_Id;
    If nvl(l_Proc,'#') <> '#' Then
      Sp_Dxp_Dynexecproc(l_Proc,v_Seqno);
    End If;
    Update Sys_File_Import_Master
    Set    Imstatus = 'Y'
    Where  Imseq = v_Seqno And v_Id = Imid;
  End;

Begin
  -- Initialization
  Null;
End Sysfileimport;
