CREATE OR REPLACE PACKAGE BODY "SALEFLMS" Is
  Function Checklpisable(p_Lpbm In Varchar2, p_mkt in varchar2) Return Char As
    l_Rtn    Char(1);
    l_Kc     Number;
    l_Type   Char(1);
    l_zpchr2 Varchar2(20);
    l_dblink Varchar2(20);
    ls_sql   Varchar2(200);
    l_Cursor Integer;
    l_Result Integer;
  Begin
    Select Nvl(Zpflag, 'N'),
           Decode(Zpchr1, '2', Nvl(Zpsl, 0), 0),
           Zpchr1,
           Zpchr2
      Into l_Rtn, l_Kc, l_Type, l_Zpchr2
      From Crmzpbm
     Where Zpbm = p_Lpbm;
    If l_Rtn <> 'Y' Then
      Return 'N';
    End If;
    If l_Type = '2' Then
      Return 'Y';
    Elsif l_Type = '3' Then
      If l_kc >= 0 Then
        Return 'Y';
      Else
        Return 'N';
      End If;
    Else
      If Trim(l_Zpchr2) Is Null Then
        Return 'N';
      End If;
      Begin
        Select MuServerGroup
          Into l_dblink
          From ManaUnit
         Where MuCode = p_mkt;
        ls_sql := 'select dbusrmkt.fgetstock@' || l_dblink ||
                  '(:1,:2,'''','''') from dual';
        Execute Immediate ls_sql
          Into l_Kc
          Using l_Zpchr2, p_mkt;
        If l_Kc <= 0 Then
          Return 'N';
        End If;
      Exception
        When Others Then
          Return 'N';
      End;
    End If;

    Return 'Y';
  Exception
    When Others Then
      Return 'N';
  End;

  Procedure Sp_Salemsrule(p_Billno  In Number,
                          p_Transno In Number,
                          o_Retcode Out Number,
                          o_Retmsg  Out Varchar2) As
    Lispop       Char(1);
    l_Crmmkt     Varchar2(20); --CRM MKT
    l_Xssj       Char(5); --销售时间
    l_Totfqje    Number; --总可返券金额 根据行递减
    l_Valid_Fqje Number; --行可返券金额
    l_Totwxje    Number; --总无效金额 根据行递减
    l_Totlimitje Number; --合计 受限券金额
    l_Rowlimitje Number; --行 受限券金额
    l_Jesign     Number;
    Row_Fqrule   Tktrule%Rowtype; --小票明细的返券规则
    Row_Flrue    Poprule%Rowtype; --满赠规则
    Row_h        Sellhead%Rowtype;
    Row_d        Selldetail%Rowtype;
    l_i          Number; --规则促销小票行数
    l_Exists     Number;
    --lisgk        NUMBER;
    Llever   Number;
    Lmje     Number;
    Lfqje    Number;
    Lmk      Number;
    l_Daynum Number;
    l_Tpnum  Number;
    Row_Card Cardmain%Rowtype;
    ----
    --lrownum          NUMBER;
    l_Act_Invalid_Je Number; --实际无效金额
    l_Temp1          Number; --
    --l_rowkc   NUMBER; --行扣除
    l_Flflag2 Char(1); --满赠 付款方式控制参数
    --返券有效金额分摊方法
    --1返券力度降序分摊有效金额 2行号降序分摊无效金额(卓展需求)
    l_Fqvalidjeft Char(1) := Trim(Fgetmanapara('FQ_VALIDJEFT'));
    l_Overnight   Date; --通宵营业时间
    b_Ismf        Char(1) := 'Y';
    b_Ispp        Char(1) := 'Y';
    b_Isgd        Char(1) := 'Y';
    l_Custid      Varchar2(60);
    l_Type        Number := 1;
    l_Flag        Boolean := False;
    l_Seq         Number;
    l_Flrule      Varchar2(60);
    l_Tpiid       Varchar2(10);
    Cursor Cur_d Is
      Select *
        From Selldetail
       Where Sdbillno = p_Billno
       Order By Decode(l_Fqvalidjeft, '2', Sdrowno, 1) Desc,
                Sdflrule Desc,
                Sdrowno;
    --
    Row_Tsfd Tktsellfldetail%Rowtype;
    --
    r_Limit Mztplimit%Rowtype;
    b_Limit Boolean := False;
  Begin
    Select Mgtltype
      Into l_Type
      From Mzgifttranslist
     Where Mgtltransno = p_Transno
       And Rownum = 1;

    --查找小票信息
    Begin
      Select * Into Row_h From Sellhead Where Shbillno = p_Billno;
    Exception
      When No_Data_Found Then
        o_Retcode := 1;
        o_Retmsg  := '未找到该小票!||小票序列号:' || To_Char(p_Billno);
        Return;
    End;

    Begin
      Select Tpiid
        Into l_Tpiid
        From Tktpopinfo, Billmktdetail
       Where Tpiid = Bmdbillno
         And Bmdbillid = 'RRS'
         And Bmdmkt = Row_h.Shmarket
         And Tpienddate >= Trunc(Row_h.Shdate)
         And Tpistartdate <= Trunc(Row_h.Shdate);
    Exception
      When No_Data_Found Then
        o_Retcode := 1;
        o_Retmsg  := '未找到该小票档期!||小票序列号:' || To_Char(p_Billno);
        Return;
    End;
    --销售、红冲销售才返券
    -- 红冲销售走扣回 salethhc 过程 20090417
    If Fvirtualsaletype(Row_h.Shdjlb) <> '1' Then
      o_Retcode := 3;
      o_Retmsg  := '不支持的小票类型!小票序列号:' || To_Char(p_Billno) || ', 小票类别:' ||
                   Fvirtualsaletype(Row_h.Shdjlb);
      Return;
    End If;
    --换货小票
    If Row_h.Shishh = 'Y' Then
      o_Retcode := 3;
      o_Retmsg  := '此为换货小票!小票序列号:' || To_Char(p_Billno) || ', 小票类别:' ||
                   Fvirtualsaletype(Row_h.Shdjlb);
      Return;
    End If;

    l_Crmmkt := Fgetcrmmkt(Row_h.Shmanaunit, Row_h.Shmarket);
    l_Jesign := 1;
    For Vr In (Select Rowid, t.*
                 From Sellpayment t
                Where Spbillno = p_Billno) Loop
      Lispop := Least(Salepopms.Fgetispoppay(Row_h.Shmanaunit,
                                             l_Crmmkt,
                                             Vr.Sppmcode,
                                             Vr.Sppayerid),
                      Salepopms.Fgetisczkpop(Row_h.Shmanaunit,
                                             Vr.Sppmcode,
                                             Vr.Spno));
      Update Sellpayment Set Spispop = Lispop Where Rowid = Vr.Rowid;
    End Loop;
    Update Sellpaygoods T1
       Set Spgispop =
           (Select Spispop
              From Sellpayment
             Where Spbillno = T1.Spgbillno
               And Sprowno = T1.Spgrowno)
     Where Spgbillno = p_Billno;
    --计算小票中不允许返券的付款金额
    If l_Fqvalidjeft = '2' Then
      Select Round(Nvl(Sum(Spggdmoney * Spgerate), 0), 2)
        Into l_Totlimitje --合计 受限付款金额
        From Sellpaygoods
       Where Spgbillno = p_Billno
         And Spgispop = 'N';
    End If;
    --
    Select Round(Nvl(Sum(Spmoney * Sperate), 0), 2)
      Into l_Temp1 --无效金额
      From Sellpayment
     Where Spbillno = p_Billno
       And Spispop = 'N';
    --
    l_Act_Invalid_Je := l_Temp1 - Row_h.Shsysy;
    If l_Act_Invalid_Je < 0 Then
      l_Act_Invalid_Je := 0;
    End If;
    l_Totfqje := Row_h.Shoughtpay - l_Act_Invalid_Je; --总可返券金额
    l_Totwxje := l_Act_Invalid_Je;
    --扣除 指定行受限
    l_Totwxje := l_Totwxje - l_Totlimitje;
    If l_Totwxje < 0 Then
      l_Totwxje := 0;
    End If;
    -- 考虑余额累计 就低 此时不RETURN 2009.04.16
    -- 小票当天返券余额累计
    l_Overnight := Fgetbusinessdate(Row_h.Shdate);

    --销售时间
    Select To_Char(Row_h.Shdate, 'HH24:MI') Into l_Xssj From Dual;
    --根据会员卡号找到会员账号
    If Nvl(Row_h.Shcustno, '#') <> '#' Then
      Select Cdmcid
        Into l_Custid
        From Cardmain
       Where Cdmno = Row_h.Shcustno;
    Else
      l_Custid := '#'; --若是未刷卡的小票记#
    End If;
    --找出小票中所有可以返券的商品
    l_i := 0;
    Open Cur_d;
    Loop
      Fetch Cur_d
        Into Row_d;
      Exit When Cur_d%Notfound Or Cur_d%Notfound Is Null;
      l_Flag := False;
      --档期表 tktpopinfo
      --档期规则 tktruledet trrulelb->规则类型1:返券 3:追券 4:买券 B:满赠 C:满减
      --确认商品是否参与返券满赠,前台已获取
      --
      --行受限付款金额
      l_Rowlimitje := Salepopms.Fgetrowlimitje(Row_d.Sdbillno,
                                               Row_d.Sdrowno);
      --0000 打折||满减||返券||满赠
      If l_Type = 1 Then
        If Substr(Trim(Row_d.Sdpoprule), 5, 1) <> '0' Then
          l_Flag   := True;
          l_Flrule := Row_d.Sdflrule;
        End If;
      End If;
      If l_Type = 2 Then
        --查找追送设置单
        Select Nvl(Max(Sriseqno), 0)
          Into l_Seq
          From Saruleinfo
         Where Sriksrq <= Trunc(Row_h.Shdate)
           And Srijsrq >= Trunc(Row_h.Shdate)
           And Srtype = 'A'
           And Srijygs = Row_h.Shmanaunit --经营公司
           And Srimkt = Row_h.Shmarket
           And ((Srigdid = Row_d.Sdgdid And
               (Srimfid = Row_d.Sdmfid Or Srimfid = 'ALL') And
               Srimode = '1') Or
               ((Srimfid = Row_d.Sdmfid Or Srimfid = 'ALL') And
               (Srippcode = Row_d.Sdppcode Or Srippcode = 'ALL') And
               (Sricatid = Row_d.Sdcatid Or Sricatid = 'ALL') And
               (Srispsx = Row_d.Sdspsx Or Srispsx = '0' Or
               Srispsx Is Null) And Srimode = '2'));
        If l_Seq > 0 Then
          l_Flag := True;
          Select Sriruleid
            Into l_Flrule
            From Saruleinfo
           Where Sriseqno = l_Seq;
        End If;
      End If;

      If l_Flag Then
        Row_Fqrule := Null;
        --popcurinfo popzpbarcode='DMJ' 满赠规则 popbillno
        l_Flflag2 := '#';
        Row_Flrue := Null;
        If l_Flag Then
          Begin
            Select *
              Into Row_Flrue
              From Poprule
             Where Prid = l_Flrule
               And Prmode = 'B';
            l_Flflag2 := Row_Flrue.Prflag2;
            b_Limit   := False;
            Begin
              Select *
                Into r_Limit
                From Mztplimit
               Where Mtltpid = l_Tpiid
                 And Mtlruleno = l_Flrule
                 And Mtlcustid =
                     Decode(Nvl(Row_Flrue.Pstr1, 'N'), 'N', '#', l_Custid);
              If Row_Flrue.Pstr1 = 'Y' Then
                If Nvl(Row_h.Shcustno, '#') = '#' Then
                  /*b_Limit := False; --如果是按会员限次，对于未刷卡的如何限制？*/
                  o_Retcode := 1;
                  o_Retmsg  := '满赠规则[' || l_Flrule || ']非会员不参于！小票序列号:' ||
                               To_Char(p_Billno);
                  Return;
                Else
                  If (Row_Flrue.Pnum1 <> 0 And
                     Row_Flrue.Pnum1 <= r_Limit.Mtldaynum) Or
                     (Row_Flrue.Pnum3 <> 0 And
                     Row_Flrue.Pnum3 <= r_Limit.Mtltpnum) Then
                    b_Limit := True;
                  End If;
                End If;
              Else
                Select Sum(Nvl(Mtldaynum, 0)), Sum(Nvl(Mtltpnum, 0))
                  Into l_Daynum, l_Tpnum
                  From Mztplimit
                 Where Mtltpid = l_Tpiid
                   And Mtlruleno = l_Flrule;
                If (Row_Flrue.Pnum1 <> 0 And Row_Flrue.Pnum1 <= l_Daynum) Or
                   (Row_Flrue.Pnum3 <> 0 And Row_Flrue.Pnum3 <= l_Tpnum) Then
                  b_Limit := True;
                End If;
              End If;
            Exception
              When Others Then
                Insert Into Mztplimit
                  Select l_Tpiid,
                         l_Flrule,
                         0,
                         0,
                         Row_Flrue.Pnum1,
                         Row_Flrue.Pnum3,
                         Decode(Nvl(Row_Flrue.Pstr1, 'N'),
                                'N',
                                '#',
                                l_Custid),
                         p_Transno
                    From Dual;
            End;
          Exception
            When Others Then
              o_Retcode := 1;
              o_Retmsg  := '满赠规则[' || l_Flrule || ']不存在!小票序列号:' ||
                           To_Char(p_Billno);
              Return;
          End;
          --row_flrue.prflag1
          --相同赠送条件的赠品是全送或任选其一
          --row_flrue.prflag2
          --有限制的付款方式或参加了满减活动就整单都不能送
          --1.按整笔销售金额赠送
          --2.按受限付款方式外的金额赠送
          --3.有受限付款方式则整笔不赠送
        End If;
        --
        If b_Limit Then
          o_Retcode := 1;
          o_Retmsg  := '已达到最大限送次数，不能继续赠送!小票序列号:' || To_Char(p_Billno);
          Return;
        End If;
        If Not b_Limit And Nvl(Row_Flrue.Prflag3, 'N') = 'N' Then
          If l_Fqvalidjeft = '1' Then
            --
            l_Valid_Fqje := Least(l_Totfqje, Row_d.Sdcjje - l_Rowlimitje);
            l_Totfqje    := l_Totfqje - l_Valid_Fqje;
            If l_Totfqje < 0 Then
              l_Totfqje := 0;
            End If;
          Elsif l_Fqvalidjeft = '2' Then
            l_Temp1      := Least(l_Totwxje, Row_d.Sdcjje - l_Rowlimitje);
            l_Valid_Fqje := Row_d.Sdcjje - l_Temp1 - l_Rowlimitje;
            l_Totwxje    := l_Totwxje - l_Temp1;
            If l_Totwxje < 0 Then
              l_Totwxje := 0;
            End If;
          End If;
          If Least(Nvl(Row_Flrue.Pnum2, 0)) <> 0 Then
            l_Valid_Fqje := Least(Row_Flrue.Pnum2 * Row_d.Sdsl,
                                  l_Valid_Fqje);
          End If;
          Row_Tsfd              := Null;
          Row_Tsfd.Tsfdbillno   := Row_d.Sdbillno; --小票billno
          Row_Tsfd.Tsfdrowno    := Row_d.Sdrowno; --小票商品行号
          Row_Tsfd.Tsfdpgmfid   := Row_d.Sdmfid; --商品柜组
          Row_Tsfd.Tsfdpggdid   := Row_d.Sdgdid; --商品编码
          Row_Tsfd.Tsfdcjje     := Row_d.Sdcjje * l_Jesign; --成交金额
          Row_Tsfd.Tsfdpid      := l_Tpiid; --Row_d.Sdnum1; --档期
          Row_Tsfd.Tsfdfqrule   := Nvl(Row_Fqrule.Trcode, '0'); --返券规则
          Row_Tsfd.Tsfdfqxfje   := Nvl(Row_Fqrule.Trxfje, 0); --返券规则满金额
          Row_Tsfd.Tsfdfqje     := Nvl(Row_Fqrule.Trfaqje, 0); --返券规则返总金额
          Row_Tsfd.Tsfdisjd     := Nvl(Trim(Row_Fqrule.Trstr2), '1'); --返券规则是否就低 1/0
          Row_Tsfd.Tsfdpayje    := l_Valid_Fqje * l_Jesign; --行可返券金额
          Row_Tsfd.Tsfdsyjid    := Row_h.Shsyjid; --收银机
          Row_Tsfd.Tsfdinvno    := Row_h.Shinvno; --发票号
          Row_Tsfd.Tsfdno       := Row_h.Shcustno; --sellhead.shcustno 会员卡号
          Row_Tsfd.Tsfdtransno  := p_Transno; --返券事务号
          Row_Tsfd.Tsfdyfqje    := 0; --应返券总金额
          Row_Tsfd.Tsfdrfqye1   := l_Valid_Fqje * l_Jesign; --行余额1  当前行计算
          Row_Tsfd.Tsfdrfqye2   := l_Valid_Fqje * l_Jesign; --行余额2  实际余额
          Row_Tsfd.Tsfdmaxsbqje := 0; --row_d.sdn2; --最大收券额
          Row_Tsfd.Tsfdftqtfkkc := 0; --分摊其他付款扣除
          Row_Tsfd.Tsfdptransno := 0;
          Begin
            Select Pgmsummf, Pgmsumpp, Pgmsumgd
              Into b_Ismf, b_Ispp, b_Isgd
              From Popgroupmode
             Where Pgmcode = Row_Flrue.Prflag4;
          Exception
            When Others Then
              Null;
          End;
          If b_Ismf = 'Y' Then
            Row_Tsfd.Tsfdmfid := '#';
          Else
            Row_Tsfd.Tsfdmfid := Row_Tsfd.Tsfdpgmfid;
          End If;
          If b_Ispp = 'Y' Then
            Row_Tsfd.Tsfdmfid := Row_Tsfd.Tsfdmfid || ',#';
          Else
            Row_Tsfd.Tsfdmfid := Row_Tsfd.Tsfdmfid || ',' || Row_d.Sdppcode;
          End If;
          If b_Isgd = 'Y' Then
            Row_Tsfd.Tsfdmfid := Row_Tsfd.Tsfdmfid || ',#';
          Else
            Row_Tsfd.Tsfdmfid := Row_Tsfd.Tsfdmfid || ',' || Row_d.Sdgdid;
          End If;

          -- 满赠用------------> start
          --row_flrue.prflag1
          --相同赠送条件的赠品是全送或任选其一
          --row_flrue.prflag2
          --有限制的付款方式或参加了满减活动就整单都不能送
          --1.按整笔销售金额赠送
          --2.按受限付款方式外的金额赠送
          --3.有受限付款方式则整笔不赠送
          Select Decode(l_Flflag2, '#', Null, l_Flrule),
                 --满赠规则(返礼)
                 --l_jesign * decode(l_flflag2, '#', 0, '1', 0, l_rowkc), --满赠_实扣金额
                 l_Jesign *
                 Decode(l_Flflag2,
                        '#',
                        0,
                        '1',
                        Row_d.Sdcjje,
                        '2',
                        l_Valid_Fqje,
                        '3',
                        Decode(Sign(l_Act_Invalid_Je), 0, Row_d.Sdcjje, 0)) --满赠有效金额
            Into Row_Tsfd.Tsfdflrule,
                 --满赠规则(返礼)
                 --row_tsfd.tsfdflskje, --满赠_实扣金额
                 Row_Tsfd.Tsfdflyxje --满赠有效金额
            From Dual;
            --[[[[EORV-767 单件商品最大参加活动金额 =  =P~ 2017.11.29
            Select Least(Row_Tsfd.Tsfdflyxje,
                         Decode(Row_Flrue.Pnum2,
                                0,
                                Row_Tsfd.Tsfdflyxje,
                                Row_Flrue.Pnum2))
              Into Row_Tsfd.Tsfdflyxje
              From Dual;
            --]]]]-------------------------------------------------------
          Row_Tsfd.Tsfdflskje := Row_d.Sdcjje - Row_Tsfd.Tsfdflyxje;
          --
          Row_Tsfd.Tsfdflflag1 := Nvl(Row_Flrue.Prflag1, 'N'); --相同赠送条件的赠品
          Row_Tsfd.Tsfdflflag2 := Nvl(Row_Flrue.Prflag2, 'N'); --受限付款方式控制
          -- 满赠用<----------- end
          Row_Tsfd.Tsfdmarket   := Row_h.Shmarket; --门店
          Row_Tsfd.Tsfdmu       := Row_h.Shmanaunit; --经营公司
          Row_Tsfd.Tsfdflflag   := 'N'; --是否已满赠
          Row_Tsfd.Tsfdnum1     := 0;
          Row_Tsfd.Tsfdnum2     := 0;
          Row_Tsfd.Tsfdnum3     := 0;
          Row_Tsfd.Tsfdchr1     := Fgetmanapara('Q2'); --返券模式,对应manapara[q2]参数
          Row_Tsfd.Tsfdchr2     := l_Custid;
          Row_Tsfd.Tsfdchr3     := '';
          Row_Tsfd.Tsfdbcje     := l_Valid_Fqje * l_Jesign; --本次可返券金额
          Row_Tsfd.Tsfdfqye     := l_Valid_Fqje * l_Jesign; --小票返券余额
          Row_Tsfd.Tsfdcbillno  := Row_d.Sdbillno; --当前小票号
          Row_Tsfd.Tsfdcrowno   := Row_d.Sdrowno; --当前行号
          Row_Tsfd.Tsfddate     := l_Overnight; --小票日期 trunc(sellhead.shdate)
          Row_Tsfd.Tsfdyefqrule := Nvl(Row_Fqrule.Trcode, '0'); --余额返券规则
          Row_Tsfd.Tsfdpriority := 0; -- 返券力度优先级 降序
          --插入到小票返券明细表
          Insert Into Tktsellfldetail Values Row_Tsfd;
          If l_Fqvalidjeft = '1' And l_Totfqje = 0 Then
            Exit; --有效付款金额已分摊完
          End If;
        End If;
      Else
        If l_Fqvalidjeft = '1' Then
          --
          l_Valid_Fqje := Least(l_Totfqje, Row_d.Sdcjje - l_Rowlimitje);
          l_Totfqje    := l_Totfqje - Row_d.Sdcjje;
          If l_Totfqje < 0 Then
            l_Totfqje := 0;
          End If;
        Elsif l_Fqvalidjeft = '2' Then
          --l_temp1 := least(l_totwxje, row_d.sdcjje - l_rowlimitje);
          l_Temp1      := Row_d.Sdcjje - l_Rowlimitje;
          l_Valid_Fqje := Row_d.Sdcjje - l_Temp1 - l_Rowlimitje;
          l_Totwxje    := l_Totwxje - l_Temp1;
          If l_Totwxje < 0 Then
            l_Totwxje := 0;
          End If;
        End If;
        --
      End If;
    End Loop;
    Close Cur_d;

    l_Exists := Sql%Rowcount;
    Dbms_Output.Put_Line(l_Exists);
    --
    <<sp_Salemsruleskip>>
    o_Retcode := 0;
    o_Retmsg  := '执行成功';
  Exception
    When Others Then
      If Cur_d%Isopen Then
        Close Cur_d;
      End If;
      Rollback;
      o_Retcode := -1;
      o_Retmsg  := '执行失败,小票序列号:' || To_Char(p_Billno) || ',' ||
                   To_Char(Sqlcode) || ',' || Sqlerrm;
  End;

  Procedure Mzrulecale(p_Transno In Number,
                       o_Retcode Out Number,
                       o_Retmsg  Out Varchar2) Is
    l_Shmu  Varchar2(20);

  Procedure Calmzlist(p_Transno In Number) Is
    r_Poprule    Poprule%Rowtype;
    l_Mje        Number; --满金额
    l_Yzsnum     Number; --已赠送数量
    l_Djxe       Number; --单件限额
    l_Zdxe       Number; --整单限额
    r_Mzgiftlist Mzgiftlist%Rowtype;
    l_Maxje      Number;
    l_Maxnum     Number;
    l_Ljye       Number; --累计余额
    l_Yxje       Number;
    r_Mzgifttemp Mzgifttemp%Rowtype;
    l_Level      Number;
  Begin
    For Vr In (Select Tsfdmfid,
                      Tsfdflrule,
                      Tsfdpid,
                      Tsfdmarket,
                      Tsfdmu,
                      Tsfdchr2 Custid, --会员账号
                      Sum(Tsfdcjje) Cjje,
                      Sum(Tsfdflyxje) Yxje
                 From Tktsellfldetail
                Where Tsfdtransno = p_Transno
                  And Tsfdcrowno > 0
                  And Tsfdflrule Is Not Null --过滤掉 无返礼规则的记录
                Group By Tsfdmfid,
                         Tsfdflrule,
                         Tsfdpid,
                         Tsfdmarket,
                         Tsfdmu,
                         Tsfdchr2) Loop
      Begin
        Select * Into r_Poprule From Poprule Where Prid = Vr.Tsfdflrule;
      Exception
        When Others Then
          Raise_Application_Error(-20000,
                                  '执行calmzlist失败,' || To_Char(Sqlcode) || ',' ||
                                  Sqlerrm);
      End;
      Select Max(Prdrowno)
        Into l_Level
        From Poprulelevel
       Where Prdid = Vr.Tsfdflrule
         And Prlxfje <= Vr.Yxje;
      If l_Level Is Not Null Then
        l_Yzsnum := 0;
        l_Ljye   := 0;
        For r_Ruledet In (Select *
                            From Popruledet
                           Where Prdid = Vr.Tsfdflrule
                             And
                                --Prdlevel = l_Level
                                --列出低级别的赠品供选择
                                 Prdlevel <= l_Level
                           Order By Prdlevel) Loop
          If Checklpisable(r_Ruledet.Prdrcode, vr.Tsfdmarket) = 'N' Then
            Goto Nextzp;
          End If;
          r_Mzgifttemp.Msttransno   := p_Transno;
          r_Mzgifttemp.Mstflrule    := Vr.Tsfdflrule;
          r_Mzgifttemp.Mstflrowno   := r_Ruledet.Prdrowno;

  Procedure Approve(p_Billno In Char,
                    p_Person In Char,
                    p_Billid In Char,
                    p_Djlb   In Char) Is
  Begin
    Bapprove(p_Billno, p_Person, p_Billid, p_Djlb);
    Commit;

    When Others Then
      Rollback;
      Raise;
  End;
  Procedure Bapprove(p_Billno In Varchar2,
                     p_Person In Varchar2,
                     p_Billid In Varchar2,
                     p_Djlb   In Varchar2) Is
    r_h       Bchangezphead%Rowtype;
    r_d_Old   Bchangezpdetail%Rowtype;
    r_d_New   Bchangezpdetail%Rowtype;
    r_Zpjxc   Crmzpjxclist%Rowtype;
    l_Transno Number;
    r_Zpbm    Crmzpbm%Rowtype;
  Begin
    Begin
      Select *
        Into r_h
        From Bchangezphead
       Where Bczhbillno = p_Billno
         For Update;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode, p_Billno || '->1 单据不存在');
    End;
    If r_h.Bczhdjlb <> '1' Then
      Raise_Application_Error(Errcode, p_Billno || '->2.1 单据类别不一致');
    End If;
    If p_Billid <> 'QS7' Then
      Raise_Application_Error(Errcode, p_Billno || '->2.2 单据编号不一致');
    End If;
    If r_h.Bczhflag <> 'N' Then
      Raise_Application_Error(Errcode,
                              p_Billno || '->2.3 单据已审核，不能修改！');
    End If;
    If Trim(r_h.Bczhpid) Is Null Then
      Raise_Application_Error(Errcode, p_Billno || '->3 请指定档期！');
    End If;
    Select Seq_Popms.Nextval Into l_Transno From Dual;
    --增加原库存
    For v_r_o In (Select *
                    From Bchangezpdetail
                   Where Bczdbillno = p_Billno
                     And Bczdtype = 'O') Loop
      Crmzpjxc.p_Init(r_Zpjxc);
      Select * Into r_Zpbm From Crmzpbm Where Zpbm = v_r_o.Bczdgdid;
      r_Zpjxc.Czjgdid   := r_Zpbm.Zpbm; --VARCHAR2(20) N     商品编码
      r_Zpjxc.Czjtrans  := 'C'; --CHAR(1)  N     事务
      r_Zpjxc.Czjqckcsl := Nvl(r_Zpbm.Zpsl, 0); --NUMBER(16,4)  Y
      r_Zpjxc.Czjsl     := v_r_o.Bczdsl; --NUMBER(16,4) Y
      r_Zpjxc.Czjqmkcsl := Nvl(r_Zpbm.Zpsl, 0) + v_r_o.Bczdsl; -- NUMBER(16,4)  Y
      r_Zpjxc.Czjhsjjje := 1 * r_Zpbm.Zpgmje; --  NUMBER(16,4)  Y
      r_Zpjxc.Czjvc1    := To_Char(l_Transno);
      r_Zpjxc.Czjvc2    := l_Transno;
      r_Zpjxc.Czjvc4    := '【满赠换货处理】库存数量 ' || To_Char(r_Zpjxc.Czjqckcsl) ||
                           '=> ' || To_Char(r_Zpjxc.Czjqmkcsl);
      /*dbusrvip.*/
      r_zpjxc.czjfsmkt   := r_h.bczhmkt;
      r_zpjxc.czjvc3     := r_Zpbm.Zpchr2;
      r_zpjxc.czjerptran := 'D'; --D 礼品退回;
      Crmzpjxc.p_Ins(r_Zpjxc);
    End Loop;
    --减新库存
    For v_r_n In (Select *
                    From Bchangezpdetail
                   Where Bczdbillno = p_Billno
                     And Bczdtype = 'N') Loop
      Crmzpjxc.p_Init(r_Zpjxc);
      Select * Into r_Zpbm From Crmzpbm Where Zpbm = v_r_n.Bczdgdid;
      r_Zpjxc.Czjgdid   := r_Zpbm.Zpbm; --VARCHAR2(20) N     商品编码
      r_Zpjxc.Czjtrans  := 'C'; --CHAR(1)  N     事务
      r_Zpjxc.Czjqckcsl := Nvl(r_Zpbm.Zpsl, 0); --NUMBER(16,4)  Y
      r_Zpjxc.Czjsl     := -1 * v_r_n.Bczdsl; --NUMBER(16,4) Y
      r_Zpjxc.Czjqmkcsl := Nvl(r_Zpbm.Zpsl, 0) + -1 * v_r_n.Bczdsl; -- NUMBER(16,4)  Y
      r_Zpjxc.Czjhsjjje := 1 * r_Zpbm.Zpgmje; --  NUMBER(16,4)  Y
      r_Zpjxc.Czjvc1    := To_Char(l_Transno);
      r_Zpjxc.Czjvc2    := l_Transno;
      r_Zpjxc.Czjvc4    := '【满赠换货处理】库存数量 ' || To_Char(r_Zpjxc.Czjqckcsl) ||
                           '=> ' || To_Char(r_Zpjxc.Czjqmkcsl);
      /*dbusrvip.*/
      r_zpjxc.czjfsmkt   := r_h.bczhmkt;
      r_zpjxc.czjvc3     := r_Zpbm.Zpchr2;
      r_zpjxc.czjerptran := 'A'; --A 礼品发放;
      Crmzpjxc.p_Ins(r_Zpjxc);
    End Loop;
    Update Bchangezphead Set Bczhflag = 'Y' Where Bczhbillno = p_Billno;
  End;

Begin
  Null;
End Saleflms;
