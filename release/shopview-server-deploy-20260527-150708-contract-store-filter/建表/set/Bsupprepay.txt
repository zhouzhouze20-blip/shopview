CREATE OR REPLACE Package Body Bsupprepay Is
  Currentgrg  Varchar2(20);
  Currentmkt  Varchar2(20);
  Currentdate Date;

  Procedure Genprepaybycont(i_Date In Date, i_Mkt In Varchar2) Is
    --根据合同中定义的结算位置生成预付款单
    Cursor c_Contprepay Is
      Select Contprepay.*,
             Cmjsmkt,
             Cmsupid,
             Cmwmid,
             Cmchar1,
             Cmchar2,
             Sbtaxno
        From Contprepay, Contmain, Supplierbase
       Where Cppcontno = Cmcontno
         And Cmsupid = Sbid
         And Cppmkt = i_Mkt
         And Cppeffdate <= i_Date
         And Cpplapdate >= i_Date
         And Cpppayday =
             Substr(Cpppayday, 1, 2) || To_Char(i_Date - 1, 'dd')
         And Cppflag = 'Y'
         And Cmstatus In ('Y', 'Q')
         And Cmeffdate < i_Date
         And Sbstatus = 'Y'
         And Sbflag In ('0', '1', '2')
      /*AND cppmkt IN (SELECT mucode FROM manaunit WHERE muservergroup = CurrentGrg AND mustatus = 'Y')*/
      ;
    Row_Cpp     c_Contprepay%Rowtype;
    Lxssr       Number;
    Lxscb       Number;
    Lxstc       Number;
    Row_Prepay  Supprepay%Rowtype;
    Lbillno     Supprepay.Sppbillno%Type;
    Ldjbh       Supprepay.Sppdjbh%Type;
    Lmktservgrp Varchar2(20);
  Begin
    Open c_Contprepay;
    Loop
      Fetch c_Contprepay
        Into Row_Cpp;
      Exit When c_Contprepay%Notfound Or c_Contprepay%Notfound Is Null;
      Begin
        Select Muservergroup
          Into Lmktservgrp
          From Manaunit
         Where Mucode = Row_Cpp.Cppmkt
           And Mustatus = 'Y';
      Exception
        When Others Then
          Raise_Application_Error(Errcode,
                                  '合同中定义预付款门店[' || Row_Cpp.Cppmkt ||
                                  ']没有有效的门店定义，请检查！');
      End;
      --当前位置为结算位置时生成预付款单
      If (Currentgrg = 'HQ' And Row_Cpp.Cmjsmkt = '0') Or
         (Currentgrg = Lmktservgrp And Row_Cpp.Cmjsmkt <> '0') Then
        If Fgetsysplace Is Null Then
          If Currentgrg = 'HQ' Then
            Fterp_Oper.Set_Oper_Market('0');
          Else
            Fterp_Oper.Set_Oper_Market(Row_Cpp.Cppmkt);
          End If;
        End If;
        Select Round(Nvl(Sum((Xssr + Pfsr)), 0), 2) Sr, --销售收入
               Round(Nvl(Sum((Xsjjml + Pfjjml)), 0), 2) Tc, --销售提成
               Round(Nvl(Sum(Xshsjjje + Xshsjjad + Pfhsjjje + Pfhsjjad), 0),
                     2) Cb --销售成本
          Into Lxssr, Lxstc, Lxscb
          From Salecostday
         Where Scdmkt = Row_Cpp.Cppmkt
           And Scdmfid = Row_Cpp.Cppmfid
           And Scdsupid = Row_Cpp.Cmsupid
           And Scdwmid = Row_Cpp.Cmwmid
           And Scdhsrq >= To_Date(To_Char(Currentdate, 'yyyymm') ||
                                  Substr(Row_Cpp.Cpppayday, 1, 2),
                                  'yyyymmdd')
           And Scdhsrq <= To_Date(To_Char(Currentdate, 'yyyymm') ||
                                  Substr(Row_Cpp.Cpppayday, -2),
                                  'yyyymmdd');
        If Lxscb > 0 And Row_Cpp.Cpprate > 0 Then
          Sp_Billseq('PRE', Lbillno, 'N');
          Sp_Billnum('PRE', Lbillno, Ldjbh, 'N');
          If Trim(Lbillno) Is Null Then
            Raise_Application_Error(Errcode, '生成预付款单号失败');
          End If;
          Row_Prepay               := Null;
          Row_Prepay.Sppbillno     := Lbillno;
          Row_Prepay.Sppdjbh       := Ldjbh;
          Row_Prepay.Sppsource     := Source_Cont;
          Row_Prepay.Sppstatus     := Flagauto;
          Row_Prepay.Sppsupid      := Row_Cpp.Cmsupid;
          Row_Prepay.Sppwmid       := Row_Cpp.Cmwmid;
          Row_Prepay.Sppcontno     := Row_Cpp.Cppcontno;
          Row_Prepay.Sppmkt        := Row_Cpp.Cppmkt;
          Row_Prepay.Sppmfid       := Row_Cpp.Cppmfid;
          Row_Prepay.Sppsdate      := To_Date(To_Char(Currentdate, 'yyyymm') ||
                                              Substr(Row_Cpp.Cpppayday,
                                                     1,
                                                     2),
                                              'yyyymmdd');
          Row_Prepay.Sppedate      := To_Date(To_Char(Currentdate, 'yyyymm') ||
                                              Substr(Row_Cpp.Cpppayday, -2),
                                              'yyyymmdd');
          Row_Prepay.Spprate       := Row_Cpp.Cpprate;
          Row_Prepay.Sppsaleamt    := Lxssr;
          Row_Prepay.Sppsalegp     := Lxstc;
          Row_Prepay.Sppsalecost   := Lxscb;
          Row_Prepay.Sppsalepay    := Round(Lxscb * Row_Cpp.Cpprate, 2);
          Row_Prepay.Sppactpay     := Round(Lxscb * Row_Cpp.Cpprate, 2);
          Row_Prepay.Spppaychn     := Fmoney2chn(Row_Prepay.Sppactpay);
          Row_Prepay.Sppsettleno   := Null;
          Row_Prepay.Spppayno      := Null;
          Row_Prepay.Sppbank       := Row_Cpp.Cmchar1;
          Row_Prepay.Sppaccntno    := Row_Cpp.Cmchar2;
          Row_Prepay.Spptaxno      := Row_Cpp.Sbtaxno;
          Row_Prepay.Sppmktbank    := Fgetprivateinfo('03');
          Row_Prepay.Sppmktaccntno := Fgetprivateinfo('04');
          Row_Prepay.Sppmkttaxno   := Fgetprivateinfo('06');
          Row_Prepay.Inputor       := 'System';
          Row_Prepay.Inputdate     := Sysdate;
          Row_Prepay.Auditor       := Null;
          Row_Prepay.Auditdate     := Null;
          Row_Prepay.Memo          := Null;
          Row_Prepay.Spptype       := '1';
          Insert Into Supprepay Values Row_Prepay;
          If Sql%Rowcount <> 1 Then
            Raise_Application_Error(Errcode, '生成预付款单失败！');
          End If;
        End If;
      End If;
    End Loop;
    Close c_Contprepay;
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End;
  Procedure Sp_Gettcje(v_Contno In Varchar2,
                       v_Mfid   In Varchar2,
                       v_Mkt    In Varchar2,
                       o_Amount Out Number,
                       o_Xssr   Out Number,
                       o_Xscb   Out Number,
                       o_Xstc   Out Number,
                       o_Bl     Out Number) Is
    l_Sdate  Date;
    l_Edate  Date;
    r_Cont   Contmain%Rowtype;
    l_Tc     Number;
    l_Prepay Number;
  Begin
    Select * Into r_Cont From Contmain Where Cmcontno = v_Contno;
    l_Sdate := Nvl(r_Cont.Cmlastsetdate, r_Cont.Cmeffdate);
    If r_Cont.Cmlastsetdate Is Null Then
      l_Sdate := r_Cont.Cmeffdate;
    Else
      l_Sdate := r_Cont.Cmlastsetdate + 1;
    End If;
    If r_Cont.Cmlapdate < Trunc(Sysdate) Then
      l_Edate := r_Cont.Cmlapdate;
    Else
      l_Edate := Trunc(Sysdate - 1);
    End If;
    Begin
      Select Cpprate
        Into o_Bl
        From Contprepay
       Where Cppcontno = v_Contno
         And Cppmkt = v_Mkt
         And Cppmfid = v_Mfid
         And Cppeffdate <= Trunc(Sysdate - 1)
         And Cpplapdate >= Trunc(Sysdate - 1)
         And Rownum = 1;
    Exception
      When Others Then
        o_Bl := 1;
    End;
  
    Select Round(Nvl(Sum((Xssr + Pfsr)), 0), 2) Sr, --销售收入
           Round(Nvl(Sum((Xsjjml + Pfjjml)), 0), 2) Tc, --销售提成
           Round(Nvl(Sum(Xshsjjje + Xshsjjad + Pfhsjjje + Pfhsjjad), 0), 2) Cb --销售成本
      Into o_Xssr, o_Xstc, o_Xscb
      From Salecostday
     Where Nvl(Fgetcontno(Scdsupid, Scdwmid, Scddate, Scdmfid, Scdbrand),
               '0') = v_Contno
       And Scdmkt = v_Mkt
       And Scdmfid = v_Mfid
       And Scdhsrq >= l_Sdate
       And Scdhsrq <= Trunc(Sysdate - 1);
  
    Select Sum(Sppactpay)
      Into l_Prepay
      From Supprepay
     Where Sppstatus = 'Y'
       And Sppcontno = v_Contno
       And Sppmfid = v_Mfid;
    o_Amount := Nvl(o_Xscb, 0) - Nvl(l_Prepay, 0);
  Exception
    When Others Then
      Raise;
  End;
  Procedure Sp_Gettcje(v_Contno In Varchar2,
                       v_Mfid   In Varchar2,
                       v_Mkt    In Varchar2,
                       m_Sdate  In Date,
                       m_Edate  In Date,
                       o_Amount Out Number,
                       o_Xssr   Out Number,
                       o_Xscb   Out Number,
                       o_Xstc   Out Number,
                       o_Bl     Out Number) Is
    l_Sdate  Date;
    l_Edate  Date;
    r_Cont   Contmain%Rowtype;
    l_Tc     Number;
    l_Prepay Number;
  Begin
    Select * Into r_Cont From Contmain Where Cmcontno = v_Contno;
    l_Sdate := Nvl(r_Cont.Cmlastsetdate, r_Cont.Cmeffdate);
    If r_Cont.Cmlastsetdate Is Null Then
      l_Sdate := r_Cont.Cmeffdate;
    Else
      l_Sdate := r_Cont.Cmlastsetdate + 1;
    End If;
    If r_Cont.Cmlapdate < Trunc(Sysdate) Then
      l_Edate := r_Cont.Cmlapdate;
    Else
      l_Edate := Trunc(Sysdate - 1);
    End If;
    Begin
      Select Cpprate
        Into o_Bl
        From Contprepay
       Where Cppcontno = v_Contno
         And Cppmkt = v_Mkt
         And Cppmfid = v_Mfid
         And Cppeffdate <= Trunc(Sysdate - 1)
         And Cpplapdate >= Trunc(Sysdate - 1)
         And Rownum = 1;
    Exception
      When Others Then
        o_Bl := 1;
    End;
    if l_Sdate = m_Sdate and l_Edate = m_Edate then
      Select Round(Nvl(Sum((Xssr + Pfsr)), 0), 2) Sr, --销售收入
             Round(Nvl(Sum((Xsjjml + Pfjjml)), 0), 2) Tc, --销售提成
             Round(Nvl(Sum(Xshsjjje + Xshsjjad + Pfhsjjje + Pfhsjjad), 0),
                   2) Cb --销售成本
        Into o_Xssr, o_Xstc, o_Xscb
        From Salecostday
       Where Nvl(Fgetcontno(Scdsupid, Scdwmid, Scddate, Scdmfid, Scdbrand),
                 '0') = v_Contno
         And Scdmkt = v_Mkt
         And Scdmfid = v_Mfid
         And Scdhsrq >= l_Sdate
         And Scdhsrq <= Trunc(Sysdate - 1);
    else
      Select Round(Nvl(Sum((Xssr + Pfsr)), 0), 2) Sr, --销售收入
             Round(Nvl(Sum((Xsjjml + Pfjjml)), 0), 2) Tc, --销售提成
             Round(Nvl(Sum(Xshsjjje + Xshsjjad + Pfhsjjje + Pfhsjjad), 0),
                   2) Cb --销售成本
        Into o_Xssr, o_Xstc, o_Xscb
        From Salecostday
       Where Nvl(Fgetcontno(Scdsupid, Scdwmid, Scddate, Scdmfid, Scdbrand),
                 '0') = v_Contno
         And Scdmkt = v_Mkt
         And Scdmfid = v_Mfid
         And Scdhsrq >= m_Sdate
         And Scdhsrq <= m_Edate;
    end if;
    Select Sum(Sppactpay)
      Into l_Prepay
      From Supprepay
     Where Sppstatus = 'Y'
       And Sppcontno = v_Contno
       And Sppmfid = v_Mfid;
    o_Amount := Nvl(o_Xscb, 0) - Nvl(l_Prepay, 0);
  Exception
    When Others Then
      Raise;
  End;
  Procedure Approve(p_Billno In Char,
                    p_Person In Char,
                    p_Billid In Char,
                    p_Djlb   In Char) Is
    b_Hd       Supprepay%Rowtype;
    Row_Cont   Contmain%Rowtype;
    Lmsg       Varchar2(20);
    Lpaybillno Suppayhead.Sphbillno%Type;
    Ljsbillno  Supsettlehead.sshbillno%Type;
    row_sb     supplierbase%RowType;
    Lcount     Number;
    row_p      nc_temp_cashpay%Rowtype;
    Lseq       Number;
    lsupsub    varchar2(20);
    lmfsub     varchar2(20);
    lzt        varchar2(20);
  Begin
    Begin
      Select * Into b_Hd From Supprepay Where Sppbillno = p_Billno;
    Exception
      When No_Data_Found Then
        Raise_Application_Error(Errcode,
                                '预付款单[' || p_Billno || ']不存在，可能已被删除！');
    End;
    If Fgetsysplace <> '0' And Fgetsysplace <> b_Hd.Sppmkt Then
      Raise_Application_Error(Errcode,
                              '预付款单[' || p_Billno || ']审核操作地点错误！');
    End If;
    If b_Hd.Sppstatus Not In (Flagauto, Flagediting) Then
      Raise_Application_Error(Errcode,
                              '预付款单[' || p_Billno || ']已审核！');
    End If;
    Begin
      Select * Into Row_Cont From Contmain Where Cmcontno = b_Hd.Sppcontno;
    Exception
      When Others Then
        Raise_Application_Error(Errcode,
                                '合同主体[' || b_Hd.Sppcontno || ']不存在!');
    End;
    If Row_Cont.Cmstatus Not In ('Y', 'Q') Then
      Raise_Application_Error(Errcode,
                              '合同[' || b_Hd.Sppcontno || ']不是有效合同!');
    End If;
    /*If Row_Cont.Cmwmid = '1' Then
      Raise_Application_Error(Errcode,
                              '合同[' || b_Hd.Sppcontno || ']为经销合同，不允许支付预付款!');
    End If;*/
    If Row_Cont.Cmjsmkt <> Currentmkt Then
      If Row_Cont.Cmjsmkt = '0' Then
        Lmsg := '总部';
      Else
        Lmsg := '门店[' || Row_Cont.Cmjsmkt || ']';
      End If;
      Raise_Application_Error(Errcode,
                              '审核地点错误,合同[' || b_Hd.Sppcontno || ']的结算位置是' || Lmsg ||
                              '需要在结算位置进行审核!');
    End If;
    Begin
      Select Sshbillno
        Into Ljsbillno
        From Supsettlehead
       Where Sshpayflag <> Flagapproved
         And Sshsupid = b_Hd.Sppsupid
         And Sshcontno = b_Hd.Sppcontno
         And Sshsetje - sshtotkk - sshyfkje <> 0 --古玩城需求，为零的结算单不影响预付款
         And Rownum = 1;
      Raise_Application_Error(Errcode,
                              '->供应商[' || b_Hd.Sppsupid || ']在门店[' ||
                              b_Hd.Sppmkt || ']存在未付款的结算单单[' || Ljsbillno || ']!');
    Exception
      When No_Data_Found Then
        Null;
    End;
    --是否存在未付款的付款单
    Begin
      Select Sphbillno
        Into Lpaybillno
        From Suppayhead
       Where Sphflag <> Suppayment.Flagapproved
         And Sphsupid = b_Hd.Sppsupid
         And Sphmkt = b_Hd.Sppmkt
         And Rownum = 1;
      If Lpaybillno Is Not Null Then
        Raise_Application_Error(Errcode,
                                '->供应商[' || b_Hd.Sppsupid || ']在门店[' ||
                                b_Hd.Sppmkt || ']存在未付款的付款单[' || Lpaybillno ||
                                '],请先进行付款或取消!');
      End If;
    Exception
      When No_Data_Found Then
        Null;
    End;
  
    Update Supprepay
       Set Sppstatus = Flagapproved,
           Auditor   = p_Person,
           Auditdate = Currentdate,
           SPPYE     = DECODE(SPPTYPE, '1', 0, SPPACTPAY)
     Where Sppbillno = p_Billno;
  
    select seq_nc_temp_cashpay.nextval into lseq from dual;
    row_p.id := lseq;
  
    begin
      select mfsubject
        into lzt
        from manaframe
       where mfcode = b_Hd.Sppmkt
         and rownum = 1;
    exception
      when others then
        Raise_Application_Error(Errcode, '->08.门店找不到对应核算信息!');
    end;
    row_p.corp       := lzt;
    row_p.tradetype  := '0';
    row_p.settlecode := '01';
    row_p.billdate   := to_char(sysdate, 'yyyy-mm-dd');
    row_p.headid     := lseq;
    row_p.bodyid     := 1;
    select nvl(sbsubject, sbid)
      into lsupsub
      from supplierbase
     where sbid = b_Hd.Sppsupid;
    select * into row_sb from supplierbase where sbid = b_Hd.sppsupid;
    row_p.cvcode := lsupsub;
    /*select count(distinct pbmfid) into lcount from paybatch where pbpaybillno = p_billno;
    if lcount > 1 then
      Raise_Application_Error(Errcode, '->08.付款单不允许包含多个柜组数据!');
    end if;*/
    select nvl(mfsubject, mfcode)
      into lmfsub
      from manaframe
     where mfcode = b_Hd.Sppmfid;
    row_p.deptcode       := lmfsub;
    row_p.paybkaccount   := b_Hd.Sppmktaccntno;
    row_p.paybkname      := b_Hd.Sppmktbank;
    row_p.recebkaccount  := nvl(b_Hd.Sppaccntno, row_sb.sbaccntno);
    row_p.recebkname     := nvl(b_Hd.Sppbank, row_sb.sbbank);
    row_p.crc            := '人民币';
    row_p.rate           := 1;
    row_p.digest         := '付款';
    row_p.payoriginalmny := b_Hd.Sppactpay;
    row_p.paylocalmny    := b_Hd.Sppactpay;
    row_p.billflag       := '0';
    row_p.sourcebillcode := b_Hd.Sppbillno;
    row_p.iflag          := 'N';
    row_p.note           := '';
    row_p.ts             := to_char(sysdate, 'yyyy-mm-dd hh24:mi:ss');
    row_p.billcode       := null;
    row_p.prepaydate     := to_char(sysdate, 'yyyy-mm-dd');
    row_p.cashflowcode   := null;
    --insert into nc_temp_cashpay values row_p;     
  
    Supcamana.Yfkgen(p_Billno, 'PREPAY');
    Commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End;

  Procedure Settle_pre(p_SettleNo In Varchar2) Is
    l_TotPrePay Number;
    l_Ye        Number;
    row_Head    SupSettleHead%Rowtype;
    l_Totonepay Number;
  Begin
    Begin
      Select *
        Into row_Head
        From SupSettleHead
       Where SshBillNo = p_SettleNo;
    Exception
      When Others Then
        Raise;
    End;
    If Row_head.SshFlag <> 'M' Then
      Raise_Application_Error(ErrCode,
                              '结算单[' || p_SettleNo || ']已提交，不能核销预付款！');
    End If;
    Select nvl(Sum(Sppuhxmoney), 0) --未查到资料时的空值处理 =  =P~ 2015.12.09
      Into l_Totprepay
      From Supprepay_Usage
     Where Sppusetno = p_Settleno;
    Select Nvl(Sum(Sppactpay), 0)
      Into l_Totonepay
      From Supprepay
     Where Sppsettleno = p_Settleno
          -- And Sppstatus = Flagsettled
       And Spptype = '1';
    l_Totprepay := l_Totprepay + l_Totonepay;
    If Row_Head.Sshsjfkje + Row_Head.Sshyfkje - l_Totprepay < 0 Then
      Raise_Application_Error(Errcode,
                              '预付款核销金额超过了结算单[' || p_Settleno ||
                              ']的应付款金额，不能核销！');
    End If;
    Update SupSettleHead
       Set SshYfkJe  = l_TotPrePay,
           SshSjFkje = SshSjfkje + Row_Head.Sshyfkje - l_TotPrePay
     Where SshBillNo = p_SettleNo;
    COMMIT;
  Exception
    When Others Then
      Rollback;
      Raise;
  End Settle_pre;

  Procedure Settle_Hx(p_SettleNo In Varchar2) Is
    l_TotPrePay Number;
    l_Ye        Number;
    row_Head    SupSettleHead%Rowtype;
    l_Totonepay NUMBER;
  Begin
    Begin
      Select *
        Into row_Head
        From SupSettleHead
       Where SshBillNo = p_SettleNo;
    Exception
      When Others Then
        Raise;
    End;
    If Row_head.SshFlag <> 'Y' Then
      Raise_Application_Error(ErrCode,
                              '结算单[' || p_SettleNo || ']未审核，不能核销！');
    End If;
    If Row_head.SshPayFlag = 'Y' Then
      Raise_Application_Error(ErrCode,
                              '结算单[' || p_SettleNo || ']已付款，不能核销！');
    End If;
    Select Nvl(Sum(Sppactpay), 0)
      Into l_Totonepay
      From Supprepay
     Where Sppsettleno = p_Settleno
          -- And Sppstatus = Flagsettled
       And Spptype = '1';
    Select Sum(SPPUHxMoney)
      Into l_TotPrePay
      From SupPrePay_Usage
     Where SppuSetNo = p_SettleNo;
    l_Totprepay := l_Totprepay + l_Totonepay;
    If Row_Head.SshSjFkje + Row_Head.SshYfkje - l_TotPrePay < 0 Then
      Raise_Application_Error(ErrCode,
                              '预付款核销金额超过了结算单[' || p_SettleNo ||
                              ']的应付款金额，不能核销！');
    End If;
    /*
    If row_Head.SshYfkJe<>l_TotPrePay Then
       Raise_Application_Error(ErrCode,
                               '预付款核销金额与结算单[' || p_SettleNo || ']的预付款金额不符，请检查并重新核销！');
    End If;*/
    For Vr In (Select * From SupPrePay_Usage Where SppuSetNo = p_SettleNo) Loop
      Update SupPrePay
         Set Sppye = Sppye - Vr.Sppuhxmoney
       Where SppBillNo = Vr.Sppubillno Return Sppye Into l_ye;
      update SupPrePay_Usage
         set sppuflag  = 'Y',
             sppuhxye  = l_ye,
             sppuvchr1 = to_char(sysdate, 'YYYY-MM-DD')
       where sppubillno = vr.sppubillno
         and sppusetno = vr.sppusetno;
    
      If l_ye < 0 Then
        Raise_Application_Error(ErrCode,
                                '预付款单[' || Vr.Sppubillno ||
                                ']的余额不足，请检查并重新核销！');
      End If;
    End Loop;
    Settlesup.Gensettlepaydet(p_Settleno);
    commit;
  Exception
    When Others Then
      Rollback;
      Raise;
  End Settle_hx;
