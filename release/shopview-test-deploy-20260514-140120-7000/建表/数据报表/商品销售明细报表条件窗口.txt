  Select class2 as "ВҘІг",
                      spbm as "ҝвЗш" ,
                     sglmfid as "№сЧй",
                     sglsupid as "№©УҰЙМ",
                     sglgdid as "ЙМЖ·ұаВл",
                     sglbarcode as "ЙМЖ·МхВл",
                      gbcname as "ЙМЖ·ГыіЖ" ,
                     sglbasekl as "ФӯҝЫВК",
                     kl as "ПъКЫҝЫВК",
                     yhkl as "УЕ»ЭҝЫВК",
                     sum(rkje) as "ИГҝЫҪр¶о",
                     sum(JGLSL) as "ПъКЫКэБҝ", 
                      sum(sjje) as "КЫјЫҪр¶о",
                       sum(JGLN3) as "ПъКЫКХИл" ,
                      sum(ML) as "Г«Аы",
                     decode(sum(JGLN3),0,0,round(sum(ML)/sum(JGLN3),4)) as "Г«АыВК",
                      sum(XSJEJE) as "ПъКЫҫ»¶о", 
                      sum(NETML) as "ҫ»Г«Аы",
                      decode(sum(XSJEJE),0,0,round(sum(NETML)/sum(XSJEJE),4))  as "ҫ»Г«АыВК" ,
                     sum(xscb) as "ПъКЫіЙұҫ",
                      sum(jxscb) as "ҫ»ПъКЫіЙұҫ",
                      sum(JGLN4) as "ЧЬХЫҝЫ", 
                       sum(JGLZK1) as "»бФұХЫҝЫ",
                      sum(JGLZK2) as "ҙЩПъХЫҝЫ",
                       sum(JGLZK3) as "КЪИЁХЫҝЫ" ,
                      sum(JGLZK4) as "ЖдЛыХЫҝЫ"
                FROM (SELECT VIEW_MFRAME_ALL.CLASS2,
       VIEW_MFRAME_ALL.SPBM,
       SGLMFID,
       SGLSUPID,
       SGLGDID,
       SGLBARCODE,
       GOODSBASE.GBCNAME,
       SGLBILLNO,SGLROWNO,
       SUM(SGLSL) JGLSL,sglbasekl,sglkl kl,decode(sglkl,sglbasekl,0,sglkl) yhkl,sum(sglsjje*(sglbasekl-sglkl))   rkje,
       sum(sglsjje)sjje,
       SUM(SGLXSSR) JGLN3, 
       sum(case   when sglwmid = '1' then sglxssr-nvl(sgln13,0)+(case when sglxssr = '0' and (sgln14  is null or sgln14 = '0') then sgln2 else sgln14 end) else SGLN2 end) ML,
       DECODE(  SUM(SGLXSSR) ,0,0,sum(case   when sglwmid = '1' then sglxssr-nvl(sgln13,0)+(case when sglxssr = '0' and (sgln14  is null or sgln14 = '0') then sgln2 else sgln14 end) else SGLN2 end) /  SUM(SGLXSSR) )SGLKL,
       SUM(sglxssr-sglgcert+sglsysy) XSJEJE,  
       SUM((case   when sglwmid = '1' then sglxssr-nvl(sgln13,0)+(case when sglxssr = '0' and (sgln14  is null or sgln14 = '0') then sgln2 else sgln14 end) else SGLN2 end)-sglgcert+sglsysy) NETML,   
       DECODE(SUM(sglxssr-sglgcert+sglsysy), 0, 0, SUM((case   when sglwmid = '1' then sglxssr-nvl(sgln13,0)+(case when sglxssr = '0' and (sgln14  is null or sgln14 = '0') then sgln2 else sgln14 end) else SGLN2 end)-sglgcert+sglsysy)/ SUM(sglxssr-sglgcert+sglsysy)) JMLL,
       sum(nvl(sgln13,0)+nvl(sgln14,0)) xscb ,
sum((case   when sglwmid in '4,5' then sglxssr*(1-sglkl) else nvl(sgln13,0)+nvl(sgln14,0) end)) jxscb,
       SUM(SGLTOTZK) JGLN4,
       SUM(SGLCUSTZK) JGLZK1,
       SUM(SGLPOPZK) JGLZK2,
       SUM(SGLGRANTZK) JGLZK3,
       SUM(SGLTOTZK-SGLCUSTZK-SGLPOPZK-SGLGRANTZK) JGLZK4,
       SUM(SGLSYSY) JGLN1  
FROM SALEGOODSLIST,GOODSBASE,VIEW_MFRAME_ALL
WHERE  GOODSBASE.GBID=SALEGOODSLIST.SGLGDID  AND
      VIEW_MFRAME_ALL.CLASS3=SALEGOODSLIST.SGLMFID AND
    
         SGLDATE>= TO_DATE('2026-05-01','YYYY-MM-DD')   AND  SGLDATE < TO_DATE('2026-05-08','YYYY-MM-DD')+1 AND
          sglsjje<>0 and
 'SALEGOODSLIST.SGLHSRQ'='SALEGOODSLIST.SGLHSRQ' AND
      'VIEW_MFRAME_ALL.CLASS2'='VIEW_MFRAME_ALL.CLASS2' AND
      'SALEGOODSLIST.SGLWMID'='SALEGOODSLIST.SGLWMID' AND
      'VIEW_MFRAME_ALL.SPBM'='VIEW_MFRAME_ALL.SPBM' AND
      'VIEW_MFRAME_ALL.CLASS3'='VIEW_MFRAME_ALL.CLASS3' AND
      'SALEGOODSLIST.SGLGDID'='SALEGOODSLIST.SGLGDID' AND
 'SALEGOODSLIST.SGLSUPID'='SALEGOODSLIST.SGLSUPID' 
GROUP BY VIEW_MFRAME_ALL.CLASS2,
         VIEW_MFRAME_ALL.SPBM,
         SGLMFID,
        SGLSUPID ,
         SGLGDID,
         SGLBARCODE,
         GOODSBASE.GBCNAME,SGLBASEKL,SGLKL,SGLBILLNO,SGLROWNO) TA,
         (SELECT SPGBILLNO,SPGgdROW  ,ROM SELLPAYGOODS,SELLHEAD
WHERE SPGBILLNO=SHBILLNO AND SHDATE>= TO_DATE('2026-05-01','YYYY-MM-DD')   AND  SHDATE < TO_DATE('2026-05-08','YYYY-MM-DD')+1
GROUP BY SPGBILLNO,SPGgdROW) TB 
WHERE TA.SGLBILLNO=TB.SPGBILLNO(+) AND TA.SGLROWNO=TB.SPGgdROW
group by class2,spbm,sglmfid,sglsupid,sglgdid ,sglbarcode ,gbcname, sglbasekl,kl,yhkl
